    34  func main() {
    35  	if len(os.Args) != 3 {
    36  		fmt.Println("migrator <old-yaml> <new-yaml>")
    37  		os.Exit(1)
    38  	}
    39  	data, err := ioutil.ReadFile(os.Args[1])
    40  	if err != nil {
    41  		fmt.Println(err)
    42  		os.Exit(1)
    43  	}
    44  	var (
    45  		old Item
    46  		new Item
    47  	)
    48  	err = yaml.Unmarshal(data, &old)
    49  	if err != nil {
    50  		fmt.Println(err)
    51  		os.Exit(1)
    52  	}
    53  	{
    54  		var (
    55  			schedule     yaml.MapSlice
    56  			findSchedule bool
    57  		)
    58  
    59  		if isIncompatibleChaos(old) {
    60  			fmt.Printf("the define of %s is changed in v2.0, please modify it according to the document, refer to https://chaos-mesh.org/docs/\n", old.Kind)
    61  			os.Exit(1)
    62  		}
    63  
    64  		for _, item := range old.Spec {
    65  			if item.Key == "scheduler" {
    66  				schedule = item.Value.(yaml.MapSlice)
    67  				findSchedule = true
    68  			}
    69  		}
    70  
    71  		if findSchedule {
    72  			new = toScheduleObject(old, schedule)
    73  		} else {
    74  			new = old
    75  			new.Spec = transformChaosSpec(old)
    76  		}
    77  	}
    78  	data, err = yaml.Marshal(new)
    79  	if err != nil {
    80  		fmt.Println(err)
    81  		os.Exit(1)
    82  	}
    83  	err = ioutil.WriteFile(os.Args[2], data, 0644)
    84  	if err != nil {
    85  		fmt.Println(err)
    86  		os.Exit(1)
    87  	}
    88  }

   102  func toScheduleObject(old Item, schedule yaml.MapSlice) Item {
   103  	var new Item
   104  	var cron string
   105  	for _, item := range schedule {
   106  		if item.Key == "cron" {
   107  			cron = item.Value.(string)
   108  		}
   109  	}
   110  	new.ApiVersion = old.ApiVersion
   111  	new.Metadata = old.Metadata
   112  	new.Kind = "Schedule"
   113  	new.Spec = append(new.Spec, yaml.MapItem{Key: "schedule", Value: cron})
   114  	new.Spec = append(new.Spec, yaml.MapItem{Key: "type", Value: toNewKind(old.Kind)})
   115  	new.Spec = append(new.Spec, yaml.MapItem{Key: "historyLimit", Value: 5})
   116  	new.Spec = append(new.Spec, yaml.MapItem{Key: "concurrencyPolicy", Value: "Forbid"})
   117  
   118  	newSpec := transformChaosSpec(old)
   119  	new.Spec = append(new.Spec, yaml.MapItem{Key: getKeyName(old.Kind), Value: newSpec})
   120  	return new
   121  }

   123  func transformChaosSpec(item Item) yaml.MapSlice {
   124  	var (
   125  		newSpec        yaml.MapSlice
   126  		containerNames []string
   127  	)
   128  	for _, kv := range item.Spec {
   129  		if kv.Key == "scheduler" {
   130  			continue
   131  		}
   132  
   133  		if kv.Key == "containerName" {
   134  			containerNames = append(containerNames, kv.Value.(string))
   135  			continue
   136  		}
   137  
   138  		if item.Kind == "DNSChaos" {
   139  			// 'scope' is obsolete in v2.0, and instead with 'patterns'
   140  			// if 'scope' is 'all', means chaos applies to all the host
   141  			// so it equal to pattern "*". otherwise, can't transform.
   142  			if kv.Key == "scope" && kv.Value == "all" {
   143  				patterns := []string{"*"}
   144  				kv = yaml.MapItem{Key: "patterns", Value: patterns}
   145  			}
   146  		}
   147  
   148  		newSpec = append(newSpec, kv)
   149  	}
   150  
   151  	if len(containerNames) != 0 {
   152  		newSpec = append(newSpec, yaml.MapItem{Key: "containerNames", Value: containerNames})
   153  	}
   154  
   155  	return newSpec
   156  }

   158  func isIncompatibleChaos(item Item) bool {
   159  	incompatible := false
   160  	switch item.Kind {
   161  	case "DNSChaos":
   162  		for _, kv := range item.Spec {
   163  			// 'scope' is obsolete in v2.0, and instead with 'patterns'
   164  			// if 'scope' is 'all', means chaos applies to all the host
   165  			// so it equal to pattern "*". otherwise, can't transform.
   166  			if kv.Key == "scope" && kv.Value != "all" {
   167  				incompatible = true
   168  			}
   169  		}
   170  	default:
   171  	}
   172  
   173  	return incompatible
   174  }
