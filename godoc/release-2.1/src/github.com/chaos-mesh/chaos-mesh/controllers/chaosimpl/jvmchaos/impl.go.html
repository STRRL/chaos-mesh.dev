    68  // Apply applies jvm-chaos
    69  func (impl *Impl) Apply(ctx context.Context, index int, records []*v1alpha1.Record, obj v1alpha1.InnerObject) (v1alpha1.Phase, error) {
    70  	impl.Log.Info("jvm chaos apply", "record", records[index])
    71  	if impl.decoder == nil {
    72  		return v1alpha1.NotInjected, fmt.Errorf("impl decoder is nil")
    73  	}
    74  	decodedContainer, err := impl.decoder.DecodeContainerRecord(ctx, records[index], obj)
    75  	if decodedContainer.PbClient != nil {
    76  		defer func() {
    77  			err := decodedContainer.PbClient.Close()
    78  			if err != nil {
    79  				impl.Log.Error(err, "fail to close pb client")
    80  			}
    81  		}()
    82  	}
    83  	if err != nil {
    84  		return v1alpha1.NotInjected, err
    85  	}
    86  
    87  	jvmChaos := obj.(*v1alpha1.JVMChaos)
    88  	err = generateRuleData(&jvmChaos.Spec)
    89  	if err != nil {
    90  		return v1alpha1.Injected, err
    91  	}
    92  
    93  	_, err = decodedContainer.PbClient.InstallJVMRules(ctx, &pb.InstallJVMRulesRequest{
    94  		ContainerId: decodedContainer.ContainerId,
    95  		Rule:        jvmChaos.Spec.RuleData,
    96  		Port:        jvmChaos.Spec.Port,
    97  		EnterNS:     true,
    98  	})
    99  	if err != nil {
   100  		impl.Log.Error(err, "install jvm rules")
   101  		return v1alpha1.NotInjected, err
   102  	}
   103  
   104  	return v1alpha1.Injected, nil
   105  }

   109  	if impl.decoder == nil {

   111  	}
   112  	decodedContainer, err := impl.decoder.DecodeContainerRecord(ctx, records[index], obj)
   113  	if decodedContainer.PbClient != nil {
   114  		defer func() {
   115  			err := decodedContainer.PbClient.Close()
   116  			if err != nil {
   117  				impl.Log.Error(err, "fail to close pb client")
   118  			}
   119  		}()
   120  	}
   121  	if err != nil {
   122  		return v1alpha1.Injected, err
   123  	}
   124  
   125  	jvmChaos := obj.(*v1alpha1.JVMChaos)

   128  		return v1alpha1.Injected, err

   137  	if err != nil {

   139  		return v1alpha1.Injected, err
   140  	}
   141  
   142  	return v1alpha1.NotInjected, nil
   143  }

   155  func generateRuleData(spec *v1alpha1.JVMChaosSpec) error {
   156  	if len(spec.RuleData) != 0 {
   157  		return nil
   158  	}
   159  
   160  	ruleParameter := &JVMRuleParameter{
   161  		JVMParameter: spec.JVMParameter,
   162  	}
   163  
   164  	switch spec.Action {
   165  	case v1alpha1.JVMLatencyAction:
   166  		ruleParameter.Do = fmt.Sprintf("Thread.sleep(%d)", ruleParameter.LatencyDuration)
   167  	case v1alpha1.JVMExceptionAction:
   168  		ruleParameter.Do = fmt.Sprintf("throw new %s", ruleParameter.ThrowException)
   169  	case v1alpha1.JVMReturnAction:
   170  		ruleParameter.Do = fmt.Sprintf("return %s", ruleParameter.ReturnValue)
   171  	case v1alpha1.JVMStressAction:
   172  		if ruleParameter.CPUCount > 0 {
   173  			ruleParameter.StressType = "CPU"
   174  			ruleParameter.StressValueName = "CPUCOUNT"
   175  			ruleParameter.StressValue = fmt.Sprintf("%d", ruleParameter.CPUCount)
   176  		} else {
   177  			ruleParameter.StressType = "MEMORY"
   178  			ruleParameter.StressValueName = "MEMORYTYPE"
   179  			ruleParameter.StressValue = ruleParameter.MemoryType
   180  		}
   181  	}
   182  
   183  	buf := new(bytes.Buffer)
   184  	var t *template.Template
   185  	switch spec.Action {
   186  	case v1alpha1.JVMStressAction:
   187  		t = template.Must(template.New("byteman rule").Parse(StressRuleTemplate))
   188  	case v1alpha1.JVMExceptionAction, v1alpha1.JVMLatencyAction, v1alpha1.JVMReturnAction:
   189  		t = template.Must(template.New("byteman rule").Parse(CommonRuleTemplate))
   190  	case v1alpha1.JVMGCAction:
   191  		t = template.Must(template.New("byteman rule").Parse(GcRuleTemplate))
   192  	default:
   193  		return errors.Errorf("jvm action %s not supported", spec.Action)
   194  	}
   195  	if t == nil {
   196  		return errors.Errorf("parse byeman rule template failed")
   197  	}
   198  	err := t.Execute(buf, ruleParameter)
   199  	if err != nil {
   200  		return err
   201  	}
   202  
   203  	spec.RuleData = buf.String()
   204  	return nil
   205  }
