    50  func (impl *Impl) Apply(ctx context.Context, index int, records []*v1alpha1.Record, obj v1alpha1.InnerObject) (v1alpha1.Phase, error) {
    51  	kernelChaos := obj.(*v1alpha1.KernelChaos)
    52  	record := records[index]
    53  
    54  	log := impl.Log.WithValues("chaos", kernelChaos, "record", record)
    55  	podId, err := controller.ParseNamespacedName(record.Id)
    56  	if err != nil {
    57  		return v1alpha1.NotInjected, err
    58  	}
    59  	var pod v1.Pod
    60  	err = impl.Client.Get(ctx, podId, &pod)
    61  	if err != nil {
    62  		log.Error(err, "fail to get pod by record")
    63  		// TODO: handle this error
    64  		if k8sError.IsNotFound(err) {
    65  			return v1alpha1.NotInjected, nil
    66  		}
    67  		return v1alpha1.NotInjected, err
    68  	}
    69  
    70  	log = log.WithValues("pod", pod)
    71  
    72  	if err = impl.applyPod(ctx, &pod, kernelChaos); err != nil {
    73  		log.Error(err, "failed to apply chaos on pod")
    74  		return v1alpha1.NotInjected, err
    75  	}
    76  
    77  	return v1alpha1.Injected, nil

   114  func (impl *Impl) recoverPod(ctx context.Context, pod *v1.Pod, somechaos v1alpha1.InnerObject) error {
   115  	// judged type in `Recover` already so no need to judge again
   116  	chaos, _ := somechaos.(*v1alpha1.KernelChaos)
   117  	impl.Log.Info("try to recover pod", "namespace", pod.Namespace, "name", pod.Name)
   118  
   119  	pbClient, err := impl.chaosDaemonClientBuilder.Build(ctx, pod, &types.NamespacedName{
   120  		Namespace: chaos.Namespace,
   121  		Name:      chaos.Name,
   122  	})
   123  	if err != nil {
   124  		return err
   125  	}
   126  	defer pbClient.Close()
   127  
   128  	if len(pod.Status.ContainerStatuses) == 0 {
   129  		return fmt.Errorf("%s %s can't get the state of container", pod.Namespace, pod.Name)
   130  	}
   131  
   132  	containerResponse, err := pbClient.ContainerGetPid(ctx, &pb.ContainerRequest{
   133  		Action: &pb.ContainerAction{
   134  			Action: pb.ContainerAction_GETPID,
   135  		},
   136  		ContainerId: pod.Status.ContainerStatuses[0].ContainerID,
   137  	})
   138  
   139  	if err != nil {
   140  		impl.Log.Error(err, "Get container pid error", "namespace", pod.Namespace, "name", pod.Name)
   141  		return err
   142  	}
   143  
   144  	impl.Log.Info("Get container pid", "namespace", pod.Namespace, "name", pod.Name)
   145  	conn, err := impl.CreateBPFKIConnection(ctx, impl.Client, pod)
   146  	if err != nil {
   147  		return err
   148  	}
   149  	defer conn.Close()
   150  
   151  	var callchain []*pb_kernel.FailKernRequestFrame
   152  	for _, frame := range chaos.Spec.FailKernRequest.Callchain {
   153  		callchain = append(callchain, &pb_kernel.FailKernRequestFrame{
   154  			Funcname:   frame.Funcname,
   155  			Parameters: frame.Parameters,
   156  			Predicate:  frame.Predicate,
   157  		})
   158  	}
   159  
   160  	bpfClient := pb_kernel.NewBPFKIServiceClient(conn)
   161  	_, err = bpfClient.RecoverMMOrBIO(ctx, &pb_kernel.FailKernRequest{
   162  		Pid:       containerResponse.Pid,
   163  		Callchain: callchain,
   164  	})
   165  
   166  	return err
   167  }

   169  func (impl *Impl) applyPod(ctx context.Context, pod *v1.Pod, chaos *v1alpha1.KernelChaos) error {
   170  	impl.Log.Info("Try to inject kernel on pod", "namespace", pod.Namespace, "name", pod.Name)
   171  
   172  	pbClient, err := impl.chaosDaemonClientBuilder.Build(ctx, pod, &types.NamespacedName{
   173  		Namespace: chaos.Namespace,
   174  		Name:      chaos.Name,
   175  	})
   176  	if err != nil {
   177  		return err
   178  	}
   179  	defer pbClient.Close()
   180  
   181  	if len(pod.Status.ContainerStatuses) == 0 {
   182  		return fmt.Errorf("%s %s can't get the state of container", pod.Namespace, pod.Name)
   183  	}
   184  
   185  	containerResponse, err := pbClient.ContainerGetPid(ctx, &pb.ContainerRequest{
   186  		Action: &pb.ContainerAction{
   187  			Action: pb.ContainerAction_GETPID,
   188  		},
   189  		ContainerId: pod.Status.ContainerStatuses[0].ContainerID,
   190  	})
   191  	if err != nil {
   192  		impl.Log.Error(err, "Get container pid error", "namespace", pod.Namespace, "name", pod.Name)
   193  		return err
   194  	}
   195  
   196  	impl.Log.Info("Get container pid", "namespace", pod.Namespace, "name", pod.Name)
   197  	conn, err := impl.CreateBPFKIConnection(ctx, impl.Client, pod)
   198  	if err != nil {
   199  		return err
   200  	}
   201  	defer conn.Close()
   202  
   203  	var callchain []*pb_kernel.FailKernRequestFrame
   204  	for _, frame := range chaos.Spec.FailKernRequest.Callchain {
   205  		callchain = append(callchain, &pb_kernel.FailKernRequestFrame{
   206  			Funcname:   frame.Funcname,
   207  			Parameters: frame.Parameters,
   208  			Predicate:  frame.Predicate,
   209  		})
   210  	}
   211  
   212  	bpfClient := pb_kernel.NewBPFKIServiceClient(conn)
   213  	_, err = bpfClient.FailMMOrBIO(ctx, &pb_kernel.FailKernRequest{
   214  		Pid:         containerResponse.Pid,
   215  		Ftype:       pb_kernel.FailKernRequest_FAILTYPE(chaos.Spec.FailKernRequest.FailType),
   216  		Headers:     chaos.Spec.FailKernRequest.Headers,
   217  		Callchain:   callchain,
   218  		Probability: float32(chaos.Spec.FailKernRequest.Probability) / 100,
   219  		Times:       chaos.Spec.FailKernRequest.Times,
   220  	})
   221  
   222  	return err
   223  }

   225  // CreateBPFKIConnection create a grpc connection with bpfki
   226  func (impl *Impl) CreateBPFKIConnection(ctx context.Context, c client.Client, pod *v1.Pod) (*grpc.ClientConn, error) {
   227  	daemonIP, err := impl.chaosDaemonClientBuilder.FindDaemonIP(ctx, pod)
   228  	if err != nil {
   229  		return nil, err
   230  	}
   231  	builder := grpcUtils.Builder(daemonIP, config.ControllerCfg.BPFKIPort).
   232  		WithDefaultTimeout().
   233  		Insecure()
   234  	return builder.Build()
   235  }
