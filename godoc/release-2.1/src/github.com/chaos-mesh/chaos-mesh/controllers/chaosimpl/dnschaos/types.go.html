    48  func (impl *Impl) Apply(ctx context.Context, index int, records []*v1alpha1.Record, obj v1alpha1.InnerObject) (v1alpha1.Phase, error) {
    49  	decodedContainer, err := impl.decoder.DecodeContainerRecord(ctx, records[index], obj)
    50  	if decodedContainer.PbClient != nil {
    51  		defer decodedContainer.PbClient.Close()
    52  	}
    53  	if err != nil {
    54  		return v1alpha1.NotInjected, err
    55  	}
    56  
    57  	service, err := pod.GetService(ctx, impl.Client, "", config.ControllerCfg.Namespace, config.ControllerCfg.DNSServiceName)
    58  	if err != nil {
    59  		impl.Log.Error(err, "fail to get service")
    60  		return v1alpha1.NotInjected, err
    61  	}
    62  
    63  	dnschaos := obj.(*v1alpha1.DNSChaos)
    64  	err = impl.setDNSServerRules(service.Spec.ClusterIP, config.ControllerCfg.DNSServicePort, dnschaos.Name, decodedContainer.Pod, dnschaos.Spec.Action, dnschaos.Spec.DomainNamePatterns)
    65  	if err != nil {
    66  		impl.Log.Error(err, "fail to set DNS server rules")
    67  		return v1alpha1.NotInjected, err
    68  	}
    69  
    70  	_, err = decodedContainer.PbClient.SetDNSServer(ctx, &pb.SetDNSServerRequest{
    71  		ContainerId: decodedContainer.ContainerId,
    72  		DnsServer:   service.Spec.ClusterIP,
    73  		Enable:      true,
    74  		EnterNS:     true,
    75  	})
    76  	if err != nil {
    77  		impl.Log.Error(err, "set dns server")
    78  		return v1alpha1.NotInjected, err
    79  	}
    80  
    81  	return v1alpha1.Injected, nil
    82  }

    84  func (impl *Impl) setDNSServerRules(dnsServerIP string, port int, name string, pod *v1.Pod, action v1alpha1.DNSChaosAction, patterns []string) error {
    85  	impl.Log.Info("setDNSServerRules", "name", name)
    86  
    87  	pbPods := make([]*dnspb.Pod, 1)
    88  	pbPods[0] = &dnspb.Pod{
    89  		Name:      pod.Name,
    90  		Namespace: pod.Namespace,
    91  	}
    92  
    93  	conn, err := grpc.Dial(net.JoinHostPort(dnsServerIP, fmt.Sprintf("%d", port)), grpc.WithInsecure())
    94  	if err != nil {
    95  		return err
    96  	}
    97  	defer conn.Close()
    98  
    99  	c := dnspb.NewDNSClient(conn)
   100  	request := &dnspb.SetDNSChaosRequest{
   101  		Name:     name,
   102  		Action:   string(action),
   103  		Pods:     pbPods,
   104  		Patterns: patterns,
   105  	}
   106  
   107  	ctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
   108  	defer cancel()
   109  	response, err := c.SetDNSChaos(ctx, request)
   110  	if err != nil {
   111  		return err
   112  	}
   113  
   114  	if !response.Result {
   115  		return fmt.Errorf("set dns chaos to dns server error %s", response.Msg)
   116  	}
   117  
   118  	return nil
   119  }

   121  func (impl *Impl) Recover(ctx context.Context, index int, records []*v1alpha1.Record, obj v1alpha1.InnerObject) (v1alpha1.Phase, error) {
   122  	decodedContainer, err := impl.decoder.DecodeContainerRecord(ctx, records[index], obj)
   123  	if decodedContainer.PbClient != nil {
   124  		defer decodedContainer.PbClient.Close()
   125  	}
   126  	if err != nil {
   127  		if utils.IsFailToGet(err) {
   128  			// pretend the disappeared container has been recovered
   129  			return v1alpha1.NotInjected, nil
   130  		}
   131  		return v1alpha1.Injected, err
   132  	}
   133  
   134  	dnschaos := obj.(*v1alpha1.DNSChaos)
   135  
   136  	// get dns server's ip used for chaos
   137  	service, err := pod.GetService(ctx, impl.Client, "", config.ControllerCfg.Namespace, config.ControllerCfg.DNSServiceName)
   138  	if err != nil {
   139  		impl.Log.Error(err, "fail to get service")
   140  		return v1alpha1.Injected, err
   141  	}
   142  	impl.Log.Info("Cancel DNS chaos to DNS service", "ip", service.Spec.ClusterIP)
   143  
   144  	err = impl.cancelDNSServerRules(service.Spec.ClusterIP, config.ControllerCfg.DNSServicePort, dnschaos.Name)
   145  	if err != nil {
   146  		impl.Log.Error(err, "fail to cancelDNSServerRules")
   147  		return v1alpha1.Injected, err
   148  	}
   149  
   150  	_, err = decodedContainer.PbClient.SetDNSServer(ctx, &pb.SetDNSServerRequest{
   151  		ContainerId: decodedContainer.ContainerId,
   152  		Enable:      false,
   153  		EnterNS:     true,
   154  	})
   155  	if err != nil {
   156  		impl.Log.Error(err, "recover pod for DNS chaos")
   157  		return v1alpha1.Injected, err
   158  	}
   159  
   160  	return v1alpha1.NotInjected, err
   161  }

   163  func (impl *Impl) cancelDNSServerRules(dnsServerIP string, port int, name string) error {
   164  	conn, err := grpc.Dial(net.JoinHostPort(dnsServerIP, fmt.Sprintf("%d", port)), grpc.WithInsecure())
   165  	if err != nil {
   166  		return err
   167  	}
   168  	defer conn.Close()
   169  
   170  	c := dnspb.NewDNSClient(conn)
   171  	request := &dnspb.CancelDNSChaosRequest{
   172  		Name: name,
   173  	}
   174  
   175  	ctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
   176  	defer cancel()
   177  	response, err := c.CancelDNSChaos(ctx, request)
   178  	if err != nil {
   179  		return err
   180  	}
   181  
   182  	if !response.Result {
   183  		return fmt.Errorf("set dns chaos to dns server error %s", response.Msg)
   184  	}
   185  
   186  	return nil
   187  }
