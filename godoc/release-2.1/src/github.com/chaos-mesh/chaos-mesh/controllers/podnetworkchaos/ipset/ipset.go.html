    34  // BuildIPSet builds an ipset with provided pod ip list
    35  func BuildIPSet(pods []v1.Pod, externalCidrs []string, networkchaos *v1alpha1.NetworkChaos, namePostFix string, source string) v1alpha1.RawIPSet {
    36  	name := GenerateIPSetName(networkchaos, namePostFix)
    37  	cidrs := externalCidrs
    38  
    39  	for _, pod := range pods {
    40  		if len(pod.Status.PodIP) > 0 {
    41  			cidrs = append(cidrs, netutils.IPToCidr(pod.Status.PodIP))
    42  		}
    43  	}
    44  
    45  	return v1alpha1.RawIPSet{
    46  		Name:  name,
    47  		Cidrs: cidrs,
    48  		RawRuleSource: v1alpha1.RawRuleSource{
    49  			Source: source,
    50  		},
    51  	}
    52  }

    59  // FlushIPSets makes grpc calls to chaosdaemon to save ipset
    60  func FlushIPSets(ctx context.Context, pbClient chaosdaemonclient.ChaosDaemonClientInterface, pod *v1.Pod, ipsets []*pb.IPSet) error {
    61  	var err error
    62  
    63  	if len(pod.Status.ContainerStatuses) == 0 {
    64  		return fmt.Errorf("%s %s can't get the state of container", pod.Namespace, pod.Name)
    65  	}
    66  
    67  	log.Info("Flushing IP Sets....")
    68  	for _, containerStatus := range pod.Status.ContainerStatuses {
    69  		containerID := containerStatus.ContainerID
    70  		log.Info("attempting to flush ip set", "containerID", containerID)
    71  
    72  		_, err = pbClient.FlushIPSets(ctx, &pb.IPSetsRequest{
    73  			Ipsets:      ipsets,
    74  			ContainerId: containerID,
    75  			EnterNS:     true,
    76  		})
    77  
    78  		if err != nil {
    79  			log.Error(err, fmt.Sprintf("error while flushing ip sets for containerID %s", containerID))
    80  		} else {
    81  			log.Info("Successfully flushed ip set")
    82  			return nil
    83  		}
    84  	}
    85  
    86  	return fmt.Errorf("unable to flush ip sets for pod %s", pod.Name)
    87  }
