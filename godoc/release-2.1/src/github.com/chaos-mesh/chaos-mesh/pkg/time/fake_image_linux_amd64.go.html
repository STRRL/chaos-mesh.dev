    28  // AttachToProcess would use ptrace to replace the VDSO ELF entry with FakeImage.
    29  // Each item in parameter "variables" needs a corresponding entry in FakeImage.offset.
    30  func (it *FakeImage) AttachToProcess(pid int, variables map[string]uint64) error {
    31  	if len(variables) != len(it.offset) {
    32  		return errors.New("fake image: extern variable number not match")
    33  	}
    34  
    35  	runtime.LockOSThread()
    36  	defer func() {
    37  		runtime.UnlockOSThread()
    38  	}()
    39  
    40  	program, err := ptrace.Trace(pid)
    41  	if err != nil {
    42  		return errors.Wrapf(err, "ptrace on target process, pid: %d", pid)
    43  	}
    44  	defer func() {
    45  		err = program.Detach()
    46  		if err != nil {
    47  			it.logger.Error(err, "fail to detach program", "pid", program.Pid())
    48  		}
    49  	}()
    50  
    51  	var vdsoEntry *mapreader.Entry
    52  	for index := range program.Entries {
    53  		// reverse loop is faster
    54  		e := program.Entries[len(program.Entries)-index-1]
    55  		if e.Path == vdsoEntryName {
    56  			vdsoEntry = &e
    57  			break
    58  		}
    59  	}
    60  	if vdsoEntry == nil {
    61  		return errors.Errorf("cannot find [vdso] entry, pid: %d", pid)
    62  	}
    63  
    64  	// minus tailing variable part
    65  	// every variable has 8 bytes
    66  	constImageLen := len(it.content) - 8*len(it.offset)
    67  	var fakeEntry *mapreader.Entry
    68  
    69  	// find injected image to avoid redundant inject (which will lead to memory leak)
    70  	for _, e := range program.Entries {
    71  		e := e
    72  
    73  		image, err := program.ReadSlice(e.StartAddress, uint64(constImageLen))
    74  		if err != nil {
    75  			continue
    76  		}
    77  
    78  		if bytes.Equal(*image, it.content[0:constImageLen]) {
    79  			fakeEntry = &e
    80  			it.logger.Info("found injected image", "addr", fakeEntry.StartAddress, "pid", pid)
    81  			break
    82  		}
    83  	}
    84  
    85  	// target process has not been injected yet
    86  	if fakeEntry == nil {
    87  		fakeEntry, err = program.MmapSlice(it.content)
    88  		if err != nil {
    89  			return errors.Wrapf(err, "mmap fake image, pid: %d", pid)
    90  		}
    91  
    92  		originAddr, err := program.FindSymbolInEntry(it.symbolName, vdsoEntry)
    93  		if err != nil {
    94  			return errors.Wrapf(err, "find origin clock_gettime in vdso, pid: %d", pid)
    95  		}
    96  
    97  		err = program.JumpToFakeFunc(originAddr, fakeEntry.StartAddress)
    98  		if err != nil {
    99  			return errors.Wrapf(err, "override origin clock_gettime, pid: %d", pid)
   100  		}
   101  	}
   102  
   103  	for k, v := range variables {
   104  		if offset, ok := it.offset[k]; ok {
   105  			err = program.WriteUint64ToAddr(fakeEntry.StartAddress+uint64(offset), v)
   106  			if err != nil {
   107  				return errors.Wrapf(err, "set %s for time skew, pid: %d", k, pid)
   108  			}
   109  		} else {
   110  			return errors.Errorf("no such extern variable in fake image: %s", k)
   111  		}
   112  	}
   113  
   114  	return nil
   115  }
