    33  func componentLabels(component model.Component) map[string]string {
    34  	var componentLabel string
    35  	switch component {
    36  	case model.ComponentManager:
    37  		componentLabel = "controller-manager"
    38  	case model.ComponentDaemon:
    39  		componentLabel = "chaos-daemon"
    40  	case model.ComponentDashboard:
    41  		componentLabel = "chaos-dashboard"
    42  	case model.ComponentDNSServer:
    43  		componentLabel = "chaos-dns-server"
    44  	default:
    45  		return nil
    46  	}
    47  	return map[string]string{
    48  		"app.kubernetes.io/component": componentLabel,
    49  	}
    50  }

    52  func parseNamespacedName(namespacedName string) types.NamespacedName {
    53  	parts := strings.Split(namespacedName, "/")
    54  	return types.NamespacedName{
    55  		Namespace: parts[0],

    60  // getDaemonMap returns a map of node name to daemon pod
    61  func getDaemonMap(ctx context.Context, c client.Client) (map[string]v1.Pod, error) {
    62  	var list v1.PodList
    63  	labels := componentLabels(model.ComponentDaemon)
    64  	if err := c.List(ctx, &list, client.MatchingLabels(labels)); err != nil {
    65  		return nil, errors.Wrapf(err, "list daemons by label %v", labels)
    66  	}
    67  
    68  	daemonMap := map[string]v1.Pod{}
    69  	for _, d := range list.Items {
    70  		if d.Spec.NodeName != "" {
    71  			daemonMap[d.Spec.NodeName] = d
    72  		}
    73  	}
    74  	return daemonMap, nil
    75  }
