    40  func (s *Selector) Select(ctx context.Context, spec interface{}) ([]Target, error) {
    41  	if spec == nil {
    42  		return []Target{}, nil
    43  	}
    44  
    45  	var targets []Target
    46  	impl, ok := s.selectorMap[reflect.TypeOf(spec)]
    47  	if ok {
    48  		vals := reflect.ValueOf(impl).MethodByName("Select").Call([]reflect.Value{reflect.ValueOf(ctx), reflect.ValueOf(spec)})
    49  		ret := vals[0]
    50  
    51  		for i := 0; i < ret.Len(); i++ {
    52  			targets = append(targets, ret.Index(i).Interface().(Target))
    53  		}
    54  
    55  		err := vals[1].Interface()
    56  		if err == nil {
    57  			return targets, nil
    58  		}
    59  
    60  		return targets, err.(error)
    61  	}
    62  
    63  	return nil, errors.Errorf("specification type not found: %s", reflect.TypeOf(spec))
    64  }

    76  func New(p SelectorParams) *Selector {
    77  	selectorMap := make(map[reflect.Type]interface{})
    78  
    79  	val := reflect.ValueOf(p)
    80  	for i := 0; i < val.NumField(); i++ {
    81  		method := val.Field(i).MethodByName("Select")
    82  		if method.IsValid() && method.Type().NumIn() > 1 {
    83  			typ := method.Type().In(1)
    84  			selectorMap[typ] = val.Field(i).Interface()
    85  		}
    86  	}
    87  
    88  	return &Selector{
    89  		selectorMap,
    90  	}
    91  }
