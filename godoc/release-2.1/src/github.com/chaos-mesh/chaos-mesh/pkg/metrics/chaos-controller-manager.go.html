   134  func (collector *ChaosControllerManagerMetricsCollector) collectChaosExperiments() {
   135  	// TODO(yeya24) if there is an error in List
   136  	// the experiment status will be lost
   137  	collector.chaosExperiments.Reset()
   138  
   139  	for kind, obj := range v1alpha1.AllKinds() {
   140  		expCache := map[string]map[string]int{}
   141  		chaosList := obj.SpawnList()
   142  		if err := collector.store.List(context.TODO(), chaosList); err != nil {
   143  			collector.logger.Error(err, "failed to list chaos", "kind", kind)
   144  			return
   145  		}
   146  
   147  		items := chaosList.GetItems()
   148  		for _, item := range items {
   149  			if _, ok := expCache[item.GetNamespace()]; !ok {
   150  				// There is only 4 supported phases
   151  				expCache[item.GetNamespace()] = make(map[string]int, 4)
   152  			}
   153  			innerObject := reflect.ValueOf(item).Interface().(v1alpha1.InnerObject)
   154  			expCache[item.GetNamespace()][string(status.GetChaosStatus(innerObject))]++
   155  		}
   156  
   157  		for ns, v := range expCache {
   158  			for phase, count := range v {
   159  				collector.chaosExperiments.WithLabelValues(ns, kind, phase).Set(float64(count))
   160  			}
   161  		}
   162  	}
   163  }

   165  func (collector *ChaosControllerManagerMetricsCollector) collectChaosSchedules() {
   166  	collector.chaosSchedules.Reset()
   167  
   168  	schedules := &v1alpha1.ScheduleList{}
   169  	if err := collector.store.List(context.TODO(), schedules); err != nil {
   170  		collector.logger.Error(err, "failed to list schedules")
   171  		return
   172  	}
   173  
   174  	countByNamespace := make(map[string]int)
   175  	items := schedules.GetItems()
   176  	for _, item := range items {
   177  		countByNamespace[item.GetNamespace()]++
   178  	}
   179  
   180  	for namespace, count := range countByNamespace {
   181  		collector.chaosSchedules.WithLabelValues(namespace).Set(float64(count))
   182  	}
   183  }
