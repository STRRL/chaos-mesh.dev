    96  // StartProcess manages a process in manager
    97  func (m *BackgroundProcessManager) StartProcess(cmd *ManagedProcess) (*process.Process, error) {
    98  	var identifierLock *sync.Mutex
    99  	if cmd.Identifier != nil {
   100  		lock, _ := m.identifiers.LoadOrStore(*cmd.Identifier, &sync.Mutex{})
   101  
   102  		identifierLock = lock.(*sync.Mutex)
   103  
   104  		identifierLock.Lock()
   105  	}
   106  
   107  	err := cmd.Start()
   108  	if err != nil {
   109  		m.rootLogger.Error(err, "fail to start process")
   110  		return nil, err
   111  	}
   112  
   113  	pid := cmd.Process.Pid
   114  	procState, err := process.NewProcess(int32(cmd.Process.Pid))
   115  	if err != nil {
   116  		return nil, err
   117  	}
   118  	ct, err := procState.CreateTime()
   119  	if err != nil {
   120  		return nil, err
   121  	}
   122  
   123  	pair := ProcessPair{
   124  		Pid:        pid,
   125  		CreateTime: ct,
   126  	}
   127  
   128  	channel, _ := m.deathSig.LoadOrStore(pair, make(chan bool, 1))
   129  	deathChannel := channel.(chan bool)
   130  
   131  	stdio := &Stdio{Locker: &sync.Mutex{}}
   132  	if cmd.Stdin != nil {
   133  		if stdin, ok := cmd.Stdin.(io.ReadWriteCloser); ok {
   134  			stdio.Stdin = stdin
   135  		}
   136  	}
   137  
   138  	if cmd.Stdout != nil {
   139  		if stdout, ok := cmd.Stdout.(io.ReadWriteCloser); ok {
   140  			stdio.Stdout = stdout
   141  		}
   142  	}
   143  
   144  	if cmd.Stderr != nil {
   145  		if stderr, ok := cmd.Stderr.(io.ReadWriteCloser); ok {
   146  			stdio.Stderr = stderr
   147  		}
   148  	}
   149  
   150  	m.stdio.Store(pair, stdio)
   151  
   152  	logger := m.rootLogger.WithValues("pid", pid)
   153  
   154  	go func() {
   155  		err := cmd.Wait()
   156  		if err != nil {
   157  			if exitErr, ok := err.(*exec.ExitError); ok {
   158  				status := exitErr.Sys().(syscall.WaitStatus)
   159  				if status.Signaled() && status.Signal() == syscall.SIGTERM {
   160  					logger.Info("process stopped with SIGTERM signal")
   161  				}
   162  			} else {
   163  				logger.Error(err, "process exited accidentally")
   164  			}
   165  		}
   166  
   167  		logger.Info("process stopped")
   168  
   169  		deathChannel <- true
   170  		m.deathSig.Delete(pair)
   171  		if io, loaded := m.stdio.LoadAndDelete(pair); loaded {
   172  			if stdio, ok := io.(*Stdio); ok {
   173  				stdio.Lock()
   174  				if stdio.Stdin != nil {
   175  					if err = stdio.Stdin.Close(); err != nil {
   176  						logger.Error(err, "stdin fails to be closed")
   177  					}
   178  				}
   179  				if stdio.Stdout != nil {
   180  					if err = stdio.Stdout.Close(); err != nil {
   181  						logger.Error(err, "stdout fails to be closed")
   182  					}
   183  				}
   184  				if stdio.Stderr != nil {
   185  					if err = stdio.Stderr.Close(); err != nil {
   186  						logger.Error(err, "stderr fails to be closed")
   187  					}
   188  				}
   189  				stdio.Unlock()
   190  			}
   191  		}
   192  
   193  		if identifierLock != nil {
   194  			identifierLock.Unlock()
   195  			m.identifiers.Delete(*cmd.Identifier)
   196  		}
   197  	}()
   198  
   199  	return procState, nil
   200  }

   202  // KillBackgroundProcess sends SIGTERM to process
   203  func (m *BackgroundProcessManager) KillBackgroundProcess(ctx context.Context, pid int, startTime int64) error {
   204  	logger := m.rootLogger.WithValues("pid", pid)
   205  
   206  	p, err := os.FindProcess(int(pid))
   207  	if err != nil {
   208  		logger.Error(err, "unreachable path. `os.FindProcess` will never return an error on unix")
   209  		return err
   210  	}
   211  
   212  	procState, err := process.NewProcess(int32(pid))
   213  	if err != nil {
   214  		// return successfully as the process has exited
   215  		return nil
   216  	}
   217  	ct, err := procState.CreateTime()
   218  	if err != nil {
   219  		logger.Error(err, "fail to read create time")
   220  		// return successfully as the process has exited
   221  		return nil
   222  	}
   223  
   224  	// There is a bug in calculating CreateTime in the new version of
   225  	// gopsutils. This is a temporary solution before the upstream fixes it.
   226  	if startTime-ct > 1000 || ct-startTime > 1000 {
   227  		logger.Info("process has already been killed", "startTime", ct, "expectedStartTime", startTime)
   228  		// return successfully as the process has exited
   229  		return nil
   230  	}
   231  
   232  	ppid, err := procState.Ppid()
   233  	if err != nil {
   234  		logger.Error(err, "fail to read parent id")
   235  		// return successfully as the process has exited
   236  		return nil
   237  	}
   238  	if ppid != int32(os.Getpid()) {
   239  		logger.Info("process has already been killed", "ppid", ppid)
   240  		// return successfully as the process has exited
   241  		return nil
   242  	}
   243  
   244  	err = p.Signal(syscall.SIGTERM)
   245  
   246  	if err != nil && err.Error() != "os: process already finished" {
   247  		logger.Error(err, "error while killing process")
   248  		return err
   249  	}
   250  
   251  	pair := ProcessPair{
   252  		Pid:        pid,
   253  		CreateTime: startTime,
   254  	}
   255  	channel, ok := m.deathSig.Load(pair)
   256  	if ok {
   257  		deathChannel := channel.(chan bool)
   258  		select {
   259  		case <-deathChannel:
   260  		case <-ctx.Done():
   261  			return ctx.Err()
   262  		}
   263  	}
   264  
   265  	logger.Info("Successfully killed process")
   266  	return nil
   267  }

   269  func (m *BackgroundProcessManager) Stdio(pid int, startTime int64) *Stdio {
   270  	logger := m.rootLogger.WithValues("pid", pid)
   271  
   272  	procState, err := process.NewProcess(int32(pid))
   273  	if err != nil {
   274  		logger.Info("fail to get process information", "pid", pid)
   275  		// return successfully as the process has exited
   276  		return nil
   277  	}
   278  	ct, err := procState.CreateTime()
   279  	if err != nil {
   280  		logger.Error(err, "fail to read create time")
   281  		// return successfully as the process has exited
   282  		return nil
   283  	}
   284  
   285  	// There is a bug in calculating CreateTime in the new version of
   286  	// gopsutils. This is a temporary solution before the upstream fixes it.
   287  	if startTime-ct > 1000 || ct-startTime > 1000 {
   288  		logger.Info("process has exited", "startTime", ct, "expectedStartTime", startTime)
   289  		// return successfully as the process has exited
   290  		return nil
   291  	}
   292  
   293  	pair := ProcessPair{
   294  		Pid:        pid,
   295  		CreateTime: startTime,
   296  	}
   297  
   298  	io, ok := m.stdio.Load(pair)
   299  	if !ok {
   300  		logger.Info("fail to load with pair", "pair", pair)
   301  		// stdio is not stored
   302  		return nil
   303  	}
   304  
   305  	return io.(*Stdio)
   306  }
