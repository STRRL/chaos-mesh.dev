    33  func (s *DaemonServer) ExecStressors(ctx context.Context,
    34  	req *pb.ExecStressRequest) (*pb.ExecStressResponse, error) {
    35  	log := s.getLoggerFromContext(ctx)
    36  	log.Info("Executing stressors", "request", req)
    37  	pid, err := s.crClient.GetPidFromContainerID(ctx, req.Target)
    38  	if err != nil {
    39  		return nil, err
    40  	}
    41  	control, err := cgroups.Load(daemonCgroups.V1, daemonCgroups.PidPath(int(pid)))
    42  	if err != nil {
    43  		return nil, err
    44  	}
    45  
    46  	processBuilder := bpm.DefaultProcessBuilder("stress-ng", strings.Fields(req.Stressors)...).
    47  		EnablePause()
    48  	if req.EnterNS {
    49  		processBuilder = processBuilder.SetNS(pid, bpm.PidNS)
    50  	}
    51  	cmd := processBuilder.Build(ctx)
    52  
    53  	procState, err := s.backgroundProcessManager.StartProcess(cmd)
    54  	if err != nil {
    55  		return nil, err
    56  	}
    57  	log.Info("Start process successfully")
    58  	ct, err := procState.CreateTime()
    59  	if err != nil {
    60  		return nil, err
    61  	}
    62  
    63  	if err = control.Add(cgroups.Process{Pid: cmd.Process.Pid}); err != nil {
    64  		if kerr := cmd.Process.Kill(); kerr != nil {
    65  			log.Error(kerr, "kill stressors failed", "request", req)
    66  		}
    67  		return nil, err
    68  	}
    69  
    70  	for {
    71  		// TODO: find a better way to resume pause process
    72  		if err := cmd.Process.Signal(syscall.SIGCONT); err != nil {
    73  			return nil, err
    74  		}
    75  
    76  		log.Info("send signal to resume process")
    77  		time.Sleep(time.Millisecond)
    78  
    79  		comm, err := ReadCommName(cmd.Process.Pid)
    80  		if err != nil {
    81  			return nil, err
    82  		}
    83  		if comm != "pause\n" {
    84  			log.Info("pause has been resumed", "comm", comm)
    85  			break
    86  		}
    87  		log.Info("the process hasn't resumed, step into the following loop", "comm", comm)
    88  	}
    89  
    90  	return &pb.ExecStressResponse{
    91  		Instance:  strconv.Itoa(cmd.Process.Pid),
    92  		StartTime: ct,
    93  	}, nil
    94  }

    96  func (s *DaemonServer) CancelStressors(ctx context.Context,
    97  	req *pb.CancelStressRequest) (*empty.Empty, error) {
    98  	log := s.getLoggerFromContext(ctx)
    99  	pid, err := strconv.Atoi(req.Instance)
   100  	if err != nil {
   101  		return nil, err
   102  	}
   103  	log.Info("Canceling stressors", "request", req)
   104  
   105  	err = s.backgroundProcessManager.KillBackgroundProcess(ctx, pid, req.StartTime)
   106  	if err != nil {
   107  		return nil, err
   108  	}
   109  	log.Info("killing stressor successfully")
   110  	return &empty.Empty{}, nil
   111  }
