    40  var _ = Describe("containerd client", func() {
    41  	Context("ContainerdClient GetPidFromContainerID", func() {
    42  		It("should return the magic number 9527", func() {
    43  			defer mock.With("pid", int(9527))()
    44  
    45  			m := &test.MockClient{}
    46  			c := ContainerdClient{client: m}
    47  			pid, err := c.GetPidFromContainerID(context.TODO(), "containerd://valid-container-id")
    48  			Expect(err).To(BeNil())
    49  			Expect(pid).To(Equal(uint32(9527)))
    50  		})
    51  
    52  		It("should error with wrong protocol", func() {
    53  			m := &test.MockClient{}
    54  			c := ContainerdClient{client: m}
    55  			_, err := c.GetPidFromContainerID(context.TODO(), "docker://this-is-a-wrong-protocol")
    56  			Expect(err).NotTo(BeNil())
    57  			Expect(fmt.Sprintf("%s", err)).To(ContainSubstring(fmt.Sprintf("expected %s but got", containerdProtocolPrefix)))
    58  		})
    59  
    60  		It("should error with specified string", func() {
    61  			errorStr := "this is a mocked error"
    62  			mock.With("LoadContainerError", errors.New(errorStr))
    63  			m := &test.MockClient{}
    64  			c := ContainerdClient{client: m}
    65  			_, err := c.GetPidFromContainerID(context.TODO(), "containerd://valid-container-id")
    66  			Expect(err).NotTo(BeNil())
    67  			Expect(fmt.Sprintf("%s", err)).To(Equal(errorStr))
    68  			mock.Reset("LoadContainerError")
    69  
    70  			mock.With("TaskError", errors.New(errorStr))
    71  			m = &test.MockClient{}
    72  			c = ContainerdClient{client: m}
    73  			_, err = c.GetPidFromContainerID(context.TODO(), "containerd://valid-container-id")
    74  			Expect(err).NotTo(BeNil())
    75  			Expect(fmt.Sprintf("%s", err)).To(Equal(errorStr))
    76  			mock.Reset("TaskError")
    77  		})
    78  	})
    79  
    80  	Context("ContainerdClient ContainerKillByContainerID", func() {
    81  		It("should work", func() {
    82  			m := &test.MockClient{}
    83  			c := ContainerdClient{client: m}
    84  			err := c.ContainerKillByContainerID(context.TODO(), "containerd://valid-container-id")
    85  			Expect(err).To(BeNil())
    86  		})
    87  
    88  		errorPoints := []string{"LoadContainer", "Task", "Kill"}
    89  		for _, e := range errorPoints {
    90  			It(fmt.Sprintf("should error on %s", e), func() {
    91  				errorStr := fmt.Sprintf("this is a mocked error on %s", e)
    92  				m := &test.MockClient{}
    93  				c := ContainerdClient{client: m}
    94  				defer mock.With(e+"Error", errors.New(errorStr))()
    95  				err := c.ContainerKillByContainerID(context.TODO(), "containerd://valid-container-id")
    96  				Expect(err).ToNot(BeNil())
    97  				Expect(fmt.Sprintf("%s", err)).To(Equal(errorStr))
    98  			})
    99  		}
