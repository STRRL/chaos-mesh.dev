   110  func newGRPCServer(containerRuntime string, reg prometheus.Registerer, tlsConf tlsConfig, logger logr.Logger) (*grpc.Server, error) {
   111  	ds, err := newDaemonServer(containerRuntime, logger)
   112  	if err != nil {
   113  		return nil, err
   114  	}
   115  
   116  	grpcMetrics := grpc_prometheus.NewServerMetrics()
   117  	grpcMetrics.EnableHandlingTimeHistogram(
   118  		grpc_prometheus.WithHistogramBuckets([]float64{0.001, 0.01, 0.1, 0.3, 0.6, 1, 3, 6, 10}),
   119  	)
   120  	reg.MustRegister(grpcMetrics)
   121  
   122  	grpcOpts := []grpc.ServerOption{
   123  		grpc_middleware.WithUnaryServerChain(
   124  			grpcUtils.TimeoutServerInterceptor,
   125  			grpcMetrics.UnaryServerInterceptor(),
   126  			MetadataExtractor(log.MetaNamespacedName),
   127  		),
   128  	}
   129  
   130  	if tlsConf != (tlsConfig{}) {
   131  		caCert, err := ioutil.ReadFile(tlsConf.CaCert)
   132  		if err != nil {
   133  			return nil, err
   134  		}
   135  		caCertPool := x509.NewCertPool()
   136  		caCertPool.AppendCertsFromPEM(caCert)
   137  
   138  		serverCert, err := tls.LoadX509KeyPair(tlsConf.Cert, tlsConf.Key)
   139  		if err != nil {
   140  			return nil, err
   141  		}
   142  
   143  		creds := credentials.NewTLS(&tls.Config{
   144  			Certificates: []tls.Certificate{serverCert},
   145  			ClientCAs:    caCertPool,
   146  			ClientAuth:   tls.RequireAndVerifyClientCert,
   147  			MinVersion:   tls.VersionTLS13,
   148  		})
   149  
   150  		grpcOpts = append(grpcOpts, grpc.Creds(creds))
   151  	}
   152  
   153  	s := grpc.NewServer(grpcOpts...)
   154  	grpcMetrics.InitializeMetrics(s)
   155  
   156  	pb.RegisterChaosDaemonServer(s, ds)
   157  	reflection.Register(s)
   158  
   159  	return s, nil
   160  }

   162  func MetadataExtractor(keys ...log.Metadatkey) grpc.UnaryServerInterceptor {
   163  	return func(ctx context.Context, req interface{}, info *grpc.UnaryServerInfo, handler grpc.UnaryHandler) (resp interface{}, err error) {
   164  		// Get the metadata from the incoming context
   165  		md, ok := metadata.FromIncomingContext(ctx)
   166  		if !ok {
   167  			return nil, errors.New("couldn't parse incoming context metadata")
   168  		}
   169  		for _, key := range keys {
   170  			values := md.Get(string(key))
   171  			if len(values) > 0 {
   172  				ctx = context.WithValue(ctx, key, values[0])
   173  			}
   174  		}
   175  
   176  		return handler(ctx, req)
   177  	}
   178  }

   186  // StartServer starts chaos-daemon.
   187  func StartServer(conf *Config, reg RegisterGatherer, logger logr.Logger) error {
   188  	g := &errgroup.Group{}
   189  
   190  	httpBindAddr := conf.HttpAddr()
   191  	httpServer := newHTTPServerBuilder().Addr(httpBindAddr).Metrics(reg).Profiling(conf.Profiling).Build()
   192  
   193  	grpcBindAddr := conf.GrpcAddr()
   194  	grpcListener, err := net.Listen("tcp", grpcBindAddr)
   195  	if err != nil {
   196  		logger.Error(err, "failed to listen grpc address", "grpcBindAddr", grpcBindAddr)
   197  		return err
   198  	}
   199  
   200  	grpcServer, err := newGRPCServer(conf.Runtime, reg, conf.tlsConfig, logger)
   201  	if err != nil {
   202  		logger.Error(err, "failed to create grpc server")
   203  		return err
   204  	}
   205  
   206  	g.Go(func() error {
   207  		logger.Info("Starting http endpoint", "address", httpBindAddr)
   208  		if err := httpServer.ListenAndServe(); err != nil {
   209  			logger.Error(err, "failed to start http endpoint")
   210  			httpServer.Shutdown(context.Background())
   211  			return err
   212  		}
   213  		return nil
   214  	})
   215  
   216  	g.Go(func() error {
   217  		logger.Info("Starting grpc endpoint", "address", grpcBindAddr, "runtime", conf.Runtime)
   218  		if err := grpcServer.Serve(grpcListener); err != nil {
   219  			logger.Error(err, "failed to start grpc endpoint")
   220  			grpcServer.Stop()
   221  			return err
   222  		}
   223  		return nil
   224  	})
   225  
   226  	return g.Wait()
   227  }
