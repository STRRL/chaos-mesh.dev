    38  func (s *DaemonServer) ApplyIOChaos(ctx context.Context, in *pb.ApplyIOChaosRequest) (*pb.ApplyIOChaosResponse, error) {
    39  	log := s.getLoggerFromContext(ctx)
    40  	log.Info("applying io chaos", "Request", in)
    41  
    42  	if in.Instance != 0 {
    43  		err := s.killIOChaos(ctx, in.Instance, in.StartTime)
    44  		if err != nil {
    45  			return nil, err
    46  		}
    47  	}
    48  
    49  	actions := []v1alpha1.IOChaosAction{}
    50  	err := json.Unmarshal([]byte(in.Actions), &actions)
    51  	if err != nil {
    52  		log.Error(err, "error while unmarshal json bytes")
    53  		return nil, err
    54  	}
    55  
    56  	log.Info("the length of actions", "length", len(actions))
    57  	if len(actions) == 0 {
    58  		return &pb.ApplyIOChaosResponse{
    59  			Instance:  0,
    60  			StartTime: 0,
    61  		}, nil
    62  	}
    63  
    64  	pid, err := s.crClient.GetPidFromContainerID(ctx, in.ContainerId)
    65  	if err != nil {
    66  		log.Error(err, "error while getting PID")
    67  		return nil, err
    68  	}
    69  
    70  	// TODO: make this log level configurable
    71  	args := fmt.Sprintf("--path %s --verbose info", in.Volume)
    72  	log.Info("executing", "cmd", todaBin+" "+args)
    73  
    74  	processBuilder := bpm.DefaultProcessBuilder(todaBin, strings.Split(args, " ")...).
    75  		EnableLocalMnt().
    76  		SetIdentifier(fmt.Sprintf("toda-%s", in.ContainerId))
    77  
    78  	if in.EnterNS {
    79  		processBuilder = processBuilder.SetNS(pid, bpm.MountNS).SetNS(pid, bpm.PidNS)
    80  	}
    81  
    82  	ctx, cancel := context.WithCancel(context.Background())
    83  	defer cancel()
    84  	caller, receiver := bpm.NewBlockingBuffer(), bpm.NewBlockingBuffer()
    85  	defer caller.Close()
    86  	defer receiver.Close()
    87  	client, err := jrpc.DialIO(ctx, receiver, caller)
    88  	if err != nil {
    89  		return nil, err
    90  	}
    91  
    92  	cmd := processBuilder.Build(ctx)
    93  	cmd.Stdin = caller
    94  	cmd.Stdout = io.MultiWriter(receiver, os.Stdout)
    95  	cmd.Stderr = os.Stderr
    96  	procState, err := s.backgroundProcessManager.StartProcess(cmd)
    97  	if err != nil {
    98  		return nil, err
    99  	}
   100  	var ret string
   101  	ct, err := procState.CreateTime()
   102  	if err != nil {
   103  		log.Error(err, "get create time failed")
   104  		if kerr := cmd.Process.Kill(); kerr != nil {
   105  			log.Error(kerr, "kill toda failed", "request", in)
   106  		}
   107  		return nil, err
   108  	}
   109  
   110  	log.Info("Waiting for toda to start")
   111  	var rpcError error
   112  	maxWaitTime := time.Millisecond * 2000
   113  	timeOut, cancel := context.WithTimeout(ctx, maxWaitTime)
   114  	defer cancel()
   115  	_ = client.CallContext(timeOut, &ret, "update", actions)
   116  	rpcError = client.CallContext(timeOut, &ret, "get_status", "ping")
   117  	if rpcError != nil || ret != "ok" {
   118  		log.Info("Starting toda takes too long or encounter an error")
   119  		caller.Close()
   120  		receiver.Close()
   121  		if kerr := s.killIOChaos(ctx, int64(cmd.Process.Pid), ct); kerr != nil {
   122  			log.Error(kerr, "kill toda failed", "request", in)
   123  		}
   124  		return nil, fmt.Errorf("toda startup takes too long or an error occurs: %s", ret)
   125  	}
   126  
   127  	return &pb.ApplyIOChaosResponse{
   128  		Instance:  int64(cmd.Process.Pid),
   129  		StartTime: ct,
   130  	}, nil
   131  }

   133  func (s *DaemonServer) killIOChaos(ctx context.Context, pid int64, startTime int64) error {
   134  	log := s.getLoggerFromContext(ctx)
   135  	log.Info("killing toda", "pid", pid)
   136  
   137  	err := s.backgroundProcessManager.KillBackgroundProcess(ctx, int(pid), startTime)
   138  	if err != nil {
   139  		return err
   140  	}
   141  	log.Info("kill toda successfully")
   142  	return nil
   143  }
