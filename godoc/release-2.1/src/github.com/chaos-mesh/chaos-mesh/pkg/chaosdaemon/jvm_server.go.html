    36  func (s *DaemonServer) InstallJVMRules(ctx context.Context,
    37  	req *pb.InstallJVMRulesRequest) (*empty.Empty, error) {
    38  	log := s.getLoggerFromContext(ctx)
    39  	log.Info("InstallJVMRules", "request", req)
    40  	pid, err := s.crClient.GetPidFromContainerID(ctx, req.ContainerId)
    41  	if err != nil {
    42  		log.Error(err, "GetPidFromContainerID")
    43  		return nil, err
    44  	}
    45  
    46  	containerPids := []uint32{pid}
    47  	childPids, err := GetChildProcesses(pid)
    48  	if err != nil {
    49  		log.Error(err, "GetChildProcesses")
    50  	}
    51  	containerPids = append(containerPids, childPids...)
    52  	for _, containerPid := range containerPids {
    53  		name, err := ReadCommName(int(containerPid))
    54  		if err != nil {
    55  			log.Error(err, "ReadCommName")
    56  			continue
    57  		}
    58  		if name == "java\n" {
    59  			pid = containerPid
    60  			break
    61  		}
    62  	}
    63  
    64  	bytemanHome := os.Getenv("BYTEMAN_HOME")
    65  	if len(bytemanHome) == 0 {
    66  		return nil, fmt.Errorf("environment variable BYTEMAN_HOME not set")
    67  	}
    68  
    69  	// copy agent.jar to container's namespace
    70  	if req.EnterNS {
    71  		processBuilder := bpm.DefaultProcessBuilder("sh", "-c", fmt.Sprintf("mkdir -p %s/lib/", bytemanHome)).SetContext(ctx).SetNS(pid, bpm.MountNS)
    72  		output, err := processBuilder.Build(ctx).CombinedOutput()
    73  		if err != nil {
    74  			return nil, err
    75  		}
    76  		if len(output) > 0 {
    77  			log.Info("mkdir", "output", string(output))
    78  		}
    79  
    80  		agentFile, err := os.Open(fmt.Sprintf("%s/lib/byteman.jar", bytemanHome))
    81  		if err != nil {
    82  			return nil, err
    83  		}
    84  		processBuilder = bpm.DefaultProcessBuilder("sh", "-c", "cat > /usr/local/byteman/lib/byteman.jar").SetContext(ctx)
    85  		processBuilder = processBuilder.SetNS(pid, bpm.MountNS).SetStdin(agentFile)
    86  		output, err = processBuilder.Build(ctx).CombinedOutput()
    87  		if err != nil {
    88  			return nil, err
    89  		}
    90  		if len(output) > 0 {
    91  			log.Info("copy agent.jar", "output", string(output))
    92  		}
    93  	}
    94  
    95  	bmInstallCmd := fmt.Sprintf(bmInstallCommand, req.Port, pid)
    96  	processBuilder := bpm.DefaultProcessBuilder("sh", "-c", bmInstallCmd).SetContext(ctx)
    97  	if req.EnterNS {
    98  		processBuilder = processBuilder.EnableLocalMnt()
    99  	}
   100  
   101  	cmd := processBuilder.Build(ctx)
   102  	output, err := cmd.CombinedOutput()
   103  	if err != nil {
   104  		// this error will occured when install agent more than once, and will ignore this error and continue to submit rule
   105  		errMsg1 := "Agent JAR loaded but agent failed to initialize"
   106  
   107  		// these two errors will occured when java version less or euqal to 1.8, and don't know why
   108  		// but it can install agent success even with this error, so just ignore it now.
   109  		// TODO: Investigate the cause of these two error
   110  		errMsg2 := "Provider sun.tools.attach.LinuxAttachProvider not found"
   111  		errMsg3 := "install java.io.IOException: Non-numeric value found"
   112  
   113  		// this error is caused by the different attach result codes in different java versions. In fact, the agent has attached success, just ignore it here.
   114  		// refer to https://stackoverflow.com/questions/54340438/virtualmachine-attach-throws-com-sun-tools-attach-agentloadexception-0-when-usi/54454418#54454418
   115  		errMsg4 := "install com.sun.tools.attach.AgentLoadException"
   116  		if !strings.Contains(string(output), errMsg1) && !strings.Contains(string(output), errMsg2) &&
   117  			!strings.Contains(string(output), errMsg3) && !strings.Contains(string(output), errMsg4) {
   118  			log.Error(err, string(output))
   119  			return nil, err
   120  		}
   121  		log.Info("exec comamnd", "cmd", cmd.String(), "output", string(output), "error", err.Error())
   122  	}
   123  
   124  	// submit rules
   125  	filename, err := writeDataIntoFile(req.Rule, "rule.btm")
   126  	if err != nil {
   127  		return nil, err
   128  	}
   129  
   130  	bmSubmitCmd := fmt.Sprintf(bmSubmitCommand, req.Port, "l", filename)
   131  	processBuilder = bpm.DefaultProcessBuilder("sh", "-c", bmSubmitCmd).SetContext(ctx)
   132  	if req.EnterNS {
   133  		processBuilder = processBuilder.SetNS(pid, bpm.NetNS)
   134  	}
   135  	output, err = processBuilder.Build(ctx).CombinedOutput()
   136  	if err != nil {
   137  		log.Error(err, string(output))
   138  		return nil, err
   139  	}
   140  	if len(output) > 0 {
   141  		log.Info("submit rules", "output", string(output))
   142  	}
   143  
   144  	return &empty.Empty{}, nil
   145  }

   147  func (s *DaemonServer) UninstallJVMRules(ctx context.Context,
   148  	req *pb.UninstallJVMRulesRequest) (*empty.Empty, error) {
   149  	log := s.getLoggerFromContext(ctx)
   150  	log.Info("InstallJVMRules", "request", req)
   151  	pid, err := s.crClient.GetPidFromContainerID(ctx, req.ContainerId)
   152  	if err != nil {
   153  		log.Error(err, "GetPidFromContainerID")
   154  		return nil, err
   155  	}
   156  
   157  	filename, err := writeDataIntoFile(req.Rule, "rule.btm")
   158  	if err != nil {
   159  		return nil, err
   160  	}
   161  	log.Info("create btm file", "file", filename)
   162  
   163  	bmSubmitCmd := fmt.Sprintf(bmSubmitCommand, req.Port, "u", filename)
   164  	processBuilder := bpm.DefaultProcessBuilder("sh", "-c", bmSubmitCmd).SetContext(ctx)
   165  	if req.EnterNS {
   166  		processBuilder = processBuilder.SetNS(pid, bpm.NetNS)
   167  	}
   168  	output, err := processBuilder.Build(ctx).CombinedOutput()
   169  	if err != nil {
   170  		log.Error(err, string(output))
   171  		if strings.Contains(string(output), "No rule scripts to remove") {
   172  			return &empty.Empty{}, nil
   173  		}
   174  		return nil, err
   175  	}
   176  
   177  	if len(output) > 0 {
   178  		log.Info(string(output))
   179  	}
   180  
   181  	return &empty.Empty{}, nil
   182  }

   184  func writeDataIntoFile(data string, filename string) (string, error) {
   185  	tmpfile, err := ioutil.TempFile("", filename)
   186  	if err != nil {
   187  		return "", err
   188  	}
   189  
   190  	if _, err := tmpfile.WriteString(data); err != nil {
   191  		return "", err
   192  	}
   193  
   194  	if err := tmpfile.Close(); err != nil {
   195  		return "", err
   196  	}
   197  
   198  	return tmpfile.Name(), err
   199  }
