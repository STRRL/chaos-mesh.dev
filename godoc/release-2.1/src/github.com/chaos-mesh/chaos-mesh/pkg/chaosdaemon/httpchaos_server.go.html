    50  func (t stdioTransport) RoundTrip(req *http.Request) (resp *http.Response, err error) {
    51  	t.stdio.Lock()
    52  	defer t.stdio.Unlock()
    53  
    54  	if t.stdio.Stdin == nil {
    55  		return nil, errors.New("fail to get stdin of process")
    56  	}
    57  	if t.stdio.Stdout == nil {
    58  		return nil, errors.New("fail to get stdout of process")
    59  	}
    60  
    61  	err = req.Write(t.stdio.Stdin)
    62  	if err != nil {
    63  		return
    64  	}
    65  
    66  	resp, err = http.ReadResponse(bufio.NewReader(t.stdio.Stdout), req)
    67  	return
    68  }

    70  func (s *DaemonServer) ApplyHttpChaos(ctx context.Context, in *pb.ApplyHttpChaosRequest) (*pb.ApplyHttpChaosResponse, error) {
    71  	logger := s.getLoggerFromContext(ctx)
    72  	log := logger.WithValues("Request", in)
    73  	log.Info("applying http chaos")
    74  
    75  	if s.backgroundProcessManager.Stdio(int(in.Instance), in.StartTime) == nil {
    76  		// chaos daemon may restart, create another tproxy instance
    77  		if err := s.backgroundProcessManager.KillBackgroundProcess(ctx, int(in.Instance), in.StartTime); err != nil {
    78  			return nil, errors.Wrapf(err, "kill background process(%d)", in.Instance)
    79  		}
    80  		if err := s.createHttpChaos(ctx, in); err != nil {
    81  			return nil, err
    82  		}
    83  	}
    84  
    85  	resp, err := s.applyHttpChaos(ctx, log, in)
    86  	if err != nil {
    87  		killError := s.backgroundProcessManager.KillBackgroundProcess(ctx, int(in.Instance), in.StartTime)
    88  		log.Error(killError, "kill tproxy", "instance", in.Instance)
    89  		return nil, errors.Wrap(err, "apply config")
    90  	}
    91  	return resp, err
    92  }

    94  func (s *DaemonServer) applyHttpChaos(ctx context.Context, logger logr.Logger, in *pb.ApplyHttpChaosRequest) (*pb.ApplyHttpChaosResponse, error) {
    95  	log := s.getLoggerFromContext(ctx)
    96  	stdio := s.backgroundProcessManager.Stdio(int(in.Instance), in.StartTime)
    97  	if stdio == nil {
    98  		return nil, errors.Errorf("fail to get stdio of process")
    99  	}
   100  
   101  	transport := stdioTransport{stdio: stdio}
   102  
   103  	rules := []v1alpha1.PodHttpChaosBaseRule{}
   104  	err := json.Unmarshal([]byte(in.Rules), &rules)
   105  	if err != nil {
   106  		return nil, errors.Wrap(err, "unmarshal rules")
   107  	}
   108  
   109  	log.Info("the length of actions", "length", len(rules))
   110  
   111  	httpChaosSpec := tproxyConfig{
   112  		ProxyPorts: append([]uint32{}, in.ProxyPorts...),
   113  		Rules:      rules,
   114  	}
   115  
   116  	config, err := json.Marshal(&httpChaosSpec)
   117  	if err != nil {
   118  		return nil, err
   119  	}
   120  
   121  	log.Info("ready to apply", "config", string(config))
   122  
   123  	req, err := http.NewRequest(http.MethodPut, "/", bytes.NewReader(config))
   124  	if err != nil {
   125  		return nil, errors.Wrap(err, "create http request")
   126  	}
   127  
   128  	resp, err := transport.RoundTrip(req)
   129  	if err != nil {
   130  		return nil, errors.Wrap(err, "send http request")
   131  	}
   132  
   133  	log.Info("http chaos applied")
   134  
   135  	body, err := ioutil.ReadAll(resp.Body)
   136  	if err != nil {
   137  		return nil, errors.Wrap(err, "read response body")
   138  	}
   139  
   140  	return &pb.ApplyHttpChaosResponse{
   141  		Instance:   int64(in.Instance),
   142  		StartTime:  in.StartTime,
   143  		StatusCode: int32(resp.StatusCode),
   144  		Error:      string(body),
   145  	}, nil
   146  }

   148  func (s *DaemonServer) createHttpChaos(ctx context.Context, in *pb.ApplyHttpChaosRequest) error {
   149  	log := s.getLoggerFromContext(ctx)
   150  	pid, err := s.crClient.GetPidFromContainerID(ctx, in.ContainerId)
   151  	if err != nil {
   152  		return errors.Wrapf(err, "get PID of container(%s)", in.ContainerId)
   153  	}
   154  	processBuilder := bpm.DefaultProcessBuilder(tproxyBin, "-i", "-vv").
   155  		EnableLocalMnt().
   156  		SetIdentifier(fmt.Sprintf("tproxy-%s", in.ContainerId)).
   157  		SetEnv(pathEnv, os.Getenv(pathEnv)).
   158  		SetStdin(bpm.NewBlockingBuffer()).
   159  		SetStdout(bpm.NewBlockingBuffer())
   160  
   161  	if in.EnterNS {
   162  		processBuilder = processBuilder.SetNS(pid, bpm.PidNS).SetNS(pid, bpm.NetNS)
   163  	}
   164  
   165  	cmd := processBuilder.Build(ctx)
   166  	cmd.Stderr = os.Stderr
   167  
   168  	procState, err := s.backgroundProcessManager.StartProcess(cmd)
   169  	if err != nil {
   170  		return errors.Wrapf(err, "execute command(%s)", cmd)
   171  	}
   172  	ct, err := procState.CreateTime()
   173  	if err != nil {
   174  		if kerr := cmd.Process.Kill(); kerr != nil {
   175  			log.Error(kerr, "kill tproxy", "request", in)
   176  		}
   177  		return errors.Wrap(err, "get create time")
   178  	}
   179  
   180  	in.Instance = int64(cmd.Process.Pid)
   181  	in.StartTime = ct
   182  	return nil
   183  }
