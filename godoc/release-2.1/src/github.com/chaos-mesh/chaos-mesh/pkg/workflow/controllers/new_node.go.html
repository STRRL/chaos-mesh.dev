    35  // renderNodesByTemplates will render the nodes one by one, will setup owner by given parent. If parent is nil, it will use workflow as its owner.
    36  func renderNodesByTemplates(workflow *v1alpha1.Workflow, parent *v1alpha1.WorkflowNode, templates ...string) ([]*v1alpha1.WorkflowNode, error) {
    37  	templateNameSet := make(map[string]v1alpha1.Template)
    38  	for _, template := range workflow.Spec.Templates {
    39  		templateNameSet[template.Name] = template
    40  	}
    41  	var result []*v1alpha1.WorkflowNode
    42  	for _, name := range templates {
    43  		if template, ok := templateNameSet[name]; ok {
    44  
    45  			now := metav1.NewTime(time.Now())
    46  			var deadline *metav1.Time = nil
    47  
    48  			if template.Deadline != nil {
    49  				duration, err := time.ParseDuration(*template.Deadline)
    50  				if err != nil {
    51  					// TODO: logger
    52  					return nil, err
    53  				}
    54  				copiedDuration := metav1.NewTime(now.DeepCopy().Add(duration))
    55  				deadline = &copiedDuration
    56  			}
    57  
    58  			renderedNode := v1alpha1.WorkflowNode{
    59  				ObjectMeta: metav1.ObjectMeta{
    60  					Namespace:    workflow.Namespace,
    61  					GenerateName: fmt.Sprintf("%s-", template.Name),
    62  				},
    63  				Spec: v1alpha1.WorkflowNodeSpec{
    64  					TemplateName:        template.Name,
    65  					WorkflowName:        workflow.Name,
    66  					Type:                template.Type,
    67  					StartTime:           &now,
    68  					Deadline:            deadline,
    69  					Children:            template.Children,
    70  					Task:                template.Task,
    71  					ConditionalBranches: template.ConditionalBranches,
    72  					EmbedChaos:          template.EmbedChaos,
    73  					Schedule:            conversionSchedule(template.Schedule),
    74  				},
    75  			}
    76  
    77  			// if parent is specified, use parent as owner, else use workflow as owner.
    78  			if parent != nil {
    79  				renderedNode.OwnerReferences = append(renderedNode.OwnerReferences, metav1.OwnerReference{
    80  					APIVersion:         ApiVersion,
    81  					Kind:               KindWorkflowNode,
    82  					Name:               parent.Name,
    83  					UID:                parent.UID,
    84  					Controller:         &isController,
    85  					BlockOwnerDeletion: &blockOwnerDeletion,
    86  				})
    87  				if renderedNode.Labels == nil {
    88  					renderedNode.Labels = make(map[string]string)
    89  				}
    90  				renderedNode.Labels[v1alpha1.LabelControlledBy] = parent.Name
    91  			} else {
    92  				renderedNode.OwnerReferences = append(renderedNode.OwnerReferences, metav1.OwnerReference{
    93  					APIVersion:         ApiVersion,
    94  					Kind:               KindWorkflow,
    95  					Name:               workflow.Name,
    96  					UID:                workflow.UID,
    97  					Controller:         &isController,
    98  					BlockOwnerDeletion: &blockOwnerDeletion,
    99  				})
   100  				if renderedNode.Labels == nil {
   101  					renderedNode.Labels = make(map[string]string)
   102  				}
   103  				renderedNode.Labels[v1alpha1.LabelControlledBy] = workflow.Name
   104  			}
   105  
   106  			renderedNode.Labels[v1alpha1.LabelWorkflow] = workflow.Name
   107  			renderedNode.Finalizers = append(renderedNode.Finalizers, metav1.FinalizerDeleteDependents)
   108  
   109  			result = append(result, &renderedNode)
   110  			continue
   111  		}
   112  		return nil, fmt.Errorf(
   113  			"workflow %s do not contains template called %s",
   114  			workflow.Name,
   115  			name,
   116  		)
   117  	}
   118  	return result, nil
   119  }

   121  func conversionSchedule(origin *v1alpha1.ChaosOnlyScheduleSpec) *v1alpha1.ScheduleSpec {
   122  	if origin == nil {
   123  		return nil
   124  	}
   125  	return &v1alpha1.ScheduleSpec{
   126  		Schedule:                origin.Schedule,
   127  		StartingDeadlineSeconds: origin.StartingDeadlineSeconds,
   128  		ConcurrencyPolicy:       origin.ConcurrencyPolicy,
   129  		HistoryLimit:            origin.HistoryLimit,
   130  		Type:                    origin.Type,
   131  		ScheduleItem: v1alpha1.ScheduleItem{
   132  			EmbedChaos: v1alpha1.EmbedChaos{
   133  				AWSChaos:             origin.EmbedChaos.AWSChaos,
   134  				DNSChaos:             origin.EmbedChaos.DNSChaos,
   135  				GCPChaos:             origin.EmbedChaos.GCPChaos,
   136  				HTTPChaos:            origin.EmbedChaos.HTTPChaos,
   137  				IOChaos:              origin.EmbedChaos.IOChaos,
   138  				JVMChaos:             origin.EmbedChaos.JVMChaos,
   139  				KernelChaos:          origin.EmbedChaos.KernelChaos,
   140  				NetworkChaos:         origin.EmbedChaos.NetworkChaos,
   141  				PodChaos:             origin.EmbedChaos.PodChaos,
   142  				StressChaos:          origin.EmbedChaos.StressChaos,
   143  				TimeChaos:            origin.EmbedChaos.TimeChaos,
   144  				PhysicalMachineChaos: origin.EmbedChaos.PhysicalMachineChaos,
   145  			},
   146  			Workflow: nil,
   147  		},
   148  	}
