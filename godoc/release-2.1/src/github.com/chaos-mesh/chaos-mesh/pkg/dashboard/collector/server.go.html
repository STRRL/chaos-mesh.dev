    50  // NewServer returns a CollectorServer and Client.
    51  func NewServer(
    52  	conf *config.ChaosDashboardConfig,
    53  	experimentArchive core.ExperimentStore,
    54  	scheduleArchive core.ScheduleStore,
    55  	event core.EventStore,
    56  	workflowStore core.WorkflowStore,
    57  ) (*Server, client.Client, client.Reader, *runtime.Scheme) {
    58  	s := &Server{}
    59  
    60  	// namespace scoped
    61  	options := ctrl.Options{
    62  		Scheme:             scheme,
    63  		MetricsBindAddress: conf.MetricAddress,
    64  		LeaderElection:     conf.EnableLeaderElection,
    65  		Port:               9443,
    66  	}
    67  	if conf.ClusterScoped {
    68  		log.Info("Chaos controller manager is running in cluster scoped mode.")
    69  	} else {
    70  		log.Info("Chaos controller manager is running in namespace scoped mode.", "targetNamespace", conf.TargetNamespace)
    71  		options.Namespace = conf.TargetNamespace
    72  	}
    73  
    74  	var err error
    75  
    76  	cfg := ctrl.GetConfigOrDie()
    77  
    78  	if conf.QPS > 0 {
    79  		cfg.QPS = conf.QPS
    80  		cfg.Burst = conf.Burst
    81  	}
    82  
    83  	s.Manager, err = ctrl.NewManager(cfg, options)
    84  	if err != nil {
    85  		log.Error(err, "unable to start collector")
    86  		os.Exit(1)
    87  	}
    88  
    89  	if conf.SecurityMode {
    90  		clientpool.K8sClients, err = clientpool.NewClientPool(cfg, scheme, 100)
    91  		if err != nil {
    92  			// this should never happen
    93  			log.Error(err, "fail to create client pool")
    94  			os.Exit(1)
    95  		}
    96  	} else {
    97  		clientpool.K8sClients, err = clientpool.NewLocalClient(cfg, scheme)
    98  		if err != nil {
    99  			log.Error(err, "fail to create client pool")
   100  			os.Exit(1)
   101  		}
   102  	}
   103  
   104  	for kind, chaosKind := range v1alpha1.AllKinds() {
   105  		if err = (&ChaosCollector{
   106  			Client:  s.Manager.GetClient(),
   107  			Log:     ctrl.Log.WithName("collector").WithName(kind),
   108  			archive: experimentArchive,
   109  			event:   event,
   110  		}).Setup(s.Manager, chaosKind.SpawnObject()); err != nil {
   111  			log.Error(err, "unable to create collector", "collector", kind)
   112  			os.Exit(1)
   113  		}
   114  	}
   115  
   116  	if err = (&ScheduleCollector{
   117  		Client:  s.Manager.GetClient(),
   118  		Log:     ctrl.Log.WithName("schedule-collector").WithName(v1alpha1.KindSchedule),
   119  		archive: scheduleArchive,
   120  	}).Setup(s.Manager, &v1alpha1.Schedule{}); err != nil {
   121  		log.Error(err, "unable to create collector", "collector", v1alpha1.KindSchedule)
   122  		os.Exit(1)
   123  	}
   124  
   125  	if err = (&EventCollector{
   126  		Client: s.Manager.GetClient(),
   127  		Log:    ctrl.Log.WithName("event-collector").WithName("Event"),
   128  		event:  event,
   129  	}).Setup(s.Manager, &v1.Event{}); err != nil {
   130  		log.Error(err, "unable to create collector", "collector", v1alpha1.KindSchedule)
   131  		os.Exit(1)
   132  	}
   133  
   134  	if err = (&WorkflowCollector{
   135  		kubeClient: s.Manager.GetClient(),
   136  		Log:        ctrl.Log.WithName("workflow-collector").WithName(v1alpha1.KindWorkflow),
   137  		store:      workflowStore,
   138  	}).Setup(s.Manager, &v1alpha1.Workflow{}); err != nil {
   139  		log.Error(err, "unable to create collector", "collector", v1alpha1.KindWorkflow)
   140  		os.Exit(1)
   141  	}
   142  
   143  	return s, s.Manager.GetClient(), s.Manager.GetAPIReader(), s.Manager.GetScheme()
   144  }
