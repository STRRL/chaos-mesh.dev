    20  // MergeCommands merges injected commands and original commands for injecting commands to containers,
    21  // eg: inject: []string{"bash", "-c", "/check.sh"}, origin: []string{"bash", "-c", "/run.sh"}
    22  // merged commands: []string{"/bin/sh", "-ec", "/check.sh\n/run.sh"}
    23  func MergeCommands(inject []string, origin []string, args []string) []string {
    24  	// merge injected commands
    25  	scripts := mergeCommandsAction(inject)
    26  
    27  	// merge original commands
    28  	scripts += mergeOriginCommandsAndArgs(origin, args)
    29  
    30  	return []string{"/bin/sh", "-ec", scripts}
    31  }
    32  
    33  func mergeCommandsAction(commands []string) string {
    34  	scripts := ""
    35  
    36  	for i := 0; i < len(commands); i++ {
    37  		cmd := commands[i]
    38  		if isCommonScripts(cmd) {
    39  			if len(commands) <= i+1 {
    40  				scripts += cmd
    41  				scripts += "\n"
    42  				break
    43  			}
    44  
    45  			if strings.HasPrefix(commands[i+1], "-") {
    46  				i++
    47  				continue
    48  			}
    49  
    50  			tempScripts := cmd + " "
    51  			for j := i + 1; j < len(commands); j++ {
    52  				c := commands[j]
    53  				if j < len(commands)-1 {
    54  					c += " "
    55  				}
    56  
    57  				tempScripts += c
    58  			}
    59  
    60  			scripts += tempScripts
    61  			scripts += "\n"
    62  			break
    63  		}
    64  
    65  		if len(commands) <= i+1 {
    66  			scripts += cmd
    67  			scripts += "\n"
    68  			break
    69  		}
    70  
    71  		if strings.HasPrefix(commands[i+1], "-") {
    72  			tempScripts := cmd + " "
    73  			for j := i + 1; j < len(commands); j++ {
    74  				if strings.HasPrefix(commands[j], "-") {
    75  					c := commands[j]
    76  					if j < len(commands)-1 {
    77  						c += " "
    78  					}
    79  					tempScripts += c
    80  					continue
    81  				}
    82  			}
    83  			scripts += tempScripts
    84  			scripts += "\n"
    85  			break
    86  		}
    87  
    88  		scripts += cmd
    89  		scripts += "\n"
    90  	}
    91  
    92  	return scripts
    93  }

    95  func mergeOriginCommandsAndArgs(origin []string, args []string) string {
    96  	commands := append(origin, args...)
    97  
    98  	return mergeCommandsAction(commands)
    99  }

   101  func isCommonScripts(cmd string) bool {
   102  	if isShellScripts(cmd) || isPythonScripts(cmd) {
   103  		return true
   104  	}
   105  
   106  	return false
   107  }
   108  
   109  func isShellScripts(cmd string) bool {
   110  	if cmd == "bash" || cmd == "sh" ||
   111  		strings.HasPrefix(cmd, "bash ") || strings.HasPrefix(cmd, "sh ") ||
   112  		strings.HasPrefix(cmd, "/bin/sh") || strings.HasPrefix(cmd, "/bin/bash") ||
   113  		strings.HasPrefix(cmd, "/usr/bin/sh") || strings.HasPrefix(cmd, "/usr/bin/bash") ||
   114  		strings.HasPrefix(cmd, "/usr/share/bin/sh") || strings.HasPrefix(cmd, "/usr/share/bin/bash") {
   115  		return true
   116  	}
   117  
   118  	return false
   119  }
   120  
   121  func isPythonScripts(cmd string) bool {
   122  	if cmd == "python" || cmd == "python3" ||
   123  		strings.HasPrefix(cmd, "python ") || strings.HasPrefix(cmd, "python3 ") ||
   124  		strings.HasPrefix(cmd, "/bin/python") || strings.HasPrefix(cmd, "/bin/python3") ||
   125  		strings.HasPrefix(cmd, "/usr/bin/python") || strings.HasPrefix(cmd, "/usr/bin/python3") ||
   126  		strings.HasPrefix(cmd, "/usr/share/bin/python") || strings.HasPrefix(cmd, "/usr/share/bin/python3") {
   127  		return true
   128  	}
   129  
   130  	return false
   131  }
