    40  // ExecAction describes a "run in container" action.
    41  type ExecAction struct {
    42  	// Command is the command line to execute inside the container, the working directory for the
    43  	// command  is root ('/') in the container's filesystem. The command is simply exec'd, it is
    44  	// not run inside a shell, so traditional shell instructions ('|', etc) won't work. To use
    45  	// a shell, you need to explicitly call out to that shell.
    46  	// Exit status of 0 is treated as live/healthy and non-zero is unhealthy.
    47  	// +optional
    48  	Command []string `json:"command,omitempty"`
    49  }

   110  // GetRequestedConfig returns the InjectionConfig given a requested key
   111  func (c *Config) GetRequestedConfig(namespace, key string) (*InjectionConfig, error) {
   112  	c.RLock()
   113  	defer c.RUnlock()
   114  
   115  	if _, ok := c.Injections[namespace]; !ok {
   116  		return nil, fmt.Errorf("no injection config at ns %s", namespace)
   117  	}
   118  
   119  	for _, conf := range c.Injections[namespace] {
   120  		if key == conf.Name {
   121  			return conf, nil
   122  		}
   123  	}
   124  
   125  	return nil, fmt.Errorf("no injection config found for key %s at ns %s", key, namespace)
   126  }

   128  // LoadTemplateArgs takes an io.Reader and parses out an template args
   129  func LoadTemplateArgs(reader io.Reader) (*TemplateArgs, error) {
   130  	data, err := ioutil.ReadAll(reader)
   131  	if err != nil {
   132  		return nil, err
   133  	}
   134  
   135  	var cfg TemplateArgs
   136  	if err := yaml.Unmarshal(data, &cfg); err != nil {
   137  		return nil, err
   138  	}
   139  
   140  	if cfg.Name == "" {
   141  		return nil, errMissingName
   142  	}
   143  
   144  	if cfg.Template == "" {
   145  		return nil, errMissingTemplateName
   146  	}
   147  
   148  	return &cfg, nil
   149  }
