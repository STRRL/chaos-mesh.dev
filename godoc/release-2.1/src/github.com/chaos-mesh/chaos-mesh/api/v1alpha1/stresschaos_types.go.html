   101  // Normalize the stressors to comply with stress-ng
   102  func (in *Stressors) Normalize() (string, error) {
   103  	stressors := ""
   104  	if in.MemoryStressor != nil && in.MemoryStressor.Workers != 0 {
   105  		stressors += fmt.Sprintf(" --vm %d --vm-keep", in.MemoryStressor.Workers)
   106  		if len(in.MemoryStressor.Size) != 0 {
   107  			if in.MemoryStressor.Size[len(in.MemoryStressor.Size)-1] != '%' {
   108  				size, err := units.FromHumanSize(string(in.MemoryStressor.Size))
   109  				if err != nil {
   110  					return "", err
   111  				}
   112  				stressors += fmt.Sprintf(" --vm-bytes %d", size)
   113  			} else {
   114  				stressors += fmt.Sprintf(" --vm-bytes %s",
   115  					in.MemoryStressor.Size)
   116  			}
   117  		}
   118  
   119  		if in.MemoryStressor.Options != nil {
   120  			for _, v := range in.MemoryStressor.Options {
   121  				stressors += fmt.Sprintf(" %v ", v)
   122  			}
   123  		}
   124  	}
   125  	if in.CPUStressor != nil && in.CPUStressor.Workers != 0 {
   126  		// Without these two args, we may not reach the resource limit of pod,
   127  		// especially when we set cpu workers > 1
   128  		// More details see: https://github.com/chaos-mesh/chaos-mesh/issues/3100
   129  		stressors += " --cpu-load-slice 10 --cpu-method sqrt"
   130  		stressors += fmt.Sprintf(" --cpu %d", in.CPUStressor.Workers)
   131  		if in.CPUStressor.Load != nil {
   132  			stressors += fmt.Sprintf(" --cpu-load %d",
   133  				*in.CPUStressor.Load)
   134  		}
   135  
   136  		if in.CPUStressor.Options != nil {
   137  			for _, v := range in.CPUStressor.Options {
   138  				stressors += fmt.Sprintf(" %v ", v)
   139  			}
   140  		}
   141  	}
   142  	return stressors, nil
   143  }
