    44  type IOChaosSpec struct {

    46  
    47  	// Action defines the specific pod chaos action.

    52  	// Delay defines the value of I/O chaos action delay.
    53  	// A delay string is a possibly signed sequence of
    54  	// decimal numbers, each with optional fraction and a unit suffix,

    56  	// Valid time units are "ns", "us" (or "µs"), "ms", "s", "m", "h".

    58  	Delay string `json:"delay,omitempty" webhook:"Duration"`
    59  
    60  	// Errno defines the error code that returned by I/O action.

    62  	// +optional
    63  	Errno uint32 `json:"errno,omitempty" webhook:"IOErrno"`
    64  
    65  	// Attr defines the overrided attribution
    66  	// +optional
    67  	Attr *AttrOverrideSpec `json:"attr,omitempty"`
    68

    70  	// +optional

    74  	// +optional
