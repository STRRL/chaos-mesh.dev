    42  	klog.Infof("sending request to http://localhost:%d/network/recv", port)

    44  	if err != nil {
    45  		return "", err
    46  	}

    49  	defer resp.Body.Close()
    50  	if err != nil {
    51  		return "", err

    83  	body := []byte(fmt.Sprintf("{\"targetIP\":\"%s\"}", targetIP))

    85

    87  	if err != nil {
    88  		return 0, err
    89  	}
    90  
    91  	out, err := ioutil.ReadAll(resp.Body)
    92  	defer resp.Body.Close()
    93  	if err != nil {
    94  		return 0, err
    95  	}
    96  
    97  	result := string(out)
    98  	parts := strings.Split(result, " ")
    99  	if len(parts) != 2 {
   100  		return 0, fmt.Errorf("the length of parts is not 2 %v", parts)
   101  	}
   102  
   103  	if parts[0] != "OK" {
   104  		return 0, fmt.Errorf("the first part of response is not OK")
   105  	}
   106  
   107  	return strconv.ParseInt(parts[1], 10, 64)

   193  func probeNetworkCondition(c http.Client, peers []*corev1.Pod, ports []uint16, bidirection bool) map[string][][]int {
   194  	result := make(map[string][][]int)
   195  
   196  	testDelay := func(from int, to int) (int64, error) {
   197  		return testNetworkDelay(c, ports[from], peers[to].Status.PodIP)
   198  	}
   199  
   200  	for source := 0; source < len(peers); source++ {
   201  		initialTarget := source + 1
   202  		if bidirection {
   203  			initialTarget = 0
   204  		}
   205  		for target := initialTarget; target < len(peers); target++ {
   206  			if target == source {
   207  				continue
   208  			}
   209  
   210  			connectable := true
   211  
   212  			var (
   213  				wg           sync.WaitGroup
   214  				link1, link2 bool
   215  			)
   216  			wg.Add(2)
   217  			go func() {
   218  				defer wg.Done()
   219  				// case 1-1: source to target blocked?
   220  				klog.Infof("testing connectivity from %s to %s", peers[source].Name, peers[target].Name)
   221  				link1 = couldConnect(c, ports[source], peers[target].Status.PodIP, ports[target])
   222  
   223  			}()
   224  
   225  			go func() {
   226  				defer wg.Done()
   227  				// case 1-2: target to source blocked?
   228  				klog.Infof("testing connectivity from %s to %s", peers[target].Name, peers[source].Name)
   229  				link2 = couldConnect(c, ports[target], peers[source].Status.PodIP, ports[source])
   230  			}()
   231  			wg.Wait()
   232  
   233  			if !link1 {
   234  				klog.Infof("%s could not connect to %s", peers[source].Name, peers[target].Name)
   235  				result[networkConditionBlocked] = append(result[networkConditionBlocked], []int{source, target})
   236  				connectable = false
   237  			}
   238  			if !link2 {
   239  				klog.Infof("%s could not connect to %s", peers[target].Name, peers[source].Name)
   240  				result[networkConditionBlocked] = append(result[networkConditionBlocked], []int{target, source})
   241  				connectable = false
   242  			}
   243  
   244  			if !connectable {
   245  				continue
   246  			}
   247  
   248  			// case 2: slow network
   249  			klog.Infof("testing delay from %s to %s", peers[source].Name, peers[target].Name)
   250  			delay, err := testDelay(source, target)
   251  			if err != nil {
   252  				klog.Errorf("error from %d to %d: %v", source, target, err)
   253  				continue
   254  			}
   255  
   256  			klog.Infof("delay from %d to %d: %d", source, target, delay)
   257  			if delay > 100*1e6 {
   258  				klog.Infof("detect slow network from %s to %s", peers[source].Name, peers[target].Name)
   259  				result[networkConditionSlow] = append(result[networkConditionSlow], []int{source, target})
   260  				continue
   261  			}
   262  
   263  			// case 3: otherwise, good network
   264  			klog.Infof("good network from %d to %d", source, target)
   265  			result[networkConditionGood] = append(result[networkConditionGood], []int{source, target})
   266  		}
   267  	}
   268  
   269  	return result
   270  }

   272  func couldConnect(c http.Client, sourcePort uint16, targetPodIP string, targetPort uint16) bool {
   273  	err := sendUDPPacket(c, sourcePort, targetPodIP)
   274  	if err != nil {
   275  		klog.Infof("Error: %v", err)
   276  		return false
   277  	}
   278  
   279  	time.Sleep(time.Second)
   280  
   281  	data, err := recvUDPPacket(c, targetPort)
   282  	if err != nil {
   283  		klog.Infof("Error: %v, Data: %s", err, data)
   284  		return false
   285  	}
   286  
   287  	// As this function ignores the data corruption, it will only return false when the
   288  	// e2e test cannot connect to the helper, or the helper failed to send (the iptables
   289  	// rules drop the sending packet on the sender side, which will give an "operation not permitted" error)
   290  	// FIXME: slow network may also make this happens
   291  	if data != "ping\n" {
   292  		klog.Infof("mismatch data return: %s, it may happens under bad network", data)
   293  	}
   294  
   295  	return true
   296  }
