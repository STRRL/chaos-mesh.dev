    65  func main() {
    66  	parseFlags()
    67  	version.PrintVersionInfo("Controller manager")
    68  	if printVersion {
    69  		os.Exit(0)
    70  	}
    71  
    72  	rootLogger, err := log.NewDefaultZapLogger()
    73  	if err != nil {
    74  		stdlog.Fatal("failed to create root logger", err)
    75  	}
    76  	log.ReplaceGlobals(rootLogger)
    77  	ctrl.SetLogger(rootLogger)
    78  
    79  	// set RPCTimeout config
    80  	grpcUtils.RPCTimeout = ccfg.ControllerCfg.RPCTimeout
    81  	app := fx.New(
    82  		fx.Logger(log.NewLogrPrinter(rootLogger.WithName("fx"))),
    83  		fx.Supply(controllermetrics.Registry),
    84  		fx.Supply(rootLogger),
    85  		fx.Provide(metrics.NewChaosControllerManagerMetricsCollector),
    86  		fx.Options(
    87  			provider.Module,
    88  			controllers.Module,
    89  			selector.Module,
    90  			types.ChaosObjects,
    91  		),
    92  		fx.Invoke(Run),
    93  	)
    94  
    95  	app.Run()
    96  }

   110  func Run(params RunParams) error {
   111  	mgr := params.Mgr
   112  	authCli := params.AuthCli
   113  
   114  	var err error
   115  	for _, obj := range params.Objs {
   116  		if !ccfg.ShouldStartWebhook(obj.Name) {
   117  			continue
   118  		}
   119  
   120  		err = ctrl.NewWebhookManagedBy(mgr).
   121  			For(obj.Object).
   122  			Complete()
   123  		if err != nil {
   124  			return err
   125  		}
   126  	}
   127  
   128  	if ccfg.ShouldStartWebhook("schedule") {
   129  		// setup schedule webhook
   130  		err = ctrl.NewWebhookManagedBy(mgr).
   131  			For(&v1alpha1.Schedule{}).
   132  			Complete()
   133  		if err != nil {
   134  			return err
   135  		}
   136  	}
   137  
   138  	if ccfg.ShouldStartWebhook("workflow") {
   139  		err = ctrl.NewWebhookManagedBy(mgr).
   140  			For(&v1alpha1.Workflow{}).
   141  			Complete()
   142  		if err != nil {
   143  			return err
   144  		}
   145  	}
   146  
   147  	// Init metrics collector
   148  	metricsCollector := metrics.NewChaosControllerManagerMetricsCollector(mgr.GetCache(), controllermetrics.Registry, params.Logger.WithName("controller-manager-metrics"))
   149  
   150  	setupLog.Info("Setting up webhook server")
   151  	hookServer := mgr.GetWebhookServer()
   152  	hookServer.CertDir = ccfg.ControllerCfg.CertsDir
   153  	conf := config.NewConfigWatcherConf()
   154  
   155  	controllerRuntimeSignalHandler := ctrl.SetupSignalHandler()
   156  
   157  	if ccfg.ControllerCfg.PprofAddr != "0" {
   158  		go func() {
   159  			if err := http.ListenAndServe(ccfg.ControllerCfg.PprofAddr, nil); err != nil {
   160  				setupLog.Error(err, "unable to start pprof server")
   161  				os.Exit(1)
   162  			}
   163  		}()
   164  	}
   165  
   166  	if ccfg.ControllerCfg.CtrlAddr != "" {
   167  		go func() {
   168  			mutex := http.NewServeMux()
   169  			mutex.Handle("/", playground.Handler("GraphQL playground", "/query"))
   170  			mutex.Handle("/query", ctrlserver.Handler(params.Logger, mgr.GetClient(), params.Clientset, params.DaemonClientBuilder))
   171  			setupLog.Info("setup ctrlserver", "addr", ccfg.ControllerCfg.CtrlAddr)
   172  			setupLog.Error(http.ListenAndServe(ccfg.ControllerCfg.CtrlAddr, mutex), "unable to start ctrlserver")
   173  		}()
   174  	}
   175  
   176  	if err = ccfg.ControllerCfg.WatcherConfig.Verify(); err != nil {
   177  		setupLog.Error(err, "invalid environment configuration")
   178  		os.Exit(1)
   179  	}
   180  	configWatcher, err := watcher.New(*ccfg.ControllerCfg.WatcherConfig, metricsCollector)
   181  	if err != nil {
   182  		setupLog.Error(err, "unable to create config watcher")
   183  		os.Exit(1)
   184  	}
   185  
   186  	go watchConfig(configWatcher, conf, controllerRuntimeSignalHandler.Done())
   187  	hookServer.Register("/inject-v1-pod", &webhook.Admission{
   188  		Handler: &apiWebhook.PodInjector{
   189  			Config:        conf,
   190  			ControllerCfg: ccfg.ControllerCfg,
   191  			Metrics:       metricsCollector,
   192  		}},
   193  	)
   194  	hookServer.Register("/validate-auth", &webhook.Admission{
   195  		Handler: apiWebhook.NewAuthValidator(ccfg.ControllerCfg.SecurityMode, authCli,
   196  			ccfg.ControllerCfg.ClusterScoped, ccfg.ControllerCfg.TargetNamespace, ccfg.ControllerCfg.EnableFilterNamespace),
   197  	},
   198  	)
   199  
   200  	setupLog.Info("Starting manager")
   201  	if err := mgr.Start(controllerRuntimeSignalHandler); err != nil {
   202  		setupLog.Error(err, "unable to start manager")
   203  		os.Exit(1)
   204  	}
   205  
   206  	return nil
   207  }

   209  func setupWatchQueue(stopCh <-chan struct{}, configWatcher *watcher.K8sConfigMapWatcher) workqueue.Interface {
   210  	// watch for reconciliation signals, and grab configmaps, then update the running configuration
   211  	// for the server
   212  	sigChan := make(chan interface{}, 10)
   213  
   214  	queue := workqueue.NewRateLimitingQueue(&workqueue.BucketRateLimiter{Limiter: rate.NewLimiter(rate.Limit(0.5), 1)})
   215  
   216  	go func() {
   217  		for {
   218  			select {
   219  			case <-stopCh:
   220  				queue.ShutDown()
   221  				return
   222  			case <-sigChan:
   223  				queue.AddRateLimited(struct{}{})
   224  			}
   225  		}
   226  	}()
   227  
   228  	go func() {
   229  		for {
   230  			setupLog.Info("Launching watcher for ConfigMaps")
   231  			if err := configWatcher.Watch(sigChan, stopCh); err != nil {
   232  				switch err {
   233  				case watcher.ErrWatchChannelClosed:
   234  					// known issue: https://github.com/kubernetes/client-go/issues/334
   235  					setupLog.Info("watcher channel has closed, restart watcher")
   236  				default:
   237  					setupLog.Error(err, "unable to watch new ConfigMaps")
   238  					os.Exit(1)
   239  				}
   240  			}
   241  
   242  			select {
   243  			case <-stopCh:
   244  				close(sigChan)
   245  				return
   246  			default:
   247  				// sleep 2 seconds to prevent excessive log due to infinite restart
   248  				time.Sleep(2 * time.Second)
   249  			}
   250  		}
   251  	}()
   252  
   253  	return queue
   254  }

   256  func watchConfig(configWatcher *watcher.K8sConfigMapWatcher, cfg *config.Config, stopCh <-chan struct{}) {
   257  	queue := setupWatchQueue(stopCh, configWatcher)
   258  
   259  	for {
   260  		item, shutdown := queue.Get()
   261  		if shutdown {
   262  			break
   263  		}
   264  		func() {
   265  			defer queue.Done(item)
   266  
   267  			setupLog.Info("Triggering ConfigMap reconciliation")
   268  			updatedInjectionConfigs, err := configWatcher.GetInjectionConfigs()
   269  			if err != nil {
   270  				setupLog.Error(err, "unable to get ConfigMaps")
   271  				return
   272  			}
   273  
   274  			setupLog.Info("Updating server with newly loaded configurations",
   275  				"original configs count", len(cfg.Injections), "updated configs count", len(updatedInjectionConfigs))
   276  			cfg.ReplaceInjectionConfigs(updatedInjectionConfigs)
   277  			setupLog.Info("Configuration replaced")
   278  		}()
   279  	}
   280  }
