    58  func NewOption(logger logr.Logger) *ctrl.Options {
    59  	setupLog := logger.WithName("setup")
    60  
    61  	leaderElectionNamespace := config.ControllerCfg.Namespace
    62  	if len(leaderElectionNamespace) == 0 {
    63  		leaderElectionNamespace = "default"
    64  	}
    65  	options := ctrl.Options{
    66  		Scheme:                     scheme,
    67  		MetricsBindAddress:         net.JoinHostPort(config.ControllerCfg.MetricsHost, strconv.Itoa(config.ControllerCfg.MetricsPort)),
    68  		LeaderElection:             config.ControllerCfg.EnableLeaderElection,
    69  		LeaderElectionNamespace:    leaderElectionNamespace,
    70  		LeaderElectionResourceLock: "configmaps",
    71  		LeaderElectionID:           "chaos-mesh",
    72  		LeaseDuration:              &config.ControllerCfg.LeaderElectLeaseDuration,
    73  		RetryPeriod:                &config.ControllerCfg.LeaderElectRetryPeriod,
    74  		RenewDeadline:              &config.ControllerCfg.LeaderElectRenewDeadline,
    75  		Port:                       config.ControllerCfg.WebhookPort,
    76  		Host:                       config.ControllerCfg.WebhookHost,
    77  		// Don't aggregate events
    78  		EventBroadcaster: record.NewBroadcasterWithCorrelatorOptions(record.CorrelatorOptions{
    79  			MaxEvents:            math.MaxInt32,
    80  			MaxIntervalInSeconds: 1,
    81  		}),
    82  	}
    83  
    84  	if config.ControllerCfg.ClusterScoped {
    85  		setupLog.Info("Chaos controller manager is running in cluster scoped mode.")
    86  		// will not specific a certain namespace
    87  	} else {
    88  		setupLog.Info("Chaos controller manager is running in namespace scoped mode.", "targetNamespace", config.ControllerCfg.TargetNamespace)
    89  		options.Namespace = config.ControllerCfg.TargetNamespace
    90  	}
    91  
    92  	return &options
    93  }

   169  func NewControlPlaneCacheReader(logger logr.Logger, cfg *rest.Config) (controlPlaneCacheReader, error) {
   170  	mapper, err := apiutil.NewDynamicRESTMapper(cfg)
   171  	if err != nil {
   172  		return controlPlaneCacheReader{}, err
   173  	}
   174  
   175  	scheme := runtime.NewScheme()
   176  	_ = clientgoscheme.AddToScheme(scheme)
   177  
   178  	// Create the cache for the cached read client and registering informers
   179  	cacheReader, err := cache.New(cfg, cache.Options{Scheme: scheme, Mapper: mapper, Resync: nil, Namespace: config.ControllerCfg.Namespace})
   180  	if err != nil {
   181  		return controlPlaneCacheReader{}, err
   182  	}
   183  	// TODO: store the channel and use it to stop
   184  	go func() {
   185  		// FIXME: get context from parameter
   186  		err := cacheReader.Start(context.TODO())
   187  		if err != nil {
   188  			logger.Error(err, "fail to start cached client")
   189  		}
   190  	}()
   191  
   192  	c, err := client.New(cfg, client.Options{Scheme: scheme, Mapper: mapper})
   193  	if err != nil {
   194  		return controlPlaneCacheReader{}, err
   195  	}
   196  
   197  	cachedClient, err := client.NewDelegatingClient(client.NewDelegatingClientInput{
   198  		CacheReader:       cacheReader,
   199  		Client:            c,
   200  		UncachedObjects:   nil,
   201  		CacheUnstructured: false,
   202  	})
   203  	if err != nil {
   204  		return controlPlaneCacheReader{}, err
   205  	}
   206  
   207  	return controlPlaneCacheReader{
   208  		Reader: cachedClient,
   209  	}, nil
   210  }
