    26  // ExperimentStore defines operations for working with experiments.
    27  type ExperimentStore interface {
    28  	// ListMeta returns experiment metadata list from the datastore.
    29  	ListMeta(ctx context.Context, kind, namespace, name string, archived bool) ([]*ExperimentMeta, error)
    30  
    31  	// FindByUID returns an experiment by UID.
    32  	FindByUID(ctx context.Context, UID string) (*Experiment, error)
    33  
    34  	// FindMetaByUID returns an experiment metadata by UID.
    35  	FindMetaByUID(context.Context, string) (*ExperimentMeta, error)
    36  
    37  	// Set saves the experiment to datastore.
    38  	Set(context.Context, *Experiment) error
    39  
    40  	// Archive archives experiments which "archived" field is false.
    41  	Archive(ctx context.Context, namespace, name string) error
    42  
    43  	// Delete deletes the archive from the datastore.
    44  	Delete(context.Context, *Experiment) error
    45  
    46  	// DeleteByFinishTime deletes archives which time difference is greater than the given time from FinishTime.
    47  	DeleteByFinishTime(context.Context, time.Duration) error
    48  
    49  	// DeleteByUIDs deletes archives by the uid list.
    50  	DeleteByUIDs(context.Context, []string) error
    51  
    52  	// DeleteIncompleteExperiments deletes all incomplete experiments.
    53  	// If the chaos-dashboard was restarted and the experiment is completed during the restart,
    54  	// which means the experiment would never save the finish_time.
    55  	// DeleteIncompleteExperiments can be used to delete all incomplete experiments to avoid this case.
    56  	DeleteIncompleteExperiments(context.Context) error
    57  }

   115  // SelectorInfo defines the selector options of the Experiment.
   116  type SelectorInfo struct {
   117  	NamespaceSelectors  []string                          `json:"namespace_selectors" binding:"NamespaceSelectorsValid"`
   118  	LabelSelectors      map[string]string                 `json:"label_selectors" binding:"MapSelectorsValid"`
   119  	ExpressionSelectors []metav1.LabelSelectorRequirement `json:"expression_selectors" binding:"RequirementSelectorsValid"`
   120  	AnnotationSelectors map[string]string                 `json:"annotation_selectors" binding:"MapSelectorsValid"`
   121  	FieldSelectors      map[string]string                 `json:"field_selectors" binding:"MapSelectorsValid"`
   122  	PhaseSelector       []string                          `json:"phase_selectors" binding:"PhaseSelectorsValid"`
   123  
   124  	// Pods is a map of string keys and a set values that used to select pods.
   125  	// The key defines the namespace which pods belong,
   126  	// and the each values is a set of pod names.
   127  	Pods map[string][]string `json:"pods" binding:"PodsValid"`
   128  }

   130  // ParseSelector parses SelectorInfo to v1alpha1.SelectorSpec
   131  func (s *SelectorInfo) ParseSelector() v1alpha1.SelectorSpec {
   132  	selector := v1alpha1.SelectorSpec{}
   133  	selector.Namespaces = append(selector.Namespaces, s.NamespaceSelectors...)
   134  
   135  	selector.LabelSelectors = make(map[string]string)
   136  	for key, val := range s.LabelSelectors {
   137  		selector.LabelSelectors[key] = val
   138  	}
   139  
   140  	selector.ExpressionSelectors = append(selector.ExpressionSelectors, s.ExpressionSelectors...)
   141  
   142  	selector.AnnotationSelectors = make(map[string]string)
   143  	for key, val := range s.AnnotationSelectors {
   144  		selector.AnnotationSelectors[key] = val
   145  	}
   146  
   147  	selector.FieldSelectors = make(map[string]string)
   148  	for key, val := range s.FieldSelectors {
   149  		selector.FieldSelectors[key] = val
   150  	}
   151  
   152  	selector.PodPhaseSelectors = append(selector.PodPhaseSelectors, s.PhaseSelector...)
   153  
   154  	if s.Pods != nil {
   155  		selector.Pods = s.Pods
   156  	}
   157  
   158  	return selector
   159  }

   249  // ParsePodChaos Parse PodChaos JSON string into ExperimentYAMLDescription.
   250  func (e *Experiment) ParsePodChaos() (ExperimentYAMLDescription, error) {
   251  	chaos := &v1alpha1.PodChaos{}
   252  	if err := json.Unmarshal([]byte(e.Experiment), &chaos); err != nil {
   253  		return ExperimentYAMLDescription{}, err
   254  	}
   255  
   256  	return ExperimentYAMLDescription{
   257  		APIVersion: chaos.APIVersion,
   258  		Kind:       chaos.Kind,
   259  		Metadata: ExperimentYAMLMetadata{
   260  			Name:        chaos.Name,

   262  			Labels:      chaos.Labels,
