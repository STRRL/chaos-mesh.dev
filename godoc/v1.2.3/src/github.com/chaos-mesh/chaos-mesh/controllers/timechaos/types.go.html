    56  // Apply applies time-chaos
    57  func (r *endpoint) Apply(ctx context.Context, req ctrl.Request, chaos v1alpha1.InnerObject) error {
    58  	timechaos, ok := chaos.(*v1alpha1.TimeChaos)
    59  	if !ok {
    60  		err := errors.New("chaos is not timechaos")
    61  		r.Log.Error(err, "chaos is not TimeChaos", "chaos", chaos)
    62  		return err
    63  	}
    64  
    65  	timechaos.SetDefaultValue()
    66  
    67  	pods, err := selector.SelectAndFilterPods(ctx, r.Client, r.Reader, &timechaos.Spec, config.ControllerCfg.ClusterScoped, config.ControllerCfg.TargetNamespace, config.ControllerCfg.EnableFilterNamespace)
    68  
    69  	if err != nil {
    70  		r.Log.Error(err, "failed to select and filter pods")
    71  		return err
    72  	}
    73  
    74  	if err = r.applyAllPods(ctx, pods, timechaos); err != nil {
    75  		r.Log.Error(err, "failed to apply chaos on all pods")
    76  		return err
    77  	}
    78  
    79  	timechaos.Status.Experiment.PodRecords = make([]v1alpha1.PodStatus, 0, len(pods))
    80  	for _, pod := range pods {
    81  		ps := v1alpha1.PodStatus{
    82  			Namespace: pod.Namespace,
    83  			Name:      pod.Name,
    84  			HostIP:    pod.Status.HostIP,
    85  			PodIP:     pod.Status.PodIP,
    86  			Message:   fmt.Sprintf(timeChaosMsg, timechaos.Spec.TimeOffset),
    87  		}
    88  
    89  		timechaos.Status.Experiment.PodRecords = append(timechaos.Status.Experiment.PodRecords, ps)
    90  	}
    91  	r.Event(timechaos, v1.EventTypeNormal, events.ChaosInjected, "")
    92  	return nil
    93  }

    96  func (r *endpoint) Recover(ctx context.Context, req ctrl.Request, chaos v1alpha1.InnerObject) error {
    97  	timechaos, ok := chaos.(*v1alpha1.TimeChaos)
    98  	if !ok {
    99  		err := errors.New("chaos is not TimeChaos")
   100  		r.Log.Error(err, "chaos is not TimeChaos", "chaos", chaos)
   101  		return err
   102  	}
   103  
   104  	rd := recover.Delegate{Client: r.Client, Log: r.Log, RecoverIntf: &recoverer{r.Client, r.Log}}
   105  
   106  	finalizers, err := rd.CleanFinalizersAndRecover(ctx, chaos, timechaos.Finalizers, timechaos.Annotations)
   107  	if err != nil {
   108  		return err
   109  	}
   110  	timechaos.Finalizers = finalizers
   111  	r.Event(timechaos, v1.EventTypeNormal, events.ChaosRecovered, "")
   112  
   113  	return nil
   114  }

   116  func (r *recoverer) RecoverPod(ctx context.Context, pod *v1.Pod, somechaos v1alpha1.InnerObject) error {
   117  	// judged type in `Recover` already so no need to judge again
   118  	chaos, _ := somechaos.(*v1alpha1.TimeChaos)
   119  	r.Log.Info("Try to recover pod", "namespace", pod.Namespace, "name", pod.Name)
   120  
   121  	pbClient, err := utils.NewChaosDaemonClient(ctx, r.Client, pod)
   122  	if err != nil {
   123  		return err
   124  	}
   125  	defer pbClient.Close()
   126  
   127  	if len(pod.Status.ContainerStatuses) == 0 {
   128  		return fmt.Errorf("%s %s can't get the state of container", pod.Namespace, pod.Name)
   129  	}
   130  
   131  	g := errgroup.Group{}
   132  	expectedNames := make(map[string]bool)
   133  	for _, name := range chaos.Spec.ContainerNames {
   134  		expectedNames[name] = true
   135  	}
   136  	for index := range pod.Status.ContainerStatuses {
   137  		container := pod.Status.ContainerStatuses[index]
   138  
   139  		if len(expectedNames) == 0 || expectedNames[container.Name] {
   140  			g.Go(func() error {
   141  				err := r.recoverContainer(ctx, pbClient, container.ContainerID)
   142  
   143  				if err != nil {
   144  					r.Log.Error(err, "recover pod error", "namespace", pod.Namespace, "name", pod.Name)
   145  				} else {
   146  					r.Log.Info("Recover pod finished", "namespace", pod.Namespace, "name", pod.Name)
   147  				}
   148  
   149  				return err
   150  			})
   151  		}
   152  	}
   153  
   154  	return g.Wait()
   155  }

   172  func (r *endpoint) applyAllPods(ctx context.Context, pods []v1.Pod, chaos *v1alpha1.TimeChaos) error {
   173  	g := errgroup.Group{}
   174  	for index := range pods {
   175  		pod := &pods[index]
   176  
   177  		key, err := cache.MetaNamespaceKeyFunc(pod)
   178  		if err != nil {
   179  			return err
   180  		}
   181  		chaos.Finalizers = finalizer.InsertFinalizer(chaos.Finalizers, key)
   182  
   183  		g.Go(func() error {
   184  			return r.applyPod(ctx, pod, chaos)
   185  		})
   186  	}
   187  
   188  	return g.Wait()
   189  }

   222  func (r *endpoint) applyContainer(ctx context.Context, client chaosdaemon.ChaosDaemonClient, containerID string, chaos *v1alpha1.TimeChaos) error {
   223  	r.Log.Info("Try to shift time on container", "id", containerID)
   224  
   225  	mask, err := timeUtils.EncodeClkIds(chaos.Spec.ClockIds)
   226  	if err != nil {
   227  		return err
   228  	}
   229  
   230  	duration, err := time.ParseDuration(chaos.Spec.TimeOffset)
   231  	if err != nil {
   232  		return err
   233  	}
   234  
   235  	sec, nsec := secAndNSecFromDuration(duration)
   236  
   237  	r.Log.Info("setting time shift", "mask", mask, "sec", sec, "nsec", nsec)
   238  	_, err = client.SetTimeOffset(ctx, &chaosdaemon.TimeRequest{
   239  		ContainerId: containerID,
   240  		Sec:         sec,
   241  		Nsec:        nsec,
   242  		ClkIdsMask:  mask,
   243  	})
   244  
   245  	return err
   246  }

   248  func secAndNSecFromDuration(duration time.Duration) (sec int64, nsec int64) {
   249  	sec = duration.Nanoseconds() / 1e9
   250  	nsec = duration.Nanoseconds() - (sec * 1e9)
   251  
   252  	return
   253  }
