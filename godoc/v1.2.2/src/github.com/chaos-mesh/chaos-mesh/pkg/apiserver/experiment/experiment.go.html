   127  // @Description Create a new chaos experiment.
   128  // @Tags experiments
   129  // @Produce json
   130  // @Param request body core.ExperimentInfo true "Request body"
   131  // @Success 200 {object} core.ExperimentInfo
   132  // @Failure 400 {object} utils.APIError

   134  // @Router /experiments/new [post]
   135  func (s *Service) createExperiment(c *gin.Context) {
   136  	kubeCli, err := clientpool.ExtractTokenAndGetClient(c.Request.Header)
   137  	if err != nil {
   138  		_ = c.Error(utils.ErrInvalidRequest.WrapWithNoMessage(err))
   139  		return
   140  	}
   141  
   142  	exp := &core.ExperimentInfo{}
   143  	if err := c.ShouldBindJSON(exp); err != nil {
   144  		c.Status(http.StatusBadRequest)
   145  		_ = c.Error(utils.ErrInvalidRequest.WrapWithNoMessage(err))
   146  		return
   147  	}
   148  
   149  	createFuncs := map[string]createExperimentFunc{
   150  		v1alpha1.KindPodChaos:     s.createPodChaos,
   151  		v1alpha1.KindNetworkChaos: s.createNetworkChaos,
   152  		v1alpha1.KindIoChaos:      s.createIOChaos,
   153  		v1alpha1.KindStressChaos:  s.createStressChaos,
   154  		v1alpha1.KindTimeChaos:    s.createTimeChaos,
   155  		v1alpha1.KindKernelChaos:  s.createKernelChaos,
   156  		v1alpha1.KindDNSChaos:     s.createDNSChaos,
   157  		v1alpha1.KindAwsChaos:     s.createAwsChaos,
   158  	}
   159  
   160  	f, ok := createFuncs[exp.Target.Kind]
   161  	if !ok {
   162  		c.Status(http.StatusBadRequest)
   163  		_ = c.Error(utils.ErrInvalidRequest.New(exp.Target.Kind + " is not supported"))
   164  		return
   165  	}
   166  
   167  	if err := f(exp, kubeCli); err != nil {
   168  		c.Status(http.StatusInternalServerError)
   169  		_ = c.Error(utils.ErrInternalServer.WrapWithNoMessage(err))
   170  		return
   171  	}
   172  
   173  	c.JSON(http.StatusOK, exp)
   174  }
