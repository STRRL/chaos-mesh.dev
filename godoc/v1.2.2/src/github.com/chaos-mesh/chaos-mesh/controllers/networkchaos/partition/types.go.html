    62  // Apply applies the chaos operation
    63  func (e *endpoint) Apply(ctx context.Context, req ctrl.Request, chaos v1alpha1.InnerObject) error {
    64  	e.Log.Info("Applying network partition")
    65  
    66  	networkchaos, ok := chaos.(*v1alpha1.NetworkChaos)
    67  	if !ok {
    68  		err := errors.New("chaos is not NetworkChaos")
    69  		e.Log.Error(err, "chaos is not NetworkChaos", "chaos", chaos)
    70  
    71  		return err
    72  	}
    73  
    74  	source := networkchaos.Namespace + "/" + networkchaos.Name
    75  	m := podnetworkchaosmanager.New(source, e.Log, e.Client)
    76  
    77  	sources, err := selector.SelectAndFilterPods(ctx, e.Client, e.Reader, &networkchaos.Spec, config.ControllerCfg.ClusterScoped, config.ControllerCfg.TargetNamespace, config.ControllerCfg.EnableFilterNamespace)
    78  
    79  	if err != nil {
    80  		e.Log.Error(err, "failed to select and filter source pods")
    81  		return err
    82  	}
    83  
    84  	var targets []v1.Pod
    85  
    86  	if networkchaos.Spec.Target != nil {
    87  		targets, err = selector.SelectAndFilterPods(ctx, e.Client, e.Reader, networkchaos.Spec.Target, config.ControllerCfg.ClusterScoped, config.ControllerCfg.TargetNamespace, config.ControllerCfg.EnableFilterNamespace)
    88  		if err != nil {
    89  			e.Log.Error(err, "failed to select and filter target pods")
    90  			return err
    91  		}
    92  	}
    93  
    94  	sourceSet := ipset.BuildIPSet(sources, []string{}, networkchaos, sourceIPSetPostFix, source)
    95  	externalCidrs, err := netutils.ResolveCidrs(networkchaos.Spec.ExternalTargets)
    96  	if err != nil {
    97  		e.Log.Error(err, "failed to resolve external targets")
    98  		return err
    99  	}
   100  	targetSet := ipset.BuildIPSet(targets, externalCidrs, networkchaos, targetIPSetPostFix, source)
   101  
   102  	allPods := append(sources, targets...)
   103  
   104  	type podPositionTuple struct {
   105  		Pod      v1.Pod
   106  		Position string
   107  	}
   108  	keyPodMap := make(map[types.NamespacedName]podPositionTuple)
   109  	for index, pod := range allPods {
   110  		position := ""
   111  		if index < len(sources) {
   112  			position = "source"
   113  		} else {
   114  			position = "target"
   115  		}
   116  		keyPodMap[types.NamespacedName{
   117  			Name:      pod.Name,
   118  			Namespace: pod.Namespace,
   119  		}] = podPositionTuple{
   120  			Pod:      pod,
   121  			Position: position,
   122  		}
   123  	}
   124  
   125  	// Set up ipset in every related pods
   126  	for index := range allPods {
   127  		pod := allPods[index]
   128  		e.Log.Info("PODS", "name", pod.Name, "namespace", pod.Namespace)
   129  
   130  		t := m.WithInit(types.NamespacedName{
   131  			Name:      pod.Name,
   132  			Namespace: pod.Namespace,
   133  		})
   134  
   135  		t.Append(sourceSet)
   136  		t.Append(targetSet)
   137  	}
   138  
   139  	sourcesChains := []v1alpha1.RawIptables{}
   140  	targetsChains := []v1alpha1.RawIptables{}
   141  	if networkchaos.Spec.Direction == v1alpha1.To || networkchaos.Spec.Direction == v1alpha1.Both {
   142  		sourcesChains = append(sourcesChains, v1alpha1.RawIptables{
   143  			Name:      iptable.GenerateName(pb.Chain_OUTPUT, networkchaos),
   144  			Direction: v1alpha1.Output,
   145  			IPSets:    []string{targetSet.Name},
   146  			RawRuleSource: v1alpha1.RawRuleSource{
   147  				Source: source,
   148  			},
   149  		})
   150  
   151  		targetsChains = append(targetsChains, v1alpha1.RawIptables{
   152  			Name:      iptable.GenerateName(pb.Chain_INPUT, networkchaos),
   153  			Direction: v1alpha1.Input,
   154  			IPSets:    []string{sourceSet.Name},
   155  			RawRuleSource: v1alpha1.RawRuleSource{
   156  				Source: source,
   157  			},
   158  		})
   159  	}
   160  
   161  	if networkchaos.Spec.Direction == v1alpha1.From || networkchaos.Spec.Direction == v1alpha1.Both {
   162  		sourcesChains = append(sourcesChains, v1alpha1.RawIptables{
   163  			Name:      iptable.GenerateName(pb.Chain_INPUT, networkchaos),
   164  			Direction: v1alpha1.Input,
   165  			IPSets:    []string{targetSet.Name},
   166  			RawRuleSource: v1alpha1.RawRuleSource{
   167  				Source: source,
   168  			},
   169  		})
   170  
   171  		targetsChains = append(targetsChains, v1alpha1.RawIptables{
   172  			Name:      iptable.GenerateName(pb.Chain_OUTPUT, networkchaos),
   173  			Direction: v1alpha1.Output,
   174  			IPSets:    []string{sourceSet.Name},
   175  			RawRuleSource: v1alpha1.RawRuleSource{
   176  				Source: source,
   177  			},
   178  		})
   179  	}
   180  	e.Log.Info("chains prepared", "sourcesChains", sourcesChains, "targetsChains", targetsChains)
   181  
   182  	err = e.SetChains(ctx, sources, sourcesChains, m, networkchaos)
   183  	if err != nil {
   184  		return err
   185  	}
   186  
   187  	err = e.SetChains(ctx, targets, targetsChains, m, networkchaos)
   188  	if err != nil {
   189  		return err
   190  	}
   191  
   192  	responses := m.Commit(ctx)
   193  
   194  	networkchaos.Status.Experiment.PodRecords = make([]v1alpha1.PodStatus, 0, len(allPods))
   195  	for _, keyErrorTuple := range responses {
   196  		key := keyErrorTuple.Key
   197  		err := keyErrorTuple.Err
   198  		if err != nil {
   199  			if err != podnetworkchaosmanager.ErrPodNotFound && err != podnetworkchaosmanager.ErrPodNotRunning {
   200  				e.Log.Error(err, "fail to commit")
   201  			} else {
   202  				e.Log.Info("pod is not found or not running", "key", key)
   203  			}
   204  			return err
   205  		}
   206  
   207  		pod := keyPodMap[keyErrorTuple.Key]
   208  		ps := v1alpha1.PodStatus{
   209  			Namespace: pod.Pod.Namespace,
   210  			Name:      pod.Pod.Name,
   211  			HostIP:    pod.Pod.Status.HostIP,
   212  			PodIP:     pod.Pod.Status.PodIP,
   213  			Action:    string(networkchaos.Spec.Action),
   214  		}
   215  		if pod.Position == "source" {
   216  			ps.Message = networkChaosSourceMsg
   217  		} else {
   218  			ps.Message = networkChaosTargetMsg
   219  		}
   220  
   221  		// TODO: add source, target and tc action message
   222  		if networkchaos.Spec.Duration != nil {
   223  			ps.Message += fmt.Sprintf(networkPartitionActionMsg, *networkchaos.Spec.Duration)
   224  		}
   225  		networkchaos.Status.Experiment.PodRecords = append(networkchaos.Status.Experiment.PodRecords, ps)
   226  	}
   227  
   228  	e.Event(networkchaos, v1.EventTypeNormal, events.ChaosInjected, "")
   229  	return nil
   230  }

   232  // SetChains sets iptables chains for pods
   233  func (e *endpoint) SetChains(ctx context.Context, pods []v1.Pod, chains []v1alpha1.RawIptables, m *podnetworkchaosmanager.PodNetworkManager, networkchaos *v1alpha1.NetworkChaos) error {
   234  	for index := range pods {
   235  		pod := &pods[index]
   236  
   237  		key, err := cache.MetaNamespaceKeyFunc(pod)
   238  		if err != nil {
   239  			return err
   240  		}
   241  
   242  		t := m.WithInit(types.NamespacedName{
   243  			Name:      pod.Name,
   244  			Namespace: pod.Namespace,
   245  		})
   246  		for _, chain := range chains {
   247  			t.Append(chain)
   248  		}
   249  
   250  		networkchaos.Finalizers = finalizer.InsertFinalizer(networkchaos.Finalizers, key)
   251  
   252  	}
   253  	return nil
   254  }

   256  // Recover means the reconciler recovers the chaos action
   257  func (e *endpoint) Recover(ctx context.Context, req ctrl.Request, chaos v1alpha1.InnerObject) error {
   258  	networkchaos, ok := chaos.(*v1alpha1.NetworkChaos)
   259  	if !ok {
   260  		err := errors.New("chaos is not NetworkChaos")
   261  		e.Log.Error(err, "chaos is not NetworkChaos", "chaos", chaos)
   262  		return err
   263  	}
   264  
   265  	if err := e.cleanFinalizersAndRecover(ctx, networkchaos); err != nil {
   266  		return err
   267  	}
   268  	e.Event(networkchaos, v1.EventTypeNormal, events.ChaosRecovered, "")
   269  
   270  	return nil
   271  }

   273  func (e *endpoint) cleanFinalizersAndRecover(ctx context.Context, chaos *v1alpha1.NetworkChaos) error {
   274  	var result error
   275  
   276  	source := chaos.Namespace + "/" + chaos.Name
   277  	m := podnetworkchaosmanager.New(source, e.Log, e.Client)
   278  
   279  	for _, key := range chaos.Finalizers {
   280  		ns, name, err := cache.SplitMetaNamespaceKey(key)
   281  		if err != nil {
   282  			result = multierror.Append(result, err)
   283  			continue
   284  		}
   285  
   286  		_ = m.WithInit(types.NamespacedName{
   287  			Namespace: ns,
   288  			Name:      name,
   289  		})
   290  	}
   291  	responses := m.Commit(ctx)
   292  	for _, response := range responses {
   293  		key := response.Key
   294  		err := response.Err
   295  		// if pod not found or not running, directly return and giveup recover.
   296  		if err != nil {
   297  			if err != podnetworkchaosmanager.ErrPodNotFound && err != podnetworkchaosmanager.ErrPodNotRunning {
   298  				e.Log.Error(err, "fail to commit", "key", key)
   299  
   300  				result = multierror.Append(result, err)
   301  				continue
   302  			}
   303  
   304  			e.Log.Info("pod is not found or not running", "key", key)
   305  		}
   306  
   307  		chaos.Finalizers = finalizer.RemoveFromFinalizer(chaos.Finalizers, response.Key.String())
   308  	}
   309  	e.Log.Info("After recovering", "finalizers", chaos.Finalizers)
   310  
   311  	if chaos.Annotations[common.AnnotationCleanFinalizer] == common.AnnotationCleanFinalizerForced {
   312  		e.Log.Info("Force cleanup all finalizers", "chaos", chaos)
   313  		chaos.Finalizers = make([]string, 0)
   314  		return nil
   315  	}
   316  
   317  	return result
   318  }
