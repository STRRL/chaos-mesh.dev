   206  // BandwidthSpec defines detail of bandwidth limit.
   207  type BandwidthSpec struct {
   208  	// Rate is the speed knob. Allows bps, kbps, mbps, gbps, tbps unit. bps means bytes per second.
   209  	Rate string `json:"rate"`
   210  	// Limit is the number of bytes that can be queued waiting for tokens to become available.
   211  	// +kubebuilder:validation:Minimum=1
   212  	Limit uint32 `json:"limit"`
   213  	// Buffer is the maximum amount of bytes that tokens can be available for instantaneously.
   214  	// +kubebuilder:validation:Minimum=1
   215  	Buffer uint32 `json:"buffer"`
   216  	// Peakrate is the maximum depletion rate of the bucket.
   217  	// The peakrate does not need to be set, it is only necessary
   218  	// if perfect millisecond timescale shaping is required.
   219  	// +optional
   220  	// +kubebuilder:validation:Minimum=0
   221  	Peakrate *uint64 `json:"peakrate,omitempty"`
   222  	// Minburst specifies the size of the peakrate bucket. For perfect
   223  	// accuracy, should be set to the MTU of the interface.  If a
   224  	// peakrate is needed, but some burstiness is acceptable, this
   225  	// size can be raised. A 3000 byte minburst allows around 3mbit/s
   226  	// of peakrate, given 1000 byte packets.
   227  	// +optional
   228  	// +kubebuilder:validation:Minimum=0
   229  	Minburst *uint32 `json:"minburst,omitempty"`
   230  }
