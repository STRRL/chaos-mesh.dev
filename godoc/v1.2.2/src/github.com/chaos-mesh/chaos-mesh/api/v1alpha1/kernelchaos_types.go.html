    78  // FailKernRequest defines the injection conditions
    79  type FailKernRequest struct {
    80  	// FailType indicates what to fail, can be set to '0' / '1' / '2'
    81  	// If `0`, indicates slab to fail (should_failslab)
    82  	// If `1`, indicates alloc_page to fail (should_fail_alloc_page)
    83  	// If `2`, indicates bio to fail (should_fail_bio)
    84  	// You can read:
    85  	//   1. https://www.kernel.org/doc/html/latest/fault-injection/fault-injection.html
    86  	//   2. http://github.com/iovisor/bcc/blob/master/tools/inject_example.txt
    87  	// to learn more
    88  	// +kubebuilder:validation:Maximum=2
    89  	// +kubebuilder:validation:Minimum=0
    90  	FailType int32 `json:"failtype"`
    91  
    92  	// Headers indicates the appropriate kernel headers you need.
    93  	// Eg: "linux/mmzone.h", "linux/blkdev.h" and so on
    94  	Headers []string `json:"headers,omitempty"`
    95  
    96  	// Callchain indicate a special call chain, such as:
    97  	//     ext4_mount
    98  	//       -> mount_subtree
    99  	//          -> ...
   100  	//             -> should_failslab
   101  	// With an optional set of predicates and an optional set of
   102  	// parameters, which used with predicates. You can read call chan
   103  	// and predicate examples from https://github.com/chaos-mesh/bpfki/tree/develop/examples
   104  	// to learn more.
   105  	// If no special call chain, just keep Callchain empty, which means it will fail at any call chain
   106  	// with slab alloc (eg: kmalloc).
   107  	Callchain []Frame `json:"callchain,omitempty"`
   108  
   109  	// Probability indicates the fails with probability.
   110  	// If you want 1%, please set this field with 1.
   111  	// +kubebuilder:validation:Minimum=0
   112  	// +kubebuilder:validation:Maximum=100
   113  	Probability uint32 `json:"probability,omitempty"`
   114  
   115  	// Times indicates the max times of fails.
   116  	// +kubebuilder:validation:Minimum=0
   117  	Times uint32 `json:"times,omitempty"`
   118  }
   119  
   120  // Frame defines the function signature and predicate in function's body
   121  type Frame struct {
   122  	// Funcname can be find from kernel source or `/proc/kallsyms`, such as `ext4_mount`
   123  	Funcname string `json:"funcname,omitempty"`
   124  
   125  	// Parameters is used with predicate, for example, if you want to inject slab error
   126  	// in `d_alloc_parallel(struct dentry *parent, const struct qstr *name)` with a special
   127  	// name `bananas`, you need to set it to `struct dentry *parent, const struct qstr *name`
   128  	// otherwise omit it.
   129  	Parameters string `json:"parameters,omitempty"`
   130  
   131  	// Predicate will access the arguments of this Frame, example with Parameters's, you can
   132  	// set it to `STRNCMP(name->name, "bananas", 8)` to make inject only with it, or omit it
   133  	// to inject for all d_alloc_parallel call chain.
   134  	Predicate string `json:"predicate,omitempty"`
   135  }
