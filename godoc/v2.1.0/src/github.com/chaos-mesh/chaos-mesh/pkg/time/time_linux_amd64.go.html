    43  func init() {
    44  	// in this function, we will load fake image from `fakeclock/*.o`
    45  	entries, err := fakeclock.ReadDir("fakeclock")
    46  	if err != nil {
    47  		log.Error(err, "readdir from embedded fs")
    48  		os.Exit(1)
    49  	}
    50  
    51  	for _, entry := range entries {
    52  		if entry.Name() == ".embed.o" {
    53  			// skip the .embed.o file, as it's used to remove the error of `go fmt`
    54  			continue
    55  		}
    56  		path := "fakeclock/" + entry.Name()
    57  		object, err := fakeclock.ReadFile(path)
    58  		if err != nil {
    59  			log.Error(err, "read file from embedded fs", "path", path)
    60  			os.Exit(1)
    61  		}
    62  
    63  		elfFile, err := elf.NewFile(bytes.NewReader(object))
    64  		if err != nil {
    65  			log.Error(err, "parse elf", "path", path)
    66  			os.Exit(1)
    67  		}
    68  
    69  		syms, err := elfFile.Symbols()
    70  		if err != nil {
    71  			log.Error(err, "get symbols")
    72  			os.Exit(1)
    73  		}
    74  
    75  		fakeImage := FakeImage{
    76  			offset: make(map[string]int),
    77  		}
    78  		for _, r := range elfFile.Sections {
    79  			if r.Type == elf.SHT_PROGBITS && r.Name == ".text" {
    80  				fakeImage.content, err = r.Data()
    81  				if err != nil {
    82  					log.Error(err, "read text section")
    83  					os.Exit(1)
    84  				}
    85  
    86  				break
    87  			}
    88  		}
    89  
    90  		for _, r := range elfFile.Sections {
    91  			if r.Type == elf.SHT_RELA && r.Name == ".rela.text" {
    92  				rela_section, err := r.Data()
    93  				if err != nil {
    94  					log.Error(err, "read rela section")
    95  					os.Exit(1)
    96  				}
    97  				rela_section_reader := bytes.NewReader(rela_section)
    98  
    99  				var rela elf.Rela64
   100  				for rela_section_reader.Len() > 0 {
   101  					binary.Read(rela_section_reader, elfFile.ByteOrder, &rela)
   102  
   103  					symNo := rela.Info >> 32
   104  					if symNo == 0 || symNo > uint64(len(syms)) {
   105  						continue
   106  					}
   107  
   108  					// The relocation of a X86 image is like:
   109  					// Relocation section '.rela.text' at offset 0x288 contains 3 entries:
   110  					// Offset          Info           Type           Sym. Value    Sym. Name + Addend
   111  					// 000000000016  000900000002 R_X86_64_PC32     0000000000000000 CLOCK_IDS_MASK - 4
   112  					// 00000000001f  000a00000002 R_X86_64_PC32     0000000000000008 TV_NSEC_DELTA - 4
   113  					// 00000000002a  000b00000002 R_X86_64_PC32     0000000000000010 TV_SEC_DELTA - 4
   114  					//
   115  					// For example, we need to write the offset of `CLOCK_IDS_MASK` - 4 in 0x16 of the section
   116  					// If we want to put the `CLOCK_IDS_MASK` at the end of the section, it will be
   117  					// len(fakeImage.content) - 4 - 0x16
   118  
   119  					sym := &syms[symNo-1]
   120  					fakeImage.offset[sym.Name] = len(fakeImage.content)
   121  					targetOffset := uint32(len(fakeImage.content)) - uint32(rela.Off) + uint32(rela.Addend)
   122  					elfFile.ByteOrder.PutUint32(fakeImage.content[rela.Off:rela.Off+4], targetOffset)
   123  
   124  					// TODO: support other length besides uint64 (which is 8 bytes)
   125  					fakeImage.content = append(fakeImage.content, make([]byte, 8)...)
   126  				}
   127  
   128  				break
   129  			}
   130  		}
   131  
   132  		fakeImages[strings.TrimSuffix(entry.Name(), ".o")] = fakeImage
   133  	}
   134  }

   136  // ModifyTime modifies time of target process
   137  func ModifyTime(pid int, deltaSec int64, deltaNsec int64, clockIdsMask uint64) error {
   138  	// Mock point to return error in unit test
   139  	if err := mock.On("ModifyTimeError"); err != nil {
   140  		if e, ok := err.(error); ok {
   141  			return e
   142  		}
   143  		if ignore, ok := err.(bool); ok && ignore {
   144  			return nil
   145  		}
   146  	}
   147  
   148  	runtime.LockOSThread()
   149  	defer func() {
   150  		runtime.UnlockOSThread()
   151  	}()
   152  
   153  	program, err := ptrace.Trace(pid)
   154  	if err != nil {
   155  		return err
   156  	}
   157  	defer func() {
   158  		err = program.Detach()
   159  		if err != nil {
   160  			log.Error(err, "fail to detach program", "pid", program.Pid())
   161  		}
   162  	}()
   163  
   164  	var vdsoEntry *mapreader.Entry
   165  	for index := range program.Entries {
   166  		// reverse loop is faster
   167  		e := program.Entries[len(program.Entries)-index-1]
   168  		if e.Path == "[vdso]" {
   169  			vdsoEntry = &e
   170  			break
   171  		}
   172  	}
   173  	if vdsoEntry == nil {
   174  		return errors.New("cannot find [vdso] entry")
   175  	}
   176  
   177  	for name, fakeImage := range fakeImages {
   178  		switch name {
   179  		case "fake_clock_gettime":
   180  			// minus tailing variable part
   181  			// every variable has 8 bytes
   182  			constImageLen := len(fakeImage.content) - 8*len(fakeImage.offset)
   183  			var fakeEntry *mapreader.Entry
   184  
   185  			// find injected image to avoid redundant inject (which will lead to memory leak)
   186  			for _, e := range program.Entries {
   187  				e := e
   188  
   189  				image, err := program.ReadSlice(e.StartAddress, uint64(constImageLen))
   190  				if err != nil {
   191  					continue
   192  				}
   193  
   194  				if bytes.Equal(*image, fakeImage.content[0:constImageLen]) {
   195  					fakeEntry = &e
   196  					log.Info("found injected image", "addr", fakeEntry.StartAddress)
   197  					break
   198  				}
   199  			}
   200  			if fakeEntry == nil {
   201  				fakeEntry, err = program.MmapSlice(fakeImage.content)
   202  				if err != nil {
   203  					return err
   204  				}
   205  
   206  				originAddr, err := program.FindSymbolInEntry("clock_gettime", vdsoEntry)
   207  				if err != nil {
   208  					return err
   209  				}
   210  
   211  				err = program.JumpToFakeFunc(originAddr, fakeEntry.StartAddress)
   212  				if err != nil {
   213  					return err
   214  				}
   215  			}
   216  
   217  			err = program.WriteUint64ToAddr(fakeEntry.StartAddress+uint64(fakeImage.offset["CLOCK_IDS_MASK"]), clockIdsMask)
   218  			if err != nil {
   219  				return err
   220  			}
   221  
   222  			err = program.WriteUint64ToAddr(fakeEntry.StartAddress+uint64(fakeImage.offset["TV_SEC_DELTA"]), uint64(deltaSec))
   223  			if err != nil {
   224  				return err
   225  			}
   226  
   227  			err = program.WriteUint64ToAddr(fakeEntry.StartAddress+uint64(fakeImage.offset["TV_NSEC_DELTA"]), uint64(deltaNsec))
   228  			if err != nil {
   229  				return err
   230  			}
   231  		}
   232  	}
   233  	return err
   234  }
