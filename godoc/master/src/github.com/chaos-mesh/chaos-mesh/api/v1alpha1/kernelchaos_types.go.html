    53  // FailKernRequest defines the injection conditions
    54  type FailKernRequest struct {
    55  	// FailType indicates what to fail, can be set to '0' / '1' / '2'
    56  	// If `0`, indicates slab to fail (should_failslab)
    57  	// If `1`, indicates alloc_page to fail (should_fail_alloc_page)
    58  	// If `2`, indicates bio to fail (should_fail_bio)
    59  	// You can read:
    60  	//   1. https://www.kernel.org/doc/html/latest/fault-injection/fault-injection.html
    61  	//   2. http://github.com/iovisor/bcc/blob/master/tools/inject_example.txt
    62  	// to learn more
    63  	// +kubebuilder:validation:Maximum=2
    64  	// +kubebuilder:validation:Minimum=0
    65  	FailType int32 `json:"failtype"`
    66  
    67  	// Headers indicates the appropriate kernel headers you need.
    68  	// Eg: "linux/mmzone.h", "linux/blkdev.h" and so on
    69  	Headers []string `json:"headers,omitempty"`
    70  
    71  	// Callchain indicate a special call chain, such as:
    72  	//     ext4_mount
    73  	//       -> mount_subtree
    74  	//          -> ...
    75  	//             -> should_failslab
    76  	// With an optional set of predicates and an optional set of
    77  	// parameters, which used with predicates. You can read call chan
    78  	// and predicate examples from https://github.com/chaos-mesh/bpfki/tree/develop/examples
    79  	// to learn more.
    80  	// If no special call chain, just keep Callchain empty, which means it will fail at any call chain
    81  	// with slab alloc (eg: kmalloc).
    82  	Callchain []Frame `json:"callchain,omitempty"`
    83  
    84  	// Probability indicates the fails with probability.
    85  	// If you want 1%, please set this field with 1.
    86  	// +kubebuilder:validation:Minimum=0
    87  	// +kubebuilder:validation:Maximum=100
    88  	Probability uint32 `json:"probability,omitempty"`
    89  
    90  	// Times indicates the max times of fails.
    91  	// +kubebuilder:validation:Minimum=0
    92  	Times uint32 `json:"times,omitempty"`
    93  }

    95  // Frame defines the function signature and predicate in function's body
    96  type Frame struct {
    97  	// Funcname can be find from kernel source or `/proc/kallsyms`, such as `ext4_mount`
    98  	Funcname string `json:"funcname,omitempty"`
    99  
   100  	// Parameters is used with predicate, for example, if you want to inject slab error
   101  	// in `d_alloc_parallel(struct dentry *parent, const struct qstr *name)` with a special
   102  	// name `bananas`, you need to set it to `struct dentry *parent, const struct qstr *name`
   103  	// otherwise omit it.
   104  	Parameters string `json:"parameters,omitempty"`
   105  
   106  	// Predicate will access the arguments of this Frame, example with Parameters's, you can
   107  	// set it to `STRNCMP(name->name, "bananas", 8)` to make inject only with it, or omit it
   108  	// to inject for all d_alloc_parallel call chain.
   109  	Predicate string `json:"predicate,omitempty"`
   110  }
