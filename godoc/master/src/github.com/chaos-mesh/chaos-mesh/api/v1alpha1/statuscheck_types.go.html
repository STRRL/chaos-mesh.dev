    58  	// Mode defines the execution mode of the status check.
    59  	// Support type: Synchronous / Continuous

    67  	// +kubebuilder:validation:Enum=HTTP
    68  	Type StatusCheckType `json:"type"`
    69  
    70  	// Duration defines the duration of the whole status check if the
    71  	// number of failed execution does not exceed the failure threshold.
    72  	// Duration is available to both `Synchronous` and `Continuous` mode.
    73  	// A duration string is a possibly signed sequence of
    74  	// decimal numbers, each with optional fraction and a unit suffix,
    75  	// such as "300ms", "-1.5h" or "2h45m".
    76  	// Valid time units are "ns", "us" (or "µs"), "ms", "s", "m", "h".
    77  	// +optional
    78  	Duration *string `json:"duration,omitempty" webhook:"Duration"`
    79  
    80  	// TimeoutSeconds defines the number of seconds after which
    81  	// an execution of status check times out.
    82  	// +optional
    83  	// +kubebuilder:default=1
    84  	// +kubebuilder:validation:Minimum=1
    85  	TimeoutSeconds int `json:"timeoutSeconds,omitempty"`
    86  
    87  	// IntervalSeconds defines how often (in seconds) to perform
    88  	// an execution of status check.
    89  	// +optional

    91  	// +kubebuilder:validation:Minimum=1
    92  	IntervalSeconds int `json:"intervalSeconds,omitempty"`
