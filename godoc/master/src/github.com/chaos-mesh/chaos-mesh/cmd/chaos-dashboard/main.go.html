    52  func main() {
    53  	var printVersion bool
    54  
    55  	flag.BoolVar(&printVersion, "version", false, "print version information and exit")
    56  	flag.Parse()
    57  
    58  	version.PrintVersionInfo("Chaos Dashboard")
    59  	if printVersion {
    60  		os.Exit(0)
    61  	}
    62  
    63  	rootLogger, err := log.NewDefaultZapLogger()
    64  	if err != nil {
    65  		stdlog.Fatal("failed to create root logger", err)
    66  	}
    67  
    68  	ctrl.SetLogger(rootLogger)
    69  	mainLog := rootLogger.WithName("main")
    70  
    71  	dashboardConfig, err := config.GetChaosDashboardEnv()
    72  	if err != nil {
    73  		mainLog.Error(err, "invalid ChaosDashboardConfig")
    74  		os.Exit(1)
    75  	}
    76  	dashboardConfig.Version = version.Get().GitVersion
    77  
    78  	persistTTLConfigParsed, err := config.ParsePersistTTLConfig(dashboardConfig.PersistTTL)
    79  	if err != nil {
    80  		mainLog.Error(err, "invalid PersistTTLConfig")
    81  		os.Exit(1)
    82  	}
    83  
    84  	controllerRuntimeSignalHandlerContext := ctrl.SetupSignalHandler()
    85  	app := fx.New(
    86  		fx.Logger(log.NewLogrPrinter(rootLogger.WithName("fx"))),
    87  		fx.Supply(rootLogger),
    88  		fx.Provide(
    89  			func() (context.Context, *config.ChaosDashboardConfig, *ttlcontroller.TTLConfig) {
    90  				return controllerRuntimeSignalHandlerContext, dashboardConfig, persistTTLConfigParsed
    91  			},
    92  			store.Bootstrap,
    93  			collector.Bootstrap,
    94  			ttlcontroller.Bootstrap,
    95  		),
    96  		store.Module,
    97  		apiserver.Module,
    98  		fx.Invoke(collector.Register),
    99  		fx.Invoke(ttlcontroller.Register),
   100  	)
   101  
   102  	app.Run()
   103  }
