    73  	err := v1alpha1.SchemeBuilder.AddToScheme(k8sScheme.Scheme)

    88  				NewOption,

    96  		fx.Populate(&mgr),
    97  		fx.Invoke(Run),
    98  	)

   100  	defer cancel()
   101  	Expect(err).ToNot(HaveOccurred())
   102  	Expect(k8sClient).ToNot(BeNil())
   103  
   104  	if err := app.Start(startCtx); err != nil {
   105  		setupLog.Error(err, "fail to start manager")
