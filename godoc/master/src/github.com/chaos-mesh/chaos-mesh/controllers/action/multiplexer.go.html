    35  // Multiplexer could combine ChaosImpl implementations into one, and route them by Action in the ChaosSpec.
    36  // Field impl should be a struct which contains several fields with struct tag "action", each field should be an implementation of ChaosImpl.
    37  // For example:
    38  //   type tempStruct struct {
    39  //     Impl1 impltypes.ChaosImpl `action:"action1"`
    40  //     Impl2 impltypes.ChaosImpl `action:"action2"`
    41  //   }
    42  // is valid to be the field in Multiplexer.
    43  //
    44  // Because we use reflect fo iterate fields in tempStruct, so fields in tempStruct should be public/exported.
    45  //
    46  // When some Chaos like:
    47  //   type SomeChaos struct {
    48  //     ***
    49  //     Spec SomeChaosSpec `json:"spec"`
    50  //     ***
    51  //   }
    52  //   type SomeChaosSpec struct {
    53  //     ***
    54  //     // available actions: action1, action2
    55  //     Action string `json:"action"`
    56  //     ***
    57  //   }
    58  // is created, the corresponding ChaosImpl(s) for each action will be invoked by struct tag.
    59  type Multiplexer struct {
    60  	impl interface{}
    61  }

    63  func (i *Multiplexer) callAccordingToAction(action, methodName string, defaultPhase v1alpha1.Phase, args ...interface{}) (v1alpha1.Phase, error) {
    64  	implType := reflect.TypeOf(i.impl).Elem()
    65  	implVal := reflect.ValueOf(i.impl)
    66  
    67  	reflectArgs := []reflect.Value{}
    68  	for _, arg := range args {
    69  		reflectArgs = append(reflectArgs, reflect.ValueOf(arg))
    70  	}
    71  	for i := 0; i < implType.NumField(); i++ {
    72  		field := implType.Field(i)
    73  
    74  		actions := strings.Split(field.Tag.Get("action"), ",")
    75  		for i := range actions {
    76  			if actions[i] == action {
    77  				rets := implVal.Elem().FieldByIndex(field.Index).MethodByName(methodName).Call(reflectArgs)
    78  
    79  				// nil.(error) will panic :(
    80  				err := rets[1].Interface()
    81  				if err == nil {
    82  					return rets[0].Interface().(v1alpha1.Phase), nil
    83  				}
    84  
    85  				return rets[0].Interface().(v1alpha1.Phase), err.(error)
    86  			}
    87  		}
    88  	}
    89  	var gvk = ""
    90  	if obj, ok := args[len(args)-1].(runtime.Object); ok {
    91  		gvk = obj.GetObjectKind().GroupVersionKind().String()
    92  	}
    93  	return defaultPhase, NewErrorUnknownAction(gvk, action)
    94  }
