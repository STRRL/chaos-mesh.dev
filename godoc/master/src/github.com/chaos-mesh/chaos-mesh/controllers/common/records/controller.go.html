    62  // Reconcile the chaos records
    63  func (r *Reconciler) Reconcile(ctx context.Context, req ctrl.Request) (ctrl.Result, error) {
    64  	obj := r.Object.DeepCopyObject().(v1alpha1.InnerObjectWithSelector)
    65  
    66  	if err := r.Client.Get(context.TODO(), req.NamespacedName, obj); err != nil {
    67  		if apierrors.IsNotFound(err) {
    68  			r.Log.Info("chaos not found")
    69  		} else {
    70  			// TODO: handle this error
    71  			r.Log.Error(err, "unable to get chaos")
    72  		}
    73  		return ctrl.Result{}, nil
    74  	}
    75  
    76  	shouldUpdate := false
    77  
    78  	desiredPhase := obj.GetStatus().Experiment.DesiredPhase
    79  	records := obj.GetStatus().Experiment.Records
    80  	selectors := obj.GetSelectorSpecs()
    81  
    82  	if records == nil {
    83  		for name, sel := range selectors {
    84  			targets, err := r.Selector.Select(context.TODO(), sel)
    85  			if err != nil {
    86  				r.Log.Error(err, "fail to select")
    87  				r.Recorder.Event(obj, recorder.Failed{
    88  					Activity: "select targets",
    89  					Err:      err.Error(),
    90  				})
    91  				return ctrl.Result{}, nil
    92  			}
    93  
    94  			if len(targets) == 0 {
    95  				r.Log.Info("no target has been selected")
    96  				r.Recorder.Event(obj, recorder.Failed{
    97  					Activity: "select targets",
    98  					Err:      "no target has been selected",
    99  				})
   100  				return ctrl.Result{}, nil
   101  			}
   102  
   103  			for _, target := range targets {
   104  				records = append(records, &v1alpha1.Record{
   105  					Id:          target.Id(),
   106  					SelectorKey: name,
   107  					Phase:       v1alpha1.NotInjected,
   108  				})
   109  				shouldUpdate = true
   110  			}
   111  		}
   112  		// TODO: dynamic upgrade the records when some of these pods/containers stopped
   113  	}
   114  
   115  	needRetry := false
   116  	for index, record := range records {
   117  		var err error
   118  		r.Log.Info("iterating record", "record", record, "desiredPhase", desiredPhase)
   119  
   120  		// The whole running logic is a cycle:
   121  		// Not Injected -> Not Injected/* -> Injected -> Injected/* -> Not Injected
   122  		// Every step should follow the cycle. For example, if it's in "Not Injected/*" status, and it wants to recover
   123  		// then it has to apply and then recover, but not recover directly.
   124  
   125  		originalPhase := record.Phase
   126  		operation := Nothing
   127  		if desiredPhase == v1alpha1.RunningPhase && originalPhase != v1alpha1.Injected {
   128  			// The originalPhase has three possible situations: Not Injected, Not Injected/* or Injected/*
   129  			// In the first two situations, it should apply, in the last situation, it should recover
   130  
   131  			if strings.HasPrefix(string(originalPhase), string(v1alpha1.NotInjected)) {
   132  				operation = Apply
   133  			} else {
   134  				operation = Recover
   135  			}
   136  		}
   137  		if desiredPhase == v1alpha1.StoppedPhase && originalPhase != v1alpha1.NotInjected {
   138  			// The originalPhase has three possible situations: Not Injected/*, Injected, or Injected/*
   139  			// In the first one situation, it should apply, in the last two situations, it should recover
   140  
   141  			if strings.HasPrefix(string(originalPhase), string(v1alpha1.NotInjected)) {
   142  				operation = Apply
   143  			} else {
   144  				operation = Recover
   145  			}
   146  		}
   147  
   148  		if operation == Apply {
   149  			r.Log.Info("apply chaos", "id", records[index].Id)
   150  			record.Phase, err = r.Impl.Apply(context.TODO(), index, records, obj)
   151  			if record.Phase != originalPhase {
   152  				shouldUpdate = true
   153  			}
   154  			if err != nil {
   155  				// TODO: add backoff and retry mechanism
   156  				// but the retry shouldn't block other resource process
   157  				r.Log.Error(err, "fail to apply chaos")
   158  				applyFailedEvent := newRecordEvent(v1alpha1.TypeFailed, v1alpha1.Apply, err.Error())
   159  				records[index].Events = append(records[index].Events, *applyFailedEvent)
   160  				r.Recorder.Event(obj, recorder.Failed{
   161  					Activity: "apply chaos",
   162  					Err:      err.Error(),
   163  				})
   164  				needRetry = true
   165  				// if the impl.Apply() failed, we need to update the status to update the records[index].Events
   166  				shouldUpdate = true
   167  				continue
   168  			}
   169  
   170  			if record.Phase == v1alpha1.Injected {
   171  				records[index].InjectedCount++
   172  				applySucceedEvent := newRecordEvent(v1alpha1.TypeSucceeded, v1alpha1.Apply, "")
   173  				records[index].Events = append(records[index].Events, *applySucceedEvent)
   174  				r.Recorder.Event(obj, recorder.Applied{
   175  					Id: records[index].Id,
   176  				})
   177  			}
   178  		} else if operation == Recover {
   179  			r.Log.Info("recover chaos", "id", records[index].Id)
   180  			record.Phase, err = r.Impl.Recover(context.TODO(), index, records, obj)
   181  			if record.Phase != originalPhase {
   182  				shouldUpdate = true
   183  			}
   184  			if err != nil {
   185  				// TODO: add backoff and retry mechanism
   186  				// but the retry shouldn't block other resource process
   187  				r.Log.Error(err, "fail to recover chaos")
   188  				recoverFailedEvent := newRecordEvent(v1alpha1.TypeFailed, v1alpha1.Recover, err.Error())
   189  				records[index].Events = append(records[index].Events, *recoverFailedEvent)
   190  				r.Recorder.Event(obj, recorder.Failed{
   191  					Activity: "recover chaos",
   192  					Err:      err.Error(),
   193  				})
   194  				needRetry = true
   195  				// if the impl.Recover() failed, we need to update the status to update the records[index].Events
   196  				shouldUpdate = true
   197  				continue
   198  			}
   199  
   200  			if record.Phase == v1alpha1.NotInjected {
   201  				records[index].RecoveredCount++
   202  				recoverSucceedEvent := newRecordEvent(v1alpha1.TypeSucceeded, v1alpha1.Recover, "")
   203  				records[index].Events = append(records[index].Events, *recoverSucceedEvent)
   204  				r.Recorder.Event(obj, recorder.Recovered{
   205  					Id: records[index].Id,
   206  				})
   207  			}
   208  		}
   209  	}
   210  
   211  	// TODO: auto generate SetCustomStatus rather than reflect
   212  	var customStatus reflect.Value
   213  	if objWithStatus, ok := obj.(v1alpha1.InnerObjectWithCustomStatus); ok {
   214  		customStatus = reflect.Indirect(reflect.ValueOf(objWithStatus.GetCustomStatus()))
   215  	}
   216  	if shouldUpdate {
   217  		updateError := retry.RetryOnConflict(retry.DefaultBackoff, func() error {
   218  			r.Log.Info("updating records", "records", records)
   219  			obj := r.Object.DeepCopyObject().(v1alpha1.InnerObjectWithSelector)
   220  
   221  			if err := r.Client.Get(context.TODO(), req.NamespacedName, obj); err != nil {
   222  				r.Log.Error(err, "unable to get chaos")
   223  				return err
   224  			}
   225  
   226  			obj.GetStatus().Experiment.Records = records
   227  			if objWithStatus, ok := obj.(v1alpha1.InnerObjectWithCustomStatus); ok {
   228  				ptrToCustomStatus := objWithStatus.GetCustomStatus()
   229  				// TODO: auto generate SetCustomStatus rather than reflect
   230  				reflect.Indirect(reflect.ValueOf(ptrToCustomStatus)).Set(reflect.Indirect(customStatus))
   231  			}
   232  			return r.Client.Update(context.TODO(), obj)
   233  		})
   234  		if updateError != nil {
   235  			r.Log.Error(updateError, "fail to update")
   236  			r.Recorder.Event(obj, recorder.Failed{
   237  				Activity: "update records",
   238  				Err:      updateError.Error(),
   239  			})
   240  			return ctrl.Result{Requeue: true}, nil
   241  		}
   242  
   243  		r.Recorder.Event(obj, recorder.Updated{
   244  			Field: "records",
   245  		})
   246  	}
   247  	return ctrl.Result{Requeue: needRetry}, nil
   248  }
