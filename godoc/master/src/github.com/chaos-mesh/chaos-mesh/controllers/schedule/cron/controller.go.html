    50  func (r *Reconciler) Reconcile(ctx context.Context, req ctrl.Request) (ctrl.Result, error) {
    51  
    52  	schedule := &v1alpha1.Schedule{}
    53  	err := r.Get(ctx, req.NamespacedName, schedule)
    54  	if err != nil {
    55  		r.Log.Error(err, "unable to get chaos")
    56  		return ctrl.Result{}, nil
    57  	}
    58  
    59  	if schedule.IsPaused() {
    60  		r.Log.Info("not starting chaos as it is paused")
    61  		return ctrl.Result{}, nil
    62  	}
    63  
    64  	now := time.Now()
    65  	shouldSpawn := false
    66  	r.Log.Info("calculate schedule time", "schedule", schedule.Spec.Schedule, "lastScheduleTime", schedule.Status.LastScheduleTime, "now", now)
    67  	missedRun, nextRun, err := getRecentUnmetScheduleTime(schedule, now)
    68  	if err != nil {
    69  		r.Recorder.Event(schedule, recorder.Failed{
    70  			Activity: "get run time",
    71  			Err:      err.Error(),
    72  		})
    73  		return ctrl.Result{}, nil
    74  	}
    75  	if missedRun == nil {
    76  		r.Log.Info("requeue later to reconcile the schedule", "requeue-after", nextRun.Sub(now))
    77  		return ctrl.Result{RequeueAfter: nextRun.Sub(now)}, nil
    78  	}
    79  
    80  	if schedule.Spec.StartingDeadlineSeconds != nil {
    81  		if missedRun.Add(time.Second * time.Duration(*schedule.Spec.StartingDeadlineSeconds)).Before(now) {
    82  			r.Recorder.Event(schedule, recorder.MissedSchedule{
    83  				MissedRun: *missedRun,
    84  			})
    85  			return ctrl.Result{}, nil
    86  		}
    87  	}
    88  
    89  	r.Log.Info("schedule to spawn new chaos", "missedRun", missedRun, "nextRun", nextRun)
    90  	shouldSpawn = true
    91  
    92  	if shouldSpawn && schedule.Spec.ConcurrencyPolicy.IsForbid() {
    93  		list, err := r.ActiveLister.ListActiveJobs(ctx, schedule)
    94  		if err != nil {
    95  			r.Recorder.Event(schedule, recorder.Failed{
    96  				Activity: "list active jobs",
    97  				Err:      err.Error(),
    98  			})
    99  			return ctrl.Result{}, nil
   100  		}
   101  
   102  		items := reflect.ValueOf(list).Elem().FieldByName("Items")
   103  		for i := 0; i < items.Len(); i++ {
   104  			if schedule.Spec.Type != v1alpha1.ScheduleTypeWorkflow {
   105  				item := items.Index(i).Addr().Interface().(v1alpha1.InnerObject)
   106  				if !controller.IsChaosFinished(item, now) {
   107  					shouldSpawn = false
   108  					r.Recorder.Event(schedule, recorder.ScheduleForbid{
   109  						RunningName: item.GetName(),
   110  					})
   111  					r.Log.Info("forbid to spawn new chaos", "running", item.GetName())
   112  					break
   113  				}
   114  			} else {
   115  				workflow := items.Index(i).Addr().Interface().(*v1alpha1.Workflow)
   116  				if !controllers.WorkflowConditionEqualsTo(workflow.Status, v1alpha1.WorkflowConditionAccomplished, corev1.ConditionTrue) {
   117  					shouldSpawn = false
   118  					r.Recorder.Event(schedule, recorder.ScheduleForbid{
   119  						RunningName: workflow.GetObjectMeta().Name,
   120  					})
   121  					r.Log.Info("forbid to spawn new workflow", "running", workflow.GetName())
   122  					break
   123  				}
   124  			}
   125  		}
   126  	}
   127  
   128  	if shouldSpawn {
   129  		newObj, err := schedule.Spec.ScheduleItem.SpawnNewObject(schedule.Spec.Type)
   130  		if err != nil {
   131  			r.Recorder.Event(schedule, recorder.Failed{
   132  				Activity: "generate new object",
   133  				Err:      err.Error(),
   134  			})
   135  			return ctrl.Result{}, nil
   136  		}
   137  
   138  		newObj.SetOwnerReferences([]metav1.OwnerReference{
   139  			{
   140  				APIVersion:         schedule.APIVersion,
   141  				Kind:               schedule.Kind,
   142  				Name:               schedule.Name,
   143  				UID:                schedule.UID,
   144  				Controller:         &t,
   145  				BlockOwnerDeletion: &t,
   146  			},
   147  		})
   148  		newObj.SetLabels(map[string]string{
   149  			v1alpha1.LabelManagedBy: schedule.Name,
   150  		})
   151  		newObj.SetNamespace(schedule.Namespace)
   152  		newObj.SetName(names.SimpleNameGenerator.GenerateName(schedule.Name + "-"))
   153  
   154  		err = r.Create(ctx, newObj)
   155  		if err != nil {
   156  			r.Recorder.Event(schedule, recorder.Failed{
   157  				Activity: "create new object",
   158  				Err:      err.Error(),
   159  			})
   160  			r.Log.Error(err, "fail to create new object", "obj", newObj)
   161  			return ctrl.Result{}, nil
   162  		}
   163  		r.Recorder.Event(schedule, recorder.ScheduleSpawn{
   164  			Name: newObj.GetName(),
   165  		})
   166  		r.Log.Info("create new object", "namespace", newObj.GetNamespace(), "name", newObj.GetName())
   167  
   168  		lastScheduleTime := now
   169  		updateError := retry.RetryOnConflict(retry.DefaultBackoff, func() error {
   170  			r.Log.Info("updating lastScheduleTime", "time", lastScheduleTime)
   171  			schedule = schedule.DeepCopyObject().(*v1alpha1.Schedule)
   172  
   173  			if err := r.Client.Get(ctx, req.NamespacedName, schedule); err != nil {
   174  				r.Log.Error(err, "unable to get schedule")
   175  				return err
   176  			}
   177  
   178  			schedule.Status.LastScheduleTime.Time = lastScheduleTime
   179  			return r.Client.Update(ctx, schedule)
   180  		})
   181  		if updateError != nil {
   182  			r.Log.Error(updateError, "fail to update")
   183  			r.Recorder.Event(schedule, recorder.Failed{
   184  				Activity: "update lastScheduleTime",
   185  				Err:      updateError.Error(),
   186  			})
   187  			return ctrl.Result{}, nil
   188  		}
   189  
   190  		r.Recorder.Event(schedule, recorder.Updated{
   191  			Field: "lastScheduleTime",
   192  		})
   193  	}
   194  
   195  	return ctrl.Result{}, nil
   196  }

   200  func Bootstrap(mgr ctrl.Manager, client client.Client, log logr.Logger, lister *utils.ActiveLister, recorderBuilder *recorder.RecorderBuilder) error {
   201  	if !config.ShouldSpawnController(controllerName) {
   202  		return nil
   203  	}
   204  
   205  	return builder.Default(mgr).
   206  		For(&v1alpha1.Schedule{}).
   207  		Named(controllerName).
   208  		Complete(&Reconciler{
   209  			client,
   210  			log.WithName(controllerName),
   211  			lister,
   212  			recorderBuilder.Build(controllerName),
   213  		})
   214  }
