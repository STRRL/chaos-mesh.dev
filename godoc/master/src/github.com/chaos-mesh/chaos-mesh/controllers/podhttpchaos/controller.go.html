    48  func (r *Reconciler) Reconcile(ctx context.Context, req ctrl.Request) (ctrl.Result, error) {
    49  	obj := &v1alpha1.PodHttpChaos{}
    50  
    51  	if err := r.Client.Get(ctx, req.NamespacedName, obj); err != nil {
    52  		if apierrors.IsNotFound(err) {
    53  			r.Log.Info("chaos not found")
    54  		} else {
    55  			// TODO: handle this error
    56  			r.Log.Error(err, "unable to get chaos")
    57  		}
    58  		return ctrl.Result{}, nil
    59  	}
    60  
    61  	r.Log.Info("updating http chaos", "pod", obj.Namespace+"/"+obj.Name, "spec", obj.Spec)
    62  
    63  	pod := &v1.Pod{}
    64  
    65  	err := r.Client.Get(ctx, types.NamespacedName{
    66  		Name:      obj.Name,
    67  		Namespace: obj.Namespace,
    68  	}, pod)
    69  	if err != nil {
    70  		r.Log.Error(err, "fail to find pod")
    71  		return ctrl.Result{}, nil
    72  	}
    73  
    74  	observedGeneration := obj.ObjectMeta.Generation
    75  	pid := obj.Status.Pid
    76  	startTime := obj.Status.StartTime
    77  
    78  	defer func() {
    79  		var failedMessage string
    80  		if err != nil {
    81  			failedMessage = err.Error()
    82  		}
    83  
    84  		updateError := retry.RetryOnConflict(retry.DefaultBackoff, func() error {
    85  			obj := &v1alpha1.PodHttpChaos{}
    86  
    87  			if err := r.Client.Get(context.TODO(), req.NamespacedName, obj); err != nil {
    88  				r.Log.Error(err, "unable to get chaos")
    89  				return err
    90  			}
    91  
    92  			obj.Status.FailedMessage = failedMessage
    93  			obj.Status.ObservedGeneration = observedGeneration
    94  			obj.Status.Pid = pid
    95  			obj.Status.StartTime = startTime
    96  
    97  			return r.Client.Status().Update(context.TODO(), obj)
    98  		})
    99  
   100  		if updateError != nil {
   101  			r.Log.Error(updateError, "fail to update")
   102  			r.Recorder.Eventf(obj, "Normal", "Failed", "Failed to update status: %s", updateError.Error())
   103  		}
   104  	}()
   105  
   106  	pbClient, err := r.ChaosDaemonClientBuilder.Build(ctx, pod)
   107  	if err != nil {
   108  		r.Recorder.Event(obj, "Warning", "Failed", err.Error())
   109  		return ctrl.Result{Requeue: true}, nil
   110  	}
   111  	defer pbClient.Close()
   112  
   113  	if len(pod.Status.ContainerStatuses) == 0 {
   114  		err = errors.Wrapf(utils.ErrContainerNotFound, "pod %s/%s has empty container status", pod.Namespace, pod.Name)
   115  		r.Recorder.Event(obj, "Warning", "Failed", err.Error())
   116  		return ctrl.Result{}, nil
   117  	}
   118  
   119  	containerID := pod.Status.ContainerStatuses[0].ContainerID
   120  
   121  	rules := make([]v1alpha1.PodHttpChaosBaseRule, 0)
   122  	proxyPortsMap := make(map[uint32]bool)
   123  
   124  	for _, rule := range obj.Spec.Rules {
   125  		proxyPortsMap[uint32(rule.Port)] = true
   126  		rules = append(rules, rule.PodHttpChaosBaseRule)
   127  	}
   128  
   129  	var proxyPorts []uint32
   130  	for port := range proxyPortsMap {
   131  		proxyPorts = append(proxyPorts, port)
   132  	}
   133  
   134  	input, err := json.Marshal(rules)
   135  	if err != nil {
   136  		r.Recorder.Event(obj, "Warning", "Failed", err.Error())
   137  		return ctrl.Result{}, nil
   138  	}
   139  
   140  	r.Log.Info("input with", "rules", string(input))
   141  
   142  	res, err := pbClient.ApplyHttpChaos(ctx, &pb.ApplyHttpChaosRequest{
   143  		Rules:       string(input),
   144  		ProxyPorts:  proxyPorts,
   145  		ContainerId: containerID,
   146  
   147  		Instance:  obj.Status.Pid,
   148  		StartTime: obj.Status.StartTime,
   149  		EnterNS:   true,
   150  	})
   151  	if err != nil {
   152  		r.Recorder.Event(obj, "Warning", "Failed", err.Error())
   153  		return ctrl.Result{Requeue: true}, nil
   154  	}
   155  
   156  	if res.StatusCode != http.StatusOK {
   157  		err = errors.Errorf("status(%d), apply fail: %s", res.StatusCode, res.Error)
   158  		r.Recorder.Event(obj, "Warning", "Failed", err.Error())
   159  		return ctrl.Result{Requeue: true}, nil
   160  	}
   161  
   162  	pid = res.Instance
   163  	startTime = res.StartTime
   164  
   165  	return ctrl.Result{}, nil
   166  }
