    49  func (r *Reconciler) getRestConfig(ctx context.Context, secretRef v1alpha1.RemoteClusterSecretRef) (clientcmd.ClientConfig, error) {
    50  	var secret corev1.Secret
    51  	err := r.Client.Get(ctx, types.NamespacedName{
    52  		Namespace: secretRef.Namespace,
    53  		Name:      secretRef.Name,
    54  	}, &secret)
    55  	if err != nil {
    56  		return nil, errors.Wrapf(err, "get secret %s/%s", secretRef.Namespace, secretRef.Name)
    57  	}
    58  
    59  	kubeconfig := secret.Data[secretRef.Key]
    60  
    61  	config, err := clientcmd.Load(kubeconfig)
    62  	if err != nil {
    63  		return nil, errors.Wrap(err, "load kubeconfig")
    64  	}
    65  
    66  	return clientcmd.NewDefaultClientConfig(*config, nil), nil
    67  }

    69  func (r *Reconciler) Reconcile(ctx context.Context, req ctrl.Request) (ctrl.Result, error) {
    70  	var obj v1alpha1.RemoteCluster
    71  	err := r.Client.Get(ctx, req.NamespacedName, &obj)
    72  	if err != nil {
    73  		if apierrors.IsNotFound(err) {
    74  			r.Log.Info("remote cluster not found", "namespace", req.Namespace, "name", req.Name)
    75  		} else {
    76  			// TODO: handle this error
    77  			r.Log.Error(err, "unable to get remote cluster", "namespace", req.Namespace, "name", req.Name)
    78  		}
    79  		return ctrl.Result{}, nil
    80  	}
    81  
    82  	clientConfig, err := r.getRestConfig(ctx, obj.Spec.KubeConfig.SecretRef)
    83  	if err != nil {
    84  		r.Log.Error(err, "fail to get clientConfig from secret")
    85  		return ctrl.Result{Requeue: true}, nil
    86  	}
    87  
    88  	// if the remoteCluster itself is being deleted, we should remove the cluster controller manager
    89  	if !obj.DeletionTimestamp.IsZero() {
    90  		err := r.registry.Stop(ctx, obj.Name)
    91  		if err != nil {
    92  			if !errors.Is(err, clusterregistry.ErrNotExist) {
    93  				r.Log.Error(err, "fail to stop cluster")
    94  				return ctrl.Result{Requeue: true}, nil
    95  			}
    96  		}
    97  
    98  		err = r.uninstallHelmRelease(ctx, &obj, clientConfig)
    99  		if err != nil {
   100  			r.Log.Error(err, "fail to uninstall helm release")
   101  			return ctrl.Result{Requeue: true}, nil
   102  		}
   103  
   104  		err = retry.RetryOnConflict(retry.DefaultRetry, func() error {
   105  			var newObj v1alpha1.RemoteCluster
   106  			r.Client.Get(ctx, req.NamespacedName, &newObj)
   107  
   108  			newObj.Finalizers = []string{}
   109  			setRemoteClusterCondition(&newObj, v1alpha1.RemoteClusterConditionInstalled, corev1.ConditionFalse, "")
   110  
   111  			return r.Client.Update(ctx, &newObj)
   112  		})
   113  		if err != nil {
   114  			r.Log.Error(err, "fail to update finalizer", "name", obj.Name)
   115  			return ctrl.Result{Requeue: true}, nil
   116  		}
   117  		return ctrl.Result{}, nil
   118  	}
   119  
   120  	currentVersion, err := r.ensureHelmRelease(ctx, &obj, clientConfig)
   121  	if err != nil {
   122  		r.Log.Error(err, "fail to list or install remote helm release")
   123  		return ctrl.Result{Requeue: true}, nil
   124  	}
   125  
   126  	err = r.ensureClusterControllerManager(ctx, &obj, clientConfig)
   127  	if err != nil {
   128  		r.Log.Error(err, "fail to boot remote cluster controller manager")
   129  		return ctrl.Result{Requeue: true}, nil
   130  	}
   131  	obj.Finalizers = []string{remoteClusterControllerFinalizer}
   132  
   133  	if err != nil {
   134  		r.Log.Error(err, "fail to operate the helm release in remote cluster")
   135  		return ctrl.Result{Requeue: true}, nil
   136  	}
   137  
   138  	err = retry.RetryOnConflict(retry.DefaultRetry, func() error {
   139  		var newObj v1alpha1.RemoteCluster
   140  		r.Client.Get(ctx, req.NamespacedName, &newObj)
   141  
   142  		newObj.Finalizers = obj.Finalizers
   143  		setRemoteClusterCondition(&newObj, v1alpha1.RemoteClusterConditionInstalled, corev1.ConditionTrue, "")
   144  
   145  		if err = r.Client.Update(ctx, &newObj); err != nil {
   146  			return err
   147  		}
   148  		newObj.Status.CurrentVersion = currentVersion
   149  		err = r.Client.Status().Update(ctx, &newObj)
   150  		return err
   151  	})
   152  	if err != nil {
   153  		r.Log.Error(err, "fail to update finalizer", "name", obj.Name)
   154  		return ctrl.Result{Requeue: true}, nil
   155  	}
   156  
   157  	return ctrl.Result{}, nil
   158  }

   160  func (r *Reconciler) ensureClusterControllerManager(ctx context.Context, obj *v1alpha1.RemoteCluster, config clientcmd.ClientConfig) error {
   161  	restConfig, err := config.ClientConfig()
   162  	if err != nil {
   163  		return errors.Wrap(err, "get rest config from client config")
   164  	}
   165  
   166  	err = r.registry.Spawn(obj.Name, restConfig)
   167  	if err != nil {
   168  		if !errors.Is(err, clusterregistry.ErrAlreadyExist) {
   169  			return err
   170  		}
   171  	}
   172  
   173  	return nil
   174  }

   176  func (r *Reconciler) getHelmClient(ctx context.Context, clientConfig clientcmd.ClientConfig) (*helm.HelmClient, error) {
   177  	restClientGetter := helm.NewRESTClientGetter(clientConfig)

   187  func (r *Reconciler) ensureHelmRelease(ctx context.Context, obj *v1alpha1.RemoteCluster, clientConfig clientcmd.ClientConfig) (string, error) {
   188  	helmClient, err := r.getHelmClient(ctx, clientConfig)
   189  	if err != nil {
   190  		return "", err
   191  	}
   192  	release, err := helmClient.GetRelease(obj.Spec.Namespace, chaosMeshReleaseName)
   193  	if err != nil {
   194  		if errors.Is(err, driver.ErrReleaseNotFound) {
   195  			chart, err := helm.FetchChaosMeshChart(ctx, obj.Spec.Version, config.ControllerCfg.LocalHelmChartPath)
   196  			if err != nil {
   197  				return "", err
   198  			}
   199  
   200  			values := make(map[string]interface{})
   201  			if obj.Spec.ConfigOverride != nil {
   202  				err = json.Unmarshal(obj.Spec.ConfigOverride, &values)
   203  				if err != nil {
   204  					return "", err
   205  				}
   206  			}
   207  			release, err = helmClient.UpgradeOrInstall(obj.Spec.Namespace, chaosMeshReleaseName, chart, values)
   208  			if err != nil {
   209  				return "", err
   210  			}
   211  		} else {
   212  			return "", err
   213  		}
   214  	}
   215  
   216  	return release.Chart.AppVersion(), nil
   217  }

   219  func (r *Reconciler) uninstallHelmRelease(ctx context.Context, obj *v1alpha1.RemoteCluster, clientConfig clientcmd.ClientConfig) error {
   220  	helmClient, err := r.getHelmClient(ctx, clientConfig)
   221  	if err != nil {
   222  		return err
   223  	}
   224  
   225  	_, err = helmClient.GetRelease(obj.Spec.Namespace, chaosMeshReleaseName)
   226  	if err != nil {
   227  		if errors.Is(err, driver.ErrReleaseNotFound) {
   228  			return nil
   229  		}
   230  
   231  		return err
   232  	}
   233  
   234  	// the release still exist
   235  	_, err = helmClient.UninstallRelease(obj.Spec.Namespace, chaosMeshReleaseName)
   236  	if err != nil {
   237  		if errors.Is(err, driver.ErrReleaseNotFound) {
   238  			return nil
   239  		}
   240  
   241  		return err
   242  	}
   243  
   244  	return nil
   245  }
