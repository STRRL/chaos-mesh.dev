    44  // RemoteClusterRegistry will manage all controllers running on a remote
    45  // cluster. The construction of these controllers (managers) will be managed by
    46  // `fx`. The main process of constructing a controller manage is nearly the same
    47  // with the main one. The only difference is that we'll need to provide a new
    48  // `RestConfig`, and `Populate` the client to allow others to use its client.
    49  type RemoteClusterRegistry struct {
    50  	clusters map[string]*remoteCluster
    51  	logger   logr.Logger
    52  	client   client.Client
    53  
    54  	lock *sync.Mutex
    55  }

    57  func New(logger logr.Logger, client client.Client) *RemoteClusterRegistry {
    58  	return &RemoteClusterRegistry{
    59  		clusters: make(map[string]*remoteCluster),
    60  		logger:   logger.WithName("clusterregistry"),
    61  		client:   client,
    62  
    63  		lock: &sync.Mutex{},
    64  	}
    65  }

    67  // run will start the controller manager of the remote cluster
    68  func run(lc fx.Lifecycle, mgr ctrl.Manager, logger logr.Logger) error {
    69  	// TODO: use the global signal context with a cancel function
    70  	executionCtx, cancel := context.WithCancel(context.TODO())
    71  	stopChan := make(chan struct{})
    72  
    73  	go func() {
    74  		setupLog := logger.WithName("setup")
    75  		setupLog.Info("Starting manager")
    76  
    77  		if err := mgr.Start(executionCtx); err != nil {
    78  			setupLog.Error(err, "unable to start manager")
    79  			os.Exit(1)
    80  		}
    81  
    82  		stopChan <- struct{}{}
    83  	}()
    84  
    85  	lc.Append(fx.Hook{
    86  		OnStop: func(ctx context.Context) error {
    87  			cancel()
    88  			<-stopChan
    89  			return nil
    90  		},
    91  	})
    92  
    93  	return nil
    94  }

   113  // WithClient enables developer getting a client of remote cluster to operate
   114  // inside the remote cluster.
   115  //
   116  // TODO: add more kinds of client, like `no-cache` into this registry, if they
   117  // are needed
   118  func (r *RemoteClusterRegistry) WithClient(name string, f func(c client.Client) error) error {
   119  	r.lock.Lock()
   120  	defer r.lock.Unlock()
   121  
   122  	cluster, ok := r.clusters[name]
   123  	if !ok {
   124  		return errors.Wrapf(ErrNotExist, "lookup cluster: %s", name)
   125  	}
   126  
   127  	return f(cluster.Client)
   128  }

   130  // Stop stops the running controller-manager which watches the remote cluster.
   131  func (r *RemoteClusterRegistry) Stop(ctx context.Context, name string) error {
   132  	r.lock.Lock()
   133  	defer r.lock.Unlock()
   134  
   135  	cluster, ok := r.clusters[name]
   136  	if !ok {
   137  		return errors.Wrapf(ErrNotExist, "lookup cluster: %s", name)
   138  	}
   139  
   140  	err := cluster.app.Stop(ctx)
   141  	if err != nil {
   142  		return errors.Wrapf(err, "stop fx app: %s", name)
   143  	}
   144  	delete(r.clusters, name)
   145  
   146  	r.logger.Info("controller manager stopped", "name", name)
   147  
   148  	return nil
   149  }

   151  // Spawn starts the controller-manager and watches the remote cluster
   152  func (r *RemoteClusterRegistry) Spawn(name string, config *rest.Config) error {
   153  	r.lock.Lock()
   154  	defer r.lock.Unlock()
   155  
   156  	if _, ok := r.clusters[name]; ok {
   157  		return errors.Wrapf(ErrAlreadyExist, "spawn cluster: %s", name)
   158  	}
   159  
   160  	localClient := r.client
   161  	var remoteClient client.Client
   162  	app := fx.New(
   163  		fx.Logger(log.NewLogrPrinter(r.logger.WithName("remotecluster-fx-"+name))),
   164  		fx.Supply(controllermetrics.Registry),
   165  		fx.Supply(r.logger.WithName("remotecluster-"+name)),
   166  		fx.Supply(config),
   167  		fx.Supply(fx.Annotated{
   168  			Name:   "cluster-name",
   169  			Target: name,
   170  		}),
   171  		fx.Provide(
   172  			fx.Annotate(func() client.Client {
   173  				return localClient
   174  			}, fx.ResultTags(`name:"manage-client"`)),
   175  		),
   176  		fx.Provide(
   177  			controllerManagerOption,
   178  			provider.NewClient,
   179  			provider.NewManager,
   180  			provider.NewScheme,
   181  		),
   182  		// more reconcilers can be listed here to add themselves to the
   183  		// controller manager
   184  		remotepodreconciler.Module,
   185  		fx.Populate(&remoteClient),
   186  		fx.Invoke(run),
   187  	)
   188  
   189  	err := app.Start(context.TODO())
   190  	if err != nil {
   191  		return errors.Wrapf(err, "start controller-manager of remote cluster %s", name)
   192  	}
   193  
   194  	r.clusters[name] = &remoteCluster{
   195  		app:    app,
   196  		Client: remoteClient,
   197  	}
   198  
   199  	return nil
   200  }
