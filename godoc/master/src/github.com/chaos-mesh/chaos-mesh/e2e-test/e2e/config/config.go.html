    46  func LoadClientRawConfig() (clientcmdapi.Config, error) {
    47  	loadingRules := clientcmd.NewDefaultClientConfigLoadingRules()
    48  	loadingRules.ExplicitPath = framework.TestContext.KubeConfig
    49  	overrides := &clientcmd.ConfigOverrides{ClusterDefaults: clientcmd.ClusterDefaults}

    51  		overrides.CurrentContext = framework.TestContext.KubeContext
