    43  // handleFlags sets up all flags and parses the command line.
    44  func handleFlags() {
    45  	config.CopyFlags(config.Flags, flag.CommandLine)
    46  	framework.RegisterCommonFlags(flag.CommandLine)
    47  	hackRegisterClusterFlags(flag.CommandLine)
    48  	e2econfig.RegisterOperatorFlags(flag.CommandLine)
    49  	flag.Parse()
    50  }

    52  func TestMain(m *testing.M) {
    53  	// Register test flags, then parse flags.
    54  	handleFlags()
    55  
    56  	framework.AfterReadingAllFlags(&framework.TestContext)
    57  
    58  	if framework.TestContext.RepoRoot != "" {
    59  		testfiles.AddFileSource(testfiles.RootFileSource{Root: framework.TestContext.RepoRoot})
    60  	}
    61  
    62  	rand.Seed(time.Now().UnixNano())
    63  	os.Exit(m.Run())
    64  }

    70  func RunE2ETests(t *testing.T) {
    71  	runtimeutils.ReallyCrash = true
    72  	logs.InitLogs()
    73  	defer logs.FlushLogs()
    74  
    75  	gomega.RegisterFailHandler(framework.Fail)
    76  
    77  	// Run tests through the Ginkgo runner with output to console + JUnit for Jenkins
    78  	var r []ginkgo.Reporter
    79  	r = append(r, reporters.NewJUnitReporter(path.Join(framework.TestContext.ReportDir, fmt.Sprintf("junit_%v%02d.xml", framework.TestContext.ReportPrefix, ginkgoconfig.GinkgoConfig.ParallelNode))))
    80  	klog.Infof("Starting e2e run %q on Ginkgo node %d", framework.RunID, ginkgoconfig.GinkgoConfig.ParallelNode)
    81  
    82  	ginkgo.RunSpecsWithDefaultAndCustomReporters(t, "chaosmesh e2e suit", r)
    83  }
