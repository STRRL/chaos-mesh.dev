    47  // Server defines a server to manage collectors.
    48  type Server struct {
    49  	Manager ctrl.Manager
    50  	logger  logr.Logger

    53  // NewServer returns a CollectorServer and Client.
    54  func NewServer(
    55  	conf *config.ChaosDashboardConfig,
    56  	experimentArchive core.ExperimentStore,
    57  	scheduleArchive core.ScheduleStore,
    58  	event core.EventStore,
    59  	workflowStore core.WorkflowStore,
    60  	logger logr.Logger,
    61  ) (*Server, client.Client, client.Reader, *runtime.Scheme) {
    62  	s := &Server{logger: logger}
    63  
    64  	// namespace scoped
    65  	options := ctrl.Options{
    66  		Scheme:             scheme,
    67  		MetricsBindAddress: net.JoinHostPort(conf.MetricHost, strconv.Itoa(conf.MetricPort)),
    68  		LeaderElection:     conf.EnableLeaderElection,
    69  		Port:               9443,
    70  	}
    71  	if conf.ClusterScoped {
    72  		logger.Info("Chaos controller manager is running in cluster scoped mode.")
    73  	} else {
    74  		logger.Info("Chaos controller manager is running in namespace scoped mode.", "targetNamespace", conf.TargetNamespace)
    75  		options.Namespace = conf.TargetNamespace
    76  	}
    77  
    78  	var err error
    79  
    80  	cfg := ctrl.GetConfigOrDie()
    81  
    82  	if conf.QPS > 0 {
    83  		cfg.QPS = conf.QPS
    84  		cfg.Burst = conf.Burst
    85  	}
    86  
    87  	s.Manager, err = ctrl.NewManager(cfg, options)
    88  	if err != nil {
    89  		logger.Error(err, "unable to start collector")
    90  		os.Exit(1)
    91  	}
    92  
    93  	if conf.SecurityMode {
    94  		clientpool.K8sClients, err = clientpool.NewClientPool(cfg, scheme, 100)
    95  		if err != nil {
    96  			// this should never happen
    97  			logger.Error(err, "fail to create client pool")
    98  			os.Exit(1)
    99  		}
   100  	} else {
   101  		clientpool.K8sClients, err = clientpool.NewLocalClient(cfg, scheme)
   102  		if err != nil {
   103  			logger.Error(err, "fail to create client pool")
   104  			os.Exit(1)
   105  		}
   106  	}
   107  
   108  	for kind, chaosKind := range v1alpha1.AllKinds() {
   109  		if err = (&ChaosCollector{
   110  			Client:  s.Manager.GetClient(),
   111  			Log:     logger.WithName(kind),
   112  			archive: experimentArchive,
   113  			event:   event,
   114  		}).Setup(s.Manager, chaosKind.SpawnObject()); err != nil {
   115  			logger.Error(err, "unable to create collector", "collector", kind)
   116  			os.Exit(1)
   117  		}
   118  	}
   119  
   120  	if err = (&ScheduleCollector{
   121  		Client:  s.Manager.GetClient(),
   122  		Log:     logger.WithName("schedule-collector").WithName(v1alpha1.KindSchedule),
   123  		archive: scheduleArchive,
   124  	}).Setup(s.Manager, &v1alpha1.Schedule{}); err != nil {
   125  		logger.Error(err, "unable to create collector", "collector", v1alpha1.KindSchedule)
   126  		os.Exit(1)
   127  	}
   128  
   129  	if err = (&EventCollector{
   130  		Client: s.Manager.GetClient(),
   131  		Log:    logger.WithName("event-collector").WithName("Event"),
   132  		event:  event,
   133  	}).Setup(s.Manager, &v1.Event{}); err != nil {
   134  		logger.Error(err, "unable to create collector", "collector", v1alpha1.KindSchedule)
   135  		os.Exit(1)
   136  	}
   137  
   138  	if err = (&WorkflowCollector{
   139  		kubeClient: s.Manager.GetClient(),
   140  		Log:        logger.WithName("workflow-collector").WithName(v1alpha1.KindWorkflow),
   141  		store:      workflowStore,
   142  	}).Setup(s.Manager, &v1alpha1.Workflow{}); err != nil {
   143  		logger.Error(err, "unable to create collector", "collector", v1alpha1.KindWorkflow)
   144  		os.Exit(1)
   145  	}
   146  
   147  	return s, s.Manager.GetClient(), s.Manager.GetAPIReader(), s.Manager.GetScheme()
   148  }

   150  // Register starts collectors manager.
   151  func Register(ctx context.Context, s *Server) {
   152  	go func() {
   153  		s.logger.Info("Starting collector")
   154  		if err := s.Manager.Start(ctx); err != nil {
   155  			s.logger.Error(err, "could not start collector")
   156  			os.Exit(1)
   157  		}
   158  	}()
   159  }
