   110  func (it *WorkflowStore) DeleteByFinishTime(ctx context.Context, ttl time.Duration) error {

   112  	if err != nil {
   113  		return err

   115  
   116  	nowTime := time.Now()
   117  	for _, wfl := range workflows {

   120  				return err

   122  		}
