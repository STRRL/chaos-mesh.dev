   157  // @Description Get pods from Kubernetes cluster.

   164  func (s *Service) listPods(c *gin.Context) {
   165  	kubeCli, err := clientpool.ExtractTokenAndGetClient(c.Request.Header)
   166  	if err != nil {
   167  		_ = c.Error(utils.ErrBadRequest.WrapWithNoMessage(err))
   168  		return
   169  	}
   170  
   171  	selector := v1alpha1.PodSelectorSpec{}
   172  	if err := c.ShouldBindJSON(&selector); err != nil {
   173  		c.Status(http.StatusBadRequest)
   174  		_ = c.Error(utils.ErrBadRequest.WrapWithNoMessage(err))
   175  		return
   176  	}
   177  	ctx := context.TODO()
   178  	filteredPods, err := pod.SelectPods(ctx, kubeCli, nil, selector, s.conf.ClusterScoped, s.conf.TargetNamespace, s.conf.EnableFilterNamespace)
   179  	if err != nil {
   180  		c.Status(http.StatusInternalServerError)
   181  		_ = c.Error(utils.ErrInternalServer.WrapWithNoMessage(err))
   182  		return
   183  	}
   184  
   185  	pods := make([]Pod, 0, len(filteredPods))
   186  	for _, pod := range filteredPods {
   187  		pods = append(pods, Pod{
   188  			Name:      pod.Name,
   189  			IP:        pod.Status.PodIP,
   190  			Namespace: pod.Namespace,
   191  			State:     string(pod.Status.Phase),
   192  		})
   193  	}
   194  
   195  	c.JSON(http.StatusOK, pods)
   196  }

   392  // @Summary Get the rbac config according to the user's choice.
   393  // @Description Get the rbac config according to the user's choice.
   394  // @Tags common
   395  // @Produce json
   396  // @Success 200 {object} MapSlice
   397  // @Router /common/rbac-config [get]
   398  // @Failure 500 {object} utils.APIError
   399  func (s *Service) getRbacConfig(c *gin.Context) {
   400  	namespace := c.Query("namespace")
   401  	roleType := c.Query("role")
   402  
   403  	var serviceAccount, role, roleBinding, verbs string
   404  	randomStr := randomStringWithCharset(5, charset)
   405  
   406  	scope := namespace
   407  	if len(namespace) == 0 {
   408  		namespace = "default"
   409  		scope = "cluster"
   410  	}
   411  	if roleType == roleManager {
   412  		verbs = `"get", "list", "watch", "create", "delete", "patch", "update"`
   413  	} else if roleType == roleViewer {
   414  		verbs = `"get", "list", "watch"`
   415  	} else {
   416  		c.Status(http.StatusBadRequest)
   417  		_ = c.Error(utils.ErrBadRequest.WrapWithNoMessage(errors.New("roleType is neither manager nor viewer")))
   418  		return
   419  	}
   420  
   421  	serviceAccountName := fmt.Sprintf("account-%s-%s-%s", scope, roleType, randomStr)
   422  	roleName := fmt.Sprintf("role-%s-%s-%s", scope, roleType, randomStr)
   423  	roleBindingName := fmt.Sprintf("bind-%s-%s-%s", scope, roleType, randomStr)
   424  
   425  	serviceAccount = fmt.Sprintf(serviceAccountTemplate, namespace, serviceAccountName)
   426  	if scope == "cluster" {
   427  		role = fmt.Sprintf(clusterRoleTemplate, roleName, verbs)
   428  		roleBinding = fmt.Sprintf(clusterRoleBindingTemplate, roleBindingName, serviceAccountName, namespace, roleName)
   429  	} else {
   430  		role = fmt.Sprintf(roleTemplate, namespace, roleName, verbs)
   431  		roleBinding = fmt.Sprintf(roleBindingTemplate, roleBindingName, namespace, serviceAccountName, namespace, roleName)
   432  	}
   433  
   434  	rbacMap := make(map[string]string)
   435  	rbacMap[serviceAccountName] = serviceAccount + "\n---\n" + role + "\n---\n" + roleBinding
   436  
   437  	c.JSON(http.StatusOK, rbacMap)
   438  }
