   149  func (s *Service) listPods(c *gin.Context) {
   150  	kubeCli, err := clientpool.ExtractTokenAndGetClient(c.Request.Header)
   151  	if err != nil {
   152  		_ = c.Error(u.ErrBadRequest.WrapWithNoMessage(err))
   153  		return
   154  	}
   155  
   156  	selector := v1alpha1.PodSelectorSpec{}
   157  	if err := c.ShouldBindJSON(&selector); err != nil {
   158  		c.Status(http.StatusBadRequest)
   159  		_ = c.Error(u.ErrBadRequest.WrapWithNoMessage(err))
   160  		return
   161  	}
   162  	ctx := context.TODO()
   163  	filteredPods, err := pod.SelectPods(ctx, kubeCli, nil, selector, s.conf.ClusterScoped, s.conf.TargetNamespace, s.conf.EnableFilterNamespace)
   164  	if err != nil {
   165  		c.Status(http.StatusInternalServerError)
   166  		_ = c.Error(u.ErrInternalServer.WrapWithNoMessage(err))
   167  		return
   168  	}
   169  
   170  	pods := make([]apiservertypes.Pod, 0, len(filteredPods))
   171  	for _, pod := range filteredPods {
   172  		pods = append(pods, apiservertypes.Pod{
   173  			Name:      pod.Name,

   175  			Namespace: pod.Namespace,
   176  			State:     string(pod.Status.Phase),

   373  // @Summary Get the rbac config according to the user's choice.
   374  // @Description Get the rbac config according to the user's choice.
   375  // @Tags common
   376  // @Produce json
   377  // @Param namespace query string false "The namespace of RBAC"
   378  // @Param role query string false "The role of RBAC"
   379  // @Success 200 {object} map[string]string
   380  // @Router /common/rbac-config [get]
   381  // @Failure 500 {object} u.APIError
   382  func (s *Service) getRbacConfig(c *gin.Context) {
   383  	namespace := c.Query("namespace")
   384  	roleType := c.Query("role")
   385  
   386  	var serviceAccount, role, roleBinding, verbs string
   387  	randomStr := randomStringWithCharset(5, charset)
   388  
   389  	scope := namespace
   390  	if len(namespace) == 0 {
   391  		namespace = "default"
   392  		scope = "cluster"
   393  	}
   394  	if roleType == roleManager {
   395  		verbs = `"get", "list", "watch", "create", "delete", "patch", "update"`
   396  	} else if roleType == roleViewer {
   397  		verbs = `"get", "list", "watch"`
   398  	} else {
   399  		c.Status(http.StatusBadRequest)
   400  		_ = c.Error(u.ErrBadRequest.WrapWithNoMessage(errors.New("roleType is neither manager nor viewer")))
   401  		return
   402  	}
   403  
   404  	serviceAccountName := fmt.Sprintf("account-%s-%s-%s", scope, roleType, randomStr)
   405  	roleName := fmt.Sprintf("role-%s-%s-%s", scope, roleType, randomStr)
   406  	roleBindingName := fmt.Sprintf("bind-%s-%s-%s", scope, roleType, randomStr)
   407  
   408  	serviceAccount = fmt.Sprintf(serviceAccountTemplate, namespace, serviceAccountName)
   409  	if scope == "cluster" {
   410  		role = fmt.Sprintf(clusterRoleTemplate, roleName, verbs)
   411  		roleBinding = fmt.Sprintf(clusterRoleBindingTemplate, roleBindingName, serviceAccountName, namespace, roleName)
   412  	} else {
   413  		role = fmt.Sprintf(roleTemplate, namespace, roleName, verbs)
   414  		roleBinding = fmt.Sprintf(roleBindingTemplate, roleBindingName, namespace, serviceAccountName, namespace, roleName)
   415  	}
   416  
   417  	rbacMap := make(map[string]string)
   418  	rbacMap[serviceAccountName] = serviceAccount + "\n---\n" + role + "\n---\n" + roleBinding
   419  
   420  	c.JSON(http.StatusOK, rbacMap)
   421  }
