   395  func (r *loggerResolver) Pod(ctx context.Context, ns string, name string) (<-chan string, error) {
   396  	logs, err := r.Clientset.CoreV1().Pods(ns).GetLogs(name, &v1.PodLogOptions{Follow: true}).Stream(ctx)
   397  	if err != nil {
   398  		return nil, err
   399  	}
   400  	logChan := make(chan string)
   401  	go func() {
   402  		defer logs.Close()
   403  		reader := bufio.NewReader(logs)
   404  		for {
   405  			line, err := reader.ReadString('\n')
   406  			if err != nil {
   407  				r.Log.Error(err, fmt.Sprintf("fail to read log of pod(%s/%s)", ns, name))
   408  				break
   409  			}
   410  			select {
   411  			case logChan <- string(line):
   412  				continue
   413  			case <-time.NewTimer(time.Minute).C:
   414  				r.Log.Info(fmt.Sprintf("client has not read log of pod(%s/%s) for 1m, close channel", ns, name))
   415  				return
   416  			}
   417  		}
   418  	}()
   419  	return logChan, nil
   420  }
