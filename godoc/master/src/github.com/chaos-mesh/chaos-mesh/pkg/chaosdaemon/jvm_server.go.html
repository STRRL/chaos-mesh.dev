    39  func (s *DaemonServer) InstallJVMRules(ctx context.Context,
    40  	req *pb.InstallJVMRulesRequest) (*empty.Empty, error) {
    41  	log := s.getLoggerFromContext(ctx)
    42  	log.Info("InstallJVMRules", "request", req)
    43  	pid, err := s.crClient.GetPidFromContainerID(ctx, req.ContainerId)
    44  	if err != nil {
    45  		log.Error(err, "GetPidFromContainerID")
    46  		return nil, err
    47  	}
    48  
    49  	containerPids := []uint32{pid}
    50  	childPids, err := util.GetChildProcesses(pid, log)
    51  	if err != nil {
    52  		log.Error(err, "GetChildProcesses")
    53  	}
    54  	containerPids = append(containerPids, childPids...)
    55  	for _, containerPid := range containerPids {
    56  		name, err := util.ReadCommName(int(containerPid))
    57  		if err != nil {
    58  			log.Error(err, "ReadCommName")
    59  			continue
    60  		}
    61  		if name == "java\n" {
    62  			pid = containerPid
    63  			break
    64  		}
    65  	}
    66  
    67  	bytemanHome := os.Getenv("BYTEMAN_HOME")
    68  	if len(bytemanHome) == 0 {
    69  		return nil, errors.New("environment variable BYTEMAN_HOME not set")
    70  	}
    71  
    72  	// copy byteman.jar and byteman-helper.jar to container's namespace
    73  	if req.EnterNS {
    74  		processBuilder := bpm.DefaultProcessBuilder("sh", "-c", fmt.Sprintf("mkdir -p %s/lib/", bytemanHome)).SetContext(ctx).SetNS(pid, bpm.MountNS)
    75  		output, err := processBuilder.Build(ctx).CombinedOutput()
    76  		if err != nil {
    77  			return nil, err
    78  		}
    79  		if len(output) > 0 {
    80  			log.Info("mkdir", "output", string(output))
    81  		}
    82  
    83  		err = copyFileAcrossNS(ctx, fmt.Sprintf("%s/lib/byteman.jar", bytemanHome), "/usr/local/byteman/lib/byteman.jar", pid)
    84  		if err != nil {
    85  			return nil, err
    86  		}
    87  
    88  		err = copyFileAcrossNS(ctx, fmt.Sprintf("%s/lib/byteman-helper.jar", bytemanHome), "/usr/local/byteman/lib/byteman-helper.jar", pid)
    89  		if err != nil {
    90  			return nil, err
    91  		}
    92  		err = copyFileAcrossNS(ctx, fmt.Sprintf("%s/lib/chaos-agent.jar", bytemanHome), "/usr/local/byteman/lib/chaos-agent.jar", pid)
    93  		if err != nil {
    94  			return nil, err
    95  		}
    96  	}
    97  
    98  	bmInstallCmd := fmt.Sprintf(bmInstallCommand, req.Port, pid)
    99  	processBuilder := bpm.DefaultProcessBuilder("sh", "-c", bmInstallCmd).SetContext(ctx)
   100  	if req.EnterNS {
   101  		processBuilder = processBuilder.EnableLocalMnt()
   102  	}
   103  
   104  	cmd := processBuilder.Build(ctx)
   105  	output, err := cmd.CombinedOutput()
   106  	if err != nil {
   107  		// this error will occured when install agent more than once, and will ignore this error and continue to submit rule
   108  		errMsg1 := "Agent JAR loaded but agent failed to initialize"
   109  
   110  		// these two errors will occured when java version less or euqal to 1.8, and don't know why
   111  		// but it can install agent success even with this error, so just ignore it now.
   112  		// TODO: Investigate the cause of these two error
   113  		errMsg2 := "Provider sun.tools.attach.LinuxAttachProvider not found"
   114  		errMsg3 := "install java.io.IOException: Non-numeric value found"
   115  
   116  		// this error is caused by the different attach result codes in different java versions. In fact, the agent has attached success, just ignore it here.
   117  		// refer to https://stackoverflow.com/questions/54340438/virtualmachine-attach-throws-com-sun-tools-attach-agentloadexception-0-when-usi/54454418#54454418
   118  		errMsg4 := "com.sun.tools.attach.AgentLoadException"
   119  		if !strings.Contains(string(output), errMsg1) && !strings.Contains(string(output), errMsg2) &&
   120  			!strings.Contains(string(output), errMsg3) && !strings.Contains(string(output), errMsg4) {
   121  			log.Error(err, string(output))
   122  			return nil, errors.Wrap(err, string(output))
   123  		}
   124  		log.Info("exec comamnd", "cmd", cmd.String(), "output", string(output), "error", err.Error())
   125  	}
   126  
   127  	// submit helper jar
   128  	bmSubmitCmd := fmt.Sprintf(bmSubmitCommand, req.Port, "b", fmt.Sprintf("%s/lib/byteman-helper.jar", os.Getenv("BYTEMAN_HOME")))
   129  	processBuilder = bpm.DefaultProcessBuilder("sh", "-c", bmSubmitCmd).SetContext(ctx)
   130  	if req.EnterNS {
   131  		processBuilder = processBuilder.SetNS(pid, bpm.NetNS)
   132  	}
   133  	output, err = processBuilder.Build(ctx).CombinedOutput()
   134  	if err != nil {
   135  		log.Error(err, string(output))
   136  		return nil, err
   137  	}
   138  	if len(output) > 0 {
   139  		log.Info("submit helper jar", "output", string(output))
   140  	}
   141  
   142  	// submit rules
   143  	filename, err := writeDataIntoFile(req.Rule, "rule.btm")
   144  	if err != nil {
   145  		return nil, err
   146  	}
   147  
   148  	bmSubmitCmd = fmt.Sprintf(bmSubmitCommand, req.Port, "l", filename)
   149  	processBuilder = bpm.DefaultProcessBuilder("sh", "-c", bmSubmitCmd).SetContext(ctx)
   150  	if req.EnterNS {
   151  		processBuilder = processBuilder.SetNS(pid, bpm.NetNS)
   152  	}
   153  	output, err = processBuilder.Build(ctx).CombinedOutput()
   154  	if err != nil {
   155  		log.Error(err, string(output))
   156  		return nil, errors.Wrap(err, string(output))
   157  	}
   158  	if len(output) > 0 {
   159  		log.Info("submit rules", "output", string(output))
   160  	}
   161  
   162  	return &empty.Empty{}, nil
   163  }

   165  func (s *DaemonServer) UninstallJVMRules(ctx context.Context,
   166  	req *pb.UninstallJVMRulesRequest) (*empty.Empty, error) {
   167  	log := s.getLoggerFromContext(ctx)
   168  	log.Info("InstallJVMRules", "request", req)
   169  	pid, err := s.crClient.GetPidFromContainerID(ctx, req.ContainerId)
   170  	if err != nil {
   171  		log.Error(err, "GetPidFromContainerID")
   172  		return nil, err
   173  	}
   174  
   175  	filename, err := writeDataIntoFile(req.Rule, "rule.btm")
   176  	if err != nil {
   177  		return nil, err
   178  	}
   179  	log.Info("create btm file", "file", filename)
   180  
   181  	bmSubmitCmd := fmt.Sprintf(bmSubmitCommand, req.Port, "u", filename)
   182  	processBuilder := bpm.DefaultProcessBuilder("sh", "-c", bmSubmitCmd).SetContext(ctx)
   183  	if req.EnterNS {
   184  		processBuilder = processBuilder.SetNS(pid, bpm.NetNS)
   185  	}
   186  	output, err := processBuilder.Build(ctx).CombinedOutput()
   187  	if err != nil {
   188  		log.Error(err, string(output))
   189  		if strings.Contains(string(output), "No rule scripts to remove") {
   190  			return &empty.Empty{}, nil
   191  		}
   192  		return nil, errors.Wrap(err, string(output))
   193  	}
   194  
   195  	if len(output) > 0 {
   196  		log.Info(string(output))
   197  	}
   198  
   199  	return &empty.Empty{}, nil
   200  }

   202  func writeDataIntoFile(data string, filename string) (string, error) {
   203  	tmpfile, err := os.CreateTemp("", filename)
   204  	if err != nil {
   205  		return "", err
   206  	}
   207  
   208  	if _, err := tmpfile.WriteString(data); err != nil {
   209  		return "", err
   210  	}
   211  
   212  	if err := tmpfile.Close(); err != nil {
   213  		return "", err
   214  	}
   215  
   216  	return tmpfile.Name(), err
   217  }

   219  func copyFileAcrossNS(ctx context.Context, source string, dest string, pid uint32) error {
   220  	sourceFile, err := os.Open(source)
   221  	if err != nil {
   222  		return err
   223  	}
   224  	defer sourceFile.Close()
   225  
   226  	processBuilder := bpm.DefaultProcessBuilder("sh", "-c", fmt.Sprintf("cat > %s", dest)).SetContext(ctx)
   227  	processBuilder = processBuilder.SetNS(pid, bpm.MountNS).SetStdin(sourceFile)
   228  	output, err := processBuilder.Build(ctx).CombinedOutput()
   229  	if err != nil {
   230  		return err
   231  	}
   232  	if len(output) > 0 {
   233  		log.Info("copy file", zap.String("source", source), zap.String("destination", dest), zap.String("output", string(output)))
   234  	}
   235  
   236  	return nil
   237  }
