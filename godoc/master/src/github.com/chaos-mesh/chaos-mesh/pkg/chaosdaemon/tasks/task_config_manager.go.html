    69  func (m TaskConfigManager) AddTaskConfig(id UID, task TaskConfig) error {
    70  	if m.TaskConfigMap == nil {
    71  		return ErrTaskConfigMapNotInit
    72  	}
    73  	if _, ok := m.TaskConfigMap[id]; ok {
    74  		return errors.Wrapf(chaoserr.ErrDuplicateEntity, "uid: %s, task: %v", id, task)
    75  	}
    76  	m.TaskConfigMap[id] = task
    77  	return nil
    78  }

    80  func (m TaskConfigManager) UpdateTaskConfig(id UID, task TaskConfig) (TaskConfig, error) {
    81  	if m.TaskConfigMap == nil {
    82  		return TaskConfig{}, ErrTaskConfigMapNotInit
    83  	}
    84  	taskOld, ok := m.TaskConfigMap[id]
    85  	if !ok {
    86  		return TaskConfig{}, errors.Wrapf(ErrUIDNotFound, "uid: %s, task: %v", id, task)
    87  	}
    88  	if taskOld.main != task.main {
    89  		return TaskConfig{}, errors.Wrapf(ErrUpdateTaskConfigWithPIDChanges, "uid: %s, task: %v", id, task)
    90  	}
    91  	m.TaskConfigMap[id] = task
    92  	return taskOld, nil
    93  }

    98  		return ErrTaskConfigMapNotInit

   116  func (m TaskConfigManager) GetUIDsWithPID(id PID) []UID {
   117  	uIds := make([]UID, 0)
   118  	for uid, task := range m.TaskConfigMap {
   119  		if task.main == id {
   120  			uIds = append(uIds, uid)
   121  		}
   122  	}
   123  	return uIds
   124  }

   126  // SumTaskConfig will sum the TaskConfig with a same TaskConfig.main.
   127  // If developers want to use it with type T, they must implement Addable for *T.
   128  // IMPORTANT: Just here , we do not assume A.Add(B) == B.Add(A).
   129  // What SumTaskConfig do : A := new(TaskConfig), A.Add(B).Add(C).Add(D)... , A marked as uid.
   130  func (m TaskConfigManager) SumTaskConfig(uid UID) (TaskConfig, error) {
   131  	if m.TaskConfigMap == nil {
   132  		return TaskConfig{}, ErrTaskConfigMapNotInit
   133  	}
   134  	taskRaw, ok := m.TaskConfigMap[uid]
   135  	if !ok {
   136  		return TaskConfig{}, ErrUIDNotFound
   137  	}
   138  
   139  	task := TaskConfig{
   140  		main: taskRaw.main,
   141  		data: taskRaw.data.DeepCopy(),
   142  	}
   143  	uids := m.GetUIDsWithPID(task.main)
   144  
   145  	for _, uidTemp := range uids {
   146  		if uid == uidTemp {
   147  			continue
   148  		}
   149  		taskTemp, ok := m.TaskConfigMap[uidTemp]
   150  		if !ok {
   151  			return TaskConfig{}, ErrTaskConfigNotFound
   152  		}
   153  		AddableData, ok := task.data.(Addable)
   154  		if !ok {
   155  			return TaskConfig{}, errors.Wrapf(chaoserr.NotImplemented("Addable"), "task.Data")
   156  		}
   157  		AddableTempData, ok := taskTemp.data.(Addable)
   158  		if !ok {
   159  			return TaskConfig{}, errors.Wrapf(chaoserr.NotImplemented("Addable"), "taskTemp.Data")
   160  		}
   161  		err := AddableData.Add(AddableTempData)
   162  		if err != nil {
   163  			return TaskConfig{}, err
   164  		}
   165  	}
   166  	return task, nil
   167  }
