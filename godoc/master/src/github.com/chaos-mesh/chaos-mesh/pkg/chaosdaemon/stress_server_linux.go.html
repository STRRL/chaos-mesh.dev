    33  func (s *DaemonServer) ExecStressors(ctx context.Context,
    34  	req *pb.ExecStressRequest) (*pb.ExecStressResponse, error) {
    35  	log.Info("Executing stressors", "request", req)
    36  
    37  	// cpuStressors

    39  	if err != nil {

    50  		CpuInstance:     cpuInstance,

    57  func (s *DaemonServer) CancelStressors(ctx context.Context,
    58  	req *pb.CancelStressRequest) (*empty.Empty, error) {
    59  	CpuPid, err := strconv.Atoi(req.CpuInstance)
    60  	if err != nil {
    61  		return nil, err
    62  	}
    63  
    64  	MemoryPid, err := strconv.Atoi(req.MemoryInstance)
    65  	if err != nil {
    66  		return nil, err
    67  	}
    68  
    69  	log.Info("Canceling stressors", "request", req)
    70  
    71  	err = s.backgroundProcessManager.KillBackgroundProcess(ctx, CpuPid, req.CpuStartTime)
    72  	if err != nil {
    73  		return nil, err
    74  	}
    75  
    76  	err = s.backgroundProcessManager.KillBackgroundProcess(ctx, MemoryPid, req.MemoryStartTime)
    77  	if err != nil {
    78  		return nil, err
    79  	}
    80  
    81  	log.Info("killing stressor successfully")
    82  	return &empty.Empty{}, nil
    83  }

    85  func (s *DaemonServer) ExecCPUStressors(ctx context.Context,
    86  	req *pb.ExecStressRequest) (string, int64, error) {
    87  	pid, err := s.crClient.GetPidFromContainerID(ctx, req.Target)
    88  	if err != nil {
    89  		return "", 0, err
    90  	}
    91  	control, err := cgroups.Load(daemonCgroups.V1, daemonCgroups.PidPath(int(pid)))
    92  	if err != nil {
    93  		return "", 0, err
    94  	}
    95  
    96  	processBuilder := bpm.DefaultProcessBuilder("stress-ng", strings.Fields(req.CpuStressors)...).
    97  		EnablePause()
    98  	if req.EnterNS {
    99  		processBuilder = processBuilder.SetNS(pid, bpm.PidNS)
   100  	}
   101  	cmd := processBuilder.Build()
   102  
   103  	procState, err := s.backgroundProcessManager.StartProcess(cmd)
   104  	if err != nil {
   105  		return "", 0, err
   106  	}
   107  	log.Info("Start stress-ng successfully")
   108  	ct, err := procState.CreateTime()
   109  	if err != nil {
   110  		return "", 0, err
   111  	}
   112  
   113  	if err = control.Add(cgroups.Process{Pid: cmd.Process.Pid}); err != nil {
   114  		if kerr := cmd.Process.Kill(); kerr != nil {
   115  			log.Error(kerr, "kill stress-ng failed", "request", req)
   116  		}
   117  		return "", 0, err
   118  	}
   119  
   120  	for {
   121  		// TODO: find a better way to resume pause process
   122  		if err := cmd.Process.Signal(syscall.SIGCONT); err != nil {
   123  			return "", 0, err
   124  		}
   125  
   126  		log.Info("send signal to resume process")
   127  		time.Sleep(time.Millisecond)
   128  
   129  		comm, err := ReadCommName(cmd.Process.Pid)
   130  		if err != nil {
   131  			return "", 0, err
   132  		}
   133  		if comm != "pause\n" {
   134  			log.Info("pause has been resumed", "comm", comm)
   135  			break
   136  		}
   137  		log.Info("the process hasn't resumed, step into the following loop", "comm", comm)
   138  	}
   139  
   140  	return strconv.Itoa(cmd.Process.Pid), ct, nil
   141  }
