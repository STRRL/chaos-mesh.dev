    36  func (s *DaemonServer) ExecStressors(ctx context.Context,
    37  	req *pb.ExecStressRequest) (*pb.ExecStressResponse, error) {
    38  	log := s.getLoggerFromContext(ctx)
    39  	log.Info("Executing stressors", "request", req)
    40  
    41  	// cpuStressors
    42  	cpuProc, err := s.ExecCPUStressors(ctx, req)
    43  	if err != nil {
    44  		return nil, err
    45  	}
    46  
    47  	// memoryStressor
    48  	memoryProc, err := s.ExecMemoryStressors(ctx, req)
    49  	if err != nil {
    50  		return nil, err
    51  	}
    52  
    53  	resp := new(pb.ExecStressResponse)
    54  	if cpuProc != nil {
    55  		resp.CpuInstance = strconv.Itoa(cpuProc.Pair.Pid)
    56  		resp.CpuStartTime = cpuProc.Pair.CreateTime
    57  		resp.CpuInstanceUid = cpuProc.Uid
    58  	}
    59  	if memoryProc != nil {
    60  		resp.MemoryInstance = strconv.Itoa(memoryProc.Pair.Pid)
    61  		resp.MemoryStartTime = memoryProc.Pair.CreateTime
    62  		resp.MemoryInstanceUid = memoryProc.Uid
    63  	}
    64  
    65  	return resp, nil
    66  }

    68  func (s *DaemonServer) CancelStressors(ctx context.Context,
    69  	req *pb.CancelStressRequest) (*empty.Empty, error) {
    70  	log := s.getLoggerFromContext(ctx)
    71  	CpuPid, err := strconv.Atoi(req.CpuInstance)
    72  	if req.CpuInstance != "" && err != nil {
    73  		return nil, err
    74  	}
    75  
    76  	MemoryPid, err := strconv.Atoi(req.MemoryInstance)
    77  	if req.MemoryInstance != "" && err != nil {
    78  		return nil, err
    79  	}
    80  
    81  	if req.CpuInstanceUid == "" && CpuPid != 0 {
    82  		if uid, ok := s.backgroundProcessManager.GetUID(bpm.ProcessPair{Pid: CpuPid, CreateTime: req.CpuStartTime}); ok {
    83  			req.CpuInstanceUid = uid
    84  		}
    85  	}
    86  
    87  	if req.MemoryInstanceUid == "" && MemoryPid != 0 {
    88  		if uid, ok := s.backgroundProcessManager.GetUID(bpm.ProcessPair{Pid: MemoryPid, CreateTime: req.MemoryStartTime}); ok {
    89  			req.MemoryInstanceUid = uid
    90  		}
    91  	}
    92  
    93  	log.Info("Canceling stressors", "request", req)
    94  
    95  	if req.CpuInstanceUid != "" {
    96  		err = s.backgroundProcessManager.KillBackgroundProcess(ctx, req.CpuInstanceUid)
    97  		if err != nil {
    98  			return nil, err
    99  		}
   100  	}
   101  
   102  	if req.MemoryInstanceUid != "" {
   103  		err = s.backgroundProcessManager.KillBackgroundProcess(ctx, req.MemoryInstanceUid)
   104  		if err != nil {
   105  			return nil, err
   106  		}
   107  	}
   108  
   109  	log.Info("killing stressor successfully")
   110  	return &empty.Empty{}, nil
   111  }

   113  func GetCGroupManagerForPID(pid int) (interface{}, error) {
   114  	if cgroups.Mode() == cgroups.Unified {
   115  		groupPath, err := cgroupsv2.PidGroupPath(pid)
   116  		if err != nil {
   117  			return nil, errors.Wrap(err, "Error detecting groupPath")
   118  		}
   119  
   120  		cgroup2, err := cgroupsv2.LoadManager("/host-sys/fs/cgroup", groupPath)
   121  		if err != nil {
   122  			return nil, errors.Wrap(err, "Error loading cgroup v2 manager")
   123  		}
   124  		return cgroup2, nil
   125  
   126  	}
   127  	// By default it's cgroup v1
   128  	cgroup1, err := cgroups.Load(daemonCgroups.V1, daemonCgroups.PidPath(pid))
   129  	if err != nil {
   130  		return nil, errors.Wrap(err, "Error loading cgroup v1 manager")
   131  	}
   132  	return cgroup1, nil
   133  }

   135  func AttachProcessToCGroup(pid int, control interface{}) error {
   136  	if cgroups.Mode() == cgroups.Unified {
   137  		var cgroup2 = control.(*cgroupsv2.Manager)
   138  		return cgroup2.AddProc(uint64(pid))
   139  	}
   140  	// By default it's cgroup v1
   141  	var cgroup1 = control.(cgroups.Cgroup)
   142  	return cgroup1.Add(cgroups.Process{Pid: pid})
   143  }

   145  func (s *DaemonServer) ExecCPUStressors(ctx context.Context,
   146  	req *pb.ExecStressRequest) (*bpm.Process, error) {
   147  	log := s.getLoggerFromContext(ctx)
   148  	if req.CpuStressors == "" {
   149  		return nil, nil
   150  	}
   151  	pid, err := s.crClient.GetPidFromContainerID(ctx, req.Target)
   152  	if err != nil {
   153  		return nil, err
   154  	}
   155  
   156  	cgroupManager, err := GetCGroupManagerForPID(int(pid))
   157  	if err != nil {
   158  		return nil, err
   159  	}
   160  
   161  	processBuilder := bpm.DefaultProcessBuilder("stress-ng", strings.Fields(req.CpuStressors)...).
   162  		EnablePause()
   163  	if req.EnterNS {
   164  		processBuilder = processBuilder.SetNS(pid, bpm.PidNS)
   165  	}
   166  	cmd := processBuilder.Build(ctx)
   167  
   168  	proc, err := s.backgroundProcessManager.StartProcess(ctx, cmd)
   169  	if err != nil {
   170  		return nil, err
   171  	}
   172  	log.Info("Start stress-ng successfully", "command", cmd, "pid", proc.Pair.Pid, "uid", proc.Uid)
   173  
   174  	if err = AttachProcessToCGroup(proc.Pair.Pid, cgroupManager); err != nil {
   175  		if kerr := cmd.Process.Kill(); kerr != nil {
   176  			log.Error(kerr, "kill stress-ng failed", "request", req)
   177  		}
   178  		return nil, err
   179  	}
   180  
   181  	for {
   182  		// TODO: find a better way to resume pause process
   183  		if err := cmd.Process.Signal(syscall.SIGCONT); err != nil {
   184  			return nil, err
   185  		}
   186  
   187  		log.Info("send signal to resume process")
   188  		time.Sleep(time.Millisecond)
   189  
   190  		// TODO: integrate the resume process into the bpm
   191  		comm, err := util.ReadCommName(cmd.Process.Pid)
   192  		if err != nil {
   193  			return nil, err
   194  		}
   195  		if comm != "pause\n" {
   196  			log.Info("pause has been resumed", "comm", comm)
   197  			break
   198  		}
   199  		log.Info("the process hasn't resumed, step into the following loop", "comm", comm)
   200  	}
   201  
   202  	return proc, nil
   203  }
