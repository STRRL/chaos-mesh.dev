    52  // TaskExecutor indicate that the type can be used
    53  // for execute task here as a task config.
    54  // Addable means we can sum many task config in to one for apply.
    55  // Creator means we can use the task config to create a running task
    56  // which can Inject on PID.
    57  // Assign means we can use the task config to update an existing running task.
    58  type TaskExecutor interface {
    59  	Object
    60  	Addable
    61  	Creator
    62  	Assign
    63  }

    65  // TaskManager is a Manager for chaos tasks.
    66  // A task base on a target marked with its PID.
    67  // We assume task should implement Injectable.
    68  // We use TaskConfig.data which implement TaskExecutor to:
    69  //	Sum task configs on same PID into one.
    70  // 	Create task.
    71  // 	Assign or update task.
    72  // SO if developers wants to use functions in TaskManager ,
    73  // their imported TaskConfig need to implement interface TaskExecutor.
    74  // If developers wants to recover task successfully,
    75  // the task must implement Recoverable.
    76  // If not implement ,
    77  // the Recover function will return a ErrNotImplement("Recoverable") error.
    78  // IMPORTANT: We assume task config obey that TaskConfig.data A,B. A.Add(B)
    79  // is approximately equal to B.Add(A)
    80  type TaskManager struct {
    81  	taskConfigManager TaskConfigManager
    82  	taskMap           map[PID]Injectable
    83  
    84  	logger logr.Logger
    85  }

   127  // Update the task with a same UID, PID and new task config.
   128  // If it comes a UID not injected , Update will return ChaosErr.NotFound("UID").
   129  // If it comes the import PID of task do not equal to the last one,
   130  // Update will return ErrUpdateTaskConfigWithPIDChanges.
   131  func (cm TaskManager) Update(uid UID, pid PID, config TaskExecutor) error {
   132  	oldTask, err := cm.taskConfigManager.UpdateTaskConfig(uid, NewTaskConfig(pid, config))
   133  	if err != nil {
   134  		return err
   135  	}
   136  	err = cm.commit(uid, pid)
   137  	if err != nil {
   138  		_, _ = cm.taskConfigManager.UpdateTaskConfig(uid, oldTask)
   139  		return err
   140  	}
   141  	return nil
   142  }

   144  // Create the first task,
   145  // the New function of TaskExecutor:Creator will only be used here.
   146  // values is only the import parameter of New function in TaskExecutor:Creator.
   147  // If it comes a task are already be injected on the PID,
   148  // Create will return ChaosErr.ErrDuplicateEntity.
   149  func (cm TaskManager) Create(uid UID, pid PID, config TaskExecutor, values interface{}) error {
   150  	if _, ok := cm.taskMap[pid]; ok {
   151  		return errors.Wrapf(chaoserr.ErrDuplicateEntity, "create")
   152  	}
   153  
   154  	err := cm.taskConfigManager.AddTaskConfig(uid, NewTaskConfig(pid, config))
   155  	if err != nil {
   156  		return err
   157  	}
   158  
   159  	processTask, err := config.New(values)
   160  	if err != nil {
   161  		_ = cm.taskConfigManager.DeleteTaskConfig(uid)
   162  		return errors.Wrapf(err, "New task: %v", config)
   163  	}
   164  
   165  	cm.taskMap[pid] = processTask
   166  	err = cm.commit(uid, pid)
   167  	if err != nil {
   168  		_ = cm.taskConfigManager.DeleteTaskConfig(uid)
   169  		delete(cm.taskMap, pid)
   170  		return errors.Wrapf(err, "update new task")
   171  	}
   172  	return nil
   173  }

   175  // Apply the task when the target pid of task is already be Created.
   176  // If it comes a UID injected , Apply will return ChaosErr.ErrDuplicateEntity.
   177  // If the Process has not been Created , Apply will return ChaosErr.NotFound("PID").
   178  func (cm TaskManager) Apply(uid UID, pid PID, config TaskExecutor) error {
   179  	err := cm.taskConfigManager.AddTaskConfig(uid, NewTaskConfig(pid, config))
   180  	if err != nil {
   181  		return err
   182  	}
   183  	err = cm.commit(uid, pid)
   184  	if err != nil {
   185  		_ = cm.taskConfigManager.DeleteTaskConfig(uid)
   186  		return err
   187  	}
   188  	return nil
   189  }

   191  // Recover the task when there is no task config on PID or
   192  // recovering the task with last task config on PID.
   193  // Recover in Recoverable will be used here,
   194  // if it runs failed it will just return the error.
   195  // If Recover is failed but developer wants to clear it,
   196  // just run : TaskManager.ClearTask(pid, true).
   197  // If PID is already recovered successfully, Recover will return ChaosErr.NotFound("PID").
   198  // If UID is not Applied or Created or the target PID of UID is not the import pid,
   199  // Recover will return ChaosErr.NotFound("UID").
   200  func (cm TaskManager) Recover(uid UID, pid PID) error {
   201  	uIDs := cm.taskConfigManager.GetUIDsWithPID(pid)
   202  	if len(uIDs) == 0 {
   203  		return ErrPIDNotFound
   204  	}
   205  	if len(uIDs) == 1 {
   206  		if uIDs[0] != uid {
   207  			return ErrUIDNotFound
   208  		}
   209  		err := cm.ClearTask(pid, false)
   210  		if err != nil {
   211  			return err
   212  		}
   213  		err = cm.taskConfigManager.DeleteTaskConfig(uid)
   214  		if err != nil {
   215  			cm.logger.Error(err, "recover task with error")
   216  		}
   217  		return nil
   218  	}
   219  
   220  	err := cm.taskConfigManager.DeleteTaskConfig(uid)
   221  	if err != nil {
   222  		cm.logger.Error(err, "recover task with error")
   223  		return nil
   224  	}
   225  
   226  	err = cm.commit(uIDs[0], pid)
   227  	if err != nil {
   228  		return errors.Wrapf(err, "update new task")
   229  	}
   230  	return nil
   231  }

   233  func (cm TaskManager) commit(uid UID, pid PID) error {
   234  	task, err := cm.taskConfigManager.SumTaskConfig(uid)
   235  	if err != nil {
   236  		return errors.Wrapf(err, "unknown recovering in the taskConfigManager, UID: %v", uid)
   237  	}
   238  	process, ok := cm.taskMap[pid]
   239  	if !ok {
   240  		return errors.Wrapf(ErrPIDNotFound, "PID : %d", pid)
   241  	}
   242  	tasker, ok := task.data.(TaskExecutor)
   243  	if !ok {
   244  		return errors.New("task.Data here must implement TaskExecutor")
   245  	}
   246  	err = tasker.Assign(process)
   247  	if err != nil {
   248  		return err
   249  	}
   250  	err = process.Inject(pid)
   251  	if err != nil {
   252  		return errors.Wrapf(err, "inject existing process PID : %d", pid)
   253  	}
   254  	return nil
   255  }

   257  // ClearTask clear the task totally.
   258  // IMPORTANT: Developer should only use this function
   259  // when want to force clear task with ignoreRecoverErr==true.
   260  func (cm TaskManager) ClearTask(pid PID, ignoreRecoverErr bool) error {
   261  	if process, ok := cm.taskMap[pid]; ok {
   262  		pRecover, ok := process.(Recoverable)
   263  		if !ok {
   264  			return errors.Wrapf(chaoserr.NotImplemented("Recoverable"), "process")
   265  		}
   266  		err := pRecover.Recover(pid)
   267  		if err != nil {
   268  			if ignoreRecoverErr {
   269  				cm.logger.Error(errors.Wrapf(err, "recover chaos"), "ERR IGNORED")
   270  			} else {
   271  				return errors.Wrapf(err, "recover chaos")
   272  			}
   273  
   274  		}
   275  		delete(cm.taskMap, pid)
   276  		return nil
   277  	}
   278  	return errors.Wrapf(ErrPIDNotFound, "recovering task")
   279  }
