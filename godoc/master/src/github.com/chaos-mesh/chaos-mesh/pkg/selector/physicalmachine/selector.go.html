    70  func (impl *SelectImpl) Select(ctx context.Context, physicalMachineSelector *v1alpha1.PhysicalMachineSelector) ([]*PhysicalMachine, error) {
    71  	if physicalMachineSelector == nil {
    72  		return []*PhysicalMachine{}, nil
    73  	}
    74  
    75  	physicalMachines, err := SelectAndFilterPhysicalMachines(ctx, impl.c, impl.r, physicalMachineSelector, impl.ClusterScoped, impl.TargetNamespace, impl.EnableFilterNamespace)
    76  	if err != nil {
    77  		return nil, err
    78  	}
    79  
    80  	filtered, err := filterPhysicalMachinesByMode(physicalMachines, physicalMachineSelector.Mode, physicalMachineSelector.Value)
    81  	if err != nil {
    82  		return nil, err
    83  	}
    84  	return filtered, nil
    85  }

    99  // SelectAndFilterPhysicalMachines returns the list of physical machines that filtered by selector and SelectorMode
   100  func SelectAndFilterPhysicalMachines(ctx context.Context, c client.Client, r client.Reader, spec *v1alpha1.PhysicalMachineSelector, clusterScoped bool, targetNamespace string, enableFilterNamespace bool) ([]*PhysicalMachine, error) {
   101  	if len(spec.Address) > 0 {
   102  		var result []*PhysicalMachine
   103  		for _, address := range spec.Address {
   104  			result = append(result, &PhysicalMachine{
   105  				Address: address,
   106  			})
   107  		}
   108  		return result, nil
   109  	}
   110  
   111  	physicalMachines, err := SelectPhysicalMachines(ctx, c, r, spec.Selector, clusterScoped, targetNamespace, enableFilterNamespace)
   112  	if err != nil {
   113  		return nil, err
   114  	}
   115  
   116  	if len(physicalMachines) == 0 {
   117  		err = errors.New("no physical machine is selected")
   118  		return nil, err
   119  	}
   120  
   121  	var result []*PhysicalMachine
   122  	for _, physicalMachine := range physicalMachines {
   123  		result = append(result, &PhysicalMachine{
   124  			PhysicalMachine: physicalMachine,
   125  		})
   126  	}
   127  	return result, nil
   128  }

   130  func SelectPhysicalMachines(ctx context.Context, c client.Client, r client.Reader,
   131  	selector v1alpha1.PhysicalMachineSelectorSpec,
   132  	clusterScoped bool, targetNamespace string, enableFilterNamespace bool) ([]v1alpha1.PhysicalMachine, error) {
   133  	if len(selector.PhysicalMachines) > 0 {
   134  		return selectSpecifiedPhysicalMachines(ctx, c, selector, clusterScoped, targetNamespace, enableFilterNamespace)
   135  	}
   136  
   137  	selectorRegistry := newSelectorRegistry()
   138  	selectorChain, err := registry.Parse(selectorRegistry, selector.GenericSelectorSpec, generic.Option{
   139  		ClusterScoped:         clusterScoped,
   140  		TargetNamespace:       targetNamespace,
   141  		EnableFilterNamespace: enableFilterNamespace,
   142  	})
   143  	if err != nil {
   144  		return nil, err
   145  	}
   146  
   147  	return listPhysicalMachines(ctx, c, r, selector, selectorChain, enableFilterNamespace)
   148  }

   150  func listPhysicalMachines(ctx context.Context, c client.Client, r client.Reader, spec v1alpha1.PhysicalMachineSelectorSpec,
   151  	selectorChain generic.SelectorChain, enableFilterNamespace bool) ([]v1alpha1.PhysicalMachine, error) {
   152  	var physicalMachines []v1alpha1.PhysicalMachine
   153  	namespaceCheck := make(map[string]bool)
   154  
   155  	if err := selectorChain.ListObjects(c, r,
   156  		func(listFunc generic.ListFunc, opts client.ListOptions) error {
   157  			var pmList v1alpha1.PhysicalMachineList
   158  			if len(spec.Namespaces) > 0 {
   159  				logger, err := stdLog.NewDefaultZapLogger()
   160  				if err != nil {
   161  					return errors.Wrap(err, "failed to create logger")
   162  				}
   163  				for _, namespace := range spec.Namespaces {
   164  					if enableFilterNamespace {
   165  						allow, ok := namespaceCheck[namespace]
   166  						if !ok {
   167  							allow = genericnamespace.CheckNamespace(ctx, c, namespace, logger)
   168  							namespaceCheck[namespace] = allow
   169  						}
   170  						if !allow {
   171  							continue
   172  						}
   173  					}
   174  
   175  					opts.Namespace = namespace
   176  					if err := listFunc(ctx, &pmList, &opts); err != nil {
   177  						return err
   178  					}
   179  					physicalMachines = append(physicalMachines, pmList.Items...)
   180  				}
   181  			} else {
   182  				// in fact, this will never happen
   183  				if err := listFunc(ctx, &pmList, &opts); err != nil {
   184  					return err
   185  				}
   186  				physicalMachines = append(physicalMachines, pmList.Items...)
   187  			}
   188  			return nil
   189  		}); err != nil {
   190  		return nil, err
   191  	}
   192  
   193  	filterList := make([]v1alpha1.PhysicalMachine, 0, len(physicalMachines))
   194  	for _, physicalMachine := range physicalMachines {
   195  		physicalMachine := physicalMachine
   196  		if selectorChain.Match(&physicalMachine) {
   197  			filterList = append(filterList, physicalMachine)
   198  		}
   199  	}
   200  	return filterList, nil
   201  }

   212  func selectSpecifiedPhysicalMachines(ctx context.Context, c client.Client, spec v1alpha1.PhysicalMachineSelectorSpec,
   213  	clusterScoped bool, targetNamespace string, enableFilterNamespace bool) ([]v1alpha1.PhysicalMachine, error) {
   214  	var physicalMachines []v1alpha1.PhysicalMachine
   215  	namespaceCheck := make(map[string]bool)
   216  	logger, err := stdLog.NewDefaultZapLogger()
   217  	if err != nil {
   218  		return physicalMachines, errors.Wrap(err, "failed to create logger")
   219  	}
   220  	for ns, names := range spec.PhysicalMachines {
   221  		if !clusterScoped {
   222  			if targetNamespace != ns {
   223  				log.Info("skip namespace because ns is out of scope within namespace scoped mode", "namespace", ns)
   224  				continue
   225  			}
   226  		}
   227  		if enableFilterNamespace {
   228  			allow, ok := namespaceCheck[ns]
   229  			if !ok {
   230  				allow = genericnamespace.CheckNamespace(ctx, c, ns, logger)
   231  				namespaceCheck[ns] = allow
   232  			}
   233  			if !allow {
   234  				continue
   235  			}
   236  		}
   237  		for _, name := range names {
   238  			var physicalMachine v1alpha1.PhysicalMachine
   239  			err := c.Get(ctx, types.NamespacedName{
   240  				Namespace: ns,
   241  				Name:      name,
   242  			}, &physicalMachine)
   243  			if err == nil {
   244  				physicalMachines = append(physicalMachines, physicalMachine)
   245  				continue
   246  			}
   247  
   248  			if apierrors.IsNotFound(err) {
   249  				log.Error(err, "PhysicalMachine is not found", "namespace", ns, "physical machine name", name)
   250  				continue
   251  			}
   252  
   253  			return nil, err
   254  		}
   255  	}
   256  	return physicalMachines, nil
   257  }

   259  // filterPhysicalMachinesByMode filters physical machines by mode from physical machine list
   260  func filterPhysicalMachinesByMode(physicalMachines []*PhysicalMachine, mode v1alpha1.SelectorMode, value string) ([]*PhysicalMachine, error) {
   261  	indexes, err := generic.FilterObjectsByMode(mode, value, len(physicalMachines))
   262  	if err != nil {
   263  		return nil, err
   264  	}
   265  
   266  	var filtered []*PhysicalMachine
   267  	for _, index := range indexes {
   268  		index := index
   269  		filtered = append(filtered, physicalMachines[index])
   270  	}
   271  	return filtered, nil
   272  }
