    63  func (f *portForwarder) forwardPorts(podKey, method string, url *url.URL, addresses []string, ports []string) (forwardedPorts []portforward.ForwardedPort, cancel context.CancelFunc, err error) {
    64  	transport, upgrader, err := spdy.RoundTripperFor(f.config)
    65  	if err != nil {
    66  		return nil, nil, err
    67  	}
    68  	dialer := spdy.NewDialer(upgrader, &http.Client{Transport: transport}, method, url)
    69  	r, w := io.Pipe()
    70  	ctx, cancel := context.WithCancel(f.ctx)
    71  	readyChan := make(chan struct{})
    72  	fw, err := portforward.NewOnAddresses(dialer, addresses, ports, ctx.Done(), readyChan, w, w)
    73  	if err != nil {
    74  		return nil, nil, err
    75  	}
    76  
    77  	// logging stdout/stderr of port forwarding
    78  	go func() {
    79  		// close pipe if the context is done
    80  		<-ctx.Done()
    81  		w.Close()
    82  	}()
    83  
    84  	go func() {
    85  		lineScanner := bufio.NewScanner(r)
    86  		for lineScanner.Scan() {
    87  			if f.enableLog {
    88  				f.logger.Info(fmt.Sprintf("log from port forwarding %q: %s", podKey, lineScanner.Text()))
    89  			}
    90  		}
    91  	}()
    92  
    93  	// run port forwarding
    94  	errChan := make(chan error)
    95  	go func() {
    96  		errChan <- fw.ForwardPorts()
    97  	}()
    98  
    99  	// wait for ready or error
   100  	select {
   101  	case <-readyChan:
   102  		break
   103  	case err := <-errChan:
   104  		cancel()
   105  		return nil, nil, err
   106  	}
   107  
   108  	forwardedPorts, err = fw.GetPorts()
   109  	if err != nil {
   110  		cancel()
   111  		return nil, nil, err
   112  	}
   113  
   114  	return forwardedPorts, cancel, nil
   115  }

   117  // Forward would port-forward to target resources
   118  func (f *portForwarder) Forward(namespace, resourceName string, addresses []string, ports []string) (forwardedPorts []portforward.ForwardedPort, cancel context.CancelFunc, err error) {
   119  	builder := resource.NewBuilder(f).
   120  		WithScheme(scheme.Scheme, scheme.Scheme.PrioritizedVersionsAllGroups()...).
   121  		ContinueOnError().
   122  		NamespaceParam(namespace).DefaultNamespace()
   123  
   124  	builder.ResourceNames("pods", resourceName)
   125  
   126  	obj, err := builder.Do().Object()
   127  	if err != nil {
   128  		return nil, nil, err
   129  	}
   130  
   131  	forwardablePod, err := polymorphichelpers.AttachablePodForObjectFn(f, obj, getPodTimeout)
   132  	if err != nil {
   133  		return nil, nil, err
   134  	}
   135  
   136  	// FIXME: get context from parameter
   137  	pod, err := f.client.CoreV1().Pods(namespace).Get(context.TODO(), forwardablePod.Name, metav1.GetOptions{})
   138  	if err != nil {
   139  		return nil, nil, err
   140  	}
   141  	return f.ForwardPod(pod, addresses, ports)
   142  }

   144  // ForwardPod would port-forward target Pod
   145  func (f *portForwarder) ForwardPod(pod *corev1.Pod, addresses []string, ports []string) (forwardedPorts []portforward.ForwardedPort, cancel context.CancelFunc, err error) {
   146  	if pod.Status.Phase != corev1.PodRunning {
   147  		return nil, nil, errors.Errorf("unable to forward port because pod is not running. Current status=%v", pod.Status.Phase)
   148  	}
   149  
   150  	req := f.client.CoreV1().RESTClient().Post().
   151  		Resource("pods").
   152  		Namespace(pod.Namespace).
   153  		Name(pod.Name).
   154  		SubResource("portforward")
   155  
   156  	return f.forwardPorts(fmt.Sprintf("%s/%s", pod.Namespace, pod.Name), "POST", req.URL(), addresses, ports)
   157  }

   160  func NewPortForwarder(ctx context.Context, restClientGetter genericclioptions.RESTClientGetter, enableLog bool, logger logr.Logger) (PortForward, error) {
   161  	config, err := restClientGetter.ToRESTConfig()

   180  // ForwardOnePort help to utility to forward one port of Kubernetes resource.
   181  func ForwardOnePort(fw PortForward, ns, resource string, port uint16) (string, uint16, context.CancelFunc, error) {
   182  	ports := []string{fmt.Sprintf("0:%d", port)}
   183  	forwardedPorts, cancel, err := fw.Forward(ns, resource, []string{"127.0.0.1"}, ports)
   184  	if err != nil {
   185  		return "", 0, nil, err
   186  	}
   187  	var localPort uint16
   188  	var found bool
   189  	for _, p := range forwardedPorts {
   190  		if p.Remote == port {
   191  			localPort = p.Local
   192  			found = true
   193  		}
   194  	}
   195  	if !found {
   196  		cancel()
   197  		return "", 0, nil, errors.New("unexpected error")
   198  	}
   199  	return "127.0.0.1", localPort, cancel, nil
   200  }
