   429  func (r *loggerResolver) Pod(ctx context.Context, ns string, name string) (<-chan string, error) {
   430  	logs, err := r.Clientset.CoreV1().Pods(ns).GetLogs(name, &v1.PodLogOptions{Follow: true}).Stream(ctx)
   431  	if err != nil {
   432  		return nil, err
   433  	}
   434  	logChan := make(chan string)
   435  	go func() {
   436  		defer logs.Close()
   437  		reader := bufio.NewReader(logs)
   438  		for {
   439  			line, err := reader.ReadString('\n')
   440  			if err != nil {
   441  				r.Log.Error(err, fmt.Sprintf("fail to read log of pod(%s/%s)", ns, name))
   442  				break
   443  			}
   444  			select {
   445  			case logChan <- string(line):
   446  				continue
   447  			case <-time.NewTimer(time.Minute).C:
   448  				r.Log.Info(fmt.Sprintf("client has not read log of pod(%s/%s) for 1m, close channel", ns, name))
   449  				return
   450  			}
   451  		}
   452  	}()
   453  	return logChan, nil
   454  }
