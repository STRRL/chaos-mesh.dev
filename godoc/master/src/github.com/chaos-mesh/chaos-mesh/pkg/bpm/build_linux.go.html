    27  // Build builds the process
    28  func (b *ProcessBuilder) Build() *ManagedProcess {
    29  	args := b.args
    30  	cmd := b.cmd
    31  
    32  	if len(b.nsOptions) > 0 {
    33  		args = append([]string{"--", cmd}, args...)
    34  		for _, option := range b.nsOptions {
    35  			args = append([]string{"-" + nsArgMap[option.Typ], option.Path}, args...)
    36  		}
    37  
    38  		if b.localMnt {
    39  			args = append([]string{"-l"}, args...)
    40  		}
    41  		cmd = nsexecPath
    42  	}
    43  
    44  	if b.pause {
    45  		args = append([]string{cmd}, args...)
    46  		cmd = pausePath
    47  	}
    48  
    49  	if c := mock.On("MockProcessBuild"); c != nil {
    50  		f := c.(func(context.Context, string, ...string) *exec.Cmd)
    51  		return &ManagedProcess{
    52  			Cmd:        f(b.ctx, cmd, args...),
    53  			Identifier: b.identifier,
    54  		}
    55  	}
    56  
    57  	log.Info("build command", "command", cmd+" "+strings.Join(args, " "))
    58  
    59  	command := exec.CommandContext(b.ctx, cmd, args...)
    60  	command.Env = b.env
    61  	command.SysProcAttr = &syscall.SysProcAttr{}
    62  	command.SysProcAttr.Pdeathsig = syscall.SIGTERM
    63  
    64  	if b.stdin != nil {
    65  		command.Stdin = b.stdin
    66  	}
    67  
    68  	if b.stdout != nil {
    69  		command.Stdout = b.stdout
    70  	}
    71  
    72  	if b.stderr != nil {
    73  		command.Stderr = b.stderr
    74  	}
    75  
    76  	return &ManagedProcess{
    77  		Cmd:        command,
    78  		Identifier: b.identifier,
    79  	}
    80  }
