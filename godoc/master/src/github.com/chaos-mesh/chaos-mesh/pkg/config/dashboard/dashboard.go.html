    27  // ChaosDashboardConfig defines the configuration for Chaos Dashboard
    28  type ChaosDashboardConfig struct {
    29  	ListenHost           string            `envconfig:"LISTEN_HOST" default:"0.0.0.0" json:"listen_host"`
    30  	ListenPort           int               `envconfig:"LISTEN_PORT" default:"2333" json:"listen_port"`
    31  	MetricHost           string            `envconfig:"METRIC_HOST" default:"0.0.0.0" json:"-"`
    32  	MetricPort           int               `envconfig:"METRIC_PORT" default:"2334" json:"-"`
    33  	EnableLeaderElection bool              `envconfig:"ENABLE_LEADER_ELECTION" json:"-"`
    34  	Database             *DatabaseConfig   `json:"-"`
    35  	PersistTTL           *PersistTTLConfig `json:"-"`
    36  	// ClusterScoped means control Chaos Object in cluster level(all namespace).
    37  	ClusterScoped bool `envconfig:"CLUSTER_SCOPED" default:"true" json:"cluster_mode"`
    38  	// TargetNamespace is the target namespace to injecting chaos.
    39  	// It only works with ClusterScoped is false.
    40  	TargetNamespace string `envconfig:"TARGET_NAMESPACE" default:"" json:"target_namespace"`
    41  	// EnableFilterNamespace will filter namespace with annotation. Only the pods/containers in namespace
    42  	// annotated with `chaos-mesh.org/inject=enabled` will be injected.
    43  	EnableFilterNamespace bool `envconfig:"ENABLE_FILTER_NAMESPACE" default:"false"`
    44  	// SecurityMode will use the token login by the user if set to true
    45  	SecurityMode bool `envconfig:"SECURITY_MODE" default:"true" json:"security_mode"`
    46  	// GcpSecurityMode will use the gcloud authentication to login to GKE user
    47  	GcpSecurityMode bool   `envconfig:"GCP_SECURITY_MODE" default:"false" json:"gcp_security_mode"`
    48  	GcpClientId     string `envconfig:"GCP_CLIENT_ID" default:"" json:"-"`
    49  	GcpClientSecret string `envconfig:"GCP_CLIENT_SECRET" default:"" json:"-"`
    50  
    51  	RootUrl string `envconfig:"ROOT_URL" default:"http://localhost:2333" json:"root_path"`
    52  
    53  	DNSServerCreate bool   `envconfig:"DNS_SERVER_CREATE" default:"false" json:"dns_server_create"`
    54  	Version         string `json:"version"`
    55  }

    79  // ParsePersistTTLConfig parse PersistTTLConfig to persistTTLConfigParsed.
    80  func ParsePersistTTLConfig(config *PersistTTLConfig) (*ttlcontroller.TTLConfig, error) {
    81  	syncPeriod, err := time.ParseDuration(config.SyncPeriod)
    82  	if err != nil {
    83  		return nil, errors.Wrap(err, "parse configuration sync period")
    84  	}
    85  
    86  	event, err := time.ParseDuration(config.Event)
    87  	if err != nil {
    88  		return nil, errors.Wrap(err, "parse configuration TTL for event")
    89  	}
    90  
    91  	experiment, err := time.ParseDuration(config.Experiment)
    92  	if err != nil {
    93  		return nil, errors.Wrap(err, "parse configuration TTL for experiment")
    94  	}
    95  
    96  	schedule, err := time.ParseDuration(config.Schedule)
    97  	if err != nil {
    98  		return nil, errors.Wrap(err, "parse configuration TTL for schedule")
    99  	}
   100  
   101  	workflow, err := time.ParseDuration(config.Workflow)
   102  	if err != nil {
   103  		return nil, errors.Wrap(err, "parse configuration TTL for workflow")
   104  	}
   105  
   106  	return &ttlcontroller.TTLConfig{
   107  		DatabaseTTLResyncPeriod: syncPeriod,
   108  		EventTTL:                event,
   109  		ArchiveTTL:              experiment,
   110  		ScheduleTTL:             schedule,
   111  		WorkflowTTL:             workflow,
   112  	}, nil
   113  }
