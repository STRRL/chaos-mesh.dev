    55  func Inject(res *v1.AdmissionRequest, cli client.Client, cfg *config.Config, controllerCfg *controllerCfg.ChaosControllerConfig, metrics *metrics.ChaosControllerManagerMetricsCollector) *v1.AdmissionResponse {
    56  	var pod corev1.Pod
    57  	if err := json.Unmarshal(res.Object.Raw, &pod); err != nil {
    58  		log.Error(err, "Could not unmarshal raw object")
    59  		return &v1.AdmissionResponse{
    60  			Result: &metav1.Status{
    61  				Message: err.Error(),
    62  			},
    63  		}
    64  	}
    65  
    66  	// Deal with potential empty fields, e.g., when the pod is created by a deployment
    67  	podName := potentialPodName(&pod.ObjectMeta)
    68  	if pod.ObjectMeta.Namespace == "" {
    69  		pod.ObjectMeta.Namespace = res.Namespace
    70  	}
    71  
    72  	log.Info("AdmissionReview for",
    73  		"Kind", res.Kind, "Namespace", res.Namespace, "Name", res.Name, "podName", podName, "UID", res.UID, "patchOperation", res.Operation, "UserInfo", res.UserInfo)
    74  	log.V(4).Info("Object", "Object", string(res.Object.Raw))
    75  	log.V(4).Info("OldObject", "OldObject", string(res.OldObject.Raw))
    76  	log.V(4).Info("Pod", "Pod", pod)
    77  
    78  	requiredKey, ok := injectRequired(&pod.ObjectMeta, cli, cfg, controllerCfg)
    79  	if !ok {
    80  		log.Info("Skipping injection due to policy check", "namespace", pod.ObjectMeta.Namespace, "name", podName)
    81  		return &v1.AdmissionResponse{
    82  			Allowed: true,
    83  		}
    84  	}
    85  
    86  	if metrics != nil {
    87  		metrics.InjectRequired.WithLabelValues(res.Namespace, requiredKey).Inc()
    88  	}
    89  	injectionConfig, err := cfg.GetRequestedConfig(pod.Namespace, requiredKey)
    90  	if err != nil {
    91  		log.Error(err, "Error getting injection config, permitting launch of pod with no sidecar injected", "injectionConfig",
    92  			injectionConfig)
    93  		// dont prevent pods from launching! just return allowed
    94  		return &v1.AdmissionResponse{
    95  			Allowed: true,
    96  		}
    97  	}
    98  
    99  	if injectionConfig.Selector != nil {
   100  		meet, err := podselector.CheckPodMeetSelector(context.Background(), cli, pod, *injectionConfig.Selector,
   101  			controllerCfg.ClusterScoped, controllerCfg.TargetNamespace, controllerCfg.EnableFilterNamespace)
   102  		if err != nil {
   103  			log.Error(err, "Failed to check pod selector", "namespace", pod.Namespace)
   104  			return &v1.AdmissionResponse{
   105  				Allowed: true,
   106  			}
   107  		}
   108  
   109  		if !meet {
   110  			log.Info("Skipping injection, this pod does not meet the selection criteria",
   111  				"namespace", pod.Namespace, "name", pod.Name)
   112  			return &v1.AdmissionResponse{
   113  				Allowed: true,
   114  			}
   115  		}
   116  	}
   117  
   118  	annotations := map[string]string{cfg.StatusAnnotationKey(): StatusInjected}
   119  
   120  	patchBytes, err := createPatch(&pod, injectionConfig, annotations)
   121  	if err != nil {
   122  		return &v1.AdmissionResponse{
   123  			Result: &metav1.Status{
   124  				Message: err.Error(),
   125  			},
   126  		}
   127  	}
   128  
   129  	log.Info("AdmissionResponse: patch", "patchBytes", string(patchBytes))
   130  	if metrics != nil {
   131  		metrics.Injections.WithLabelValues(res.Namespace, requiredKey).Inc()
   132  	}
   133  
   134  	patchType := v1.PatchTypeJSONPatch
   135  	return &v1.AdmissionResponse{
   136  		Allowed:   true,
   137  		Patch:     patchBytes,
   138  		PatchType: &patchType,
   139  	}
   140  }

   142  // Check whether the target resource need to be injected and return the required config name
   143  func injectRequired(metadata *metav1.ObjectMeta, cli client.Client, cfg *config.Config, controllerCfg *controllerCfg.ChaosControllerConfig) (string, bool) {
   144  	// skip special kubernetes system namespaces
   145  	for _, namespace := range ignoredNamespaces {
   146  		if metadata.Namespace == namespace {
   147  			log.Info("Skip mutation for it' in special namespace", "name", metadata.Name, "namespace", metadata.Namespace)
   148  			return "", false
   149  		}
   150  	}
   151  
   152  	if controllerCfg.EnableFilterNamespace {
   153  		ok, err := genericnamespace.IsAllowedNamespaces(context.Background(), cli, metadata.Namespace)
   154  		if err != nil {
   155  			log.Error(err, "fail to check whether this namespace should be injected", "namespace", metadata.Namespace)
   156  		}
   157  
   158  		if !ok {
   159  			log.Info("Skip mutation for it' in special namespace", "name", metadata.Name, "namespace", metadata.Namespace)
   160  			return "", false
   161  		}
   162  	}
   163  
   164  	log.V(4).Info("meta", "meta", metadata)
   165  
   166  	if checkInjectStatus(metadata, cfg) {
   167  		log.Info("Pod annotation indicates injection already satisfied, skipping",
   168  			"namespace", metadata.Namespace, "name", metadata.Name,
   169  			"annotationKey", cfg.StatusAnnotationKey(), "value", StatusInjected)
   170  		return "", false
   171  	}
   172  
   173  	requiredConfig, ok := injectByPodRequired(metadata, cfg)
   174  	if ok {
   175  		log.Info("Pod annotation requesting sidecar config",
   176  			"namespace", metadata.Namespace, "name", metadata.Name,
   177  			"annotation", cfg.RequestAnnotationKey(), "requiredConfig", requiredConfig)
   178  		return requiredConfig, true
   179  	}
   180  
   181  	requiredConfig, ok = injectByNamespaceRequired(metadata, cli, cfg)
   182  	if ok {
   183  		log.Info("Pod annotation requesting sidecar config",
   184  			"namespace", metadata.Namespace, "name", metadata.Name,
   185  			"annotation", cfg.RequestAnnotationKey(), "requiredConfig", requiredConfig)
   186  		return requiredConfig, true
   187  	}
   188  
   189  	requiredConfig, ok = injectByNamespaceInitRequired(metadata, cli, cfg)
   190  	if ok {
   191  		log.Info("Pod annotation init requesting sidecar config",
   192  			"namespace", metadata.Namespace, "name", metadata.Name,
   193  			"annotation", cfg.RequestAnnotationKey(), "requiredConfig", requiredConfig)
   194  		return requiredConfig, true
   195  	}
   196  
   197  	return "", false
   198  }

   200  func checkInjectStatus(metadata *metav1.ObjectMeta, cfg *config.Config) bool {
   201  	annotations := metadata.GetAnnotations()
   202  	if annotations == nil {
   203  		annotations = make(map[string]string)
   204  	}
   205  
   206  	status, ok := annotations[cfg.StatusAnnotationKey()]
   207  	if ok && strings.ToLower(status) == StatusInjected {
   208  		return true
   209  	}
   210  
   211  	return false
   212  }

   214  func injectByNamespaceRequired(metadata *metav1.ObjectMeta, cli client.Client, cfg *config.Config) (string, bool) {
   215  	var ns corev1.Namespace
   216  	if err := cli.Get(context.Background(), types.NamespacedName{Name: metadata.Namespace}, &ns); err != nil {
   217  		log.Error(err, "failed to get namespace", "namespace", metadata.Namespace)
   218  		return "", false
   219  	}
   220  	annotations := ns.GetAnnotations()
   221  	if annotations == nil {
   222  		annotations = make(map[string]string)
   223  	}
   224  
   225  	required, ok := annotations[annotation.GenKeyForWebhook(cfg.RequestAnnotationKey(), metadata.Name)]
   226  	if !ok {
   227  		log.Info("Pod annotation by namespace is missing, skipping injection",
   228  			"namespace", metadata.Namespace, "pod", metadata.Name, "config", required)
   229  		return "", false
   230  	}
   231  
   232  	log.Info("Get sidecar config from namespace annotations",
   233  		"namespace", metadata.Namespace, "pod", metadata.Name, "config", required)
   234  	return strings.ToLower(required), true
   235  }

   279  // create mutation patch for resource
   280  func createPatch(pod *corev1.Pod, inj *config.InjectionConfig, annotations map[string]string) ([]byte, error) {
   281  	var patch []patchOperation
   282  
   283  	// make sure any injected containers in our config get the EnvVars and VolumeMounts injected
   284  	// this mutates inj.Containers with our environment vars
   285  	mutatedInjectedContainers := mergeEnvVars(inj.Environment, inj.Containers)
   286  	mutatedInjectedContainers = mergeVolumeMounts(inj.VolumeMounts, mutatedInjectedContainers)
   287  
   288  	// make sure any injected init containers in our config get the EnvVars and VolumeMounts injected
   289  	// this mutates inj.InitContainers with our environment vars
   290  	mutatedInjectedInitContainers := mergeEnvVars(inj.Environment, inj.InitContainers)
   291  	mutatedInjectedInitContainers = mergeVolumeMounts(inj.VolumeMounts, mutatedInjectedInitContainers)
   292  
   293  	// patch all existing containers with the env vars and volume mounts
   294  	patch = append(patch, setVolumeMounts(pod.Spec.Containers, inj.VolumeMounts, "/spec/containers")...)
   295  	// TODO: fix set env
   296  	// setEnvironment may not work, because we replace the whole container in `setVolumeMounts`
   297  	patch = append(patch, setEnvironment(pod.Spec.Containers, inj.Environment)...)
   298  
   299  	// patch containers with our injected containers
   300  	patch = append(patch, addContainers(pod.Spec.Containers, mutatedInjectedContainers, "/spec/containers")...)
   301  
   302  	// add initContainers, hostAliases and volumes
   303  	patch = append(patch, addContainers(pod.Spec.InitContainers, mutatedInjectedInitContainers, "/spec/initContainers")...)
   304  	patch = append(patch, addHostAliases(pod.Spec.HostAliases, inj.HostAliases, "/spec/hostAliases")...)
   305  	patch = append(patch, addVolumes(pod.Spec.Volumes, inj.Volumes, "/spec/volumes")...)
   306  
   307  	// set annotations
   308  	patch = append(patch, updateAnnotations(pod.Annotations, annotations)...)
   309  
   310  	// set shareProcessNamespace
   311  	patch = append(patch, updateShareProcessNamespace(inj.ShareProcessNamespace)...)
   312  
   313  	// TODO: remove injecting commands when sidecar container supported
   314  	// set commands and args
   315  	patch = append(patch, setCommands(pod.Spec.Containers, inj.PostStart)...)
   316  
   317  	return json.Marshal(patch)
   318  }

   320  func setCommands(target []corev1.Container, postStart map[string]config.ExecAction) (patch []patchOperation) {
   321  	if postStart == nil {
   322  		return
   323  	}
   324  
   325  	for containerIndex, container := range target {
   326  		execCmd, ok := postStart[container.Name]
   327  		if !ok {
   328  			continue
   329  		}
   330  
   331  		path := fmt.Sprintf("/spec/containers/%d/command", containerIndex)
   332  
   333  		commands := MergeCommands(execCmd.Command, container.Command, container.Args)
   334  
   335  		log.Info("Inject command", "command", commands)
   336  
   337  		patch = append(patch, patchOperation{
   338  			Op:    "replace",
   339  			Path:  path,
   340  			Value: commands,
   341  		})
   342  
   343  		argsPath := fmt.Sprintf("/spec/containers/%d/args", containerIndex)
   344  		patch = append(patch, patchOperation{
   345  			Op:    "replace",
   346  			Path:  argsPath,
   347  			Value: []string{},
   348  		})
   349  	}
   350  	return patch
   351  }

   359  func setEnvironment(target []corev1.Container, addedEnv []corev1.EnvVar) (patch []patchOperation) {
   360  	var value interface{}
   361  	for containerIndex, container := range target {
   362  		// for each container in the spec, determine if we want to patch with any env vars
   363  		first := len(container.Env) == 0
   364  		for _, add := range addedEnv {
   365  			path := fmt.Sprintf("/spec/containers/%d/env", containerIndex)
   366  			hasKey := false
   367  			// make sure we dont override any existing env vars; we only add, dont replace
   368  			for _, origEnv := range container.Env {
   369  				if origEnv.Name == add.Name {
   370  					hasKey = true
   371  					break
   372  				}
   373  			}
   374  			if !hasKey {
   375  				// make a patch
   376  				value = add
   377  				if first {
   378  					first = false
   379  					value = []corev1.EnvVar{add}
   380  				} else {
   381  					path = path + "/-"
   382  				}
   383  				patch = append(patch, patchOperation{
   384  					Op:    "add",
   385  					Path:  path,
   386  					Value: value,
   387  				})
   388  			}
   389  		}
   390  	}
   391  
   392  	return patch
   393  }

   438  func setVolumeMounts(target []corev1.Container, addedVolumeMounts []corev1.VolumeMount, basePath string) (patch []patchOperation) {
   439  	for index, c := range target {
   440  		volumeMounts := map[string]corev1.VolumeMount{}
   441  		for _, vm := range c.VolumeMounts {
   442  			volumeMounts[vm.Name] = vm
   443  		}
   444  		for _, added := range addedVolumeMounts {
   445  			log.Info("volumeMount", "add", added)
   446  			volumeMounts[added.Name] = added
   447  		}
   448  
   449  		vs := []corev1.VolumeMount{}
   450  		for _, vm := range volumeMounts {
   451  			vs = append(vs, vm)
   452  		}
   453  		target[index].VolumeMounts = vs
   454  	}
   455  
   456  	patch = append(patch, patchOperation{
   457  		Op:    "replace",
   458  		Path:  basePath,
   459  		Value: target,
   460  	})
   461

   486  // for containers, add any env vars that are not already defined in the Env list.
   487  // this does _not_ return patches; this is intended to be used only on containers defined
   488  // in the injection config, so the resources do not exist yet in the k8s api (thus no patch needed)
   489  func mergeEnvVars(envs []corev1.EnvVar, containers []corev1.Container) []corev1.Container {
   490  	mutatedContainers := []corev1.Container{}
   491  	for _, c := range containers {
   492  		for _, newEnv := range envs {
   493  			// check each container for each env var by name.
   494  			// if the container has a matching name, dont override!
   495  			skip := false
   496  			for _, origEnv := range c.Env {
   497  				if origEnv.Name == newEnv.Name {
   498  					skip = true
   499  					break
   500  				}
   501  			}
   502  			if !skip {
   503  				c.Env = append(c.Env, newEnv)
   504  			}
   505  		}
   506  		mutatedContainers = append(mutatedContainers, c)
   507  	}
   508  	return mutatedContainers
   509  }

   533  func updateAnnotations(target map[string]string, added map[string]string) (patch []patchOperation) {
   534  	for key, value := range added {
   535  		if target == nil {
   536  			target = map[string]string{}
   537  			patch = append(patch, patchOperation{
   538  				Op:   "add",
   539  				Path: "/metadata/annotations",
   540  				Value: map[string]string{
   541  					key: value,
   542  				},
   543  			})
   544  		} else {
   545  			op := "add"
   546  			if target[key] != "" {
   547  				op = "replace"
   548  			}
   549  			patch = append(patch, patchOperation{
   550  				Op:    op,
   551  				Path:  "/metadata/annotations/" + escapeJSONPointerValue(key),
   552  				Value: value,
   553  			})
   554  		}
   555  	}
   556  	return patch
   557  }

   559  func escapeJSONPointerValue(in string) string {
   560  	step := strings.Replace(in, "~", "~0", -1)
   561  	return strings.Replace(step, "/", "~1", -1)
