    60  // NewChaosDaemonMetricsCollector initializes metrics for each chaos daemon
    61  func NewChaosDaemonMetricsCollector() *ChaosDaemonMetricsCollector {
    62  	return &ChaosDaemonMetricsCollector{
    63  		iptablesPackets: prometheus.NewGaugeVec(prometheus.GaugeOpts{
    64  			Name: "chaos_daemon_iptables_packets",
    65  			Help: "Total number of iptables packets",
    66  		}, []string{"namespace", "pod", "container", "table", "chain", "policy", "rule"}),
    67  		iptablesPacketBytes: prometheus.NewGaugeVec(prometheus.GaugeOpts{
    68  			Name: "chaos_daemon_iptables_packet_bytes",
    69  			Help: "Total bytes of iptables packets",
    70  		}, []string{"namespace", "pod", "container", "table", "chain", "policy", "rule"}),
    71  		ipsetMembers: prometheus.NewGaugeVec(prometheus.GaugeOpts{
    72  			Name: "chaos_daemon_ipset_members",
    73  			Help: "Total number of ipset members",
    74  		}, []string{"namespace", "pod", "container"}),
    75  		tcRules: prometheus.NewGaugeVec(prometheus.GaugeOpts{
    76  			Name: "chaos_daemon_tcs",
    77  			Help: "Total number of tc rules",
    78  		}, []string{"namespace", "pod", "container"}),
    79  	}
    80  }

   102  func (collector *ChaosDaemonMetricsCollector) collectNetworkMetrics() {
   103  	collector.iptablesPackets.Reset()
   104  	collector.iptablesPacketBytes.Reset()
   105  	collector.ipsetMembers.Reset()
   106  	collector.tcRules.Reset()
   107  
   108  	containerIDs, err := collector.crClient.ListContainerIDs(context.Background())
   109  	if err != nil {
   110  		log.Error(err, "fail to list all container process IDs")
   111  		return
   112  	}
   113  
   114  	for _, containerID := range containerIDs {
   115  		pid, err := collector.crClient.GetPidFromContainerID(context.Background(), containerID)
   116  		if err != nil {
   117  			log.Error(err, "fail to get pid from container ID")
   118  			continue
   119  		}
   120  
   121  		labels, err := collector.crClient.GetLabelsFromContainerID(context.Background(), containerID)
   122  		if err != nil {
   123  			log.Error(err, "fail to get container labels", "containerID", containerID)
   124  			continue
   125  		}
   126  
   127  		namespace, podName, containerName := labels[kubernetesPodNamespaceLabel],
   128  			labels[kubernetesPodNameLabel], labels[kubernetesContainerNameLabel]
   129  
   130  		labelValues := []string{namespace, podName, containerName}
   131  		log := log.WithValues(
   132  			"namespace", namespace,
   133  			"podName", podName,
   134  			"containerName", containerName,
   135  			"containerID", containerID,
   136  		)
   137  
   138  		tables, err := utils.GetIptablesContentByNetNS(pid)
   139  		if err != nil {
   140  			log.Error(err, "fail to collect iptables metrics")
   141  		}
   142  		for tableName, table := range tables {
   143  			for chainName, chain := range table {
   144  				for _, rule := range chain.Rules {
   145  					collector.iptablesPackets.
   146  						WithLabelValues(namespace, podName, containerName, tableName, chainName, chain.Policy, rule.Rule).
   147  						Set(float64(rule.Packets))
   148  
   149  					collector.iptablesPacketBytes.
   150  						WithLabelValues(namespace, podName, containerName, tableName, chainName, chain.Policy, rule.Rule).
   151  						Set(float64(rule.Bytes))
   152  				}
   153  			}
   154  		}
   155  
   156  		members, err := utils.GetIPSetRulesNumberByNetNS(pid)
   157  		if err != nil {
   158  			log.Error(err, "fail to collect ipset member metric")
   159  		}
   160  		collector.ipsetMembers.WithLabelValues(labelValues...).Set(float64(members))
   161  
   162  		tcRules, err := utils.GetTcRulesNumberByNetNS(pid)
   163  		if err != nil {
   164  			log.Error(err, "fail to collect tc rules metric")
   165  		}
   166  		collector.tcRules.WithLabelValues(labelValues...).Set(float64(tcRules))
   167  	}
   168  }
