   125  // @Summary Create a new chaos experiment.
   126  // @Description Create a new chaos experiment.
   127  // @Tags experiments
   128  // @Produce json
   129  // @Param request body core.ExperimentInfo true "Request body"
   130  // @Success 200 {object} core.ExperimentInfo
   131  // @Failure 400 {object} utils.APIError
   132  // @Failure 500 {object} utils.APIError
   133  // @Router /experiments/new [post]
   134  func (s *Service) createExperiment(c *gin.Context) {
   135  	kubeCli, err := clientpool.ExtractTokenAndGetClient(c.Request.Header)
   136  	if err != nil {
   137  		_ = c.Error(utils.ErrInvalidRequest.WrapWithNoMessage(err))
   138  		return
   139  	}
   140  
   141  	exp := &core.ExperimentInfo{}
   142  	if err := c.ShouldBindJSON(exp); err != nil {
   143  		c.Status(http.StatusBadRequest)
   144  		_ = c.Error(utils.ErrInvalidRequest.WrapWithNoMessage(err))
   145  		return
   146  	}
   147  
   148  	createFuncs := map[string]createExperimentFunc{
   149  		v1alpha1.KindPodChaos:     s.createPodChaos,
   150  		v1alpha1.KindNetworkChaos: s.createNetworkChaos,
   151  		v1alpha1.KindIoChaos:      s.createIOChaos,
   152  		v1alpha1.KindStressChaos:  s.createStressChaos,
   153  		v1alpha1.KindTimeChaos:    s.createTimeChaos,
   154  		v1alpha1.KindKernelChaos:  s.createKernelChaos,
   155  		v1alpha1.KindDNSChaos:     s.createDNSChaos,
   156  	}
   157  
   158  	f, ok := createFuncs[exp.Target.Kind]
   159  	if !ok {
   160  		c.Status(http.StatusBadRequest)
   161  		_ = c.Error(utils.ErrInvalidRequest.New(exp.Target.Kind + " is not supported"))
   162  		return
   163  	}
   164  
   165  	if err := f(exp, kubeCli); err != nil {
   166  		c.Status(http.StatusInternalServerError)
   167  		_ = c.Error(utils.ErrInternalServer.WrapWithNoMessage(err))
   168  		return
   169  	}
   170  
   171  	c.JSON(http.StatusOK, exp)
   172  }
