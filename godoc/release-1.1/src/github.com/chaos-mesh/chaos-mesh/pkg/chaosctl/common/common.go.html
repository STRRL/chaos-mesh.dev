    93  // PrettyPrint print with tab number and color
    94  func PrettyPrint(s string, num int, color string) {
    95  	var tabStr string
    96  	for i := 0; i < num; i++ {
    97  		tabStr += "\t"
    98  	}
    99  	str := fmt.Sprintf("%s%s\n\n", tabStr, regexp.MustCompile("\n").ReplaceAllString(s, "\n"+tabStr))
   100  	if color != "" {
   101  		if cfunc, ok := colorFunc[color]; !ok {
   102  			fmt.Printf("COLOR NOT SUPPORTED")
   103  		} else {
   104  			cfunc(str)
   105  		}
   106  	} else {
   107  		fmt.Printf(str)
   108  	}
   109  }

   111  // PrintResult prints result to users in prettier format
   112  func PrintResult(result []ChaosResult) {
   113  	for _, chaos := range result {
   114  		PrettyPrint("[Chaos]: "+chaos.Name, 0, "Blue")
   115  		for _, pod := range chaos.Pods {
   116  			PrettyPrint("[Pod]: "+pod.Name, 0, "Blue")
   117  			for i, item := range pod.Items {
   118  				PrettyPrint(fmt.Sprintf("%d. [%s]", i+1, item.Name), 1, "Cyan")
   119  				PrettyPrint(item.Value, 1, "")
   120  				if item.Status == ItemSuccess {
   121  					if item.SucInfo != "" {
   122  						PrettyPrint(item.SucInfo, 1, "Green")
   123  					} else {
   124  						PrettyPrint("Execute as expected", 1, "Green")
   125  					}
   126  				} else if item.Status == ItemFailure {
   127  					PrettyPrint(fmt.Sprintf("Failed: %s ", item.ErrInfo), 1, "Red")
   128  				}
   129  			}
   130  		}
   131  	}
   132  }

   157  func GetPods(ctx context.Context, status v1alpha1.ChaosStatus, selectorSpec v1alpha1.SelectorSpec, c client.Client) ([]v1.Pod, []v1.Pod, error) {
   158  	// get podName
   159  	failedMessage := status.FailedMessage
   160  	if failedMessage != "" {
   161  		PrettyPrint(fmt.Sprintf("chaos failed with: %s", failedMessage), 0, "Red")
   162  	}
   163  
   164  	phase := status.Experiment.Phase
   165  	nextStart := status.Scheduler.NextStart
   166  
   167  	if phase == v1alpha1.ExperimentPhaseWaiting {
   168  		waitTime := nextStart.Sub(time.Now())
   169  		fmt.Printf("Waiting for chaos to start, in %s\n", waitTime)
   170  		time.Sleep(waitTime)
   171  	}
   172  
   173  	pods, err := selector.SelectPods(ctx, c, c, selectorSpec, ctrlconfig.ControllerCfg.ClusterScoped, ctrlconfig.ControllerCfg.TargetNamespace, ctrlconfig.ControllerCfg.AllowedNamespaces, ctrlconfig.ControllerCfg.IgnoredNamespaces)
   174  	if err != nil {
   175  		return nil, nil, fmt.Errorf("failed to SelectPods with: %s", err.Error())
   176  	}
   177  	if len(pods) == 0 {
   178  		return nil, nil, fmt.Errorf("no pods found for selector: %s", selectorSpec)
   179  	}
   180  
   181  	var chaosDaemons []v1.Pod
   182  	// get chaos daemon
   183  	for _, pod := range pods {
   184  		nodeName := pod.Spec.NodeName
   185  		daemonSelector := v1alpha1.SelectorSpec{
   186  			Nodes:          []string{nodeName},
   187  			LabelSelectors: map[string]string{"app.kubernetes.io/component": "chaos-daemon"},
   188  		}
   189  		daemons, err := selector.SelectPods(ctx, c, nil, daemonSelector, ctrlconfig.ControllerCfg.ClusterScoped, ctrlconfig.ControllerCfg.TargetNamespace, ctrlconfig.ControllerCfg.AllowedNamespaces, ctrlconfig.ControllerCfg.IgnoredNamespaces)
   190  		if err != nil {
   191  			return nil, nil, fmt.Errorf("failed to SelectPods with: %s", err.Error())
   192  		}
   193  		if len(daemons) == 0 {
   194  			return nil, nil, fmt.Errorf("no daemons found for selector: %s", daemonSelector)
   195  		}
   196  		chaosDaemons = append(chaosDaemons, daemons[0])
   197  	}
   198  
   199  	return pods, chaosDaemons, nil
   200  }

   203  func GetChaosList(ctx context.Context, chaosType string, chaosName string, ns string, c client.Client) ([]runtime.Object, []string, error) {
   204  	chaosType = upperCaseChaos(strings.ToLower(chaosType))
   205  	allKinds := v1alpha1.AllKinds()
   206  	chaosListInterface := allKinds[chaosType].ChaosList
   207  
   208  	if err := c.List(ctx, chaosListInterface, client.InNamespace(ns)); err != nil {
   209  		return nil, nil, fmt.Errorf("failed to get chaosList: %s", err.Error())
   210  	}
   211  	chaosList := chaosListInterface.ListChaos()
   212  	if len(chaosList) == 0 {
   213  		return nil, nil, fmt.Errorf("no chaos is found, please check your input")
   214  	}
   215  
   216  	var retList []runtime.Object
   217  	var retNameList []string
   218  	for _, ch := range chaosList {
   219  		if chaosName == "" || chaosName == ch.Name {
   220  			chaos, err := getChaos(ctx, chaosType, ch.Name, ns, c)
   221  			if err != nil {
   222  				return nil, nil, err
   223  			}
   224  			retList = append(retList, chaos)
   225  			retNameList = append(retNameList, ch.Name)
   226  		}
   227  	}
   228  	if len(retList) == 0 {
   229  		return nil, nil, fmt.Errorf("no chaos is found, please check your input")
   230  	}
   231  
   232  	return retList, retNameList, nil
   233  }

   248  // GetPidFromPS returns pid-command pairs
   249  func GetPidFromPS(ctx context.Context, pod v1.Pod, daemon v1.Pod, c *kubernetes.Clientset) ([]string, []string, error) {
   250  	cmd := fmt.Sprintf("ps")
   251  	out, err := ExecBypass(ctx, pod, daemon, cmd, c)
   252  	if err != nil {
   253  		return nil, nil, fmt.Errorf("run command '%s' failed with: %s", cmd, err.Error())
   254  	}
   255  	outLines := strings.Split(string(out), "\n")
   256  	if len(outLines) < 2 {
   257  		return nil, nil, fmt.Errorf("ps returns empty")
   258  	}
   259  	titles := strings.Fields(outLines[0])
   260  	var pidColumn, cmdColumn int
   261  	for i, t := range titles {
   262  		if t == "PID" {
   263  			pidColumn = i
   264  		}
   265  		if t == "COMMAND" || t == "CMD" {
   266  			cmdColumn = i
   267  		}
   268  	}
   269  	if pidColumn == 0 && cmdColumn == 0 {
   270  		return nil, nil, fmt.Errorf("Parsing ps error: could not get PID and COMMAND column")
   271  	}
   272  	var pids, commands []string
   273  	for _, line := range outLines[1:] {
   274  		item := strings.Fields(line)
   275  		// break when got empty line
   276  		if len(item) == 0 {
   277  			break
   278  		}
   279  		pids = append(pids, item[pidColumn])
   280  		commands = append(commands, item[cmdColumn])
   281  	}
   282  	return pids, commands, nil
   283  }

   285  // GetPidFromPod returns pid given containerd ID in pod
   286  func GetPidFromPod(ctx context.Context, pod v1.Pod, daemon v1.Pod) (uint32, error) {
   287  	pfCancel, localPort, err := forwardPorts(ctx, daemon, uint16(ctrlconfig.ControllerCfg.ChaosDaemonPort))
   288  	if err != nil {
   289  		return 0, fmt.Errorf("forward ports failed: %s", err.Error())
   290  	}
   291  	defer func() {
   292  		pfCancel()
   293  	}()
   294  
   295  	daemonClient, err := daemonClient.NewChaosDaemonClientLocally(int(localPort))
   296  	if err != nil {
   297  		return 0, fmt.Errorf("new chaos daemon client failed: %s", err.Error())
   298  	}
   299  	defer daemonClient.Close()
   300  
   301  	if len(pod.Status.ContainerStatuses) == 0 {
   302  		return 0, fmt.Errorf("%s %s can't get the state of container", pod.Namespace, pod.Name)
   303  	}
   304  
   305  	res, err := daemonClient.ContainerGetPid(ctx, &pb.ContainerRequest{
   306  		Action: &pb.ContainerAction{
   307  			Action: pb.ContainerAction_GETPID,
   308  		},
   309  		ContainerId: pod.Status.ContainerStatuses[0].ContainerID,
   310  	})
   311  	if err != nil {
   312  		return 0, fmt.Errorf("container get pid failed: %s", err.Error())
   313  	}
   314  	return res.Pid, nil
   315  }

   317  func forwardPorts(ctx context.Context, pod v1.Pod, port uint16) (context.CancelFunc, uint16, error) {
   318  	clientRawConfig, err := e2econfig.LoadClientRawConfig()
   319  	if err != nil {
   320  		log.Fatal("failed to load raw config", err.Error())
   321  	}
   322  	fw, err := portforward.NewPortForwarder(ctx, e2econfig.NewSimpleRESTClientGetter(clientRawConfig), false)
   323  	if err != nil {
   324  		log.Fatal("failed to create port forwarder", err.Error())
   325  	}
   326  	_, localPort, pfCancel, err := portforward.ForwardOnePort(fw, pod.Namespace, pod.Name, port)
   327  	return pfCancel, localPort, err

   330  // Log print log of pod
   331  func Log(pod v1.Pod, tail int64, c *kubernetes.Clientset) (string, error) {
   332  	podLogOpts := v1.PodLogOptions{}
   333  	//use negative tail to indicate no tail limit is needed
   334  	if tail >= 0 {
   335  		podLogOpts.TailLines = func(i int64) *int64 { return &i }(tail)
   336  	}
   337  
   338  	req := c.CoreV1().Pods(pod.Namespace).GetLogs(pod.Name, &podLogOpts)
   339  	podLogs, err := req.Stream()
   340  	if err != nil {
   341  		return "", fmt.Errorf("failed to open stream: %v", err.Error())
   342  	}
   343  	defer podLogs.Close()
   344  
   345  	buf := new(bytes.Buffer)
   346  	_, err = io.Copy(buf, podLogs)
   347  	if err != nil {
   348  		return "", fmt.Errorf("failed to copy information from podLogs to buf: %v", err.Error())
   349  	}
   350  	return buf.String(), nil
   351  }

   363  func checkConnForCtrlAndDaemon(ctx context.Context, daemons []v1.Pod, c *ClientSet) error {
   364  	ctrlSelector := v1alpha1.SelectorSpec{
   365  		LabelSelectors: map[string]string{"app.kubernetes.io/component": "controller-manager"},
   366  	}
   367  	ctrlMgrs, err := selector.SelectPods(ctx, c.CtrlCli, c.CtrlCli, ctrlSelector, ctrlconfig.ControllerCfg.ClusterScoped, ctrlconfig.ControllerCfg.TargetNamespace, ctrlconfig.ControllerCfg.AllowedNamespaces, ctrlconfig.ControllerCfg.IgnoredNamespaces)
   368  	if err != nil {
   369  		return fmt.Errorf("failed to SelectPods with: %s", err.Error())
   370  	}
   371  	if len(ctrlMgrs) == 0 {
   372  		return fmt.Errorf("Could not found controller manager")
   373  	}
   374  	for _, daemon := range daemons {
   375  		daemonIP := daemon.Status.PodIP
   376  		cmd := fmt.Sprintf("ping -c 1 %s > /dev/null; echo $?", daemonIP)
   377  		out, err := Exec(ctx, ctrlMgrs[0], cmd, c.KubeCli)
   378  		if err != nil {
   379  			return fmt.Errorf("run command '%s' failed with: %s", cmd, err.Error())
   380  		}
   381  		if string(out) == "0" {
   382  			PrettyPrint(fmt.Sprintf("Connection between Controller-Manager and Daemon %s (ip address: %s) works well", daemon.Name, daemonIP), 0, "Green")
   383  		} else {
   384  			PrettyPrint(fmt.Sprintf(`Connection between Controller-Manager and Daemon %s (ip address: %s) is blocked.
   385  Please check network policy / firewall, or see FAQ on website`, daemon.Name, daemonIP), 0, "Red")
   386  		}
   387  
   388  	}
   389  	return nil
   390  }
