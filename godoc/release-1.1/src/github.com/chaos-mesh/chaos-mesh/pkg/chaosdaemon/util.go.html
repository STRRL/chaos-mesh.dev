    65  // FormatContainerID strips protocol prefix from the container ID
    66  func (c DockerClient) FormatContainerID(ctx context.Context, containerID string) (string, error) {
    67  	if len(containerID) < len(dockerProtocolPrefix) {
    68  		return "", fmt.Errorf("container id %s is not a docker container id", containerID)
    69  	}
    70  	if containerID[0:len(dockerProtocolPrefix)] != dockerProtocolPrefix {
    71  		return "", fmt.Errorf("expected %s but got %s", dockerProtocolPrefix, containerID[0:len(dockerProtocolPrefix)])
    72  	}
    73  	return containerID[len(dockerProtocolPrefix):], nil
    74  }

    76  // GetPidFromContainerID fetches PID according to container id
    77  func (c DockerClient) GetPidFromContainerID(ctx context.Context, containerID string) (uint32, error) {
    78  	id, err := c.FormatContainerID(ctx, containerID)
    79  	if err != nil {
    80  		return 0, err
    81  	}
    82  	container, err := c.client.ContainerInspect(ctx, id)
    83  	if err != nil {
    84  		return 0, err
    85  	}
    86  
    87  	if container.State.Pid == 0 {
    88  		return 0, fmt.Errorf("container is not running, status: %s", container.State.Status)
    89  	}
    90  
    91  	return uint32(container.State.Pid), nil
    92  }

   160  // CreateContainerRuntimeInfoClient creates a container runtime information client.
   161  func CreateContainerRuntimeInfoClient(containerRuntime string) (ContainerRuntimeInfoClient, error) {
   162  	// TODO: support more container runtime
   163  
   164  	var cli ContainerRuntimeInfoClient
   165  	switch containerRuntime {
   166  	case containerRuntimeDocker:
   167  		client, err := newDockerClient(defaultDockerSocket, "", nil, nil)
   168  		if err != nil {
   169  			return nil, err
   170  		}
   171  		cli = DockerClient{client}
   172  
   173  	case containerRuntimeContainerd:
   174  		// TODO(yeya24): add more options?
   175  		client, err := newContainerdClient(defaultContainerdSocket, containerd.WithDefaultNamespace(containerdDefaultNS))
   176  		if err != nil {
   177  			return nil, err
   178  		}
   179  		cli = ContainerdClient{client}
   180  
   181  	default:
   182  		return nil, fmt.Errorf("only docker and containerd is supported, but got %s", containerRuntime)
   183  	}
   184  
   185  	return cli, nil
   186  }

   216  		return err
   217  	}

   240  // GetChildProcesses will return all child processes's pid. Include all generations.
   241  // only return error when /proc/pid/tasks cannot be read
   242  func GetChildProcesses(ppid uint32) ([]uint32, error) {
   243  	procs, err := ioutil.ReadDir(bpm.DefaultProcPrefix)
   244  	if err != nil {
   245  		return nil, err
   246  	}
   247  
   248  	type processPair struct {
   249  		Pid  uint32
   250  		Ppid uint32
   251  	}
   252  
   253  	pairs := make(chan processPair)
   254  	done := make(chan bool)
   255  
   256  	go func() {
   257  		var wg sync.WaitGroup
   258  
   259  		for _, proc := range procs {
   260  			_, err := strconv.ParseUint(proc.Name(), 10, 32)
   261  			if err != nil {
   262  				continue
   263  			}
   264  
   265  			statusPath := bpm.DefaultProcPrefix + "/" + proc.Name() + "/stat"
   266  
   267  			wg.Add(1)
   268  			go func() {
   269  				defer wg.Done()
   270  
   271  				reader, err := os.Open(statusPath)
   272  				if err != nil {
   273  					log.Error(err, "read status file error", "path", statusPath)
   274  					return
   275  				}
   276  				defer reader.Close()
   277  
   278  				var (
   279  					pid    uint32
   280  					comm   string
   281  					state  string
   282  					parent uint32
   283  				)
   284  				// according to procfs's man page
   285  				fmt.Fscanf(reader, "%d %s %s %d", &pid, &comm, &state, &parent)
   286  
   287  				pairs <- processPair{
   288  					Pid:  pid,
   289  					Ppid: parent,
   290  				}
   291  			}()
   292  		}
   293  
   294  		wg.Wait()
   295  		done <- true
   296  	}()
   297  
   298  	processGraph := NewGraph()
   299  	for {
   300  		select {
   301  		case pair := <-pairs:
   302  			processGraph.Insert(pair.Ppid, pair.Pid)
   303  		case <-done:
   304  			return processGraph.Flatten(ppid), nil
   305  		}
   306  	}
   307  }
