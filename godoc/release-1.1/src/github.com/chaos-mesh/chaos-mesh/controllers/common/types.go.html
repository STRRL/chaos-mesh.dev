    57  // Reconcile the common chaos
    58  func (r *Reconciler) Reconcile(req ctrl.Request) (ctrl.Result, error) {
    59  	var err error
    60  
    61  	r.Log.Info("Reconciling a common chaos", "name", req.Name, "namespace", req.Namespace)
    62  	ctx := context.Background()
    63  
    64  	chaos := r.Object()
    65  	if err = r.Client.Get(ctx, req.NamespacedName, chaos); err != nil {
    66  		r.Log.Error(err, "unable to get chaos")
    67  		return ctrl.Result{}, err
    68  	}
    69  
    70  	status := chaos.GetStatus()
    71  
    72  	if chaos.IsDeleted() {
    73  		// This chaos was deleted
    74  		r.Log.Info("Removing self")
    75  		if err = r.Recover(ctx, req, chaos); err != nil {
    76  			r.Log.Error(err, "failed to recover chaos")
    77  			updateFailedMessage(ctx, r, chaos, err.Error())
    78  			return ctrl.Result{Requeue: true}, err
    79  		}
    80  		status.Experiment.Phase = v1alpha1.ExperimentPhaseFinished
    81  		status.FailedMessage = emptyString
    82  	} else if chaos.IsPaused() {
    83  		if status.Experiment.Phase == v1alpha1.ExperimentPhaseRunning {
    84  			r.Log.Info("Pausing")
    85  
    86  			if err = r.Recover(ctx, req, chaos); err != nil {
    87  				r.Log.Error(err, "failed to pause chaos")
    88  				updateFailedMessage(ctx, r, chaos, err.Error())
    89  				return ctrl.Result{Requeue: true}, err
    90  			}
    91  			now := time.Now()
    92  			status.Experiment.EndTime = &metav1.Time{
    93  				Time: now,
    94  			}
    95  			if status.Experiment.StartTime != nil {
    96  				status.Experiment.Duration = now.Sub(status.Experiment.StartTime.Time).String()
    97  			}
    98  		}
    99  		status.Experiment.Phase = v1alpha1.ExperimentPhasePaused
   100  		status.FailedMessage = emptyString
   101  	} else if status.Experiment.Phase == v1alpha1.ExperimentPhaseRunning {
   102  		r.Log.Info("The common chaos is already running", "name", req.Name, "namespace", req.Namespace)
   103  		return ctrl.Result{}, nil
   104  	} else {
   105  		// Start chaos action
   106  		r.Log.Info("Performing Action")
   107  
   108  		if err = r.Apply(ctx, req, chaos); err != nil {
   109  			r.Log.Error(err, "failed to apply chaos action")
   110  			updateFailedMessage(ctx, r, chaos, err.Error())
   111  
   112  			status.Experiment.Phase = v1alpha1.ExperimentPhaseFailed
   113  
   114  			updateError := retry.RetryOnConflict(retry.DefaultRetry, func() error {
   115  				return r.Update(ctx, chaos)
   116  			})
   117  			if updateError != nil {
   118  				r.Log.Error(updateError, "unable to update chaos finalizers")
   119  				updateFailedMessage(ctx, r, chaos, updateError.Error())
   120  			}
   121  
   122  			return ctrl.Result{Requeue: true}, err
   123  		}
   124  		status.Experiment.StartTime = &metav1.Time{
   125  			Time: time.Now(),
   126  		}
   127  		status.Experiment.Phase = v1alpha1.ExperimentPhaseRunning
   128  		status.FailedMessage = emptyString
   129  	}
   130  
   131  	if err := r.Update(ctx, chaos); err != nil {
   132  		r.Log.Error(err, "unable to update chaos status")
   133  		return ctrl.Result{}, err
   134  	}
   135  
   136  	return ctrl.Result{}, nil
   137  }

   139  func updateFailedMessage(
   140  	ctx context.Context,
   141  	r *Reconciler,

   148  		r.Log.Error(err, "unable to update chaos status")
   149  	}
