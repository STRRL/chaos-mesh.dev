    47  // Apply flushes network configuration on pod
    48  func (h *Handler) Apply(ctx context.Context, chaos *v1alpha1.PodNetworkChaos) error {
    49  	h.Log.Info("updating network chaos", "pod", chaos.Namespace+"/"+chaos.Name, "spec", chaos.Spec)
    50  
    51  	pod := &corev1.Pod{}
    52  
    53  	err := h.Client.Get(ctx, types.NamespacedName{
    54  		Name:      chaos.Name,
    55  		Namespace: chaos.Namespace,
    56  	}, pod)
    57  	if err != nil {
    58  		h.Log.Error(err, "fail to find pod")
    59  		return err
    60  	}
    61  
    62  	if !h.AllowHostNetworkTesting {
    63  		if pod.Spec.HostNetwork {
    64  			err := errors.Errorf("it's dangerous to inject network chaos on a pod(%s/%s) with `hostNetwork`", pod.Namespace, pod.Name)
    65  			return err
    66  		}
    67  	}
    68  
    69  	err = h.SetIPSets(ctx, pod, chaos)
    70  	if err != nil {
    71  		return err
    72  	}
    73  
    74  	err = h.SetIptables(ctx, pod, chaos)
    75  	if err != nil {
    76  		return err
    77  	}
    78  
    79  	err = h.SetTcs(ctx, pod, chaos)
    80  	if err != nil {
    81  		return err
    82  	}
    83  
    84  	return nil
    85  }

    99  // SetIptables sets iptables on pod
   100  func (h *Handler) SetIptables(ctx context.Context, pod *corev1.Pod, chaos *v1alpha1.PodNetworkChaos) error {
   101  	chains := []*pb.Chain{}
   102  	for _, chain := range chaos.Spec.Iptables {
   103  		var direction pb.Chain_Direction
   104  		if chain.Direction == v1alpha1.Input {
   105  			direction = pb.Chain_INPUT
   106  		} else if chain.Direction == v1alpha1.Output {
   107  			direction = pb.Chain_OUTPUT
   108  		} else {
   109  			err := fmt.Errorf("unknown direction %s", string(chain.Direction))
   110  			h.Log.Error(err, "unknown direction")
   111  			return err
   112  		}
   113  		chains = append(chains, &pb.Chain{
   114  			Name:      chain.Name,
   115  			Ipsets:    chain.IPSets,
   116  			Direction: direction,
   117  			Target:    "DROP",
   118  		})
   119  	}
   120  	return iptable.SetIptablesChains(ctx, h.Client, pod, chains)
   121  }

   123  // SetTcs sets traffic control related chaos on pod
   124  func (h *Handler) SetTcs(ctx context.Context, pod *corev1.Pod, chaos *v1alpha1.PodNetworkChaos) error {
   125  	tcs := []*pb.Tc{}
   126  	for _, tc := range chaos.Spec.TrafficControls {
   127  		if tc.Type == v1alpha1.Bandwidth {
   128  			tbf, err := tc.Bandwidth.ToTbf()
   129  			if err != nil {
   130  				return err
   131  			}
   132  			tcs = append(tcs, &pb.Tc{
   133  				Type:  pb.Tc_BANDWIDTH,
   134  				Tbf:   tbf,
   135  				Ipset: tc.IPSet,
   136  			})
   137  		} else if tc.Type == v1alpha1.Netem {
   138  			netem, err := mergeNetem(tc.TcParameter)
   139  			if err != nil {
   140  				return err
   141  			}
   142  			tcs = append(tcs, &pb.Tc{
   143  				Type:  pb.Tc_NETEM,
   144  				Netem: netem,
   145  				Ipset: tc.IPSet,
   146  			})
   147  		} else {
   148  			return fmt.Errorf("unknown tc type")
   149  		}
   150  	}
   151  
   152  	h.Log.Info("setting tcs", "tcs", tcs)
   153  	return tcpkg.SetTcs(ctx, h.Client, pod, tcs)
   154  }

   158  	ToNetem() (*pb.Netem, error)
   159  }

   161  // mergeNetem calls ToNetem on all non nil network emulation specs and merges them into one request.
   162  func mergeNetem(spec v1alpha1.TcParameter) (*pb.Netem, error) {
   163  	// NOTE: a cleaner way like
   164  	// emSpecs = []NetemSpec{spec.Delay, spec.Loss} won't work.
   165  	// Because in the for _, spec := range emSpecs loop,
   166  	// spec != nil would always be true.
   167  	// See https://stackoverflow.com/questions/13476349/check-for-nil-and-nil-interface-in-go
   168  	// And https://groups.google.com/forum/#!topic/golang-nuts/wnH302gBa4I/discussion
   169  	// > In short: If you never store (*T)(nil) in an interface, then you can reliably use comparison against nil
   170  	var emSpecs []NetemSpec
   171  	if spec.Delay != nil {
   172  		emSpecs = append(emSpecs, spec.Delay)
   173  	}
   174  	if spec.Loss != nil {
   175  		emSpecs = append(emSpecs, spec.Loss)
   176  	}
   177  	if spec.Duplicate != nil {
   178  		emSpecs = append(emSpecs, spec.Duplicate)
   179  	}
   180  	if spec.Corrupt != nil {
   181  		emSpecs = append(emSpecs, spec.Corrupt)
   182  	}
   183  	if len(emSpecs) == 0 {
   184  		return nil, errors.New(invalidNetemSpecMsg)
   185  	}
   186  
   187  	merged := &pb.Netem{}
   188  	for _, spec := range emSpecs {
   189  		em, err := spec.ToNetem()
   190  		if err != nil {
   191  			return nil, err
   192  		}
   193  		merged = netem.MergeNetem(merged, em)
   194  	}
   195  	return merged, nil
   196  }
