    45  // Apply implements the reconciler.InnerReconciler.Apply
    46  func (r *endpoint) Apply(ctx context.Context, req ctrl.Request, obj v1alpha1.InnerObject) error {
    47  	var err error
    48  
    49  	podchaos, ok := obj.(*v1alpha1.PodChaos)
    50  	if !ok {
    51  		err = errors.New("chaos is not PodChaos")
    52  		r.Log.Error(err, "chaos is not PodChaos", "chaos", obj)
    53  		return err
    54  	}
    55  
    56  	if podchaos.Spec.ContainerName == "" {
    57  		r.Log.Error(nil, "the name of container is empty", "name", req.Name, "namespace", req.Namespace)
    58  		return fmt.Errorf("podchaos[%s/%s] the name of container is empty", podchaos.Namespace, podchaos.Name)
    59  	}
    60  
    61  	pods, err := selector.SelectAndFilterPods(ctx, r.Client, r.Reader, &podchaos.Spec, config.ControllerCfg.ClusterScoped, config.ControllerCfg.TargetNamespace, config.ControllerCfg.AllowedNamespaces, config.ControllerCfg.IgnoredNamespaces)
    62  	if err != nil {
    63  		r.Log.Error(err, "fail to select and filter pods")
    64  		return err
    65  	}
    66  
    67  	var podsNotHaveContainer []string
    68  	for podIndex := range pods {
    69  		pod := &pods[podIndex]
    70  		haveContainer := false
    71  
    72  		for _, item := range pod.Status.ContainerStatuses {
    73  			if item.Name == podchaos.Spec.ContainerName {
    74  				haveContainer = true
    75  				break
    76  			}
    77  		}
    78  
    79  		if !haveContainer {
    80  			podsNotHaveContainer = append(podsNotHaveContainer, pod.Name)
    81  			r.Log.Error(nil, fmt.Sprintf("the pod %s doesn't have container %s", pod.Name, podchaos.Spec.ContainerName))
    82  		}
    83  	}
    84  
    85  	if len(podsNotHaveContainer) != 0 {
    86  		return fmt.Errorf("the pod %v doesn't have container %s", podsNotHaveContainer, podchaos.Spec.ContainerName)
    87  	}
    88  
    89  	g := errgroup.Group{}
    90  	for podIndex := range pods {
    91  		pod := &pods[podIndex]
    92  
    93  		for containerIndex := range pod.Status.ContainerStatuses {
    94  			containerName := pod.Status.ContainerStatuses[containerIndex].Name
    95  			containerID := pod.Status.ContainerStatuses[containerIndex].ContainerID
    96  
    97  			if containerName == podchaos.Spec.ContainerName {
    98  				g.Go(func() error {
    99  					err = r.KillContainer(ctx, pod, containerID)
   100  					if err != nil {
   101  						r.Log.Error(err, fmt.Sprintf(
   102  							"failed to kill container: %s, pod: %s, namespace: %s",
   103  							containerName, pod.Name, pod.Namespace))
   104  					}
   105  					return err
   106  				})
   107  			}
   108  		}
   109  	}
   110  
   111  	if err := g.Wait(); err != nil {
   112  		return err
   113  	}
   114  
   115  	podchaos.Status.Experiment.PodRecords = make([]v1alpha1.PodStatus, 0, len(pods))
   116  	for _, pod := range pods {
   117  		ps := v1alpha1.PodStatus{
   118  			Namespace: pod.Namespace,
   119  			Name:      pod.Name,
   120  			HostIP:    pod.Status.HostIP,
   121  			PodIP:     pod.Status.PodIP,
   122  			Action:    string(podchaos.Spec.Action),
   123  			Message:   fmt.Sprintf(containerKillActionMsg, podchaos.Spec.ContainerName),
   124  		}
   125  
   126  		podchaos.Status.Experiment.PodRecords = append(podchaos.Status.Experiment.PodRecords, ps)
   127  	}
   128  	r.Event(podchaos, v1.EventTypeNormal, events.ChaosRecovered, "")
   129  	return nil
   130  }

   142  // KillContainer kills container according to containerID
   143  // Use client in chaos-daemon
   144  func (r *endpoint) KillContainer(ctx context.Context, pod *v1.Pod, containerID string) error {
   145  	r.Log.Info("Try to kill container", "namespace", pod.Namespace, "podName", pod.Name, "containerID", containerID)
   146  
   147  	pbClient, err := client.NewChaosDaemonClient(ctx, r.Client, pod, config.ControllerCfg.ChaosDaemonPort)
   148  	if err != nil {
   149  		return err
   150  	}
   151  	defer pbClient.Close()
   152  
   153  	if len(pod.Status.ContainerStatuses) == 0 {
   154  		return fmt.Errorf("%s %s can't get the state of container", pod.Namespace, pod.Name)
   155  	}
   156  
   157  	if _, err = pbClient.ContainerKill(ctx, &pb.ContainerRequest{
   158  		Action: &pb.ContainerAction{
   159  			Action: pb.ContainerAction_KILL,
   160  		},
   161  		ContainerId: containerID,
   162  	}); err != nil {
   163  		r.Log.Error(err, "kill container error", "namespace", pod.Namespace, "podName", pod.Name, "containerID", containerID)
   164  		return err
   165  	}
   166  
   167  	return nil
   168  }
