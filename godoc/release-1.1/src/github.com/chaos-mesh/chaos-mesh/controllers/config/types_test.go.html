   113  func TestIsAllowedNamespaces(t *testing.T) {
   114  	g := NewGomegaWithT(t)
   115  	type TestCase struct {
   116  		name   string
   117  		pods   []v1.Pod
   118  		ret    []bool
   119  		allow  string
   120  		ignore string
   121  	}
   122  	pods := []v1.Pod{
   123  		newPod("p1", v1.PodRunning, "allow", nil, nil, ""),
   124  		newPod("p1", v1.PodRunning, "allow-app", nil, nil, ""),
   125  		newPod("p1", v1.PodRunning, "app-allow", nil, nil, ""),
   126  		newPod("p1", v1.PodRunning, "ignore", nil, nil, ""),
   127  		newPod("p1", v1.PodRunning, "ignore-app", nil, nil, ""),
   128  		newPod("p1", v1.PodRunning, "app-ignore", nil, nil, ""),
   129  	}
   130  
   131  	allowRet := []bool{true, true, true, false, false, false}
   132  
   133  	var tcs []TestCase
   134  	tcs = append(tcs, TestCase{
   135  		name:  "only set allow",
   136  		pods:  pods,
   137  		ret:   allowRet,
   138  		allow: "allow",
   139  	})
   140  
   141  	tcs = append(tcs, TestCase{
   142  		name:   "only set ignore",
   143  		pods:   pods,
   144  		ret:    allowRet,
   145  		ignore: "ignore",
   146  	})
   147  
   148  	tcs = append(tcs, TestCase{
   149  		name:   "only set allow",
   150  		pods:   pods,
   151  		ret:    allowRet,
   152  		allow:  "allow",
   153  		ignore: "ignore",
   154  	})
   155  
   156  	for _, tc := range tcs {
   157  		for index, pod := range tc.pods {
   158  			g.Expect(IsAllowedNamespaces(pod.Namespace, tc.allow, tc.ignore)).Should(Equal(tc.ret[index]))
   159  		}
   160  	}
   161  }
