    55  // Apply applies dns-chaos
    56  func (r *endpoint) Apply(ctx context.Context, req ctrl.Request, chaos v1alpha1.InnerObject) error {
    57  	dnschaos, ok := chaos.(*v1alpha1.DNSChaos)
    58  	if !ok {
    59  		err := errors.New("chaos is not DNSChaos")
    60  		r.Log.Error(err, "chaos is not DNSChaos", "chaos", chaos)
    61  		return err
    62  	}
    63  
    64  	pods, err := selector.SelectAndFilterPods(ctx, r.Client, r.Reader, &dnschaos.Spec, config.ControllerCfg.ClusterScoped, config.ControllerCfg.TargetNamespace, config.ControllerCfg.AllowedNamespaces, config.ControllerCfg.IgnoredNamespaces)
    65  	if err != nil {
    66  		r.Log.Error(err, "failed to select and generate pods")
    67  		return err
    68  	}
    69  
    70  	// get dns server's ip used for chaos
    71  	service, err := selector.GetService(ctx, r.Client, "", config.ControllerCfg.Namespace, config.ControllerCfg.DNSServiceName)
    72  	if err != nil {
    73  		r.Log.Error(err, "fail to get service")
    74  		return err
    75  	}
    76  	r.Log.Info("Set DNS chaos to DNS service", "ip", service.Spec.ClusterIP)
    77  
    78  	err = r.setDNSServerRules(service.Spec.ClusterIP, config.ControllerCfg.DNSServicePort, dnschaos.Name, pods, dnschaos.Spec.Action, dnschaos.Spec.Scope)
    79  	if err != nil {
    80  		r.Log.Error(err, "fail to set DNS server rules")
    81  		return err
    82  	}
    83  
    84  	if err = r.applyAllPods(ctx, pods, dnschaos, service.Spec.ClusterIP); err != nil {
    85  		r.Log.Error(err, "failed to apply chaos on all pods")
    86  		return err
    87  	}
    88  
    89  	dnschaos.Status.Experiment.PodRecords = make([]v1alpha1.PodStatus, 0, len(pods))
    90  	for _, pod := range pods {
    91  		ps := v1alpha1.PodStatus{
    92  			Namespace: pod.Namespace,
    93  			Name:      pod.Name,
    94  			HostIP:    pod.Status.HostIP,
    95  			PodIP:     pod.Status.PodIP,
    96  		}
    97  
    98  		dnschaos.Status.Experiment.PodRecords = append(dnschaos.Status.Experiment.PodRecords, ps)
    99  	}
   100  	r.Event(dnschaos, v1.EventTypeNormal, events.ChaosInjected, "")
   101  	return nil
   102  }

   104  // Recover means the reconciler recovers the chaos action
   105  func (r *endpoint) Recover(ctx context.Context, req ctrl.Request, chaos v1alpha1.InnerObject) error {
   106  	dnschaos, ok := chaos.(*v1alpha1.DNSChaos)
   107  	if !ok {
   108  		err := errors.New("chaos is not DNSChaos")
   109  		r.Log.Error(err, "chaos is not DNSChaos", "chaos", chaos)
   110  		return err
   111  	}
   112  
   113  	// get dns server's ip used for chaos
   114  	service, err := selector.GetService(ctx, r.Client, "", config.ControllerCfg.Namespace, config.ControllerCfg.DNSServiceName)
   115  	if err != nil {
   116  		r.Log.Error(err, "fail to get service")
   117  		return err
   118  	}
   119  	r.Log.Info("Cancel DNS chaos to DNS service", "ip", service.Spec.ClusterIP)
   120  
   121  	r.cancelDNSServerRules(service.Spec.ClusterIP, config.ControllerCfg.DNSServicePort, dnschaos.Name)
   122  
   123  	rd := recover.Delegate{Client: r.Client, Log: r.Log, RecoverIntf: &recoverer{r.Client, r.Log}}
   124  
   125  	finalizers, err := rd.CleanFinalizersAndRecover(ctx, chaos, dnschaos.Finalizers, dnschaos.Annotations)
   126  	if err != nil {
   127  		return err
   128  	}
   129  	dnschaos.Finalizers = finalizers
   130  	r.Event(dnschaos, v1.EventTypeNormal, events.ChaosRecovered, "")
   131  
   132  	return nil
   133  }

   135  func (r *recoverer) RecoverPod(ctx context.Context, pod *v1.Pod, somechaos v1alpha1.InnerObject) error {
   136  	r.Log.Info("Try to recover pod", "namespace", pod.Namespace, "name", pod.Name)
   137  
   138  	daemonClient, err := client.NewChaosDaemonClient(ctx, r.Client,
   139  		pod, config.ControllerCfg.ChaosDaemonPort)
   140  	if err != nil {
   141  		r.Log.Error(err, "get chaos daemon client")
   142  		return err
   143  	}
   144  	defer daemonClient.Close()
   145  	if len(pod.Status.ContainerStatuses) == 0 {
   146  		return fmt.Errorf("%s %s can't get the state of container", pod.Namespace, pod.Name)
   147  	}
   148  
   149  	target := pod.Status.ContainerStatuses[0].ContainerID
   150  
   151  	_, err = daemonClient.SetDNSServer(ctx, &pb.SetDNSServerRequest{
   152  		ContainerId: target,
   153  		Enable:      false,
   154  		EnterNS:     true,
   155  	})
   156  	if err != nil {
   157  		r.Log.Error(err, "recover pod for DNS chaos")
   158  		return err
   159  	}
   160  
   161  	return nil
   162  }

   169  func (r *endpoint) applyAllPods(ctx context.Context, pods []v1.Pod, chaos *v1alpha1.DNSChaos, dnsServerIP string) error {
   170  	g := errgroup.Group{}
   171  	for index := range pods {
   172  		pod := &pods[index]
   173  
   174  		key, err := cache.MetaNamespaceKeyFunc(pod)
   175  		if err != nil {
   176  			r.Log.Error(err, "get meta namespace key")
   177  			return err
   178  		}
   179  		chaos.Finalizers = finalizer.InsertFinalizer(chaos.Finalizers, key)
   180  
   181  		g.Go(func() error {
   182  			return r.applyPod(ctx, pod, dnsServerIP)
   183  		})
   184  	}
   185  	err := g.Wait()
   186  	if err != nil {
   187  		r.Log.Error(err, "g.Wait")
   188  		return err
   189  	}
   190  	return nil
   191  }

   195  		pod.Namespace, "name", pod.Name)

   197  		pod, config.ControllerCfg.ChaosDaemonPort)
   198  	if err != nil {
   199  		r.Log.Error(err, "get chaos daemon client")
   200  		return err
   201  	}
   202  	defer daemonClient.Close()
   203  	if len(pod.Status.ContainerStatuses) == 0 {
   204  		return fmt.Errorf("%s %s can't get the state of container", pod.Namespace, pod.Name)
   205  	}
   206  
   207  	target := pod.Status.ContainerStatuses[0].ContainerID
   208  
   209  	_, err = daemonClient.SetDNSServer(ctx, &pb.SetDNSServerRequest{
   210  		ContainerId: target,
   211  		DnsServer:   dnsServerIP,
   212  		Enable:      true,
   213  		EnterNS:     true,
   214  	})
   215  	if err != nil {
   216  		r.Log.Error(err, "set dns server")

   218  	}
   219  
   220  	return nil

   223  func (r *endpoint) setDNSServerRules(dnsServerIP string, port int, name string, pods []v1.Pod, action v1alpha1.DNSChaosAction, scope v1alpha1.DNSChaosScope) error {
   224  	r.Log.Info("setDNSServerRules", "name", name)
   225  
   226  	pbPods := make([]*dnspb.Pod, len(pods))
   227  	for i, pod := range pods {
   228  		pbPods[i] = &dnspb.Pod{
   229  			Name:      pod.Name,
   230  			Namespace: pod.Namespace,
   231  		}
   232  	}
   233  
   234  	conn, err := grpc.Dial(fmt.Sprintf("%s:%d", dnsServerIP, port), grpc.WithInsecure())
   235  	if err != nil {
   236  		return err
   237  	}
   238  	defer conn.Close()
   239  
   240  	c := dnspb.NewDNSClient(conn)
   241  	request := &dnspb.SetDNSChaosRequest{
   242  		Name:   name,
   243  		Action: string(action),
   244  		Pods:   pbPods,
   245  		Scope:  string(scope),
   246  	}
   247  
   248  	ctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
   249  	defer cancel()
   250  	response, err := c.SetDNSChaos(ctx, request)
   251  	if err != nil {
   252  		return err
   253  	}
   254  
   255  	if !response.Result {
   256  		return fmt.Errorf("set dns chaos to dns server error %s", response.Msg)
   257  	}
   258  
   259  	return nil
   260  }
