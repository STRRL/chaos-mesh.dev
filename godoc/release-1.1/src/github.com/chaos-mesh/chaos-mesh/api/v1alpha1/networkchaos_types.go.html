   195  // ToNetem implements Netem interface.
   196  func (in *DelaySpec) ToNetem() (*chaosdaemonpb.Netem, error) {
   197  	delayTime, err := time.ParseDuration(in.Latency)
   198  	if err != nil {
   199  		return nil, err
   200  	}
   201  	jitter, err := time.ParseDuration(in.Jitter)
   202  	if err != nil {
   203  		return nil, err
   204  	}
   205  
   206  	corr, err := strconv.ParseFloat(in.Correlation, 32)
   207  	if err != nil {
   208  		return nil, err
   209  	}
   210  
   211  	netem := &chaosdaemonpb.Netem{
   212  		Time:      uint32(delayTime.Nanoseconds() / 1e3),
   213  		DelayCorr: float32(corr),
   214  		Jitter:    uint32(jitter.Nanoseconds() / 1e3),
   215  	}
   216  
   217  	if in.Reorder != nil {
   218  		reorderPercentage, err := strconv.ParseFloat(in.Reorder.Reorder, 32)
   219  		if err != nil {
   220  			return nil, err
   221  		}
   222  
   223  		corr, err := strconv.ParseFloat(in.Reorder.Correlation, 32)
   224  		if err != nil {
   225  			return nil, err
   226  		}
   227  
   228  		netem.Reorder = float32(reorderPercentage)
   229  		netem.ReorderCorr = float32(corr)
   230  		netem.Gap = uint32(in.Reorder.Gap)
   231  	}
   232  
   233  	return netem, nil
   234  }

   308  // BandwidthSpec defines detail of bandwidth limit.
   309  type BandwidthSpec struct {
   310  	// Rate is the speed knob. Allows bps, kbps, mbps, gbps, tbps unit. bps means bytes per second.
   311  	Rate string `json:"rate"`
   312  	// Limit is the number of bytes that can be queued waiting for tokens to become available.
   313  	// +kubebuilder:validation:Minimum=1
   314  	Limit uint32 `json:"limit"`
   315  	// Buffer is the maximum amount of bytes that tokens can be available for instantaneously.
   316  	// +kubebuilder:validation:Minimum=1
   317  	Buffer uint32 `json:"buffer"`
   318  	// Peakrate is the maximum depletion rate of the bucket.
   319  	// The peakrate does not need to be set, it is only necessary
   320  	// if perfect millisecond timescale shaping is required.
   321  	// +optional
   322  	// +kubebuilder:validation:Minimum=0
   323  	Peakrate *uint64 `json:"peakrate,omitempty"`
   324  	// Minburst specifies the size of the peakrate bucket. For perfect
   325  	// accuracy, should be set to the MTU of the interface.  If a
   326  	// peakrate is needed, but some burstiness is acceptable, this
   327  	// size can be raised. A 3000 byte minburst allows around 3mbit/s
   328  	// of peakrate, given 1000 byte packets.
   329  	// +optional
   330  	// +kubebuilder:validation:Minimum=0
   331  	Minburst *uint32 `json:"minburst,omitempty"`
   332  }

   334  // ToTbf converts BandwidthSpec to *chaosdaemonpb.Tbf
   335  // Bandwidth action use TBF under the hood.
   336  // TBF stands for Token Bucket Filter, is a classful queueing discipline available
   337  // for traffic control with the tc command.
   338  // http://man7.org/linux/man-pages/man8/tc-tbf.8.html
   339  func (in *BandwidthSpec) ToTbf() (*chaosdaemonpb.Tbf, error) {
   340  	rate, err := convertUnitToBytes(in.Rate)
   341  
   342  	if err != nil {
   343  		return nil, err
   344  	}
   345  
   346  	tbf := &chaosdaemonpb.Tbf{
   347  		Rate:   rate,
   348  		Limit:  in.Limit,
   349  		Buffer: in.Buffer,
   350  	}
   351  
   352  	if in.Peakrate != nil && in.Minburst != nil {
   353  		tbf.PeakRate = *in.Peakrate
   354  		tbf.MinBurst = *in.Minburst
   355  	}
   356  
   357  	return tbf, nil
   358  }

   360  func convertUnitToBytes(nu string) (uint64, error) {
   361  	// normalize input
   362  	s := strings.ToLower(strings.TrimSpace(nu))
   363  
   364  	for i, u := range []string{"tbps", "gbps", "mbps", "kbps", "bps"} {
   365  		if strings.HasSuffix(s, u) {
   366  			ts := strings.TrimSuffix(s, u)
   367  			s := strings.TrimSpace(ts)
   368  
   369  			n, err := strconv.ParseUint(s, 10, 64)
   370  
   371  			if err != nil {
   372  				return 0, err
   373  			}
   374  
   375  			// convert unit to bytes
   376  			for j := 4 - i; j > 0; j-- {
   377  				n = n * 1024
   378  			}
   379  
   380  			return n, nil
   381  		}
   382  	}
   383  
   384  	return 0, errors.New("invalid unit")
   385  }
