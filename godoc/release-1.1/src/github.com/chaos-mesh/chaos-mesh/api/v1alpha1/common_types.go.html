    41  // SelectorSpec defines the some selectors to select objects.
    42  // If the all selectors are empty, all objects will be used in chaos experiment.
    43  type SelectorSpec struct {
    44  	// Namespaces is a set of namespace to which objects belong.
    45  	// +optional
    46  	Namespaces []string `json:"namespaces,omitempty"`
    47  
    48  	// Nodes is a set of node name and objects must belong to these nodes.
    49  	// +optional
    50  	Nodes []string `json:"nodes,omitempty"`
    51  
    52  	// Pods is a map of string keys and a set values that used to select pods.
    53  	// The key defines the namespace which pods belong,
    54  	// and the each values is a set of pod names.
    55  	// +optional
    56  	Pods map[string][]string `json:"pods,omitempty"`
    57  
    58  	// Map of string keys and values that can be used to select nodes.
    59  	// Selector which must match a node's labels,
    60  	// and objects must belong to these selected nodes.
    61  	// +optional
    62  	NodeSelectors map[string]string `json:"nodeSelectors,omitempty"`
    63  
    64  	// Map of string keys and values that can be used to select objects.
    65  	// A selector based on fields.
    66  	// +optional
    67  	FieldSelectors map[string]string `json:"fieldSelectors,omitempty"`
    68  
    69  	// Map of string keys and values that can be used to select objects.
    70  	// A selector based on labels.
    71  	// +optional
    72  	LabelSelectors map[string]string `json:"labelSelectors,omitempty"`
    73  
    74  	// a slice of label selector expressions that can be used to select objects.
    75  	// A list of selectors based on set-based label expressions.
    76  	// +optional
    77  	ExpressionSelectors LabelSelectorRequirements `json:"expressionSelectors,omitempty"`
    78  
    79  	// Map of string keys and values that can be used to select objects.
    80  	// A selector based on annotations.
    81  	// +optional
    82  	AnnotationSelectors map[string]string `json:"annotationSelectors,omitempty"`
    83  
    84  	// PodPhaseSelectors is a set of condition of a pod at the current time.
    85  	// supported value: Pending / Running / Succeeded / Failed / Unknown
    86  	// +optional
    87  	PodPhaseSelectors []string `json:"podPhaseSelectors,omitempty"`
    88  }

    90  // ClusterScoped returns true if the selector selects Pods in the cluster
    91  func (s SelectorSpec) ClusterScoped() bool {
    92  	// in fact, this will never happened, will add namespace if it is empty, so len(s.Namespaces) can not be 0,
    93  	// but still add judgentment here for safe
    94  	// https://github.com/chaos-mesh/chaos-mesh/blob/478d00d01bb0f9fb08a1085428a7da8c8f9df4e8/api/v1alpha1/common_webhook.go#L22
    95  	if len(s.Namespaces) == 0 && len(s.Pods) == 0 {
    96  		return true
    97  	}
    98  
    99  	return false
   100  }

   102  // AffectedNamespaces returns all the namespaces which the selector effect
   103  func (s SelectorSpec) AffectedNamespaces() []string {
   104  	affectedNamespacesMap := make(map[string]struct{})
   105  	affectedNamespacesArray := make([]string, 0, 2)
   106  
   107  	for namespace := range s.Pods {
   108  		affectedNamespacesMap[namespace] = struct{}{}
   109  	}
   110  
   111  	for _, namespace := range s.Namespaces {
   112  		affectedNamespacesMap[namespace] = struct{}{}
   113  	}
   114  
   115  	for namespace := range affectedNamespacesMap {
   116  		affectedNamespacesArray = append(affectedNamespacesArray, namespace)
   117  	}
   118  
   119  	return affectedNamespacesArray
   120  }

   135  // PodMode represents the mode to run pod chaos action.
   136  type PodMode string
   137  
   138  const (
   139  	// OnePodMode represents that the system will do the chaos action on one pod selected randomly.
   140  	OnePodMode PodMode = "one"
   141  	// AllPodMode represents that the system will do the chaos action on all pods
   142  	// regardless of status (not ready or not running pods includes).
   143  	// Use this label carefully.
   144  	AllPodMode PodMode = "all"
   145  	// FixedPodMode represents that the system will do the chaos action on a specific number of running pods.
   146  	FixedPodMode PodMode = "fixed"
   147  	// FixedPercentPodMode to specify a fixed % that can be inject chaos action.
   148  	FixedPercentPodMode PodMode = "fixed-percent"
   149  	// RandomMaxPercentPodMode to specify a maximum % that can be inject chaos action.
   150  	RandomMaxPercentPodMode PodMode = "random-max-percent"
   151  )

   171  func (in *ChaosStatus) GetNextStart() time.Time {
   172  	if in.Scheduler.NextStart == nil {
   173  		return time.Time{}
   174  	}
   175  	return in.Scheduler.NextStart.Time
   176  }

   178  func (in *ChaosStatus) SetNextStart(t time.Time) {
   179  	if t.IsZero() {
   180  		in.Scheduler.NextStart = nil
   181  		return
   182  	}
   183  
   184  	if in.Scheduler.NextStart == nil {
   185  		in.Scheduler.NextStart = &metav1.Time{}
   186  	}
   187  	in.Scheduler.NextStart.Time = t
