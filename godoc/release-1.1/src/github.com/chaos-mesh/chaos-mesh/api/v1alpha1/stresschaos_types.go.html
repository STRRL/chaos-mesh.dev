   129  // Normalize the stressors to comply with stress-ng
   130  func (in *Stressors) Normalize() (string, error) {
   131  	stressors := ""
   132  	if in.MemoryStressor != nil {
   133  		stressors += fmt.Sprintf(" --vm %d --vm-keep", in.MemoryStressor.Workers)
   134  		if len(in.MemoryStressor.Size) != 0 {
   135  			if in.MemoryStressor.Size[len(in.MemoryStressor.Size)-1] != '%' {
   136  				size, err := units.FromHumanSize(in.MemoryStressor.Size)
   137  				if err != nil {
   138  					return "", err
   139  				}
   140  				stressors += fmt.Sprintf(" --vm-bytes %d", size)
   141  			} else {
   142  				stressors += fmt.Sprintf("--vm-bytes %s",
   143  					in.MemoryStressor.Size)
   144  			}
   145  		}
   146  
   147  		if in.MemoryStressor.Options != nil {
   148  			for _, v := range in.MemoryStressor.Options {
   149  				stressors += fmt.Sprintf(" %v ", v)
   150  			}
   151  		}
   152  	}
   153  	if in.CPUStressor != nil {
   154  		stressors += fmt.Sprintf(" --cpu %d", in.CPUStressor.Workers)
   155  		if in.CPUStressor.Load != nil {
   156  			stressors += fmt.Sprintf(" --cpu-load %d",
   157  				*in.CPUStressor.Load)
   158  		}
   159  
   160  		if in.CPUStressor.Options != nil {
   161  			for _, v := range in.CPUStressor.Options {
   162  				stressors += fmt.Sprintf(" %v ", v)
   163  			}
   164  		}
   165  	}
   166  	return stressors, nil
   167  }
