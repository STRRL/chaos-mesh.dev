    20  // JVMChaosSpec defines the desired state of JVMChaos
    21  type JVMChaosSpec struct {
    22  	// Mode defines the mode to run chaos action.
    23  	// Supported mode: one / all / fixed / fixed-percent / random-max-percent
    24  	Mode PodMode `json:"mode"`
    25  
    26  	// Value is required when the mode is set to `FixedPodMode` / `FixedPercentPodMod` / `RandomMaxPercentPodMod`.
    27  	// If `FixedPodMode`, provide an integer of pods to do chaos action.
    28  	// If `FixedPercentPodMod`, provide a number from 0-100 to specify the max % of pods the server can do chaos action.
    29  	// If `RandomMaxPercentPodMod`,  provide a number from 0-100 to specify the % of pods to do chaos action
    30  	// +optional
    31  	Value string `json:"value"`
    32  
    33  	// Selector is used to select pods that are used to inject chaos action.
    34  	Selector SelectorSpec `json:"selector"`
    35  
    36  	// Duration represents the duration of the chaos action
    37  	// +optional
    38  	Duration *string `json:"duration,omitempty"`
    39

    41  	// +optional
    42  	Scheduler *SchedulerSpec `json:"scheduler,omitempty"`
    43  
    44  	// Action defines the specific jvm chaos action.
    45  	// Supported action: delay, return, script, cfl, oom, ccf, tce, delay4servlet, tce4servlet
    46  	// +kubebuilder:validation:Enum=delay;return;script;cfl;oom;ccf;tce;delay4servlet;tce4servlet
    47  	Action JVMChaosAction `json:"action"`

   164  	// Area represents JVM memory area, currently supported [HEAP, NOHEAP, OFFHEAP], required.
   165  	// Eden+Old is denoted by HEAP

   167  	// off-heap memory is denoted by OFFHEAP

   169  	Area string `json:"area"`
   170

   172  	// +optional
   173  	Block string `json:"block"`
   174  
   175  	// Interval represents unit MS, default interval between 500 OOM exceptions, only in non-violent mode, can slow down the frequency of GC without worrying about the process being unresponsive
   176  	// +optional
   177  	Interval int `json:"interval"`
   178  
   179  	// WildMode represents default false, whether to turn on wild mode or not.
   180  	// If it is wild mode, the memory created before will not be released after OOM occurrence, which may cause the application process to be unresponsive
   181  	// +optional
