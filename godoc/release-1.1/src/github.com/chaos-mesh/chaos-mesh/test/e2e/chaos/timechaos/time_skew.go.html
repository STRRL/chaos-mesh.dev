    34  func TestcaseTimeSkewOnceThenRecover(
    35  	ns string,
    36  	cli client.Client,
    37  	c http.Client,
    38  	port uint16,
    39  ) {
    40  
    41  	ctx, cancel := context.WithCancel(context.Background())
    42  	defer cancel()
    43  
    44  	By("wait e2e helper ready")
    45  	err := util.WaitE2EHelperReady(c, port)
    46  	framework.ExpectNoError(err, "wait e2e helper ready error")
    47  
    48  	By("create chaos CRD objects")
    49  	initTime, err := getPodTimeNS(c, port)
    50  	framework.ExpectNoError(err, "failed to get pod time")
    51  
    52  	timeChaos := &v1alpha1.TimeChaos{
    53  		ObjectMeta: metav1.ObjectMeta{
    54  			Name:      "timer-time-chaos",
    55  			Namespace: ns,
    56  		},
    57  		Spec: v1alpha1.TimeChaosSpec{
    58  			Selector: v1alpha1.SelectorSpec{
    59  				Namespaces:     []string{ns},
    60  				LabelSelectors: map[string]string{"app": "timer"},
    61  			},
    62  			Mode:       v1alpha1.OnePodMode,
    63  			Duration:   pointer.StringPtr("9m"),
    64  			TimeOffset: "-1h",
    65  			Scheduler: &v1alpha1.SchedulerSpec{
    66  				Cron: "@every 10m",
    67  			},
    68  		},
    69  	}
    70  	err = cli.Create(ctx, timeChaos)
    71  	framework.ExpectNoError(err, "create time chaos error")
    72  
    73  	By("waiting for assertion")
    74  	err = wait.PollImmediate(5*time.Second, 5*time.Minute, func() (done bool, err error) {
    75  		podTime, err := getPodTimeNS(c, port)
    76  		framework.ExpectNoError(err, "failed to get pod time")
    77  		if podTime.Before(*initTime) {
    78  			return true, nil
    79  		}
    80  		return false, nil
    81  	})
    82  	framework.ExpectNoError(err, "time chaos doesn't work as expected")
    83  
    84  	By("delete chaos CRD objects")
    85  	err = cli.Delete(ctx, timeChaos)
    86  	framework.ExpectNoError(err, "failed to delete time chaos")
    87  
    88  	By("waiting for assertion recovering")
    89  	err = wait.Poll(5*time.Second, 1*time.Minute, func() (done bool, err error) {
    90  		podTime, err := getPodTimeNS(c, port)
    91  		framework.ExpectNoError(err, "failed to get pod time")
    92  		// since there is no timechaos now, current pod time should not be earlier
    93  		// than the init time
    94  		if podTime.Before(*initTime) {
    95  			return true, nil
    96  		}
    97  		return false, nil
    98  	})
    99  	framework.ExpectError(err, "wait no timechaos error")
   100  	framework.ExpectEqual(err.Error(), wait.ErrWaitTimeout.Error())
   101  	By("success to perform time chaos")
   102  }
