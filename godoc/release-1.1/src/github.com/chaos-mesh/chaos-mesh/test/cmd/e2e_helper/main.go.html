    79  		return err
    80  	}

   144  func (s *server) networkPingTest(w http.ResponseWriter, r *http.Request) {
   145  	var body networkSendTestBody
   146  
   147  	err := json.NewDecoder(r.Body).Decode(&body)
   148  	if err != nil {
   149  		http.Error(w, err.Error(), http.StatusBadRequest)
   150  		return
   151  	}
   152  
   153  	c := http.Client{
   154  		Timeout: 2 * time.Second,
   155  	}
   156  	startTime := time.Now()
   157  	resp, err := c.Get(fmt.Sprintf("http://%s:8080/ping", body.TargetIP))
   158  	if err != nil {
   159  		http.Error(w, err.Error(), http.StatusBadRequest)
   160  		return
   161  	}
   162  	defer resp.Body.Close()
   163  
   164  	endTime := time.Now()
   165  	out, err := ioutil.ReadAll(resp.Body)
   166  	if err != nil {
   167  		http.Error(w, err.Error(), http.StatusBadRequest)
   168  		return
   169  	}
   170  
   171  	if string(out) != "pong" {
   172  		http.Error(w, "response is not pong", http.StatusBadRequest)
   173  		return
   174  	}
   175  
   176  	w.Write([]byte(fmt.Sprintf("OK %d", endTime.UnixNano()-startTime.UnixNano())))
   177  }
