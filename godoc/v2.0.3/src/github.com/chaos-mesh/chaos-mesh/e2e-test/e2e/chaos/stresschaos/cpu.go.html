    28  func TestcaseCPUStressInjectionOnceThenRecover(
    29  	ns string,
    30  	cli client.Client,
    31  	peers []*corev1.Pod,
    32  	ports []uint16,
    33  	c http.Client,
    34  ) {
    35  	ctx := context.Background()
    36  	By("create cpu stress chaos CRD objects")
    37  	cpuStressChaos := makeCPUStressChaos(ns, "cpu-stress", ns, "stress-peer-0", 1, 100)
    38  	err := cli.Create(ctx, cpuStressChaos.DeepCopy())
    39  	framework.ExpectNoError(err, "create stresschaos error")
    40  
    41  	lastCPUTime := make([]uint64, 2)
    42  	diff := make([]uint64, 2)
    43  	By("waiting for assertion some pods are experiencing cpu stress ")
    44  	err = wait.Poll(time.Second, 15*time.Second, func() (done bool, err error) {
    45  		conditions, err := probeStressCondition(c, peers, ports)
    46  		if err != nil {
    47  			return false, err
    48  		}
    49  
    50  		diff[0] = conditions[0].CpuTime - lastCPUTime[0]
    51  		diff[1] = conditions[1].CpuTime - lastCPUTime[1]
    52  		lastCPUTime[0] = conditions[0].CpuTime
    53  		lastCPUTime[1] = conditions[1].CpuTime
    54  		framework.Logf("get CPU: [%d, %d]", diff[0], diff[1])
    55  		// diff means the increasing CPU time (in nanosecond)
    56  		// just pick two threshold, 5e8 is a little shorter than one second
    57  		if diff[0] > 5e8 && diff[1] < 5e6 {
    58  			return true, nil
    59  		}
    60  		return false, nil
    61  	})
    62  	framework.ExpectNoError(err, "cpu stress failed")
    63  	By("delete pod failure chaos CRD objects")
    64  
    65  	err = cli.Delete(ctx, cpuStressChaos.DeepCopy())
    66  	framework.ExpectNoError(err, "delete stresschaos error")
    67  	By("waiting for assertion recovering")
    68  	lastCPUTime = make([]uint64, 2)
    69  	diff = make([]uint64, 2)
    70  	err = wait.Poll(time.Second, 15*time.Second, func() (done bool, err error) {
    71  		conditions, err := probeStressCondition(c, peers, ports)
    72  		if err != nil {
    73  			return false, err
    74  		}
    75  
    76  		diff[0] = conditions[0].CpuTime - lastCPUTime[0]
    77  		diff[1] = conditions[1].CpuTime - lastCPUTime[1]
    78  		lastCPUTime[0] = conditions[0].CpuTime
    79  		lastCPUTime[1] = conditions[1].CpuTime
    80  		framework.Logf("get CPU: [%d, %d]", diff[0], diff[1])
    81  		// diff means the increasing CPU time (in nanosecond)
    82  		// just pick two threshold, they are both much shorter than 1 second
    83  		if diff[0] < 1e7 && diff[1] < 5e6 {
    84  			return true, nil
    85  		}
    86  		return false, nil
    87  	})
    88  	framework.ExpectNoError(err, "fail to recover from cpu stress")
    89  }
