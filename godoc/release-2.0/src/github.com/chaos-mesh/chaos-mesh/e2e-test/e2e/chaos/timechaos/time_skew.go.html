    33  func TestcaseTimeSkewOnceThenRecover(
    34  	ns string,
    35  	cli client.Client,
    36  	c http.Client,
    37  	port uint16,
    38  ) {
    39  
    40  	ctx, cancel := context.WithCancel(context.Background())
    41  	defer cancel()
    42  
    43  	By("wait e2e helper ready")
    44  	err := util.WaitE2EHelperReady(c, port)
    45  	framework.ExpectNoError(err, "wait e2e helper ready error")
    46  
    47  	By("create chaos CRD objects")
    48  	initTime, err := getPodTimeNS(c, port)
    49  	framework.ExpectNoError(err, "failed to get pod time")
    50  
    51  	timeChaos := &v1alpha1.TimeChaos{
    52  		ObjectMeta: metav1.ObjectMeta{
    53  			Name:      "timer-time-chaos",
    54  			Namespace: ns,
    55  		},
    56  		Spec: v1alpha1.TimeChaosSpec{
    57  			Duration:   pointer.StringPtr("9m"),
    58  			TimeOffset: "-1h",
    59  			ContainerSelector: v1alpha1.ContainerSelector{
    60  				PodSelector: v1alpha1.PodSelector{
    61  					Selector: v1alpha1.PodSelectorSpec{
    62  						Namespaces:     []string{ns},
    63  						LabelSelectors: map[string]string{"app": "timer"},
    64  					},
    65  					Mode: v1alpha1.OnePodMode,
    66  				},
    67  			},
    68  		},
    69  	}
    70  	err = cli.Create(ctx, timeChaos)
    71  	framework.ExpectNoError(err, "create time chaos error")
    72  
    73  	By("waiting for assertion")
    74  	err = wait.PollImmediate(5*time.Second, 5*time.Minute, func() (done bool, err error) {
    75  		podTime, err := getPodTimeNS(c, port)
    76  		framework.ExpectNoError(err, "failed to get pod time")
    77  		if podTime.Before(*initTime) {
    78  			return true, nil
    79  		}
    80  		return false, nil
    81  	})
    82  	framework.ExpectNoError(err, "time chaos doesn't work as expected")
    83  
    84  	By("delete chaos CRD objects")
    85  	err = cli.Delete(ctx, timeChaos)
    86  	framework.ExpectNoError(err, "failed to delete time chaos")
    87  
    88  	By("waiting for assertion recovering")
    89  	err = wait.Poll(5*time.Second, 1*time.Minute, func() (done bool, err error) {
    90  		podTime, err := getPodTimeNS(c, port)
    91  		framework.ExpectNoError(err, "failed to get pod time")
    92  		// since there is no timechaos now, current pod time should not be earlier
    93  		// than the init time
    94  		if podTime.Before(*initTime) {
    95  			return true, nil
    96  		}
    97  		return false, nil
    98  	})
    99  	framework.ExpectError(err, "wait no timechaos error")
   100  	framework.ExpectEqual(err.Error(), wait.ErrWaitTimeout.Error())
   101  	By("success to perform time chaos")
   102  }

   104  func TestcaseTimeSkewPauseThenUnpause(
   105  	ns string,
   106  	cli client.Client,
   107  	c http.Client,
   108  	port uint16,
   109  ) {
   110  	ctx, cancel := context.WithCancel(context.Background())
   111  	defer cancel()
   112  
   113  	By("wait e2e helper ready")
   114  	err := util.WaitE2EHelperReady(c, port)
   115  	framework.ExpectNoError(err, "wait e2e helper ready error")
   116  
   117  	initTime, err := getPodTimeNS(c, port)
   118  	framework.ExpectNoError(err, "failed to get pod time")
   119  
   120  	By("create chaos CRD objects")
   121  	timeChaos := &v1alpha1.TimeChaos{
   122  		ObjectMeta: metav1.ObjectMeta{
   123  			Name:      "timer-time-chaos",
   124  			Namespace: ns,
   125  		},
   126  		Spec: v1alpha1.TimeChaosSpec{
   127  			Duration:   pointer.StringPtr("9m"),
   128  			TimeOffset: "-1h",
   129  			ContainerSelector: v1alpha1.ContainerSelector{
   130  				PodSelector: v1alpha1.PodSelector{
   131  					Selector: v1alpha1.PodSelectorSpec{
   132  						Namespaces:     []string{ns},
   133  						LabelSelectors: map[string]string{"app": "timer"},
   134  					},
   135  					Mode: v1alpha1.OnePodMode,
   136  				},
   137  			},
   138  		},
   139  	}
   140  	err = cli.Create(ctx, timeChaos)
   141  	framework.ExpectNoError(err, "create time chaos error")
   142  
   143  	By("waiting for assertion")
   144  	err = wait.PollImmediate(5*time.Second, 5*time.Minute, func() (done bool, err error) {
   145  		podTime, err := getPodTimeNS(c, port)
   146  		framework.ExpectNoError(err, "failed to get pod time")
   147  		if podTime.Before(*initTime) {
   148  			return true, nil
   149  		}
   150  		return false, nil
   151  	})
   152  	framework.ExpectNoError(err, "time chaos doesn't work as expected")
   153  
   154  	chaosKey := types.NamespacedName{
   155  		Namespace: ns,
   156  		Name:      "timer-time-chaos",
   157  	}
   158  
   159  	By("pause time skew chaos experiment")
   160  	// pause experiment
   161  	err = util.PauseChaos(ctx, cli, timeChaos)
   162  	framework.ExpectNoError(err, "pause chaos error")
   163  
   164  	By("assert pause is effective")
   165  	err = wait.Poll(5*time.Second, 5*time.Minute, func() (done bool, err error) {
   166  		chaos := &v1alpha1.TimeChaos{}
   167  		err = cli.Get(ctx, chaosKey, chaos)
   168  		framework.ExpectNoError(err, "get time chaos error")
   169  		if chaos.Status.Experiment.DesiredPhase == v1alpha1.StoppedPhase {
   170  			return true, nil
   171  		}
   172  		return false, err
   173  	})
   174  	framework.ExpectNoError(err, "check paused chaos failed")
   175  
   176  	// wait for 1 minutes and check timer
   177  	framework.ExpectNoError(err, "get timer pod error")
   178  	err = wait.Poll(5*time.Second, 1*time.Minute, func() (done bool, err error) {
   179  		podTime, err := getPodTimeNS(c, port)
   180  		framework.ExpectNoError(err, "failed to get pod time")
   181  		if podTime.Before(*initTime) {
   182  			return true, nil
   183  		}
   184  		return false, nil
   185  	})
   186  	framework.ExpectError(err, "wait time chaos paused error")
   187  	framework.ExpectEqual(err.Error(), wait.ErrWaitTimeout.Error())
   188  
   189  	By("resume time skew chaos experiment")
   190  	err = util.UnPauseChaos(ctx, cli, timeChaos)
   191  	framework.ExpectNoError(err, "resume chaos error")
   192  
   193  	By("assert chaos experiment resumed")
   194  	err = wait.Poll(5*time.Second, 5*time.Minute, func() (done bool, err error) {
   195  		chaos := &v1alpha1.TimeChaos{}
   196  		err = cli.Get(ctx, chaosKey, chaos)
   197  		framework.ExpectNoError(err, "get time chaos error")
   198  		if chaos.Status.Experiment.DesiredPhase == v1alpha1.RunningPhase {
   199  			return true, nil
   200  		}
   201  		return false, err
   202  	})
   203  	framework.ExpectNoError(err, "check resumed chaos failed")
   204  
   205  	// timechaos is running again, we want to check pod
   206  	// whether time is earlier than init time,
   207  	err = wait.PollImmediate(5*time.Second, 1*time.Minute, func() (done bool, err error) {
   208  		podTime, err := getPodTimeNS(c, port)
   209  		framework.ExpectNoError(err, "failed to get pod time")
   210  		if podTime.Before(*initTime) {
   211  			return true, nil
   212  		}
   213  		return false, nil
   214  	})
   215  	framework.ExpectNoError(err, "time chaos failed")
   216  
   217  	By("delete chaos CRD objects")
   218  	cli.Delete(ctx, timeChaos)
   219  }
