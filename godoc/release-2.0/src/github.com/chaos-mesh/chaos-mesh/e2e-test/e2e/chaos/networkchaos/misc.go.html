    81  	body := []byte(fmt.Sprintf("{\"targetIP\":\"%s\"}", targetIP))

    85  	if err != nil {
    86  		return 0, err
    87  	}
    88  
    89  	out, err := ioutil.ReadAll(resp.Body)
    90  	defer resp.Body.Close()
    91  	if err != nil {
    92  		return 0, err
    93  	}
    94  
    95  	result := string(out)
    96  	parts := strings.Split(result, " ")
    97  	if len(parts) != 2 {
    98  		return 0, fmt.Errorf("the length of parts is not 2 %v", parts)
    99  	}
   100  
   101  	if parts[0] != "OK" {
   102  		return 0, fmt.Errorf("the first part of response is not OK")
   103  	}
   104  
   105  	return strconv.ParseInt(parts[1], 10, 64)
   106  }

   183  func probeNetworkCondition(c http.Client, peers []*corev1.Pod, ports []uint16, bidirection bool) map[string][][]int {
   184  	result := make(map[string][][]int)
   185  
   186  	testDelay := func(from int, to int) (int64, error) {
   187  		return testNetworkDelay(c, ports[from], peers[to].Status.PodIP)
   188  	}
   189  
   190  	for source := 0; source < len(peers); source++ {
   191  		initialTarget := source + 1
   192  		if bidirection {
   193  			initialTarget = 0
   194  		}
   195  		for target := initialTarget; target < len(peers); target++ {
   196  			if target == source {
   197  				continue
   198  			}
   199  
   200  			connectable := true
   201  
   202  			var (
   203  				wg           sync.WaitGroup
   204  				link1, link2 bool
   205  			)
   206  			wg.Add(2)
   207  			go func() {
   208  				defer wg.Done()
   209  				// case 1-1: source to target blocked?
   210  				klog.Infof("testing connectivity from %s to %s", peers[source].Name, peers[target].Name)
   211  				link1 = couldConnect(c, ports[source], peers[target].Status.PodIP, ports[target])
   212  
   213  			}()
   214  
   215  			go func() {
   216  				defer wg.Done()
   217  				// case 1-2: target to source blocked?
   218  				klog.Infof("testing connectivity from %s to %s", peers[target].Name, peers[source].Name)
   219  				link2 = couldConnect(c, ports[target], peers[source].Status.PodIP, ports[source])
   220  			}()
   221  			wg.Wait()
   222  
   223  			if !link1 {
   224  				klog.Infof("%s could not connect to %s", peers[source].Name, peers[target].Name)
   225  				result[networkConditionBlocked] = append(result[networkConditionBlocked], []int{source, target})
   226  				connectable = false
   227  			}
   228  			if !link2 {
   229  				klog.Infof("%s could not connect to %s", peers[target].Name, peers[source].Name)
   230  				result[networkConditionBlocked] = append(result[networkConditionBlocked], []int{target, source})
   231  				connectable = false
   232  			}
   233  
   234  			if !connectable {
   235  				continue
   236  			}
   237  
   238  			// case 2: slow network
   239  			klog.Infof("testing delay from %s to %s", peers[source].Name, peers[target].Name)
   240  			delay, err := testDelay(source, target)
   241  			if err != nil {
   242  				klog.Errorf("error from %d to %d: %v", source, target, err)
   243  				continue
   244  			}
   245  
   246  			klog.Infof("delay from %d to %d: %d", source, target, delay)
   247  			if delay > 100*1e6 {
   248  				klog.Infof("detect slow network from %s to %s", peers[source].Name, peers[target].Name)
   249  				result[networkConditionSlow] = append(result[networkConditionSlow], []int{source, target})
   250  				continue
   251  			}
   252  
   253  			// case 3: otherwise, good network
   254  			klog.Infof("good network from %d to %d", source, target)
   255  			result[networkConditionGood] = append(result[networkConditionGood], []int{source, target})
   256  		}
   257  	}
   258  
   259  	return result
   260  }

   262  func couldConnect(c http.Client, sourcePort uint16, targetPodIP string, targetPort uint16) bool {
   263  	err := sendUDPPacket(c, sourcePort, targetPodIP)
   264  	if err != nil {
   265  		klog.Infof("Error: %v", err)
   266  		return false
   267  	}
   268  
   269  	time.Sleep(time.Second)
   270  
   271  	data, err := recvUDPPacket(c, targetPort)
   272  	if err != nil {
   273  		klog.Infof("Error: %v, Data: %s", err, data)
   274  		return false
   275  	}
   276  
   277  	// FIXME: slow network may also make this happens
   278  	if data != "ping\n" {
   279  		klog.Infof("mismatch data return: %s, it may happens under bad network", data)
   280  	}
   281  
   282  	return true
   283  }
