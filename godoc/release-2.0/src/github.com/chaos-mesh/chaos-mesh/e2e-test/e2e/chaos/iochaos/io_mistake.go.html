    35  	cli client.Client,

    43  
    44  	ioChaos := &v1alpha1.IOChaos{
    45  		ObjectMeta: metav1.ObjectMeta{

    47  			Namespace: ns,

    49  		Spec: v1alpha1.IOChaosSpec{
    50  			ContainerSelector: v1alpha1.ContainerSelector{
    51  				PodSelector: v1alpha1.PodSelector{
    52  					Selector: v1alpha1.PodSelectorSpec{
    53  						Namespaces:     []string{ns},
    54  						LabelSelectors: map[string]string{"app": "io"},
    55  					},
    56  					Mode: v1alpha1.OnePodMode,
    57  				},
    58  			},
    59  			Action:     v1alpha1.IoMistake,
    60  			VolumePath: "/var/run/data",
    61  			Path:       "/var/run/data/*",
    62  			Percent:    100,
    63  			Mistake: &v1alpha1.MistakeSpec{
    64  				MaxOccurrences: 1,
    65  				MaxLength:      10000,
    66  				Filling:        v1alpha1.Zero,
    67  			},
    68  			// only inject read or write method. Other method may or may not run properly, but is not recommended
    69  			Methods:  []v1alpha1.IoMethod{v1alpha1.Read, v1alpha1.Write},
    70  			Duration: pointer.StringPtr("9m"),
    71  		},
    72  	}
    73  	err = cli.Create(ctx, ioChaos)
    74  	framework.ExpectNoError(err, "create io chaos")
    75  
    76  	err = wait.PollImmediate(5*time.Second, 1*time.Minute, func() (bool, error) {
    77  		res, err := getPodIoMistake(c, port)
    78  		if err != nil {
    79  			return false, nil
    80  		}
    81  		if err == nil && res {
    82  			return true, nil
    83  		}
    84  		return false, nil
    85  	})
    86  	framework.ExpectNoError(err, "io chaos doesn't work as expected")
    87  
    88  	err = cli.Delete(ctx, ioChaos)
    89  	framework.ExpectNoError(err, "failed to delete io chaos")
    90  
    91  	klog.Infof("success to perform io chaos")
    92  	err = wait.PollImmediate(5*time.Second, 1*time.Minute, func() (bool, error) {
    93  		res, err := getPodIoMistake(c, port)
    94  		if err != nil {
    95  			return false, nil
    96  		}
    97  		if err == nil && !res {
    98  			return true, nil
    99  		}
   100  		return false, nil
   101  	})
   102  	framework.ExpectNoError(err, "fail to recover io chaos")
   103  }
