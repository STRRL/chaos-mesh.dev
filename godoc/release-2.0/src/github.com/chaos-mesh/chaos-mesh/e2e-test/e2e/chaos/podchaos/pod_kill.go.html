    62  					},

    65  			},

    69  	framework.ExpectNoError(err, "create pod chaos error")

    74  			return true, nil
    75  		}
    76  		return false, nil
    77  	})

    81  func TestcasePodKillPauseThenUnPause(ns string, kubeCli kubernetes.Interface, cli client.Client) {
    82  	ctx, cancel := context.WithCancel(context.Background())
    83  	defer cancel()
    84  
    85  	nd := fixture.NewCommonNginxDeployment("nginx", ns, 3)
    86  	_, err := kubeCli.AppsV1().Deployments(ns).Create(nd)
    87  	framework.ExpectNoError(err, "create nginx deployment error")
    88  	err = util.WaitDeploymentReady("nginx", ns, kubeCli)
    89  	framework.ExpectNoError(err, "wait nginx deployment ready error")
    90  
    91  	var pods *corev1.PodList
    92  	var newPods *corev1.PodList
    93  	listOption := metav1.ListOptions{
    94  		LabelSelector: labels.SelectorFromSet(map[string]string{
    95  			"app": "nginx",
    96  		}).String(),
    97  	}
    98  	pods, err = kubeCli.CoreV1().Pods(ns).List(listOption)
    99  	framework.ExpectNoError(err, "get nginx pods error")
   100  
   101  	podKillChaos := &v1alpha1.PodChaos{
   102  		ObjectMeta: metav1.ObjectMeta{
   103  			Name:      "nginx-kill",
   104  			Namespace: ns,
   105  		},
   106  		Spec: v1alpha1.PodChaosSpec{
   107  			Action:   v1alpha1.PodKillAction,
   108  			Duration: pointer.StringPtr("9m"),
   109  			ContainerSelector: v1alpha1.ContainerSelector{
   110  				PodSelector: v1alpha1.PodSelector{
   111  					Selector: v1alpha1.PodSelectorSpec{
   112  						Namespaces: []string{
   113  							ns,
   114  						},
   115  						LabelSelectors: map[string]string{
   116  							"app": "nginx",
   117  						},
   118  					},
   119  					Mode: v1alpha1.OnePodMode,
   120  				},
   121  			},
   122  		},
   123  	}
   124  	err = cli.Create(ctx, podKillChaos)
   125  	framework.ExpectNoError(err, "create pod chaos error")
   126  
   127  	chaosKey := types.NamespacedName{
   128  		Namespace: ns,
   129  		Name:      "nginx-kill",
   130  	}
   131  
   132  	// some pod is killed as expected
   133  	err = wait.Poll(5*time.Second, 5*time.Minute, func() (done bool, err error) {
   134  		newPods, err = kubeCli.CoreV1().Pods(ns).List(listOption)
   135  		framework.ExpectNoError(err, "get nginx pods error")
   136  		return !fixture.HaveSameUIDs(pods.Items, newPods.Items), nil
   137  	})
   138  	framework.ExpectNoError(err, "wait pod killed failed")
   139  
   140  	// pause experiment
   141  	err = util.PauseChaos(ctx, cli, podKillChaos)
   142  	framework.ExpectNoError(err, "pause chaos error")
   143  
   144  	err = wait.Poll(1*time.Second, 5*time.Second, func() (done bool, err error) {
   145  		chaos := &v1alpha1.PodChaos{}
   146  		err = cli.Get(ctx, chaosKey, chaos)
   147  		framework.ExpectNoError(err, "get pod chaos error")
   148  		if chaos.Status.Experiment.DesiredPhase == v1alpha1.StoppedPhase {
   149  			return true, nil
   150  		}
   151  		return false, err
   152  	})
   153  	framework.ExpectError(err, "chaos shouldn't enter stopped phase")
   154  
   155  	// wait for 1 minutes and no pod is killed
   156  	pods, err = kubeCli.CoreV1().Pods(ns).List(listOption)
   157  	framework.ExpectNoError(err, "get nginx pods error")
   158  	err = wait.Poll(5*time.Second, 1*time.Minute, func() (done bool, err error) {
   159  		newPods, err = kubeCli.CoreV1().Pods(ns).List(listOption)
   160  		framework.ExpectNoError(err, "get nginx pods error")
   161  		return !fixture.HaveSameUIDs(pods.Items, newPods.Items), nil
   162  	})
   163  	framework.ExpectError(err, "wait pod not killed failed")
   164  	framework.ExpectEqual(err.Error(), wait.ErrWaitTimeout.Error())
   165  
   166  }
