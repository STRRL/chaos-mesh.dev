    37  func TestcasePodFailureOnceThenDelete(ns string, kubeCli kubernetes.Interface, cli client.Client) {
    38  	ctx, cancel := context.WithCancel(context.Background())
    39  	defer cancel()
    40  
    41  	By("preparing experiment pods")
    42  	appName := "timer-pod-failure1"
    43  	nd := fixture.NewTimerDeployment(appName, ns)
    44  	_, err := kubeCli.AppsV1().Deployments(ns).Create(nd)
    45  	framework.ExpectNoError(err, "create timer deployment error")
    46  	err = util.WaitDeploymentReady(appName, ns, kubeCli)
    47  	framework.ExpectNoError(err, "wait timer deployment ready error")
    48  
    49  	By("create pod failure chaos CRD objects")
    50  	listOption := metav1.ListOptions{
    51  		LabelSelector: labels.SelectorFromSet(map[string]string{
    52  			"app": appName,
    53  		}).String(),
    54  	}
    55  	podFailureChaos := &v1alpha1.PodChaos{
    56  		ObjectMeta: metav1.ObjectMeta{
    57  			Name:      "timer-failure1",
    58  			Namespace: ns,
    59  		},
    60  		Spec: v1alpha1.PodChaosSpec{
    61  			Action: v1alpha1.PodFailureAction,
    62  			ContainerSelector: v1alpha1.ContainerSelector{
    63  				PodSelector: v1alpha1.PodSelector{
    64  					Selector: v1alpha1.PodSelectorSpec{
    65  						Namespaces: []string{
    66  							ns,
    67  						},
    68  						LabelSelectors: map[string]string{
    69  							"app": appName,
    70  						},
    71  					},
    72  					Mode: v1alpha1.OnePodMode,
    73  				},
    74  			},
    75  		},
    76  	}
    77  
    78  	err = cli.Create(ctx, podFailureChaos)
    79  	framework.ExpectNoError(err, "create pod failure chaos error")
    80  
    81  	By("waiting for assertion some pod fall into failure")
    82  	err = wait.Poll(5*time.Second, 5*time.Minute, func() (done bool, err error) {
    83  		pods, err := kubeCli.CoreV1().Pods(ns).List(listOption)
    84  		if err != nil {
    85  			return false, nil
    86  		}
    87  		if len(pods.Items) != 1 {
    88  			return false, nil
    89  		}
    90  		pod := pods.Items[0]
    91  		for _, c := range pod.Spec.Containers {
    92  			if c.Image == e2econst.PauseImage {
    93  				return true, nil
    94  			}
    95  		}
    96  		return false, nil
    97  	})
    98  	framework.ExpectNoError(err, "failed to verify PodFailure")
    99  
   100  	By("delete pod failure chaos CRD objects")
   101  	err = cli.Delete(ctx, podFailureChaos)
   102  	framework.ExpectNoError(err, "failed to delete pod failure chaos")
   103  
   104  	By("waiting for assertion recovering")
   105  	err = wait.Poll(5*time.Second, 2*time.Minute, func() (done bool, err error) {
   106  		pods, err := kubeCli.CoreV1().Pods(ns).List(listOption)
   107  		if err != nil {
   108  			return false, nil
   109  		}
   110  		if len(pods.Items) != 1 {
   111  			return false, nil
   112  		}
   113  		pod := pods.Items[0]
   114  		for _, c := range pod.Spec.Containers {
   115  			if c.Image == nd.Spec.Template.Spec.Containers[0].Image {
   116  				return true, nil
   117  			}
   118  		}
   119  		return false, nil
   120  	})
   121  	framework.ExpectNoError(err, "pod failure recover failed")
   122  }

   124  func TestcasePodFailurePauseThenUnPause(ns string, kubeCli kubernetes.Interface, cli client.Client) {
   125  	ctx, cancel := context.WithCancel(context.Background())
   126  	defer cancel()
   127  
   128  	By("preparing experiment pods")
   129  	appName := "timer-pod-failure2"
   130  	nd := fixture.NewTimerDeployment(appName, ns)
   131  	_, err := kubeCli.AppsV1().Deployments(ns).Create(nd)
   132  	framework.ExpectNoError(err, "create timer deployment error")
   133  	err = util.WaitDeploymentReady(appName, ns, kubeCli)
   134  	framework.ExpectNoError(err, "wait timer deployment ready error")
   135  
   136  	By("create pod failure chaos CRD objects")
   137  	var pods *corev1.PodList
   138  	listOption := metav1.ListOptions{
   139  		LabelSelector: labels.SelectorFromSet(map[string]string{
   140  			"app": appName,
   141  		}).String(),
   142  	}
   143  
   144  	podFailureChaos := &v1alpha1.PodChaos{
   145  		ObjectMeta: metav1.ObjectMeta{
   146  			Name:      "timer-failure2",
   147  			Namespace: ns,
   148  		},
   149  		Spec: v1alpha1.PodChaosSpec{
   150  			Action:   v1alpha1.PodFailureAction,
   151  			Duration: pointer.StringPtr("9m"),
   152  			ContainerSelector: v1alpha1.ContainerSelector{
   153  				PodSelector: v1alpha1.PodSelector{
   154  					Selector: v1alpha1.PodSelectorSpec{
   155  						Namespaces: []string{
   156  							ns,
   157  						},
   158  						LabelSelectors: map[string]string{
   159  							"app": appName,
   160  						},
   161  					},
   162  					Mode: v1alpha1.OnePodMode,
   163  				},
   164  			},
   165  		},
   166  	}
   167  	err = cli.Create(ctx, podFailureChaos)
   168  	framework.ExpectNoError(err, "create pod failure chaos error")
   169  	chaosKey := types.NamespacedName{
   170  		Namespace: ns,
   171  		Name:      "timer-failure2",
   172  	}
   173  
   174  	By("waiting for assertion some pod fall into failure")
   175  	// check whether the pod failure chaos succeeded or not
   176  	err = wait.Poll(5*time.Second, 5*time.Minute, func() (done bool, err error) {
   177  		pods, err := kubeCli.CoreV1().Pods(ns).List(listOption)
   178  		if err != nil {
   179  			return false, nil
   180  		}
   181  		pod := pods.Items[0]
   182  		for _, c := range pod.Spec.Containers {
   183  			if c.Image == e2econst.PauseImage {
   184  				return true, nil
   185  			}
   186  		}
   187  		return false, nil
   188  	})
   189  	framework.ExpectNoError(err, "image not update to pause")
   190  
   191  	// pause experiment
   192  	By("pause pod failure chaos")
   193  	err = util.PauseChaos(ctx, cli, podFailureChaos)
   194  	framework.ExpectNoError(err, "pause chaos error")
   195  
   196  	By("waiting for assertion about chaos experiment paused")
   197  	err = wait.Poll(5*time.Second, 5*time.Minute, func() (done bool, err error) {
   198  		chaos := &v1alpha1.PodChaos{}
   199  		err = cli.Get(ctx, chaosKey, chaos)
   200  		framework.ExpectNoError(err, "get pod chaos error")
   201  		if chaos.Status.Experiment.DesiredPhase == v1alpha1.StoppedPhase {
   202  			return true, nil
   203  		}
   204  		return false, err
   205  	})
   206  	framework.ExpectNoError(err, "check paused chaos failed")
   207  
   208  	By("wait for 30 seconds and no pod failure")
   209  	pods, err = kubeCli.CoreV1().Pods(ns).List(listOption)
   210  	framework.ExpectNoError(err, "get timer pod error")
   211  	err = wait.Poll(5*time.Second, 30*time.Second, func() (done bool, err error) {
   212  		pods, err = kubeCli.CoreV1().Pods(ns).List(listOption)
   213  		framework.ExpectNoError(err, "get timer pod error")
   214  		pod := pods.Items[0]
   215  		for _, c := range pod.Spec.Containers {
   216  			if c.Image == e2econst.PauseImage {
   217  				return false, nil
   218  			}
   219  		}
   220  
   221  		return true, nil
   222  	})
   223  	framework.ExpectNoError(err, "check paused chaos failed")
   224  
   225  	By("resume paused chaos experiment")
   226  	err = util.UnPauseChaos(ctx, cli, podFailureChaos)
   227  	framework.ExpectNoError(err, "resume chaos error")
   228  
   229  	By("waiting for assertion about pod failure happens again")
   230  	err = wait.Poll(5*time.Second, 5*time.Minute, func() (done bool, err error) {
   231  		chaos := &v1alpha1.PodChaos{}
   232  		err = cli.Get(ctx, chaosKey, chaos)
   233  		framework.ExpectNoError(err, "get pod chaos error")
   234  		if chaos.Status.Experiment.DesiredPhase == v1alpha1.RunningPhase {
   235  			return true, nil
   236  		}
   237  		return false, err
   238  	})
   239  	framework.ExpectNoError(err, "check resumed chaos failed")
   240  
   241  	By("waiting for assert pod failure happens again")
   242  	err = wait.Poll(5*time.Second, 5*time.Minute, func() (done bool, err error) {
   243  		pods, err = kubeCli.CoreV1().Pods(ns).List(listOption)
   244  		framework.ExpectNoError(err, "get timer pod error")
   245  		pod := pods.Items[0]
   246  		for _, c := range pod.Spec.Containers {
   247  			if c.Image == e2econst.PauseImage {
   248  				return true, nil
   249  			}
   250  		}
   251  		return false, nil
   252  	})
   253  	framework.ExpectNoError(err, "wait pod failure failed")
   254  }
