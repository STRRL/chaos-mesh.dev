    62  func (oa *operatorAction) DeployOperator(info OperatorConfig) error {
    63  	klog.Infof("create namespace chaos-testing")
    64  	cmd := fmt.Sprintf(`kubectl create ns %s`, e2econst.ChaosMeshNamespace)
    65  	klog.Infof(cmd)
    66  	output, err := exec.Command("/bin/sh", "-c", cmd).CombinedOutput()
    67  	if err != nil {
    68  		return fmt.Errorf("failed to create namespace chaos-testing: %v %s", err, string(output))
    69  	}
    70  	klog.Infof("deploying chaos-mesh:%v", info.ReleaseName)
    71  	cmd = fmt.Sprintf(`helm install %s %s --namespace %s --set %s --skip-crds`,
    72  		info.ReleaseName,
    73  		oa.operatorChartPath(info.Tag),
    74  		info.Namespace,
    75  		info.operatorHelmSetValue())
    76  	klog.Info(cmd)
    77  	res, err := exec.Command("/bin/sh", "-c", cmd).CombinedOutput()
    78  	if err != nil {
    79  		return fmt.Errorf("failed to deploy operator: %v, %s", err, string(res))
    80  	}
    81  	klog.Infof("start to waiting chaos-mesh ready")
    82  	err = wait.Poll(5*time.Second, 5*time.Minute, func() (done bool, err error) {
    83  
    84  		ls := &metav1.LabelSelector{
    85  			MatchLabels: map[string]string{
    86  				"app.kubernetes.io/instance": "chaos-mesh",
    87  			},
    88  		}
    89  		l, err := metav1.LabelSelectorAsSelector(ls)
    90  		if err != nil {
    91  			klog.Errorf("failed to get selector, err:%v", err)
    92  			return false, nil
    93  		}
    94  		pods, err := oa.kubeCli.CoreV1().Pods(info.Namespace).List(metav1.ListOptions{LabelSelector: l.String()})
    95  		if err != nil {
    96  			klog.Errorf("failed to get chaos-mesh pods, err:%v", err)
    97  			return false, nil
    98  		}
    99  		for _, pod := range pods.Items {
   100  			if pod.Status.Phase != corev1.PodRunning {
   101  				return false, nil
   102  			}
   103  		}
   104  		return true, nil
   105  	})
   106  	if err != nil {
   107  		return err
   108  	}
   109  	return e2eutil.WaitForAPIServicesAvailable(oa.aggrCli, labels.Everything())
   110  }

   112  func (oa *operatorAction) InstallCRD(info OperatorConfig) error {
   113  	klog.Infof("deploying chaos-mesh crd :%v", info.ReleaseName)
   114  	if oa.apiextensionsV1Available() {
   115  		oa.runKubectlOrDie("create", "-f", oa.manifestPath("e2e/crd.yaml"), "--validate=false")
   116  	} else {
   117  		oa.runKubectlOrDie("create", "-f", oa.manifestPath("e2e/crd-v1beta1.yaml"), "--validate=false")
   118  	}
   119  	e2eutil.WaitForCRDsEstablished(oa.apiExtCli, labels.Everything())
   120  	// workaround for https://github.com/kubernetes/kubernetes/issues/65517
   121  	klog.Infof("force sync kubectl cache")
   122  	cmdArgs := []string{"sh", "-c", "rm -rf ~/.kube/cache ~/.kube/http-cache"}
   123  	_, err := exec.Command(cmdArgs[0], cmdArgs[1:]...).CombinedOutput()
   124  	if err != nil {
   125  		klog.Fatalf("Failed to run '%s': %v", strings.Join(cmdArgs, " "), err)
   126  	}
   127  	return nil
   128  }

   130  // check apiextensions.k8s.io/v1 CustomResourceDefinition is Availabel or not
   131  func (oa *operatorAction) apiextensionsV1Available() bool {
   132  	for _, item := range oa.apiVersions() {
   133  		if item == "apiextensions.k8s.io/v1" {
   134  			return true
   135  		}
   136  	}
   137  	return false
   138  }
