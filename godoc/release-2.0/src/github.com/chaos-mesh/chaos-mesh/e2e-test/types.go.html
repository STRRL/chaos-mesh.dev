    90  func (oi *OperatorConfig) operatorHelmSetValue() string {
    91  	set := map[string]string{
    92  		"controllerManager.image":           fmt.Sprintf("%s:%s", oi.Manager.Image, oi.Manager.Tag),
    93  		"controllerManager.imagePullPolicy": oi.Manager.ImagePullPolicy,
    94  		"chaosDaemon.image":                 fmt.Sprintf("%s:%s", oi.Daemon.Image, oi.Daemon.Tag),
    95  		"chaosDaemon.runtime":               oi.Daemon.Runtime,
    96  		"chaosDaemon.socketPath":            oi.Daemon.SocketPath,
    97  		"chaosDaemon.imagePullPolicy":       oi.Daemon.ImagePullPolicy,
    98  		"dnsServer.create":                  "true",
    99  		"dnsServer.image":                   oi.DNSImage,
   100  		"dashboard.create":                  fmt.Sprintf("%t", oi.EnableDashboard),
   101  	}
   102  	arr := make([]string, 0, len(set))
   103  	for k, v := range set {
   104  		arr = append(arr, fmt.Sprintf("%s=%s", k, v))
   105  	}
   106  	return fmt.Sprintf("\"%s\"", strings.Join(arr, ","))
   107  }

   121  func (oa *operatorAction) runKubectlOrDie(args ...string) string {
   122  	cmd := "kubectl"
   123  	klog.Infof("Running '%s %s'", cmd, strings.Join(args, " "))
   124  	out, err := exec.Command(cmd, args...).CombinedOutput()
   125  	if err != nil {
   126  		klog.Fatalf("Failed to run '%s %s'\nCombined output: %q\nError: %v", cmd, strings.Join(args, " "), string(out), err)
   127  	}
   128  	klog.Infof("Combined output: %q", string(out))
   129  	return string(out)
   130  }
