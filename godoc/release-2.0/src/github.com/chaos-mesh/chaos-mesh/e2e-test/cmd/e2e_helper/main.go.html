   187

   191  		return

   194  	c := http.Client{

   198  	resp, err := c.Get(fmt.Sprintf("http://%s:8080/ping", body.TargetIP))
   199  	if err != nil {
   200  		http.Error(w, err.Error(), http.StatusBadRequest)
   201  		return
   202  	}

   205  	endTime := time.Now()

   207  	if err != nil {
   208  		http.Error(w, err.Error(), http.StatusBadRequest)

   211  
   212  	if string(out) != "pong" {

   214  		return
   215  	}
   216  
   217  	w.Write([]byte(fmt.Sprintf("OK %d", endTime.UnixNano()-startTime.UnixNano())))

   259  func (s *server) stressCondition(w http.ResponseWriter, r *http.Request) {
   260  	control, err := cgroups.Load(cgroups.V1, cgroups.PidPath(1))
   261  	if err != nil {
   262  		http.Error(w, err.Error(), http.StatusInternalServerError)
   263  		return
   264  	}
   265  
   266  	stats, err := control.Stat(cgroups.IgnoreNotExist)
   267  	if err != nil {
   268  		http.Error(w, err.Error(), http.StatusInternalServerError)
   269  		return
   270  	}
   271  
   272  	response, err := json.Marshal(map[string]uint64{
   273  		"cpuTime":     stats.CPU.Usage.Total,
   274  		"memoryUsage": stats.Memory.Usage.Usage - stats.Memory.Kernel.Usage - stats.Memory.Cache,
   275  	})
   276  	if err != nil {
   277  		http.Error(w, "fail to marshal response", http.StatusInternalServerError)
   278  		return
   279  	}
   280  
   281  	w.Write(response)
   282  }

   284  func (s *server) httpEcho(w http.ResponseWriter, r *http.Request) {
   285  	secrets := r.Header["Secret"]
   286  	if len(secrets) == 0 {
   287  		http.Error(w, "Forbidden", http.StatusForbidden)
   288  		return
   289  	}
   290  
   291  	for _, secret := range secrets {
   292  		w.Header().Add("Secret", secret)
   293  	}
   294  	defer r.Body.Close()
   295  	_, err := io.Copy(w, r.Body)
   296  	if err != nil {
   297  		http.Error(w, "fail to copy body between request and response", http.StatusInternalServerError)
   298  		return
   299  	}
   300  }
