   126  func (s *Service) createSchedule(c *gin.Context) {
   127  	kubeCli, err := clientpool.ExtractTokenAndGetClient(c.Request.Header)
   128  	if err != nil {
   129  		_ = c.Error(utils.ErrInvalidRequest.WrapWithNoMessage(err))
   130  		return
   131  	}
   132  
   133  	exp := &core.ScheduleInfo{}
   134  	if err := c.ShouldBindJSON(exp); err != nil {
   135  		c.Status(http.StatusBadRequest)
   136  		_ = c.Error(utils.ErrInvalidRequest.WrapWithNoMessage(err))
   137  		return
   138  	}
   139  
   140  	sch := &v1alpha1.Schedule{
   141  		ObjectMeta: metav1.ObjectMeta{
   142  			Name:        exp.Name,
   143  			Namespace:   exp.Namespace,
   144  			Labels:      exp.Labels,
   145  			Annotations: exp.Annotations,
   146  		},
   147  		Spec: v1alpha1.ScheduleSpec{
   148  			Schedule:          exp.Schedule,
   149  			ConcurrencyPolicy: exp.ConcurrencyPolicy,
   150  			HistoryLimit:      exp.HistoryLimit,
   151  			Type:              v1alpha1.ScheduleTemplateType(exp.Target.Kind),
   152  		},
   153  	}
   154  	if exp.StartingDeadlineSeconds != nil {
   155  		sch.Spec.StartingDeadlineSeconds = exp.StartingDeadlineSeconds
   156  	}
   157  
   158  	parseFuncs := map[string]parseScheduleFunc{
   159  		v1alpha1.KindPodChaos:     parsePodChaos,
   160  		v1alpha1.KindNetworkChaos: parseNetworkChaos,
   161  		v1alpha1.KindIOChaos:      parseIOChaos,
   162  		v1alpha1.KindStressChaos:  parseStressChaos,
   163  		v1alpha1.KindTimeChaos:    parseTimeChaos,
   164  		v1alpha1.KindKernelChaos:  parseKernelChaos,
   165  		v1alpha1.KindDNSChaos:     parseDNSChaos,
   166  		v1alpha1.KindAWSChaos:     parseAWSChaos,
   167  		v1alpha1.KindGCPChaos:     parseGCPChaos,
   168  	}
   169  
   170  	f, ok := parseFuncs[exp.Target.Kind]
   171  	if !ok {
   172  		c.Status(http.StatusBadRequest)
   173  		_ = c.Error(utils.ErrInvalidRequest.New(exp.Target.Kind + " is not supported"))
   174  		return
   175  	}
   176  	embedChaos := f(exp)
   177  	sch.Spec.ScheduleItem = embedChaos
   178  
   179  	if err := kubeCli.Create(context.Background(), sch); err != nil {
   180  		c.Status(http.StatusInternalServerError)
   181  		_ = c.Error(utils.ErrInternalServer.WrapWithNoMessage(err))
   182  		return
   183  	}
   184  
   185  	c.JSON(http.StatusOK, exp)
   186  }

   494  // @Summary Get chaos schedules from Kubernetes cluster.
   495  // @Description Get chaos schedules from Kubernetes cluster.
   496  // @Tags schedules
   497  // @Produce json
   498  // @Param namespace query string false "namespace"
   499  // @Param name query string false "name"
   500  // @Success 200 {array} Schedule
   501  // @Router /Schedules [get]
   502  // @Failure 500 {object} utils.APIError
   503  func (s *Service) listSchedules(c *gin.Context) {
   504  	kubeCli, err := clientpool.ExtractTokenAndGetClient(c.Request.Header)
   505  	if err != nil {
   506  		_ = c.Error(utils.ErrInvalidRequest.WrapWithNoMessage(err))
   507  		return
   508  	}
   509  
   510  	name := c.Query("name")
   511  	ns := c.Query("namespace")
   512  
   513  	if !s.conf.ClusterScoped {
   514  		log.Info("Overwrite namespace within namespace scoped mode", "origin", ns, "new", s.conf.TargetNamespace)
   515  		ns = s.conf.TargetNamespace
   516  	}
   517  
   518  	ScheduleList := v1alpha1.ScheduleList{}
   519  	sches := make([]*Schedule, 0)
   520  	if err := kubeCli.List(context.Background(), &ScheduleList, &client.ListOptions{Namespace: ns}); err != nil {
   521  		c.Status(http.StatusInternalServerError)
   522  		utils.SetErrorForGinCtx(c, err)
   523  		return
   524  	}
   525  	for _, schedule := range ScheduleList.Items {
   526  		if name != "" && schedule.Name != name {
   527  			continue
   528  		}
   529  		sches = append(sches, &Schedule{
   530  			Base: Base{
   531  				Kind:      string(schedule.Spec.Type),
   532  				Namespace: schedule.Namespace,
   533  				Name:      schedule.Name,
   534  			},
   535  			UID:     string(schedule.UID),
   536  			Created: schedule.CreationTimestamp.Format(time.RFC3339),
   537  			Status:  string(utils.GetScheduleState(schedule)),
   538  		})
   539  	}
   540  
   541  	sort.Slice(sches, func(i, j int) bool {
   542  		return sches[i].Created > sches[j].Created
   543  	})
   544  
   545  	c.JSON(http.StatusOK, sches)
