   143  // @Summary Get pods from Kubernetes cluster.
   144  // @Description Get pods from Kubernetes cluster.
   145  // @Tags common
   146  // @Produce json
   147  // @Param request body core.SelectorInfo true "Request body"
   148  // @Success 200 {array} Pod
   149  // @Router /common/pods [post]
   150  // @Failure 500 {object} utils.APIError
   151  func (s *Service) listPods(c *gin.Context) {
   152  	kubeCli, err := clientpool.ExtractTokenAndGetClient(c.Request.Header)
   153  	if err != nil {
   154  		_ = c.Error(utils.ErrInvalidRequest.WrapWithNoMessage(err))
   155  		return
   156  	}
   157  
   158  	exp := &core.SelectorInfo{}
   159  	if err := c.ShouldBindJSON(exp); err != nil {
   160  		c.Status(http.StatusBadRequest)
   161  		_ = c.Error(utils.ErrInvalidRequest.WrapWithNoMessage(err))
   162  		return
   163  	}
   164  	ctx := context.TODO()
   165  	filteredPods, err := pod.SelectPods(ctx, kubeCli, nil, exp.ParseSelector(), s.conf.ClusterScoped, s.conf.TargetNamespace, s.conf.EnableFilterNamespace)
   166  	if err != nil {
   167  		c.Status(http.StatusInternalServerError)
   168  		_ = c.Error(utils.ErrInternalServer.WrapWithNoMessage(err))
   169  		return
   170  	}
   171  
   172  	pods := make([]Pod, 0, len(filteredPods))
   173  	for _, pod := range filteredPods {
   174  		pods = append(pods, Pod{
   175  			Name:      pod.Name,
   176  			IP:        pod.Status.PodIP,
   177  			Namespace: pod.Namespace,
   178  			State:     string(pod.Status.Phase),
   179  		})
   180  	}
   181  
   182  	c.JSON(http.StatusOK, pods)
   183  }

   194  
   195  	var namespaces sort.StringSlice
   196  
   197  	var nsList v1.NamespaceList
   198  	if err := s.kubeCli.List(context.Background(), &nsList); err != nil {

   201  		return
   202  	}
   203  	namespaces = make(sort.StringSlice, 0, len(nsList.Items))
   204  	for _, ns := range nsList.Items {
   205  		namespaces = append(namespaces, ns.Name)
   206  	}
   207  
   208  	sort.Sort(namespaces)
   209  	c.JSON(http.StatusOK, namespaces)

   265  // @Description Get the labels of the pods in the specified namespace from Kubernetes cluster.
   266  // @Tags common
   267  // @Produce json
   268  // @Param podNamespaceList query string true "The pod's namespace list, split by ,"
   269  // @Success 200 {object} MapSlice
   270  // @Router /common/labels [get]
   271  // @Failure 500 {object} utils.APIError
   272  func (s *Service) getLabels(c *gin.Context) {
   273  
   274  	kubeCli, err := clientpool.ExtractTokenAndGetClient(c.Request.Header)
   275  	if err != nil {
   276  		_ = c.Error(utils.ErrInvalidRequest.WrapWithNoMessage(err))
   277  		return
   278  	}
   279  
   280  	podNamespaceList := c.Query("podNamespaceList")
   281  
   282  	if podNamespaceList == "" {
   283  		c.Status(http.StatusInternalServerError)
   284  		_ = c.Error(utils.ErrInternalServer.WrapWithNoMessage(fmt.Errorf("podNamespaceList cannot be empty")))
   285  		return
   286  	}
   287  
   288  	exp := &core.SelectorInfo{}
   289  	nsList := strings.Split(podNamespaceList, ",")
   290  	exp.Namespaces = nsList
   291  
   292  	ctx := context.TODO()
   293  	filteredPods, err := pod.SelectPods(ctx, kubeCli, nil, exp.ParseSelector(), s.conf.ClusterScoped, s.conf.TargetNamespace, s.conf.EnableFilterNamespace)
   294  	if err != nil {
   295  		c.Status(http.StatusInternalServerError)
   296  		_ = c.Error(utils.ErrInternalServer.WrapWithNoMessage(err))
   297  		return
   298  	}
   299  
   300  	labels := make(map[string][]string)
   301  
   302  	for _, pod := range filteredPods {
   303  		for k, v := range pod.Labels {
   304  			if _, ok := labels[k]; ok {
   305  				if !inSlice(v, labels[k]) {
   306  					labels[k] = append(labels[k], v)
   307  				}
   308  			} else {
   309  				labels[k] = []string{v}
   310  			}
   311  		}
   312  	}
   313  	c.JSON(http.StatusOK, labels)
   314  }

   379  // @Summary Get the rbac config according to the user's choice.
   380  // @Description Get the rbac config according to the user's choice.
   381  // @Tags common
   382  // @Produce json
   383  // @Success 200 {object} MapSlice
   384  // @Router /common/rbac-config [get]
   385  // @Failure 500 {object} utils.APIError
   386  func (s *Service) getRbacConfig(c *gin.Context) {
   387  	namespace := c.Query("namespace")
   388  	roleType := c.Query("role")
   389  
   390  	var serviceAccount, role, roleBinding, verbs string
   391  	randomStr := randomStringWithCharset(5, charset)
   392  
   393  	scope := namespace
   394  	if len(namespace) == 0 {
   395  		namespace = "default"
   396  		scope = "cluster"
   397  	}
   398  	if roleType == roleManager {
   399  		verbs = `"get", "list", "watch", "create", "delete", "patch", "update"`
   400  	} else if roleType == roleViewer {
   401  		verbs = `"get", "list", "watch"`
   402  	} else {
   403  		c.Status(http.StatusBadRequest)
   404  		_ = c.Error(utils.ErrInvalidRequest.WrapWithNoMessage(fmt.Errorf("roleType is neither manager nor viewer")))
   405  		return
   406  	}
   407  
   408  	serviceAccountName := fmt.Sprintf("account-%s-%s-%s", scope, roleType, randomStr)
   409  	roleName := fmt.Sprintf("role-%s-%s-%s", scope, roleType, randomStr)
   410  	roleBindingName := fmt.Sprintf("bind-%s-%s-%s", scope, roleType, randomStr)
   411  
   412  	serviceAccount = fmt.Sprintf(serviceAccountTemplate, namespace, serviceAccountName)
   413  	if scope == "cluster" {
   414  		role = fmt.Sprintf(clusterRoleTemplate, roleName, verbs)
   415  		roleBinding = fmt.Sprintf(clusterRoleBindingTemplate, roleBindingName, serviceAccountName, namespace, roleName)
   416  	} else {
   417  		role = fmt.Sprintf(roleTemplate, namespace, roleName, verbs)
   418  		roleBinding = fmt.Sprintf(roleBindingTemplate, roleBindingName, namespace, serviceAccountName, namespace, roleName)
   419  	}
   420  
   421  	rbacMap := make(map[string]string)
   422  	rbacMap[serviceAccountName] = serviceAccount + "\n---\n" + role + "\n---\n" + roleBinding
   423  
   424  	c.JSON(http.StatusOK, rbacMap)
   425  }

   427  // inSlice checks given string in string slice or not.
   428  func inSlice(v string, sl []string) bool {
   429  	for _, vv := range sl {
   430  		if vv == v {
   431  			return true
   432  		}
   433  	}
   434  	return false
   435  }

   439  func randomStringWithCharset(length int, charset string) string {
   440  	b := make([]byte, length)
   441  	for i := range b {
   442  		num, _ := rand.Int(rand.Reader, big.NewInt(int64(len(charset))))
   443  		b[i] = charset[num.Int64()]
   444  	}
   445  	return string(b)
   446  }
