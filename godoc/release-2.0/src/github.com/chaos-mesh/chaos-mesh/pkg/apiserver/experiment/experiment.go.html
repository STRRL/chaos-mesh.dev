   124  // @Summary Create a new chaos experiment.
   125  // @Description Create a new chaos experiment.
   126  // @Tags experiments
   127  // @Produce json
   128  // @Param request body core.ExperimentInfo true "Request body"
   129  // @Success 200 {object} core.ExperimentInfo
   130  // @Failure 400 {object} utils.APIError
   131  // @Failure 500 {object} utils.APIError
   132  // @Router /experiments/new [post]
   133  func (s *Service) createExperiment(c *gin.Context) {
   134  	kubeCli, err := clientpool.ExtractTokenAndGetClient(c.Request.Header)
   135  	if err != nil {
   136  		_ = c.Error(utils.ErrInvalidRequest.WrapWithNoMessage(err))
   137  		return
   138  	}
   139  
   140  	exp := &core.ExperimentInfo{}
   141  	if err := c.ShouldBindJSON(exp); err != nil {
   142  		c.Status(http.StatusBadRequest)
   143  		_ = c.Error(utils.ErrInvalidRequest.WrapWithNoMessage(err))
   144  		return
   145  	}
   146  
   147  	createFuncs := map[string]createExperimentFunc{
   148  		v1alpha1.KindPodChaos:     s.createPodChaos,
   149  		v1alpha1.KindNetworkChaos: s.createNetworkChaos,
   150  		v1alpha1.KindIOChaos:      s.createIOChaos,
   151  		v1alpha1.KindStressChaos:  s.createStressChaos,
   152  		v1alpha1.KindTimeChaos:    s.createTimeChaos,
   153  		v1alpha1.KindKernelChaos:  s.createKernelChaos,
   154  		v1alpha1.KindDNSChaos:     s.createDNSChaos,
   155  		v1alpha1.KindAWSChaos:     s.createAWSChaos,
   156  		v1alpha1.KindGCPChaos:     s.createGCPChaos,
   157  	}
   158  
   159  	f, ok := createFuncs[exp.Target.Kind]
   160  	if !ok {
   161  		c.Status(http.StatusBadRequest)
   162  		_ = c.Error(utils.ErrInvalidRequest.New(exp.Target.Kind + " is not supported"))
   163  		return
   164  	}
   165  
   166  	if err := f(exp, kubeCli); err != nil {
   167  		c.Status(http.StatusInternalServerError)
   168  		_ = c.Error(utils.ErrInternalServer.WrapWithNoMessage(err))
   169  		return
   170  	}
   171  
   172  	c.JSON(http.StatusOK, exp)
   173  }

   341  func (s *Service) createStressChaos(exp *core.ExperimentInfo, kubeCli client.Client) error {
   342  	var stressors *v1alpha1.Stressors
   343  
   344  	// Error checking
   345  	originStressors := exp.Target.StressChaos.Stressors
   346  	if originStressors == nil {
   347  		return utils.ErrInvalidRequest.New("stressor is required")
   348  	}
   349  	if originStressors.CPUStressor == nil && originStressors.MemoryStressor == nil {
   350  		return utils.ErrInvalidRequest.New("one of CPUStressor or MemoryStressor is required")
   351  	}
   352  	if originStressors.MemoryStressor != nil && originStressors.MemoryStressor.Workers > 0 &&
   353  		(originStressors.CPUStressor == nil || originStressors.CPUStressor.Workers <= 0) {
   354  		stressors = &v1alpha1.Stressors{
   355  			MemoryStressor: originStressors.MemoryStressor,
   356  		}
   357  	} else if originStressors.CPUStressor != nil && originStressors.CPUStressor.Workers > 0 &&
   358  		(originStressors.MemoryStressor == nil || originStressors.MemoryStressor.Workers <= 0) {
   359  		stressors = &v1alpha1.Stressors{
   360  			CPUStressor: originStressors.CPUStressor,
   361  		}
   362  	} else {
   363  		stressors = originStressors
   364  	}
   365  
   366  	chaos := &v1alpha1.StressChaos{
   367  		ObjectMeta: metav1.ObjectMeta{
   368  			Name:        exp.Name,
   369  			Namespace:   exp.Namespace,
   370  			Labels:      exp.Labels,

   372  		},

   480  func (s *Service) getPodChaosDetail(namespace string, name string, kubeCli client.Client) (Detail, error) {
   481  	chaos := &v1alpha1.PodChaos{}
   482  
   483  	chaosKey := types.NamespacedName{Namespace: namespace, Name: name}
   484  	if err := kubeCli.Get(context.Background(), chaosKey, chaos); err != nil {
   485  		if apierrors.IsNotFound(err) {
   486  			return Detail{}, utils.ErrNotFound.NewWithNoMessage()
   487  		}
   488  
   489  		return Detail{}, err
   490  	}
   491  
   492  	gvk, err := apiutil.GVKForObject(chaos, s.scheme)
   493  	if err != nil {
   494  		return Detail{}, err
   495  	}
   496  
   497  	return Detail{
   498  		Experiment: Experiment{
   499  			Base: Base{
   500  				Kind:      gvk.Kind,
   501  				Namespace: chaos.Namespace,
   502  				Name:      chaos.Name,
   503  			},
   504  			UID:     chaos.GetChaos().UID,
   505  			Created: chaos.GetChaos().StartTime.Format(time.RFC3339),
   506  			Status:  string(utils.GetChaosState(chaos)),
   507  		},
   508  		KubeObject: core.KubeObjectDesc{
   509  			TypeMeta: metav1.TypeMeta{
   510  				APIVersion: gvk.GroupVersion().String(),
   511  				Kind:       gvk.Kind,
   512  			},
   513  			Meta: core.KubeObjectMeta{
   514  				Name:        chaos.Name,
   515  				Namespace:   chaos.Namespace,
   516  				Labels:      chaos.Labels,
   517  				Annotations: chaos.Annotations,
   518  			},
   519  			Spec: chaos.Spec,
   520  		},
   521  	}, nil

   883  // @Param status query string false "status" Enums(Running, Paused, Failed, Finished)
   884  // @Success 200 {array} Experiment
   885  // @Router /experiments [get]
   886  // @Failure 500 {object} utils.APIError
   887  func (s *Service) listExperiments(c *gin.Context) {
   888  	kubeCli, err := clientpool.ExtractTokenAndGetClient(c.Request.Header)
   889  	if err != nil {
   890  		_ = c.Error(utils.ErrInvalidRequest.WrapWithNoMessage(err))
   891  		return
   892  	}
   893  
   894  	kind := c.Query("kind")
   895  	name := c.Query("name")
   896  	ns := c.Query("namespace")
   897  
   898  	if len(ns) == 0 && !s.conf.ClusterScoped &&
   899  		len(s.conf.TargetNamespace) != 0 {
   900  		ns = s.conf.TargetNamespace
   901  	}
   902  
   903  	exps := make([]*Experiment, 0)
   904  	for key, list := range v1alpha1.AllKinds() {
   905  		if kind != "" && key != kind {
   906  			continue
   907  		}
   908  		if err := kubeCli.List(context.Background(), list.ChaosList, &client.ListOptions{Namespace: ns}); err != nil {
   909  			c.Status(http.StatusInternalServerError)
   910  			utils.SetErrorForGinCtx(c, err)
   911  			return
   912  		}
   913  
   914  		items := reflect.ValueOf(list.ChaosList).Elem().FieldByName("Items")
   915  		for i := 0; i < items.Len(); i++ {
   916  			item := items.Index(i).Addr().Interface().(v1alpha1.InnerObject)
   917  			chaos := item.GetChaos()
   918  			if name != "" && chaos.Name != name {
   919  				continue
   920  			}
   921  			status := utils.GetChaosState(item)
   922  			exps = append(exps, &Experiment{
   923  				Base: Base{
   924  					Name:      chaos.Name,
   925  					Namespace: chaos.Namespace,
   926  					Kind:      chaos.Kind,
   927  				},
   928  				Created: chaos.StartTime.Format(time.RFC3339),
   929  				Status:  string(status),
   930  				UID:     chaos.UID,
   931  			})
   932  		}
   933  	}
   934  
   935  	sort.Slice(exps, func(i, j int) bool {
   936  		return exps[i].Created > exps[j].Created
   937  	})
   938  
   939  	c.JSON(http.StatusOK, exps)
   940  }
