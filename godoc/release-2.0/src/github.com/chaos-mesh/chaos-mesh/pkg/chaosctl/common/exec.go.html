    34  // Exec executes certain command and returns the result
    35  // Only commands in chaos-mesh components should use this way
    36  // for target pod, use ExecBypass
    37  func Exec(ctx context.Context, pod v1.Pod, cmd string, c *kubernetes.Clientset) (string, error) {
    38  	name := pod.GetObjectMeta().GetName()
    39  	namespace := pod.GetObjectMeta().GetNamespace()
    40  	// TODO: if `containerNames` is set and specific container is injected chaos,
    41  	// need to use THE name rather than the first one.
    42  	// till 20/11/10 only podchaos and kernelchaos support `containerNames`, so not set it for now
    43  	containerName := pod.Spec.Containers[0].Name
    44  
    45  	req := c.CoreV1().RESTClient().Post().
    46  		Resource("pods").
    47  		Name(name).
    48  		Namespace(namespace).
    49  		SubResource("exec")
    50  
    51  	req.VersionedParams(&v1.PodExecOptions{
    52  		Container: containerName,
    53  		Command:   []string{"/bin/sh", "-c", cmd},
    54  		Stdin:     false,
    55  		Stdout:    true,
    56  		Stderr:    true,
    57  		TTY:       false,
    58  	}, kubectlscheme.ParameterCodec)
    59  
    60  	var stdout, stderr bytes.Buffer
    61  	exec, err := remotecommand.NewSPDYExecutor(config.GetConfigOrDie(), "POST", req.URL())
    62  	if err != nil {
    63  		return "", errors.Wrapf(err, "error in creating NewSPDYExecutor for pod %s/%s", pod.GetNamespace(), pod.GetName())
    64  	}
    65  	err = exec.Stream(remotecommand.StreamOptions{
    66  		Stdin:  nil,
    67  		Stdout: &stdout,
    68  		Stderr: &stderr,
    69  	})
    70  	if err != nil {
    71  		if stderr.String() != "" {
    72  			return "", fmt.Errorf("error: %s\npod: %s\ncommand: %s", strings.TrimSuffix(stderr.String(), "\n"), pod.Name, cmd)
    73  		}
    74  		return "", errors.Wrapf(err, "error in streaming remotecommand: pod: %s/%s, command: %s", pod.GetNamespace(), pod.Name, cmd)
    75  	}
    76  	if stderr.String() != "" {
    77  		return "", fmt.Errorf("error of command %s: %s", cmd, stderr.String())
    78  	}
    79  	return stdout.String(), nil
    80  }

    82  // ExecBypass use chaos-daemon to enter namespace and execute command in target pod
    83  func ExecBypass(ctx context.Context, pod v1.Pod, daemon v1.Pod, cmd string, c *kubernetes.Clientset) (string, error) {
    84  	// To disable printing irrelevant log from grpc/clientconn.go
    85  	// See grpc/grpc-go#3918 for detail. Could be resolved in the future
    86  	grpclog.SetLoggerV2(grpclog.NewLoggerV2(ioutil.Discard, ioutil.Discard, ioutil.Discard))
    87  	pid, err := GetPidFromPod(ctx, pod, daemon)
    88  	if err != nil {
    89  		return "", err
    90  	}
    91  	// enter all possible namespaces needed, since there's no bad effect to do so
    92  	cmdBuilder := bpm.DefaultProcessBuilder(cmd).SetNS(pid, bpm.MountNS).SetNS(pid, bpm.PidNS).SetContext(ctx)
    93  	return Exec(ctx, daemon, cmdBuilder.Build().Cmd.String(), c)
    94  }
