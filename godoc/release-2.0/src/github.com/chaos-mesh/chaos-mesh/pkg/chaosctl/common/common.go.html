   111  // PrettyPrint print with tab number and color
   112  func PrettyPrint(s string, indentLevel int, color Color) {
   113  	var tabStr string
   114  	for i := 0; i < indentLevel; i++ {
   115  		tabStr += "\t"
   116  	}
   117  	str := fmt.Sprintf("%s%s\n\n", tabStr, regexp.MustCompile("\n").ReplaceAllString(s, "\n"+tabStr))
   118  	if color != NoColor {
   119  		if cfunc, ok := colorFunc[color]; !ok {
   120  			fmt.Print("COLOR NOT SUPPORTED")
   121  		} else {
   122  			cfunc(str)
   123  		}
   124  	} else {
   125  		fmt.Print(str)
   126  	}
   127  }

   129  // PrintResult prints result to users in prettier format
   130  func PrintResult(result []ChaosResult) {
   131  	for _, chaos := range result {
   132  		PrettyPrint("[Chaos]: "+chaos.Name, 0, Blue)
   133  		for _, pod := range chaos.Pods {
   134  			PrettyPrint("[Pod]: "+pod.Name, 0, Blue)
   135  			for i, item := range pod.Items {
   136  				PrettyPrint(fmt.Sprintf("%d. [%s]", i+1, item.Name), 1, Cyan)
   137  				PrettyPrint(item.Value, 1, NoColor)
   138  				if item.Status == ItemSuccess {
   139  					if item.SucInfo != "" {
   140  						PrettyPrint(item.SucInfo, 1, Green)
   141  					} else {
   142  						PrettyPrint("Execute as expected", 1, Green)
   143  					}
   144  				} else if item.Status == ItemFailure {
   145  					PrettyPrint(fmt.Sprintf("Failed: %s ", item.ErrInfo), 1, Red)
   146  				}
   147  			}
   148  		}
   149  	}
   150  }

   262  // GetPidFromPS returns pid-command pairs
   263  func GetPidFromPS(ctx context.Context, pod v1.Pod, daemon v1.Pod, c *kubernetes.Clientset) ([]string, []string, error) {
   264  	cmd := "ps"
   265  	out, err := ExecBypass(ctx, pod, daemon, cmd, c)
   266  	if err != nil {
   267  		return nil, nil, errors.Wrapf(err, "run command %s failed", cmd)
   268  	}
   269  	outLines := strings.Split(string(out), "\n")
   270  	if len(outLines) < 2 {
   271  		return nil, nil, fmt.Errorf("ps returns empty")
   272  	}
   273  	titles := strings.Fields(outLines[0])
   274  	var pidColumn, cmdColumn int
   275  	for i, t := range titles {
   276  		if t == "PID" {
   277  			pidColumn = i
   278  		}
   279  		if t == "COMMAND" || t == "CMD" {
   280  			cmdColumn = i
   281  		}
   282  	}
   283  	if pidColumn == 0 && cmdColumn == 0 {
   284  		return nil, nil, fmt.Errorf("parsing ps error: could not get PID and COMMAND column")
   285  	}
   286  	var pids, commands []string
   287  	for _, line := range outLines[1:] {
   288  		item := strings.Fields(line)
   289  		// break when got empty line
   290  		if len(item) == 0 {
   291  			break
   292  		}
   293  		pids = append(pids, item[pidColumn])
   294  		commands = append(commands, item[cmdColumn])
   295  	}
   296  	return pids, commands, nil
   297  }

   299  // GetPidFromPod returns pid given containerd ID in pod
   300  func GetPidFromPod(ctx context.Context, pod v1.Pod, daemon v1.Pod) (uint32, error) {
   301  	pfCancel, localPort, err := forwardPorts(ctx, daemon, uint16(ctrlconfig.ControllerCfg.ChaosDaemonPort))
   302  	if err != nil {
   303  		return 0, errors.Wrapf(err, "forward ports for daemon pod %s/%s failed", daemon.Namespace, daemon.Name)
   304  	}
   305  	L().WithName("GetPidFromPod").V(4).Info(fmt.Sprintf("port forwarding 127.0.0.1:%d -> pod/%s/%s:%d", localPort, daemon.Namespace, daemon.Name, ctrlconfig.ControllerCfg.ChaosDaemonPort))
   306  
   307  	defer func() {
   308  		pfCancel()
   309  	}()
   310  
   311  	daemonClient, err := ConnectToLocalChaosDaemon(int(localPort))
   312  	if err != nil {
   313  		return 0, errors.Wrapf(err, "failed to create new chaos daemon client with local port %d", localPort)
   314  	}
   315  	defer daemonClient.Close()
   316  
   317  	if len(pod.Status.ContainerStatuses) == 0 {
   318  		return 0, fmt.Errorf("%s %s can't get the state of container", pod.Namespace, pod.Name)
   319  	}
   320  
   321  	res, err := daemonClient.ContainerGetPid(ctx, &pb.ContainerRequest{
   322  		Action: &pb.ContainerAction{
   323  			Action: pb.ContainerAction_GETPID,
   324  		},
   325  		ContainerId: pod.Status.ContainerStatuses[0].ContainerID,
   326  	})
   327  	if err != nil {
   328  		return 0, errors.Wrapf(err, "failed get pid from pod %s/%s", pod.GetNamespace(), pod.GetName())
   329  	}
   330  	return res.Pid, nil
   331  }

   344  func Log(pod v1.Pod, tail int64, c *kubernetes.Clientset) (string, error) {
   345  	podLogOpts := v1.PodLogOptions{}
   346  	//use negative tail to indicate no tail limit is needed
   347  	if tail >= 0 {
   348  		podLogOpts.TailLines = func(i int64) *int64 { return &i }(tail)
   349  	}
   350  
   351  	req := c.CoreV1().Pods(pod.Namespace).GetLogs(pod.Name, &podLogOpts)
   352  	podLogs, err := req.Stream()
   353  	if err != nil {
   354  		return "", errors.Wrapf(err, "failed to open log stream for pod %s/%s", pod.GetNamespace(), pod.GetName())
   355  	}
   356  	defer podLogs.Close()
   357  
   358  	buf := new(bytes.Buffer)
   359  	_, err = io.Copy(buf, podLogs)
   360  	if err != nil {
   361  		return "", errors.Wrapf(err, "failed to copy information from podLogs to buf")
   362  	}
   363  	return buf.String(), nil
   364  }

   382  func getGrpcClient(port int) (*grpc.ClientConn, error) {
   383  	builder := grpcUtils.Builder("localhost", port)
   384  	if Insecure {
   385  		builder.Insecure()
   386  	} else {
   387  		if TLSFiles.CaCert == "" || TLSFiles.Cert == "" || TLSFiles.Key == "" {
   388  			PrettyPrint("TLS Files are not complete, fall back to use secrets.", 0, Green)
   389  			config, err := getTLSConfigFromSecrets()
   390  			if err != nil {
   391  				return nil, err
   392  			}
   393  			builder.TLSFromRaw(config.CaCert, config.Cert, config.Key)
   394  		} else {
   395  			PrettyPrint("Using TLS Files.", 0, Green)
   396  			builder.TLSFromFile(TLSFiles.CaCert, TLSFiles.Cert, TLSFiles.Key)
   397  		}
   398  	}
   399  	return builder.Build()
   400  }
