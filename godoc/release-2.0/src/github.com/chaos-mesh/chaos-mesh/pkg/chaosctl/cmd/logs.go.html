    39  func NewLogsCmd(logger logr.Logger) (*cobra.Command, error) {
    40  	o := &logsOptions{
    41  		logger: logger,
    42  	}
    43  
    44  	logsCmd := &cobra.Command{
    45  		Use:   `logs [-t LINE]`,
    46  		Short: `Print logs of controller-manager, chaos-daemon and chaos-dashboard`,
    47  		Long: `Print logs of controller-manager, chaos-daemon and chaos-dashboard, to provide debug information.
    48  
    49  Examples:
    50    # Default print all log of all chaosmesh components
    51    chaosctl logs
    52  
    53    # Print 100 log lines for chaosmesh components in node NODENAME
    54    chaosctl logs -t 100 -n NODENAME`,
    55  		RunE: func(cmd *cobra.Command, args []string) error {
    56  			return o.Run(args)
    57  		},
    58  		SilenceErrors:     true,
    59  		SilenceUsage:      true,
    60  		ValidArgsFunction: noCompletions,
    61  	}
    62  
    63  	logsCmd.Flags().Int64VarP(&o.tail, "tail", "t", -1, "Number of lines of recent log")
    64  	logsCmd.Flags().StringVarP(&o.node, "node", "n", "", "Number of lines of recent log")
    65  	err := logsCmd.RegisterFlagCompletionFunc("node", func(cmd *cobra.Command, args []string, toComplete string) ([]string, cobra.ShellCompDirective) {
    66  		clientset, err := cm.InitClientSet()
    67  		if err != nil {
    68  			return nil, cobra.ShellCompDirectiveDefault
    69  		}
    70  		return listNodes(toComplete, clientset.KubeCli)
    71  	})
    72  	if err != nil {
    73  		return nil, err
    74  	}
    75  	return logsCmd, nil
    76  }

    78  // Run logs
    79  func (o *logsOptions) Run(args []string) error {
    80  	ctx, cancel := context.WithCancel(context.Background())
    81  	defer cancel()
    82  	c, err := cm.InitClientSet()
    83  	if err != nil {
    84  		o.logger.V(4).Info("failed to initialize clientset", "err", err)
    85  		return err
    86  	}
    87  
    88  	componentsNeeded := []string{"controller-manager", "chaos-daemon", "chaos-dashboard"}
    89  	for _, name := range componentsNeeded {
    90  		selectorSpec := v1alpha1.PodSelectorSpec{
    91  			LabelSelectors: map[string]string{"app.kubernetes.io/component": name},
    92  		}
    93  		if o.node != "" {
    94  			selectorSpec.Nodes = []string{o.node}
    95  		}
    96  
    97  		// TODO: just use kubernetes native label selector
    98  		components, err := pod.SelectPods(ctx, c.CtrlCli, nil, selectorSpec, config.ControllerCfg.ClusterScoped, config.ControllerCfg.TargetNamespace, false)
    99  		if err != nil {
   100  			return errors.Wrapf(err, "failed to SelectPods for component %s", name)
   101  		}
   102  		o.logger.V(4).Info("select pods for component", "component", name, "pods", components)
   103  		for _, comp := range components {
   104  			cm.PrettyPrint(fmt.Sprintf("[%s]", comp.Name), 0, cm.Cyan)
   105  			comLog, err := cm.Log(comp, o.tail, c.KubeCli)
   106  			if err != nil {
   107  				cm.PrettyPrint(err.Error(), 1, cm.Red)
   108  			} else {
   109  				cm.PrettyPrint(comLog, 1, cm.NoColor)
   110  			}
   111  		}
   112  	}
   113  	return nil
   114  }

   116  func listNodes(toComplete string, c *kubernetes.Clientset) ([]string, cobra.ShellCompDirective) {
   117  	nodes, err := c.CoreV1().Nodes().List(v1.ListOptions{})
   118  	if err != nil {
   119  		return nil, cobra.ShellCompDirectiveDefault
   120  	}
   121  	var ret []string
   122  	for _, ns := range nodes.Items {
   123  		if strings.HasPrefix(ns.Name, toComplete) {
   124  			ret = append(ret, ns.Name)
   125  		}
   126  	}
   127  	return ret, cobra.ShellCompDirectiveNoFileComp
   128  }
