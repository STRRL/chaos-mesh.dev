    36  func (s *DaemonServer) ApplyIOChaos(ctx context.Context, in *pb.ApplyIOChaosRequest) (*pb.ApplyIOChaosResponse, error) {
    37  	log.Info("applying io chaos", "Request", in)
    38  
    39  	if in.Instance != 0 {
    40  		err := s.killIOChaos(ctx, in.Instance, in.StartTime)
    41  		if err != nil {
    42  			return nil, err
    43  		}
    44  	}
    45  
    46  	actions := []v1alpha1.IOChaosAction{}
    47  	err := json.Unmarshal([]byte(in.Actions), &actions)
    48  	if err != nil {
    49  		log.Error(err, "error while unmarshal json bytes")
    50  		return nil, err
    51  	}
    52  
    53  	log.Info("the length of actions", "length", len(actions))
    54  	if len(actions) == 0 {
    55  		return &pb.ApplyIOChaosResponse{
    56  			Instance:  0,
    57  			StartTime: 0,
    58  		}, nil
    59  	}
    60  
    61  	pid, err := s.crClient.GetPidFromContainerID(ctx, in.ContainerId)
    62  	if err != nil {
    63  		log.Error(err, "error while getting PID")
    64  		return nil, err
    65  	}
    66  
    67  	// TODO: make this log level configurable
    68  	args := fmt.Sprintf("--path %s --verbose info", in.Volume)
    69  	log.Info("executing", "cmd", todaBin+" "+args)
    70  
    71  	processBuilder := bpm.DefaultProcessBuilder(todaBin, strings.Split(args, " ")...).
    72  		EnableLocalMnt().
    73  		SetIdentifier(fmt.Sprintf("toda-%s", in.ContainerId))
    74  
    75  	if in.EnterNS {
    76  		processBuilder = processBuilder.SetNS(pid, bpm.MountNS).SetNS(pid, bpm.PidNS)
    77  	}
    78  
    79  	ctx, cancel := context.WithCancel(context.Background())
    80  	defer cancel()
    81  	caller, receiver := bpm.NewBlockingBuffer(), bpm.NewBlockingBuffer()
    82  	defer caller.Close()
    83  	defer receiver.Close()
    84  	client, err := jrpc.DialIO(ctx, receiver, caller)
    85  	if err != nil {
    86  		return nil, err
    87  	}
    88  
    89  	cmd := processBuilder.Build()
    90  	cmd.Stdin = caller
    91  	cmd.Stdout = io.MultiWriter(receiver, os.Stdout)
    92  	cmd.Stderr = os.Stderr
    93  	procState, err := s.backgroundProcessManager.StartProcess(cmd)
    94  	if err != nil {
    95  		return nil, err
    96  	}
    97  	var ret string
    98  	ct, err := procState.CreateTime()
    99  	if err != nil {
   100  		log.Error(err, "get create time failed")
   101  		if kerr := cmd.Process.Kill(); kerr != nil {
   102  			log.Error(kerr, "kill toda failed", "request", in)
   103  		}
   104  		return nil, err
   105  	}
   106  
   107  	log.Info("Waiting for toda to start")
   108  	var rpcError error
   109  	maxWaitTime := time.Millisecond * 2000
   110  	timeOut, cancel := context.WithTimeout(ctx, maxWaitTime)
   111  	defer cancel()
   112  	_ = client.CallContext(timeOut, &ret, "update", actions)
   113  	rpcError = client.CallContext(timeOut, &ret, "get_status", "ping")
   114  	if rpcError != nil || ret != "ok" {
   115  		log.Info("Starting toda takes too long or encounter an error")
   116  		caller.Close()
   117  		receiver.Close()
   118  		if kerr := s.killIOChaos(ctx, int64(cmd.Process.Pid), ct); kerr != nil {
   119  			log.Error(kerr, "kill toda failed", "request", in)
   120  		}
   121  		return nil, fmt.Errorf("toda startup takes too long or an error occurs: %s", ret)
   122  	}
   123  
   124  	return &pb.ApplyIOChaosResponse{
   125  		Instance:  int64(cmd.Process.Pid),
   126  		StartTime: ct,
   127  	}, nil
   128  }

   130  func (s *DaemonServer) killIOChaos(ctx context.Context, pid int64, startTime int64) error {
   131  	log.Info("killing toda", "pid", pid)
   132  
   133  	err := s.backgroundProcessManager.KillBackgroundProcess(ctx, int(pid), startTime)
   134  	if err != nil {
   135  		return err
   136  	}
   137  	log.Info("kill toda successfully")
   138  	return nil
   139  }
