    26  // ReadCommName returns the command name of process
    27  func ReadCommName(pid int) (string, error) {
    28  	f, err := os.Open(fmt.Sprintf("%s/%d/comm", bpm.DefaultProcPrefix, pid))
    29  	if err != nil {
    30  		return "", err
    31  	}
    32  	defer f.Close()
    33  
    34  	b, err := ioutil.ReadAll(f)
    35  	if err != nil {
    36  		return "", err
    37  	}
    38  
    39  	return string(b), nil
    40  }

    42  // GetChildProcesses will return all child processes's pid. Include all generations.
    43  // only return error when /proc/pid/tasks cannot be read
    44  func GetChildProcesses(ppid uint32) ([]uint32, error) {
    45  	procs, err := ioutil.ReadDir(bpm.DefaultProcPrefix)
    46  	if err != nil {
    47  		return nil, err
    48  	}
    49  
    50  	type processPair struct {
    51  		Pid  uint32
    52  		Ppid uint32
    53  	}
    54  
    55  	pairs := make(chan processPair)
    56  	done := make(chan bool)
    57  
    58  	go func() {
    59  		var wg sync.WaitGroup
    60  
    61  		for _, proc := range procs {
    62  			_, err := strconv.ParseUint(proc.Name(), 10, 32)
    63  			if err != nil {
    64  				continue
    65  			}
    66  
    67  			statusPath := bpm.DefaultProcPrefix + "/" + proc.Name() + "/stat"
    68  
    69  			wg.Add(1)
    70  			go func() {
    71  				defer wg.Done()
    72  
    73  				reader, err := os.Open(statusPath)
    74  				if err != nil {
    75  					log.Error(err, "read status file error", "path", statusPath)
    76  					return
    77  				}
    78  				defer reader.Close()
    79  
    80  				var (
    81  					pid    uint32
    82  					comm   string
    83  					state  string
    84  					parent uint32
    85  				)
    86  				// according to procfs's man page
    87  				fmt.Fscanf(reader, "%d %s %s %d", &pid, &comm, &state, &parent)
    88  
    89  				pairs <- processPair{
    90  					Pid:  pid,
    91  					Ppid: parent,
    92  				}
    93  			}()
    94  		}
    95  
    96  		wg.Wait()
    97  		done <- true
    98  	}()
    99  
   100  	processGraph := NewGraph()
   101  	for {
   102  		select {
   103  		case pair := <-pairs:
   104  			processGraph.Insert(pair.Ppid, pair.Pid)
   105  		case <-done:
   106  			return processGraph.Flatten(ppid), nil
   107  		}
   108  	}
   109  }
