    16  // Edge represents an edge in graph

    28  // NewGraph creates a Graph

    35  // Insert inserts an Edge into a graph from source to target
    36  func (g *Graph) Insert(source uint32, target uint32) {
    37  	newEdge := Edge{
    38  		Source: source,
    39  		Target: target,
    40  		Next:   g.head[source],
    41  	}
    42  	g.head[source] = &newEdge
    43  }
    44  
    45  // IterFrom starts iterating from source node
    46  func (g *Graph) IterFrom(source uint32) *Edge {
    47  	return g.head[source]
    48  }

    50  // Flatten flattens the subtree from source (without checking whether it's a tree)
    51  func (g *Graph) Flatten(source uint32) []uint32 {
    52  	current := g.head[source]
    53  
    54  	var flatTree []uint32
    55  	for {
    56  		if current == nil {
    57  			break
    58  		}
    59  
    60  		children := g.Flatten(current.Target)
    61  		flatTree = append(flatTree, current.Target)
    62  		flatTree = append(flatTree, children...)
    63  
    64  		current = current.Next
    65  	}
    66  
    67  	log.Info("get flatTree", "source", source, "flatTree", flatTree)
    68  	return flatTree
    69  }
