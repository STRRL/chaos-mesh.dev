    85  func (e *experimentStore) FindManagedByNamespaceName(_ context.Context, namespace, name string) ([]*core.Experiment, error) {
    86  	experiments := make([]*core.Experiment, 0)
    87  	if err := e.db.Model(core.Experiment{}).
    88  		Find(&experiments, "namespace = ? AND name = ? AND archived = ?", namespace, name, true).Error; err != nil {
    89  		return nil, err
    90  	}
    91  
    92  	managedExperiments := make([]*core.Experiment, 0)
    93  	for _, expr := range experiments {
    94  		meta := &unstructured.Unstructured{}
    95  		if err := meta.UnmarshalJSON([]byte(expr.Experiment)); err != nil {
    96  			return nil, err
    97  		}
    98  
    99  		if meta.GetLabels()["managed-by"] != "" {
   100  			managedExperiments = append(managedExperiments, expr)
   101  		}
   102  	}
   103  
   104  	if len(managedExperiments) == 0 {
   105  		return nil, gorm.ErrRecordNotFound
   106  	}
   107  
   108  	return managedExperiments, nil
   109  }

   133  // DeleteByFinishTime deletes experiments whose time difference is greater than the given time from FinishTime.
   134  func (e *experimentStore) DeleteByFinishTime(_ context.Context, ttl time.Duration) error {
   135  	experiments, err := e.ListMeta(context.Background(), "", "", "", true)
   136  	if err != nil {
   137  		return err
   138  	}
   139  
   140  	nowTime := time.Now()
   141  	for _, exp := range experiments {
   142  		if exp.FinishTime.Add(ttl).Before(nowTime) {
   143  			if err := e.db.Table("experiments").Unscoped().Delete(*exp).Error; err != nil {
   144  				return err
   145  			}
   146  		}
   147  	}
   148  
   149  	return nil
   150  }
