    24  // FromDelay convert delay to netem
    25  func FromDelay(in *v1alpha1.DelaySpec) (*chaosdaemonpb.Netem, error) {
    26  	delayTime, err := time.ParseDuration(in.Latency)
    27  	if err != nil {
    28  		return nil, err
    29  	}
    30  	jitter, err := time.ParseDuration(in.Jitter)
    31  	if err != nil {
    32  		return nil, err
    33  	}
    34  
    35  	corr, err := strconv.ParseFloat(in.Correlation, 32)
    36  	if err != nil {
    37  		return nil, err
    38  	}
    39  
    40  	netem := &chaosdaemonpb.Netem{
    41  		Time:      uint32(delayTime.Nanoseconds() / 1e3),
    42  		DelayCorr: float32(corr),
    43  		Jitter:    uint32(jitter.Nanoseconds() / 1e3),
    44  	}
    45  
    46  	if in.Reorder != nil {
    47  		reorderPercentage, err := strconv.ParseFloat(in.Reorder.Reorder, 32)
    48  		if err != nil {
    49  			return nil, err
    50  		}
    51  
    52  		corr, err := strconv.ParseFloat(in.Reorder.Correlation, 32)
    53  		if err != nil {
    54  			return nil, err
    55  		}
    56  
    57  		netem.Reorder = float32(reorderPercentage)
    58  		netem.ReorderCorr = float32(corr)
    59  		netem.Gap = uint32(in.Reorder.Gap)
    60  	}
    61  
    62  	return netem, nil
    63  }

   119  // FromBandwidth converts BandwidthSpec to *chaosdaemonpb.Tbf
   120  // Bandwidth action use TBF under the hood.
   121  // TBF stands for Token Bucket Filter, is a classful queueing discipline available
   122  // for traffic control with the tc command.
   123  // http://man7.org/linux/man-pages/man8/tc-tbf.8.html
   124  func FromBandwidth(in *v1alpha1.BandwidthSpec) (*chaosdaemonpb.Tbf, error) {
   125  	rate, err := v1alpha1.ConvertUnitToBytes(in.Rate)
   126  
   127  	if err != nil {
   128  		return nil, err
   129  	}
   130  
   131  	tbf := &chaosdaemonpb.Tbf{
   132  		Rate:   rate,
   133  		Limit:  in.Limit,
   134  		Buffer: in.Buffer,
   135  	}
   136  
   137  	if in.Peakrate != nil && in.Minburst != nil {
   138  		tbf.PeakRate = *in.Peakrate
   139  		tbf.MinBurst = *in.Minburst
   140  	}
   141  
   142  	return tbf, nil
   143  }
