    27  func BootstrapWorkflowControllers(mgr manager.Manager, logger logr.Logger, recorderBuilder *recorder.RecorderBuilder) error {
    28  	noCacheClient, err := client.New(mgr.GetConfig(), client.Options{
    29  		Scheme: mgr.GetScheme(),
    30  		Mapper: mgr.GetRESTMapper(),
    31  	})
    32  	if err != nil {
    33  		return err
    34  	}
    35  	err = ctrl.NewControllerManagedBy(mgr).
    36  		For(&v1alpha1.Workflow{}).
    37  		Owns(&v1alpha1.WorkflowNode{}).
    38  		Named("workflow-entry-reconciler").
    39  		Complete(
    40  			NewWorkflowEntryReconciler(
    41  				mgr.GetClient(),
    42  				recorderBuilder.Build("workflow-entry-reconciler"),
    43  				logger.WithName("workflow-entry-reconciler"),
    44  			),
    45  		)
    46  	if err != nil {
    47  		return err
    48  	}
    49  
    50  	// TODO: serial node reconciler restore some state in the workflow node status(the active children), it requires keep syncing in time, so we could not use the default controller-runtime client with cache
    51  	// TODO: maybe we could use select with labelSelector as instead
    52  	err = ctrl.NewControllerManagedBy(mgr).
    53  		For(&v1alpha1.WorkflowNode{}).
    54  		Owns(&v1alpha1.WorkflowNode{}).
    55  		Named("workflow-serial-node-reconciler").
    56  		Complete(
    57  			NewSerialNodeReconciler(
    58  				noCacheClient,
    59  				recorderBuilder.Build("workflow-serial-node-reconciler"),
    60  				logger.WithName("workflow-serial-node-reconciler"),
    61  			),
    62  		)
    63  	if err != nil {
    64  		return err
    65  	}
    66  
    67  	err = ctrl.NewControllerManagedBy(mgr).
    68  		For(&v1alpha1.WorkflowNode{}).
    69  		Owns(&v1alpha1.WorkflowNode{}).
    70  		Named("workflow-parallel-node-reconciler").
    71  		Complete(
    72  			NewParallelNodeReconciler(
    73  				noCacheClient,

    75  				logger.WithName("workflow-parallel-node-reconciler"),
    76  			),
    77  		)
    78  	if err != nil {
    79  		return err
    80  	}
    81  
    82  	err = ctrl.NewControllerManagedBy(mgr).
    83  		For(&v1alpha1.WorkflowNode{}).
    84  		Named("workflow-deadline-reconciler").
    85  		Complete(
    86  			NewDeadlineReconciler(
    87  				mgr.GetClient(),
    88  				recorderBuilder.Build("workflow-deadline-reconciler"),
    89  				logger.WithName("workflow-deadline-reconciler"),
    90  			),
    91  		)
    92  	if err != nil {
    93  		return err
    94  	}
    95  
    96  	err = ctrl.NewControllerManagedBy(mgr).
    97  		For(&v1alpha1.WorkflowNode{}).
    98  		Named("workflow-chaos-node-reconciler").
    99  		Complete(
   100  			NewChaosNodeReconciler(
   101  				mgr.GetClient(),
   102  				recorderBuilder.Build("workflow-chaos-node-reconciler"),
   103  				logger.WithName("workflow-chaos-node-reconciler"),
   104  			),
   105  		)
   106  	if err != nil {
   107  		return err
   108  	}
   109  	err = ctrl.NewControllerManagedBy(mgr).
   110  		For(&v1alpha1.WorkflowNode{}).
   111  		Owns(&v1alpha1.WorkflowNode{}).
   112  		Owns(&corev1.Pod{}).
   113  		Named("workflow-task-reconciler").
   114  		Complete(NewTaskReconciler(
   115  			noCacheClient,
   116  			mgr.GetConfig(),
   117  			recorderBuilder.Build("workflow-task-reconciler"),
   118  			logger.WithName("workflow-task-reconciler"),
   119  		))
   120  	return err
