    33  // renderNodesByTemplates will render the nodes one by one, will setup owner by given parent. If parent is nil, it will use workflow as its owner.
    34  func renderNodesByTemplates(workflow *v1alpha1.Workflow, parent *v1alpha1.WorkflowNode, templates ...string) ([]*v1alpha1.WorkflowNode, error) {
    35  	templateNameSet := make(map[string]v1alpha1.Template)
    36  	for _, template := range workflow.Spec.Templates {
    37  		templateNameSet[template.Name] = template
    38  	}
    39  	var result []*v1alpha1.WorkflowNode
    40  	for _, name := range templates {
    41  		if template, ok := templateNameSet[name]; ok {
    42  
    43  			now := metav1.NewTime(time.Now())
    44  			var deadline *metav1.Time = nil
    45  
    46  			if template.Deadline != nil {
    47  				duration, err := time.ParseDuration(*template.Deadline)
    48  				if err != nil {
    49  					// TODO: logger
    50  					return nil, err
    51  				}
    52  				copiedDuration := metav1.NewTime(now.DeepCopy().Add(duration))
    53  				deadline = &copiedDuration
    54  			}
    55  
    56  			renderedNode := v1alpha1.WorkflowNode{
    57  				ObjectMeta: metav1.ObjectMeta{
    58  					Namespace:    workflow.Namespace,
    59  					GenerateName: fmt.Sprintf("%s-", template.Name),
    60  				},
    61  				Spec: v1alpha1.WorkflowNodeSpec{
    62  					TemplateName:        template.Name,
    63  					WorkflowName:        workflow.Name,
    64  					Type:                template.Type,
    65  					StartTime:           &now,
    66  					Deadline:            deadline,
    67  					Children:            template.Children,
    68  					Task:                template.Task,
    69  					ConditionalBranches: template.ConditionalBranches,
    70  					EmbedChaos:          template.EmbedChaos,
    71  					Schedule:            conversionSchedule(template.Schedule),
    72  				},
    73  			}
    74  
    75  			// if parent is specified, use parent as owner, else use workflow as owner.
    76  			if parent != nil {
    77  				renderedNode.OwnerReferences = append(renderedNode.OwnerReferences, metav1.OwnerReference{
    78  					APIVersion:         ApiVersion,
    79  					Kind:               KindWorkflowNode,
    80  					Name:               parent.Name,
    81  					UID:                parent.UID,
    82  					Controller:         &isController,
    83  					BlockOwnerDeletion: &blockOwnerDeletion,
    84  				})
    85  				if renderedNode.Labels == nil {
    86  					renderedNode.Labels = make(map[string]string)
    87  				}
    88  				renderedNode.Labels[v1alpha1.LabelControlledBy] = parent.Name
    89  			} else {
    90  				renderedNode.OwnerReferences = append(renderedNode.OwnerReferences, metav1.OwnerReference{
    91  					APIVersion:         ApiVersion,
    92  					Kind:               KindWorkflow,
    93  					Name:               workflow.Name,
    94  					UID:                workflow.UID,
    95  					Controller:         &isController,
    96  					BlockOwnerDeletion: &blockOwnerDeletion,
    97  				})
    98  				if renderedNode.Labels == nil {
    99  					renderedNode.Labels = make(map[string]string)
   100  				}
   101  				renderedNode.Labels[v1alpha1.LabelControlledBy] = workflow.Name
   102  			}
   103  
   104  			renderedNode.Labels[v1alpha1.LabelWorkflow] = workflow.Name
   105  			renderedNode.Finalizers = append(renderedNode.Finalizers, metav1.FinalizerDeleteDependents)
   106  
   107  			result = append(result, &renderedNode)
   108  			continue
   109  		}
   110  		return nil, fmt.Errorf(
   111  			"workflow %s do not contains template called %s",
   112  			workflow.Name,
   113  			name,
   114  		)
   115  	}
   116  	return result, nil
   117  }

   119  func conversionSchedule(origin *v1alpha1.ChaosOnlyScheduleSpec) *v1alpha1.ScheduleSpec {
   120  	if origin == nil {
   121  		return nil
   122  	}
   123  	return &v1alpha1.ScheduleSpec{
   124  		Schedule:                origin.Schedule,
   125  		StartingDeadlineSeconds: origin.StartingDeadlineSeconds,
   126  		ConcurrencyPolicy:       origin.ConcurrencyPolicy,
   127  		HistoryLimit:            origin.HistoryLimit,
   128  		Type:                    origin.Type,
   129  		ScheduleItem: v1alpha1.ScheduleItem{
   130  			EmbedChaos: v1alpha1.EmbedChaos{
   131  				AWSChaos:     origin.EmbedChaos.AWSChaos,
   132  				DNSChaos:     origin.EmbedChaos.DNSChaos,
   133  				GCPChaos:     origin.EmbedChaos.GCPChaos,
   134  				HTTPChaos:    origin.EmbedChaos.HTTPChaos,
   135  				IOChaos:      origin.EmbedChaos.IOChaos,
   136  				JVMChaos:     origin.EmbedChaos.JVMChaos,
   137  				KernelChaos:  origin.EmbedChaos.KernelChaos,
   138  				NetworkChaos: origin.EmbedChaos.NetworkChaos,
   139  				PodChaos:     origin.EmbedChaos.PodChaos,
   140  				StressChaos:  origin.EmbedChaos.StressChaos,
   141  				TimeChaos:    origin.EmbedChaos.TimeChaos,
   142  			},
   143  			Workflow: nil,
   144  		},
   145  	}
