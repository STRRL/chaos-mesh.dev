    42  // Reconcile reconciles a chaos collector.
    43  func (r *ChaosCollector) Reconcile(req ctrl.Request) (ctrl.Result, error) {
    44  	if r.apiType == nil {
    45  		r.Log.Error(nil, "apiType has not been initialized")
    46  		return ctrl.Result{}, nil
    47  	}
    48  	ctx := context.Background()
    49  
    50  	obj, ok := r.apiType.DeepCopyObject().(v1alpha1.InnerObject)
    51  	if !ok {
    52  		r.Log.Error(nil, "it's not a stateful object")
    53  		return ctrl.Result{}, nil
    54  	}
    55  
    56  	err := r.Get(ctx, req.NamespacedName, obj)
    57  	if apierrors.IsNotFound(err) {
    58  		if err = r.archiveExperiment(req.Namespace, req.Name); err != nil {
    59  			r.Log.Error(err, "failed to archive experiment")
    60  		}
    61  
    62  		// If the experiment was created by schedule or workflow,
    63  		// it and its events will be deleted from database.
    64  		if err = r.deleteManagedExperiments(req.Namespace, req.Name); err != nil {
    65  			r.Log.Error(err, "delete managed experiments", "namespace", req.Namespace, "name", req.Name)
    66  		}
    67  
    68  		return ctrl.Result{}, nil
    69  	}
    70  
    71  	if err != nil {
    72  		r.Log.Error(err, "failed to get chaos object", "request", req.NamespacedName)
    73  		return ctrl.Result{}, nil
    74  	}
    75  
    76  	if err := r.setUnarchivedExperiment(req, obj); err != nil {
    77  		r.Log.Error(err, "failed to archive experiment")
    78  		// ignore error here
    79  	}
    80  
    81  	return ctrl.Result{}, nil
    82  }

    93  func (r *ChaosCollector) setUnarchivedExperiment(req ctrl.Request, obj v1alpha1.InnerObject) error {
    94  	archive, err := convertInnerObjectToExperiment(obj)
    95  	if err != nil {
    96  		r.Log.Error(err, "failed to covert InnerObject")
    97  		return err
    98  	}
    99  
   100  	find, err := r.archive.FindByUID(context.Background(), archive.UID)
   101  	if err != nil && !gorm.IsRecordNotFoundError(err) {
   102  		r.Log.Error(err, "failed to find experiment", "UID", archive.UID)
   103  		return err
   104  	}
   105  
   106  	if find != nil {
   107  		archive.ID = find.ID
   108  		archive.CreatedAt = find.CreatedAt
   109  		archive.UpdatedAt = find.UpdatedAt
   110  	}
   111  
   112  	if err := r.archive.Set(context.Background(), archive); err != nil {
   113  		r.Log.Error(err, "failed to update experiment", "archive", archive)
   114  		return err
   115  	}
   116  
   117  	return nil
   118  }

   120  func (r *ChaosCollector) archiveExperiment(ns, name string) error {
   121  	if err := r.archive.Archive(context.Background(), ns, name); err != nil {
   122  		r.Log.Error(err, "failed to archive experiment", "namespace", ns, "name", name)
   123  		return err
   124  	}
   125  
   126  	return nil
   127  }

   129  func (r *ChaosCollector) deleteManagedExperiments(ns, name string) error {
   130  	archives, err := r.archive.FindManagedByNamespaceName(context.Background(), ns, name)
   131  	if gorm.IsRecordNotFoundError(err) {
   132  		return nil
   133  	}
   134  
   135  	if err != nil {
   136  		return err
   137  	}
   138  
   139  	for _, expr := range archives {
   140  		if err = r.event.DeleteByUID(context.Background(), expr.UID); err != nil {
   141  			r.Log.Error(err, "failed to delete experiment related events")
   142  		}
   143  
   144  		if err = r.archive.Delete(context.Background(), expr); err != nil {
   145  			r.Log.Error(err, "failed to delete managed experiment")
   146  		}
   147  	}
   148  
   149  	return nil
   150  }

   152  func convertInnerObjectToExperiment(obj v1alpha1.InnerObject) (*core.Experiment, error) {
   153  	chaosMeta, ok := obj.(metav1.Object)
   154  	if !ok {
   155  		return nil, errors.New("chaos meta information not found")
   156  	}
   157  	UID := string(chaosMeta.GetUID())
   158  
   159  	archive := &core.Experiment{
   160  		ExperimentMeta: core.ExperimentMeta{
   161  			Namespace: chaosMeta.GetNamespace(),
   162  			Name:      chaosMeta.GetName(),
   163  			Kind:      obj.GetObjectKind().GroupVersionKind().Kind,
   164  			UID:       UID,
   165  			Archived:  false,
   166  		},
   167  	}
   168  
   169  	switch chaos := obj.(type) {
   170  	case *v1alpha1.PodChaos:
   171  		archive.Action = string(chaos.Spec.Action)
   172  	case *v1alpha1.NetworkChaos:
   173  		archive.Action = string(chaos.Spec.Action)
   174  	case *v1alpha1.IOChaos:
   175  		archive.Action = string(chaos.Spec.Action)
   176  	case *v1alpha1.TimeChaos, *v1alpha1.KernelChaos, *v1alpha1.StressChaos:
   177  		archive.Action = ""
   178  	case *v1alpha1.DNSChaos:
   179  		archive.Action = string(chaos.Spec.Action)
   180  	case *v1alpha1.AWSChaos:
   181  		archive.Action = string(chaos.Spec.Action)
   182  	case *v1alpha1.GCPChaos:
   183  		archive.Action = string(chaos.Spec.Action)
   184  	default:
   185  		return nil, errors.New("unsupported chaos type " + archive.Kind)
   186  	}
   187  
   188  	archive.StartTime = chaosMeta.GetCreationTimestamp().Time
   189  	if chaosMeta.GetDeletionTimestamp() != nil {
   190  		archive.FinishTime = chaosMeta.GetDeletionTimestamp().Time
   191  	}
   192  
   193  	data, err := json.Marshal(chaosMeta)
   194  	if err != nil {
   195  		return nil, err
   196  	}
   197  
   198  	archive.Experiment = string(data)
   199  
   200  	return archive, nil
   201  }
