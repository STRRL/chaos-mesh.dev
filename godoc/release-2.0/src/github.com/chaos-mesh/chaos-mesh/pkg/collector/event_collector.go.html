    33  // EventCollector represents a collector for Event Object.

    41  // Reconcile reconciles a Event collector.
    42  func (r *EventCollector) Reconcile(req ctrl.Request) (ctrl.Result, error) {
    43  	if r.apiType == nil {
    44  		r.Log.Error(nil, "apiType has not been initialized")
    45  		return ctrl.Result{}, nil
    46  	}
    47  	ctx := context.Background()
    48  
    49  	event := &v1.Event{}
    50  	err := r.Get(ctx, req.NamespacedName, event)
    51  	if err != nil {
    52  		if !apierrors.IsNotFound(err) {
    53  			r.Log.Error(err, "unable to get event")
    54  		}
    55  		return ctrl.Result{}, nil
    56  	}
    57  
    58  	chaosKind, ok := v1alpha1.AllKinds()[event.InvolvedObject.Kind]
    59  	if ok {
    60  		if err = r.Get(ctx, types.NamespacedName{
    61  			Namespace: event.InvolvedObject.Namespace,
    62  			Name:      event.InvolvedObject.Name,
    63  		}, chaosKind.Chaos); err != nil {
    64  			return ctrl.Result{}, nil
    65  		}
    66  	} else {
    67  		if err = r.Get(ctx, types.NamespacedName{
    68  			Namespace: event.InvolvedObject.Namespace,
    69  			Name:      event.InvolvedObject.Name,
    70  		}, &v1alpha1.Schedule{}); err != nil {
    71  			return ctrl.Result{}, nil
    72  		}
    73  	}
    74  
    75  	et := core.Event{
    76  		CreatedAt: event.CreationTimestamp.Time,
    77  		Kind:      event.InvolvedObject.Kind,
    78  		Type:      event.Type,
    79  		Reason:    event.Reason,
    80  		Message:   event.Message,
    81  		Name:      event.InvolvedObject.Name,
    82  		Namespace: event.InvolvedObject.Namespace,
    83  		ObjectID:  string(event.InvolvedObject.UID),
    84  	}
    85  	if err := r.event.Create(context.Background(), &et); err != nil {
    86  		r.Log.Error(err, "failed to save event", "event", et)
    87  	}
    88  
    89  	return ctrl.Result{}, nil
    90  }

    92  // Setup setups collectors by Manager.
    93  func (r *EventCollector) Setup(mgr ctrl.Manager, apiType runtime.Object) error {
    94  	r.apiType = apiType
    95  
    96  	return ctrl.NewControllerManagedBy(mgr).
    97  		For(apiType).
    98  		WithEventFilter(predicate.Funcs{
    99  			CreateFunc: func(e event.CreateEvent) bool {
   100  				event, ok := e.Object.(*v1.Event)
   101  				if !ok {
   102  					return false
   103  				}
   104  				flag := false
   105  				_, ok = v1alpha1.AllKinds()[event.InvolvedObject.Kind]
   106  				if ok {
   107  					flag = true
   108  				}
   109  				if event.InvolvedObject.Kind == v1alpha1.KindSchedule {
   110  					flag = true
   111  				}
   112  				return flag
   113  
   114  			},
   115  			DeleteFunc: func(e event.DeleteEvent) bool {
   116  				return false
   117  			},
   118  			UpdateFunc: func(e event.UpdateEvent) bool {
   119  				return false
   120  			},
   121  			GenericFunc: func(e event.GenericEvent) bool {
   122  				return false
   123  			},
   124  		}).
   125  		Complete(r)
   126  }
