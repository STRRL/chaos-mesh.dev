    28  // ExperimentStore defines operations for working with experiments.
    29  type ExperimentStore interface {
    30  	// ListMeta returns experiment metadata list from the datastore.
    31  	ListMeta(ctx context.Context, kind, namespace, name string, archived bool) ([]*ExperimentMeta, error)
    32  
    33  	// FindByUID returns an experiment by UID.
    34  	FindByUID(ctx context.Context, UID string) (*Experiment, error)
    35  
    36  	// FindManagedByNamespaceName returns experiment list which are managed by schedule or workflow.
    37  	FindManagedByNamespaceName(ctx context.Context, namespace, name string) ([]*Experiment, error)
    38  
    39  	// FindMetaByUID returns an experiment metadata by UID.
    40  	FindMetaByUID(context.Context, string) (*ExperimentMeta, error)
    41  
    42  	// Set saves the experiment to datastore.
    43  	Set(context.Context, *Experiment) error
    44  
    45  	// Archive archives experiments which "archived" field is false.
    46  	Archive(ctx context.Context, namespace, name string) error
    47  
    48  	// Delete deletes the archive from the datastore.
    49  	Delete(context.Context, *Experiment) error
    50  
    51  	// DeleteByFinishTime deletes archives which time difference is greater than the given time from FinishTime.
    52  	DeleteByFinishTime(context.Context, time.Duration) error
    53  
    54  	// DeleteByUIDs deletes archives by the uid list.
    55  	DeleteByUIDs(context.Context, []string) error
    56  
    57  	// DeleteIncompleteExperiments deletes all incomplete experiments.
    58  	// If the chaos-dashboard was restarted and the experiment is completed during the restart,
    59  	// which means the experiment would never save the finish_time.
    60  	// DeleteIncompleteExperiments can be used to delete all incomplete experiments to avoid this case.
    61  	DeleteIncompleteExperiments(context.Context) error
    62  }

   101  // SelectorInfo defines the selector options of the Experiment.
   102  type SelectorInfo struct {
   103  	Namespaces          []string                          `json:"namespaces,omitempty" binding:"NamespaceSelectorsValid"`
   104  	Nodes               []string                          `json:"nodes,omitempty"`
   105  	NodeSelectors       map[string]string                 `json:"node_selectors,omitempty"`
   106  	FieldSelectors      map[string]string                 `json:"field_selectors,omitempty" binding:"MapSelectorsValid"`
   107  	LabelSelectors      map[string]string                 `json:"label_selectors,omitempty" binding:"MapSelectorsValid"`
   108  	ExpressionSelectors []metav1.LabelSelectorRequirement `json:"expression_selectors,omitempty" binding:"RequirementSelectorsValid"`
   109  	AnnotationSelectors map[string]string                 `json:"annotation_selectors,omitempty" binding:"MapSelectorsValid"`
   110  	PodPhaseSelectors   []string                          `json:"phase_selectors,omitempty" binding:"PhaseSelectorsValid"`
   111  
   112  	// Pods is a map of string keys and a set values that used to select pods.
   113  	// The key defines the namespace which pods belong,
   114  	// and the each values is a set of pod names.
   115  	Pods map[string][]string `json:"pods,omitempty" binding:"PodsValid"`
   116  }

   118  // ParseSelector parses SelectorInfo to v1alpha1.SelectorSpec
   119  func (s *SelectorInfo) ParseSelector() v1alpha1.PodSelectorSpec {
   120  	selector := v1alpha1.PodSelectorSpec{}
   121  	selector.Namespaces = append(selector.Namespaces, s.Namespaces...)
   122  
   123  	selector.LabelSelectors = make(map[string]string)
   124  	for key, val := range s.LabelSelectors {
   125  		selector.LabelSelectors[key] = val
   126  	}
   127  
   128  	selector.ExpressionSelectors = append(selector.ExpressionSelectors, s.ExpressionSelectors...)
   129  
   130  	selector.AnnotationSelectors = make(map[string]string)
   131  	for key, val := range s.AnnotationSelectors {
   132  		selector.AnnotationSelectors[key] = val
   133  	}
   134  
   135  	selector.FieldSelectors = make(map[string]string)
   136  	for key, val := range s.FieldSelectors {
   137  		selector.FieldSelectors[key] = val
   138  	}
   139  
   140  	selector.PodPhaseSelectors = append(selector.PodPhaseSelectors, s.PodPhaseSelectors...)
   141  
   142  	if s.Pods != nil {
   143  		selector.Pods = s.Pods
   144  	}
   145  
   146  	return selector
   147  }
