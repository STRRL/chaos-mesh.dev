    33  func (in *Workflow) ValidateCreate() error {
    34  	var allErrs field.ErrorList
    35  	specPath := field.NewPath("spec")
    36  	allErrs = append(allErrs, entryMustExists(specPath.Child("entry"), in.Spec.Entry, in.Spec.Templates)...)

    38  	if len(allErrs) > 0 {
    39  		return fmt.Errorf(allErrs.ToAggregate().Error())
    40  	}
    41  	return nil
    42  }

    52  func entryMustExists(path *field.Path, entry string, templates []Template) field.ErrorList {
    53  	var result field.ErrorList
    54  	// name is required
    55  	if len(entry) == 0 {
    56  		result = append(result, field.Required(path, "the entry of workflow is required"))
    57  	}
    58  	founded := false
    59  	for _, item := range templates {
    60  		if item.Name == entry {
    61  			founded = true
    62  			break
    63  		}
    64  	}
    65  	if !founded {
    66  		result = append(result, field.Invalid(path, entry, fmt.Sprintf("can not find a template with name %s", entry)))
    67  	}
    68  	return result
    69  }

    71  func validateTemplates(path *field.Path, templates []Template) field.ErrorList {
    72  	var result field.ErrorList
    73  	if len(templates) == 0 {
    74  		result = append(result, field.Invalid(path, templates, "templates in workflow could not be empty"))
    75  		return result
    76  	}
    77  	var allNames []string
    78  	for _, template := range templates {
    79  		allNames = append(allNames, template.Name)
    80  	}
    81  	result = append(result, namesCouldNotBeDuplicated(path, allNames)...)
    82  
    83  	for i, item := range templates {
    84  		itemPath := path.Index(i)
    85  		result = append(result, validateTemplate(itemPath, item, templates)...)
    86  	}
    87  	return result
    88  }

    90  func validateTemplate(path *field.Path, template Template, allTemplates []Template) field.ErrorList {
    91  	var result field.ErrorList
    92  	// name is required
    93  	if len(template.Name) == 0 {
    94  		result = append(result, field.Required(path.Child("name"), "name of template is required"))
    95  	}
    96  
    97  	// name must be restricted with DNS-1123
    98  	errs := validation.IsDNS1123Subdomain(template.Name)
    99  	if len(errs) > 0 {
   100  		result = append(result, field.Invalid(path.Child("name"), template.Name, fmt.Sprintf("field name must be DNS-1123 subdomain, %s", errs)))
   101  	}
   102  
   103  	// template name could not be duplicated
   104  
   105  	switch templateType := template.Type; {
   106  	case templateType == TypeSuspend:
   107  		if template.Deadline == nil || len(*template.Deadline) == 0 {
   108  			result = append(result, field.Invalid(path.Child("deadline"), template.Deadline, "deadline in template with type Suspend could not be empty"))
   109  		}
   110  		result = append(result, shouldBeNoTask(path, template)...)
   111  		result = append(result, shouldBeNoChildren(path, template)...)
   112  		result = append(result, shouldBeNoConditionalBranches(path, template)...)
   113  		result = append(result, shouldBeNoEmbedChaos(path, template)...)
   114  		result = append(result, shouldBeNoSchedule(path, template)...)
   115  	case templateType == TypeSerial, templateType == TypeParallel:
   116  		for i, item := range template.Children {
   117  			result = append(result, templateMustExists(item, path.Child("children").Index(i), allTemplates)...)
   118  		}
   119  		result = append(result, shouldBeNoTask(path, template)...)
   120  		result = append(result, shouldBeNoConditionalBranches(path, template)...)
   121  		result = append(result, shouldBeNoEmbedChaos(path, template)...)
   122  		result = append(result, shouldBeNoSchedule(path, template)...)
   123  	case templateType == TypeSchedule:
   124  		result = append(result, shouldBeNoTask(path, template)...)
   125  		result = append(result, shouldBeNoChildren(path, template)...)
   126  		result = append(result, shouldBeNoConditionalBranches(path, template)...)
   127  		result = append(result, shouldBeNoEmbedChaos(path, template)...)
   128  	case templateType == TypeTask:
   129  		result = append(result, shouldBeNoChildren(path, template)...)
   130  		result = append(result, shouldBeNoEmbedChaos(path, template)...)
   131  		result = append(result, shouldBeNoSchedule(path, template)...)
   132  	case IsChaosTemplateType(templateType):
   133  		result = append(result, shouldNotSetupDurationInTheChaos(path, template)...)
   134  
   135  		result = append(result, shouldBeNoTask(path, template)...)
   136  		result = append(result, shouldBeNoChildren(path, template)...)
   137  		result = append(result, shouldBeNoConditionalBranches(path, template)...)
   138  		result = append(result, shouldBeNoSchedule(path, template)...)
   139  
   140  		result = append(result, template.EmbedChaos.Validate(string(templateType))...)
   141  	default:
   142  		result = append(result, field.Invalid(path.Child("templateType"), template.Type, fmt.Sprintf("unrecognized template type: %s", template.Type)))
   143  	}
   144  
   145  	return result
   146  }

   148  func namesCouldNotBeDuplicated(templatesPath *field.Path, names []string) field.ErrorList {
   149  	nameCounter := make(map[string]int)
   150  	for _, name := range names {
   151  		if count, ok := nameCounter[name]; ok {
   152  			nameCounter[name] = count + 1
   153  		} else {
   154  			nameCounter[name] = 1
   155  		}
   156  	}
   157  	var duplicatedNames []string
   158  	for name, count := range nameCounter {
   159  		if count > 1 {
   160  			duplicatedNames = append(duplicatedNames, name)
   161  		}
   162  	}
   163  	sort.Strings(duplicatedNames)
   164  	if len(duplicatedNames) > 0 {
   165  		return field.ErrorList{
   166  			field.Invalid(templatesPath, "", fmt.Sprintf("template name must be unique, duplicated names: %s", duplicatedNames)),
   167  		}
   168  	}
   169  	return nil
   170  }

   190  func shouldNotSetupDurationInTheChaos(path *field.Path, template Template) field.ErrorList {
   191  	var result field.ErrorList
   192  
   193  	if template.EmbedChaos == nil {
   194  		result = append(result, field.Invalid(path.Child(string(template.Type)), nil, fmt.Sprintf("the value of chaos %s is required", template.Type)))
   195  	}
   196  
   197  	spec := reflect.ValueOf(template.EmbedChaos).Elem().FieldByName(string(template.Type))
   198  	if !spec.IsValid() || spec.IsNil() {
   199  		result = append(result, field.Invalid(path.Child(string(template.Type)),
   200  			nil,
   201  			fmt.Sprintf("parse workflow field error: missing chaos spec %s", template.Type)))
   202  		return result
   203  	}
   204  	if commonSpec, ok := spec.Interface().(CommonSpec); !ok {
   205  		result = append(result, field.Invalid(path, "", fmt.Sprintf("Chaos: %s does not implement CommonSpec", template.Type)))
   206  	} else {
   207  		duration, err := commonSpec.GetDuration()
   208  		if err != nil {
   209  			result = append(result, field.Invalid(path, "", err.Error()))
   210  			return result
   211  		}
   212  		if duration != nil {
   213  			result = append(result, field.Invalid(path, duration, "should not define duration in chaos when using Workflow, use Template#Deadline instead."))
   214  		}
   215  	}
   216  	return result
   217  }
