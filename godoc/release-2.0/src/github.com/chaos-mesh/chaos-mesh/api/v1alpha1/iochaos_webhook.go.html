    91  func (in *IOChaosSpec) Validate() field.ErrorList {
    92  	specField := field.NewPath("spec")
    93  	allErrs := in.validateDelay(specField.Child("delay"))
    94  	allErrs = append(allErrs, validateDuration(in, specField)...)
    95  	allErrs = append(allErrs, validatePodSelector(in.PodSelector.Value, in.PodSelector.Mode, specField.Child("value"))...)
    96  	allErrs = append(allErrs, in.validateErrno(specField.Child("errno"))...)
    97  	allErrs = append(allErrs, in.validatePercent(specField.Child("percent"))...)
    98  
    99  	return allErrs
   100  }

   102  func (in *IOChaosSpec) validateDelay(delay *field.Path) field.ErrorList {
   103  	allErrs := field.ErrorList{}
   104  	if in.Action == IoLatency {
   105  		_, err := time.ParseDuration(in.Delay)
   106  		if err != nil {
   107  			allErrs = append(allErrs, field.Invalid(delay, in.Delay,
   108  				fmt.Sprintf("parse delay field error:%s for action:%s", err, in.Action)))
   109  		}
   110  	}
   111  	return allErrs
   112  }

   114  func (in *IOChaosSpec) validateErrno(errno *field.Path) field.ErrorList {
   115  	allErrs := field.ErrorList{}

   119  				fmt.Sprintf("action %s: errno 0 is not supported", in.Action)))
