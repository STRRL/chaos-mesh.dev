    29  // +kubebuilder:object:generate=false

    36  func validateDuration(spec CommonSpec, path *field.Path) field.ErrorList {
    37  	allErrs := field.ErrorList{}
    38  
    39  	durationField := path.Child("duration")
    40  	_, err := spec.GetDuration()
    41  	if err != nil {
    42  		allErrs = append(allErrs, field.Invalid(durationField, nil,
    43  			fmt.Sprintf("parse duration field error:%s", err)))
    44  	}
    45  
    46  	return allErrs
    47  }

    49  // validatePodSelector validates the value with podmode
    50  func validatePodSelector(value string, mode PodMode, valueField *field.Path) field.ErrorList {
    51  	allErrs := field.ErrorList{}
    52  
    53  	switch mode {
    54  	case FixedPodMode:
    55  		num, err := strconv.Atoi(value)
    56  		if err != nil {
    57  			allErrs = append(allErrs, field.Invalid(valueField, value,
    58  				fmt.Sprintf(ValidateValueParseError, err)))
    59  			break
    60  		}
    61  
    62  		if num <= 0 {
    63  			allErrs = append(allErrs, field.Invalid(valueField, value,
    64  				fmt.Sprintf("value must be greater than 0 with mode:%s", FixedPodMode)))
    65  		}
    66  
    67  	case RandomMaxPercentPodMode, FixedPercentPodMode:
    68  		percentage, err := strconv.Atoi(value)
    69  		if err != nil {
    70  			allErrs = append(allErrs, field.Invalid(valueField, value,
    71  				fmt.Sprintf(ValidateValueParseError, err)))
    72  			break
    73  		}
    74  
    75  		if percentage <= 0 || percentage > 100 {
    76  			allErrs = append(allErrs, field.Invalid(valueField, value,
    77  				fmt.Sprintf("value of %d is invalid, Must be (0,100] with mode:%s",
    78  					percentage, mode)))
    79  		}
    80  	}
    81  
    82  	return allErrs
    83  }
