   154  // BandwidthSpec defines detail of bandwidth limit.
   155  type BandwidthSpec struct {
   156  	// Rate is the speed knob. Allows bps, kbps, mbps, gbps, tbps unit. bps means bytes per second.
   157  	Rate string `json:"rate"`
   158  	// Limit is the number of bytes that can be queued waiting for tokens to become available.
   159  	// +kubebuilder:validation:Minimum=1
   160  	Limit uint32 `json:"limit"`
   161  	// Buffer is the maximum amount of bytes that tokens can be available for instantaneously.
   162  	// +kubebuilder:validation:Minimum=1
   163  	Buffer uint32 `json:"buffer"`
   164  	// Peakrate is the maximum depletion rate of the bucket.
   165  	// The peakrate does not need to be set, it is only necessary
   166  	// if perfect millisecond timescale shaping is required.
   167  	// +optional
   168  	// +kubebuilder:validation:Minimum=0
   169  	Peakrate *uint64 `json:"peakrate,omitempty"`
   170  	// Minburst specifies the size of the peakrate bucket. For perfect
   171  	// accuracy, should be set to the MTU of the interface.  If a
   172  	// peakrate is needed, but some burstiness is acceptable, this
   173  	// size can be raised. A 3000 byte minburst allows around 3mbit/s
   174  	// of peakrate, given 1000 byte packets.
   175  	// +optional
   176  	// +kubebuilder:validation:Minimum=0
   177  	Minburst *uint32 `json:"minburst,omitempty"`
   178  }
