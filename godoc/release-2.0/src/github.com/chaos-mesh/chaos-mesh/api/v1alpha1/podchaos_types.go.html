    24  // PodChaos is the control script`s spec.
    25  type PodChaos struct {
    26  	metav1.TypeMeta   `json:",inline"`
    27  	metav1.ObjectMeta `json:"metadata,omitempty"`
    28  
    29  	// Spec defines the behavior of a pod chaos experiment
    30  	Spec PodChaosSpec `json:"spec"`
    31  
    32  	// +optional
    33  	// Most recently observed status of the chaos experiment about pods
    34  	Status PodChaosStatus `json:"status"`
    35  }
    36  
    37  // PodChaosAction represents the chaos action about pods.
    38  type PodChaosAction string
    39  
    40  const (
    41  	// PodKillAction represents the chaos action of killing pods.
    42  	PodKillAction PodChaosAction = "pod-kill"
    43  	// PodFailureAction represents the chaos action of injecting errors to pods.
    44  	// This action will cause the pod to not be created for a while.
    45  	PodFailureAction PodChaosAction = "pod-failure"
    46  	// ContainerKillAction represents the chaos action of killing the container
    47  	ContainerKillAction PodChaosAction = "container-kill"
    48  )
    49  
    50  // PodChaosSpec defines the attributes that a user creates on a chaos experiment about pods.
    51  type PodChaosSpec struct {
    52  	ContainerSelector `json:",inline"`
    53  
    54  	// Action defines the specific pod chaos action.
    55  	// Supported action: pod-kill / pod-failure / container-kill
    56  	// Default action: pod-kill
    57  	// +kubebuilder:validation:Enum=pod-kill;pod-failure;container-kill
    58  	Action PodChaosAction `json:"action"`
    59  
    60  	// Duration represents the duration of the chaos action.
    61  	// It is required when the action is `PodFailureAction`.
    62  	// A duration string is a possibly signed sequence of
    63  	// decimal numbers, each with optional fraction and a unit suffix,
    64  	// such as "300ms", "-1.5h" or "2h45m".
    65  	// Valid time units are "ns", "us" (or "µs"), "ms", "s", "m", "h".
    66  	// +optional
    67  	Duration *string `json:"duration,omitempty"`
    68  
    69  	// GracePeriod is used in pod-kill action. It represents the duration in seconds before the pod should be deleted.
    70  	// Value must be non-negative integer. The default value is zero that indicates delete immediately.
    71  	// +optional
    72  	// +kubebuilder:validation:Minimum=0
    73  	GracePeriod int64 `json:"gracePeriod"`
    74  }
    75  
    76  // PodChaosStatus represents the current status of the chaos experiment about pods.

    79  }

    81  func (obj *PodChaos) GetSelectorSpecs() map[string]interface{} {
    82  	switch obj.Spec.Action {
    83  	case PodKillAction, PodFailureAction:
    84  		return map[string]interface{}{
    85  			".": &obj.Spec.PodSelector,
    86  		}
    87  	case ContainerKillAction:
    88  		return map[string]interface{}{
    89  			".": &obj.Spec.ContainerSelector,
    90  		}
    91  	}
    92  
    93  	return nil
    94  }
