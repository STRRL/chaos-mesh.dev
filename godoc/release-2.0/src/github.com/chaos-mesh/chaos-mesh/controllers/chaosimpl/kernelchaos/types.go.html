   100  func (impl *Impl) recoverPod(ctx context.Context, pod *v1.Pod, somechaos v1alpha1.InnerObject) error {
   101  	// judged type in `Recover` already so no need to judge again
   102  	chaos, _ := somechaos.(*v1alpha1.KernelChaos)
   103  	impl.Log.Info("try to recover pod", "namespace", pod.Namespace, "name", pod.Name)
   104  
   105  	pbClient, err := impl.chaosDaemonClientBuilder.Build(ctx, pod)
   106  	if err != nil {
   107  		return err
   108  	}
   109  	defer pbClient.Close()
   110  
   111  	if len(pod.Status.ContainerStatuses) == 0 {
   112  		return fmt.Errorf("%s %s can't get the state of container", pod.Namespace, pod.Name)
   113  	}
   114  
   115  	containerResponse, err := pbClient.ContainerGetPid(ctx, &pb.ContainerRequest{
   116  		Action: &pb.ContainerAction{
   117  			Action: pb.ContainerAction_GETPID,
   118  		},
   119  		ContainerId: pod.Status.ContainerStatuses[0].ContainerID,
   120  	})
   121  
   122  	if err != nil {
   123  		impl.Log.Error(err, "Get container pid error", "namespace", pod.Namespace, "name", pod.Name)
   124  		return err
   125  	}
   126  
   127  	impl.Log.Info("Get container pid", "namespace", pod.Namespace, "name", pod.Name)
   128  	conn, err := impl.CreateBPFKIConnection(ctx, impl.Client, pod)
   129  	if err != nil {
   130  		return err
   131  	}
   132  	defer conn.Close()
   133  
   134  	var callchain []*pb_kernel.FailKernRequestFrame
   135  	for _, frame := range chaos.Spec.FailKernRequest.Callchain {
   136  		callchain = append(callchain, &pb_kernel.FailKernRequestFrame{
   137  			Funcname:   frame.Funcname,
   138  			Parameters: frame.Parameters,
   139  			Predicate:  frame.Predicate,
   140  		})
   141  	}
   142  
   143  	bpfClient := pb_kernel.NewBPFKIServiceClient(conn)
   144  	_, err = bpfClient.RecoverMMOrBIO(ctx, &pb_kernel.FailKernRequest{
   145  		Pid:       containerResponse.Pid,
   146  		Callchain: callchain,
   147  	})
   148  
   149  	return err
   150  }

   152  func (impl *Impl) applyPod(ctx context.Context, pod *v1.Pod, chaos *v1alpha1.KernelChaos) error {
   153  	impl.Log.Info("Try to inject kernel on pod", "namespace", pod.Namespace, "name", pod.Name)
   154  
   155  	pbClient, err := impl.chaosDaemonClientBuilder.Build(ctx, pod)
   156  	if err != nil {
   157  		return err
   158  	}
   159  	defer pbClient.Close()
   160  
   161  	if len(pod.Status.ContainerStatuses) == 0 {
   162  		return fmt.Errorf("%s %s can't get the state of container", pod.Namespace, pod.Name)
   163  	}
   164  
   165  	containerResponse, err := pbClient.ContainerGetPid(ctx, &pb.ContainerRequest{
   166  		Action: &pb.ContainerAction{
   167  			Action: pb.ContainerAction_GETPID,
   168  		},
   169  		ContainerId: pod.Status.ContainerStatuses[0].ContainerID,
   170  	})
   171  	if err != nil {
   172  		impl.Log.Error(err, "Get container pid error", "namespace", pod.Namespace, "name", pod.Name)
   173  		return err
   174  	}
   175  
   176  	impl.Log.Info("Get container pid", "namespace", pod.Namespace, "name", pod.Name)
   177  	conn, err := impl.CreateBPFKIConnection(ctx, impl.Client, pod)
   178  	if err != nil {
   179  		return err
   180  	}
   181  	defer conn.Close()
   182  
   183  	var callchain []*pb_kernel.FailKernRequestFrame
   184  	for _, frame := range chaos.Spec.FailKernRequest.Callchain {
   185  		callchain = append(callchain, &pb_kernel.FailKernRequestFrame{
   186  			Funcname:   frame.Funcname,
   187  			Parameters: frame.Parameters,
   188  			Predicate:  frame.Predicate,
   189  		})
   190  	}
   191  
   192  	bpfClient := pb_kernel.NewBPFKIServiceClient(conn)
   193  	_, err = bpfClient.FailMMOrBIO(ctx, &pb_kernel.FailKernRequest{
   194  		Pid:         containerResponse.Pid,
   195  		Ftype:       pb_kernel.FailKernRequest_FAILTYPE(chaos.Spec.FailKernRequest.FailType),
   196  		Headers:     chaos.Spec.FailKernRequest.Headers,
   197  		Callchain:   callchain,
   198  		Probability: float32(chaos.Spec.FailKernRequest.Probability) / 100,
   199  		Times:       chaos.Spec.FailKernRequest.Times,
   200  	})
   201  
   202  	return err
   203  }

   207  	daemonIP, err := impl.chaosDaemonClientBuilder.FindDaemonIP(ctx, pod)
   208  	if err != nil {
   209  		return nil, err
   210  	}
   211  	builder := grpcUtils.Builder(daemonIP, config.ControllerCfg.BPFKIPort).
   212  		WithDefaultTimeout().
   213  		Insecure()
   214  	return builder.Build()
