    33  // Reconciler for common chaos
    34  type Reconciler struct {
    35  	// Object is used to mark the target type of this Reconciler
    36  	Object runtime.Object
    37  
    38  	// Client is used to operate on the Kubernetes cluster

    40  
    41  	Recorder record.EventRecorder
    42  
    43  	Log logr.Logger
    44  }

    46  type StatusAndReason struct {
    47  	Status corev1.ConditionStatus
    48  	Reason string
    49  }

    51  func (r *Reconciler) Reconcile(req ctrl.Request) (ctrl.Result, error) {
    52  	obj := r.Object.DeepCopyObject().(records.InnerObjectWithSelector)
    53  	if err := r.Client.Get(context.TODO(), req.NamespacedName, obj); err != nil {
    54  		if apierrors.IsNotFound(err) {
    55  			r.Log.Info("chaos not found")
    56  		} else {
    57  			// TODO: handle this error
    58  			r.Log.Error(err, "unable to get chaos")
    59  		}
    60  		return ctrl.Result{}, nil
    61  	}
    62  
    63  	updateError := retry.RetryOnConflict(retry.DefaultBackoff, func() error {
    64  		conditionMap := make(map[v1alpha1.ChaosConditionType]StatusAndReason)
    65  		for _, c := range obj.GetStatus().Conditions {
    66  			conditionMap[c.Type] = StatusAndReason{
    67  				Status: c.Status,
    68  				Reason: c.Reason,
    69  			}
    70  		}
    71  
    72  		newConditionMap := make(map[v1alpha1.ChaosConditionType]StatusAndReason)
    73  		if obj.GetStatus().Experiment.Records != nil {
    74  			newConditionMap[v1alpha1.ConditionSelected] = StatusAndReason{
    75  				Status: corev1.ConditionTrue,
    76  			}
    77  		} else {
    78  			newConditionMap[v1alpha1.ConditionSelected] = StatusAndReason{
    79  				Status: corev1.ConditionFalse,
    80  			}
    81  		}
    82  
    83  		allInjected := corev1.ConditionTrue
    84  		allRecovered := corev1.ConditionTrue
    85  		for _, record := range obj.GetStatus().Experiment.Records {
    86  			if record.Phase != v1alpha1.NotInjected {
    87  				allRecovered = corev1.ConditionFalse
    88  			}
    89  
    90  			if record.Phase != v1alpha1.Injected {
    91  				allInjected = corev1.ConditionFalse
    92  			}
    93  		}
    94  		newConditionMap[v1alpha1.ConditionAllInjected] = StatusAndReason{
    95  			Status: allInjected,
    96  		}
    97  		newConditionMap[v1alpha1.ConditionAllRecovered] = StatusAndReason{
    98  			Status: allRecovered,
    99  		}
   100  
   101  		if obj.IsPaused() {
   102  			newConditionMap[v1alpha1.ConditionPaused] = StatusAndReason{
   103  				Status: corev1.ConditionTrue,
   104  			}
   105  		} else {
   106  			newConditionMap[v1alpha1.ConditionPaused] = StatusAndReason{
   107  				Status: corev1.ConditionFalse,
   108  			}
   109  		}
   110  
   111  		if !reflect.DeepEqual(newConditionMap, conditionMap) {
   112  			conditions := make([]v1alpha1.ChaosCondition, 0, 5)
   113  			for k, v := range newConditionMap {
   114  				conditions = append(conditions, v1alpha1.ChaosCondition{
   115  					Type:   k,
   116  					Status: v.Status,
   117  					Reason: v.Reason,
   118  				})
   119  			}
   120  
   121  			r.Log.Info("updating conditions", "conditions", conditions)
   122  			obj := r.Object.DeepCopyObject().(records.InnerObjectWithSelector)
   123  
   124  			if err := r.Client.Get(context.TODO(), req.NamespacedName, obj); err != nil {
   125  				r.Log.Error(err, "unable to get chaos")
   126  				return err
   127  			}
   128  
   129  			obj.GetStatus().Conditions = conditions
   130  			return r.Client.Update(context.TODO(), obj)
   131  		}
   132  
   133  		return nil
   134  	})
   135  
   136  	if updateError != nil {
   137  		r.Log.Error(updateError, "fail to update")
   138  		r.Recorder.Eventf(obj, "Normal", "Failed", "Failed to update conditions: %s", updateError.Error())
   139  		return ctrl.Result{}, nil
   140  	}
   141  	return ctrl.Result{}, nil
   142  }
