    53  					Namespace: "default",
    54  				},

    56  					Schedule: "@every 10s",
    57  					ScheduleItem: v1alpha1.ScheduleItem{
    58  						EmbedChaos: v1alpha1.EmbedChaos{TimeChaos: &v1alpha1.TimeChaosSpec{
    59  							TimeOffset: "100ms",
    60  							ClockIds:   []string{"CLOCK_REALTIME"},
    61  							Duration:   &duration,
    62  							ContainerSelector: v1alpha1.ContainerSelector{

    64  									Mode: v1alpha1.OnePodMode,
    65  								},

    67  						}},
    68  					},
    69  					ConcurrencyPolicy: v1alpha1.ForbidConcurrent,
    70  					HistoryLimit:      5,

    72  				},

    74  					LastScheduleTime: metav1.NewTime(time.Time{}),
    75  				},
    76  			}
    77

    80

    82  			Expect(k8sClient.Get(context.TODO(), key, fetched)).To(Succeed())
