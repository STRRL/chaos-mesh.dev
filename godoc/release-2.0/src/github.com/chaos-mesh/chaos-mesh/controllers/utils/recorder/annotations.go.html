    31  func generateAnnotations(e ChaosEvent) (map[string]string, error) {
    32  	annotations := make(map[string]string)
    33  
    34  	if e == nil {
    35  		return annotations, nil
    36  	}
    37  
    38  	val := reflect.ValueOf(e)
    39  	val = reflect.Indirect(val)
    40  	for index := 0; index < val.NumField(); index++ {
    41  		fieldName := val.Type().Field(index).Name
    42  		key := annotationPrefix + strcase.ToKebab(fieldName)
    43  		field := val.Field(index)
    44  		switch field.Kind() {
    45  		case reflect.Invalid:
    46  			return nil, ErrInvalidType
    47  		case reflect.String:
    48  			annotations[key] = field.String()
    49  		case reflect.Int, reflect.Int8, reflect.Int16, reflect.Int32, reflect.Int64:
    50  			annotations[key] = strconv.Itoa(int(field.Int()))
    51  		default:
    52  			if marshaler, ok := field.Interface().(encoding.TextMarshaler); ok {
    53  				text, err := marshaler.MarshalText()
    54  				if err != nil {
    55  					return nil, err
    56  				}
    57  
    58  				annotations[key] = string(text)
    59  			} else {
    60  				text, err := json.Marshal(field.Interface())
    61  				if err != nil {
    62  					return nil, err
    63  				}
    64  				annotations[key] = string(text)
    65  			}
    66  		}
    67  	}
    68  	annotations[annotationPrefix+"type"] = strcase.ToKebab(val.Type().Name())
    69  
    70  	return annotations, nil
    71  }

    73  // FromAnnotations will iterate over all the registered event,
    74  // return `nil` if there is no suitable event.
    75  func FromAnnotations(annotations map[string]string) (ChaosEvent, error) {
    76  	typeName := annotations[annotationPrefix+"type"]
    77  	ev := allEvents[typeName]
    78  
    79  	if ev == nil {
    80  		return nil, ErrUknownType
    81  	}
    82  
    83  	val := reflect.ValueOf(ev)
    84  	val = reflect.Indirect(val)
    85  	newEmptyValue := reflect.Indirect(reflect.New(val.Type()))
    86  
    87  	for index := 0; index < newEmptyValue.NumField(); index++ {
    88  		fieldName := newEmptyValue.Type().Field(index).Name
    89  		key := annotationPrefix + strcase.ToKebab(fieldName)
    90  		field := newEmptyValue.Field(index)
    91  		switch field.Kind() {
    92  		case reflect.Invalid:
    93  			return nil, ErrInvalidType
    94  		case reflect.String:
    95  			field.SetString(annotations[key])
    96  		case reflect.Int, reflect.Int8, reflect.Int16, reflect.Int32, reflect.Int64:
    97  			num, err := strconv.Atoi(annotations[key])
    98  			if err != nil {
    99  				return nil, err
   100  			}
   101  			field.SetInt(int64(num))
   102  		default:
   103  			if unmarshaler, ok := field.Interface().(encoding.TextUnmarshaler); ok {
   104  				err := unmarshaler.UnmarshalText([]byte(annotations[key]))
   105  				if err != nil {
   106  					return nil, err
   107  				}
   108  			} else if unmarshaler, ok := field.Addr().Interface().(encoding.TextUnmarshaler); ok {
   109  				err := unmarshaler.UnmarshalText([]byte(annotations[key]))
   110  				if err != nil {
   111  					return nil, err
   112  				}
   113  			} else {
   114  				err := json.Unmarshal([]byte(annotations[key]), field.Addr().Interface())
   115  				if err != nil {
   116  					return nil, err
   117  				}
   118  			}
   119  		}
   120  	}
   121  	return newEmptyValue.Interface().(ChaosEvent), nil
   122  }
