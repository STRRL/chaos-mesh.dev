    55  func (p *Pipeline) AddSteps(steps ...PipelineStep) {
    56  	for _, step := range steps {
    57  		reconciler := step(p.ctx)
    58  		if reconciler == nil {
    59  			return
    60  		}
    61  		p.controllers = append(p.controllers, reconciler)
    62  	}
    63  }

    65  // Reconcile the steps
    66  func (p *Pipeline) Reconcile(req ctrl.Request) (ctrl.Result, error) {
    67  	var deadline *time.Time
    68  
    69  	for _, controller := range p.controllers {
    70  		ret, err := controller.Reconcile(req)
    71  		if err != nil {
    72  			return ctrl.Result{}, err
    73  		}
    74  
    75  		p.ctx.Logger.WithName("pipeline").Info("reconcile result", "result", ret)
    76  
    77  		if ret.Requeue || deadline != nil && deadline.Before(time.Now()) {
    78  			ret.Requeue = true
    79  			return ret, nil
    80  		}
    81  
    82  		if ret.RequeueAfter != 0 {
    83  			// The controller wants us to re-enqueue after a certain amount of time,
    84  			// and the desiredphase controller will always return a RequeueAfter before the experiment is finished.
    85  			//
    86  			// So, DO NOT re-queue immediately.
    87  			end := time.Now().Add(ret.RequeueAfter)
    88  			deadline = minTime(deadline, &end)
    89  		}
    90  	}
    91  
    92  	ret := ctrl.Result{}
    93  
    94  	if deadline != nil {
    95  		if deadline.Before(time.Now()) {
    96  			ret.Requeue = true
    97  		} else {
    98  			ret.RequeueAfter = time.Until(*deadline)
    99  		}
   100  	}
   101  
   102  	return ret, nil
   103  }

   105  func minTime(d1, d2 *time.Time) *time.Time {
   106  	if d1 == nil {
   107  		return d2
   108  	}
   109  
   110  	if d2 == nil {
   111  		return d1
   112  	}
   113  
   114  	if d1.Before(*d2) {
   115  		return d1
   116  	}
   117  
   118  	return d2
   119  }
