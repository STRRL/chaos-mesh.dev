    45  func (r *Reconciler) Reconcile(req ctrl.Request) (ctrl.Result, error) {
    46  	ctx := context.TODO()
    47  
    48  	obj := &v1alpha1.PodIOChaos{}
    49  
    50  	if err := r.Client.Get(ctx, req.NamespacedName, obj); err != nil {
    51  		if apierrors.IsNotFound(err) {
    52  			r.Log.Info("chaos not found")
    53  		} else {
    54  			// TODO: handle this error
    55  			r.Log.Error(err, "unable to get chaos")
    56  		}
    57  		return ctrl.Result{}, nil
    58  	}
    59  
    60  	if obj.ObjectMeta.Generation <= obj.Status.ObservedGeneration && obj.Status.FailedMessage == "" {
    61  		r.Log.Info("the target pod has been up to date", "pod", obj.Namespace+"/"+obj.Name)
    62  		return ctrl.Result{}, nil
    63  	}
    64  
    65  	r.Log.Info("updating io chaos", "pod", obj.Namespace+"/"+obj.Name, "spec", obj.Spec)
    66  
    67  	pod := &v1.Pod{}
    68  
    69  	err := r.Client.Get(ctx, types.NamespacedName{
    70  		Name:      obj.Name,
    71  		Namespace: obj.Namespace,
    72  	}, pod)
    73  	if err != nil {
    74  		r.Log.Error(err, "fail to find pod")
    75  		return ctrl.Result{}, nil
    76  	}
    77  
    78  	failedMessage := ""
    79  	observedGeneration := obj.ObjectMeta.Generation
    80  	pid := obj.Status.Pid
    81  	startTime := obj.Status.StartTime
    82  	defer func() {
    83  		if err != nil {
    84  			failedMessage = err.Error()
    85  		}
    86  
    87  		updateError := retry.RetryOnConflict(retry.DefaultBackoff, func() error {
    88  			obj := &v1alpha1.PodIOChaos{}
    89  
    90  			if err := r.Client.Get(context.TODO(), req.NamespacedName, obj); err != nil {
    91  				r.Log.Error(err, "unable to get chaos")
    92  				return err
    93  			}
    94  
    95  			obj.Status.FailedMessage = failedMessage
    96  			obj.Status.ObservedGeneration = observedGeneration
    97  			obj.Status.Pid = pid
    98  			obj.Status.StartTime = startTime
    99  
   100  			return r.Client.Status().Update(context.TODO(), obj)
   101  		})
   102  
   103  		if updateError != nil {
   104  			r.Log.Error(updateError, "fail to update")
   105  			r.Recorder.Eventf(obj, "Normal", "Failed", "Failed to update status: %s", updateError.Error())
   106  		}
   107  	}()
   108  
   109  	pbClient, err := r.ChaosDaemonClientBuilder.Build(ctx, pod)
   110  	if err != nil {
   111  		r.Recorder.Event(obj, "Warning", "Failed", err.Error())
   112  		return ctrl.Result{Requeue: true}, nil
   113  	}
   114  	defer pbClient.Close()
   115  
   116  	if len(pod.Status.ContainerStatuses) == 0 {
   117  		err = fmt.Errorf("%s %s can't get the state of container", pod.Namespace, pod.Name)
   118  		r.Recorder.Event(obj, "Warning", "Failed", err.Error())
   119  		return ctrl.Result{}, nil
   120  	}
   121  
   122  	containerID := pod.Status.ContainerStatuses[0].ContainerID
   123  	if obj.Spec.Container != nil &&
   124  		len(strings.TrimSpace(*obj.Spec.Container)) != 0 {
   125  		containerID = ""
   126  		for _, container := range pod.Status.ContainerStatuses {
   127  			if container.Name == *obj.Spec.Container {
   128  				containerID = container.ContainerID
   129  				break
   130  			}
   131  		}
   132  		if len(containerID) == 0 {
   133  			err = fmt.Errorf("cannot find container with name %s", *obj.Spec.Container)
   134  			r.Recorder.Event(obj, "Warning", "Failed", err.Error())
   135  			return ctrl.Result{}, nil
   136  		}
   137  	}
   138  
   139  	actions, err := json.Marshal(obj.Spec.Actions)
   140  	if err != nil {
   141  		r.Recorder.Event(obj, "Warning", "Failed", err.Error())
   142  		return ctrl.Result{Requeue: true}, nil
   143  	}
   144  	input := string(actions)
   145  	r.Log.Info("input with", "config", input)
   146  
   147  	res, err := pbClient.ApplyIOChaos(ctx, &pb.ApplyIOChaosRequest{
   148  		Actions:     input,
   149  		Volume:      obj.Spec.VolumeMountPath,
   150  		ContainerId: containerID,
   151  
   152  		Instance:  obj.Status.Pid,
   153  		StartTime: obj.Status.StartTime,
   154  		EnterNS:   true,
   155  	})
   156  	if err != nil {
   157  		r.Recorder.Event(obj, "Warning", "Failed", err.Error())
   158  		return ctrl.Result{Requeue: true}, nil
   159  	}
   160  
   161  	startTime = res.StartTime
   162  	pid = res.Instance
   163  
   164  	return ctrl.Result{}, nil
   165  }
