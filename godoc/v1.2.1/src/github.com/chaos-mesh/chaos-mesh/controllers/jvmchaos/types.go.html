    49  // Apply applies jvm-chaos
    50  func (r *endpoint) Apply(ctx context.Context, req ctrl.Request, chaos v1alpha1.InnerObject) error {
    51  	jvmchaos, ok := chaos.(*v1alpha1.JVMChaos)
    52  	if !ok {
    53  		err := errors.New("chaos is not JVMChaos")
    54  		r.Log.Error(err, "chaos is not JVMChaos", "chaos", chaos)
    55  		return err
    56  	}
    57  
    58  	pods, err := selector.SelectAndFilterPods(ctx, r.Client, r.Reader, &jvmchaos.Spec, config.ControllerCfg.ClusterScoped, config.ControllerCfg.TargetNamespace, config.ControllerCfg.EnableFilterNamespace)
    59  	if err != nil {
    60  		r.Log.Error(err, "failed to select and generate pods")
    61  		return err
    62  	}
    63  
    64  	if err = r.applyAllPods(ctx, pods, jvmchaos); err != nil {
    65  		r.Log.Error(err, "failed to apply chaos on all pods")
    66  		return err
    67  	}
    68  
    69  	jvmchaos.Status.Experiment.PodRecords = make([]v1alpha1.PodStatus, 0, len(pods))
    70  	for _, pod := range pods {
    71  		ps := v1alpha1.PodStatus{
    72  			Namespace: pod.Namespace,
    73  			Name:      pod.Name,
    74  			HostIP:    pod.Status.HostIP,
    75  			PodIP:     pod.Status.PodIP,
    76  		}
    77  
    78  		jvmchaos.Status.Experiment.PodRecords = append(jvmchaos.Status.Experiment.PodRecords, ps)
    79  	}
    80  	r.Event(jvmchaos, v1.EventTypeNormal, events.ChaosInjected, "")
    81  	return nil
    82  }

    84  func (r *endpoint) applyAllPods(ctx context.Context, pods []v1.Pod, chaos *v1alpha1.JVMChaos) error {
    85  	g := errgroup.Group{}
    86  	for index := range pods {
    87  		pod := &pods[index]
    88  
    89  		key, err := cache.MetaNamespaceKeyFunc(pod)
    90  		if err != nil {
    91  			r.Log.Error(err, "get meta namespace key")
    92  			return err
    93  		}
    94  		chaos.Finalizers = finalizer.InsertFinalizer(chaos.Finalizers, key)
    95  
    96  		g.Go(func() error {
    97  			return r.applyPod(ctx, pod, chaos)
    98  		})
    99  	}
   100  	err := g.Wait()
   101  	if err != nil {
   102  		r.Log.Error(err, "g.Wait")
   103  		return err
   104  	}
   105  	return nil
   106  }

   108  func (r *endpoint) applyPod(ctx context.Context, pod *v1.Pod, chaos *v1alpha1.JVMChaos) error {
   109  	r.Log.Info("Try to apply jvm chaos", "namespace",
   110  		pod.Namespace, "name", pod.Name)
   111  
   112  	// TODO: Custom port may be required
   113  	err := jvm.ActiveSandbox(pod.Status.PodIP, sandboxPort)
   114  	if err != nil {
   115  		return err
   116  	}
   117  
   118  	r.Log.Info("active sandbox", "pod", pod.Name)
   119  
   120  	suid := genSUID(pod, chaos)
   121  	jsonBytes, err := jvm.ToSandboxAction(suid, chaos)
   122  
   123  	if err != nil {
   124  		return err
   125  	}
   126  	// TODO: Custom port may be required
   127  	err = jvm.InjectChaos(pod.Status.PodIP, sandboxPort, jsonBytes)
   128  	if err != nil {
   129  		return err
   130  	}
   131  	r.Log.Info("Inject JVM Chaos", "pod", pod.Name, "action", chaos.Spec.Action)
   132  	return nil
   133  }

   144  // Recover means the reconciler recovers the chaos action
   145  func (r *endpoint) Recover(ctx context.Context, req ctrl.Request, chaos v1alpha1.InnerObject) error {
   146  	jvmchaos, ok := chaos.(*v1alpha1.JVMChaos)
   147  	if !ok {
   148  		err := errors.New("chaos is not JVMChaos")
   149  		r.Log.Error(err, "chaos is not JVMChaos", "chaos", chaos)
   150  		return err
   151  	}
   152  	if err := r.cleanFinalizersAndRecover(ctx, jvmchaos); err != nil {
   153  		return err
   154  	}
   155  
   156  	r.Event(jvmchaos, v1.EventTypeNormal, events.ChaosRecovered, "")
   157  
   158  	return nil
   159  }

   161  func (r *endpoint) cleanFinalizersAndRecover(ctx context.Context, chaos *v1alpha1.JVMChaos) error {
   162  	var result error
   163  
   164  	for _, key := range chaos.Finalizers {
   165  		ns, name, err := cache.SplitMetaNamespaceKey(key)
   166  		if err != nil {
   167  			result = multierror.Append(result, err)
   168  			continue
   169  		}
   170  
   171  		var pod v1.Pod
   172  		err = r.Client.Get(ctx, types.NamespacedName{
   173  			Namespace: ns,
   174  			Name:      name,
   175  		}, &pod)
   176  
   177  		if err != nil {
   178  			if !k8serror.IsNotFound(err) {
   179  				result = multierror.Append(result, err)
   180  				continue
   181  			}
   182  
   183  			r.Log.Info("Pod not found", "namespace", ns, "name", name)
   184  			chaos.Finalizers = finalizer.RemoveFromFinalizer(chaos.Finalizers, key)
   185  			continue
   186  		}
   187  
   188  		err = r.recoverPod(ctx, &pod, chaos)
   189  		if err != nil {
   190  			result = multierror.Append(result, err)
   191  			continue
   192  		}
   193  
   194  		chaos.Finalizers = finalizer.RemoveFromFinalizer(chaos.Finalizers, key)
   195  	}
   196  
   197  	if chaos.Annotations[common.AnnotationCleanFinalizer] == common.AnnotationCleanFinalizerForced {
   198  		r.Log.Info("Force cleanup all finalizers", "chaos", chaos)
   199  		chaos.Finalizers = chaos.Finalizers[:0]
   200  		return nil
   201  	}
   202  
   203  	return result
   204  }

   206  func (r *endpoint) recoverPod(ctx context.Context, pod *v1.Pod, chaos *v1alpha1.JVMChaos) error {
   207  	r.Log.Info("Try to recover pod", "namespace", pod.Namespace, "name", pod.Name)
   208  
   209  	suid := genSUID(pod, chaos)
   210  	jsonBytes, err := jvm.ToSandboxAction(suid, chaos)
   211  	if err != nil {
   212  		return err
   213  	}
   214  
   215  	// TODO: Custom port may be required
   216  	err = jvm.RecoverChaos(pod.Status.PodIP, sandboxPort, jsonBytes)
   217  	if err != nil {
   218  		return err
   219  	}
   220  	return nil
   221  }
