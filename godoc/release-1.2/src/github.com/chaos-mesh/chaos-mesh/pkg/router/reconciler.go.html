    51  // Reconcile reconciles a chaos resource
    52  func (r *Reconciler) Reconcile(req ctrl.Request) (result ctrl.Result, err error) {
    53  	if !r.ClusterScoped && req.Namespace != r.TargetNamespace {
    54  		// NOOP
    55  		r.Log.Info("ignore chaos which belongs to an unexpected namespace within namespace scoped mode",
    56  			"chaosName", req.Name, "expectedNamespace", r.TargetNamespace, "actualNamespace", req.Namespace)
    57  		return ctrl.Result{}, nil
    58  	}
    59  
    60  	ctx := r.Context.LogWithValues("reconciler", r.Name, "resource name", req.NamespacedName)
    61  
    62  	chaos, ok := r.Object.DeepCopyObject().(v1alpha1.InnerSchedulerObject)
    63  	if !ok {
    64  		err := errors.New("object is not InnerSchedulerObject")
    65  		r.Log.Error(err, "object is not InnerSchedulerObject", "object", r.Object.DeepCopyObject())
    66  		return ctrl.Result{}, err
    67  	}
    68  
    69  	if err := r.Client.Get(context.Background(), req.NamespacedName, chaos); err != nil {
    70  		if apierrors.IsNotFound(err) {
    71  			r.Log.Info("chaos not found")
    72  		} else {
    73  			r.Log.Error(err, "unable to get chaos")
    74  		}
    75  		return ctrl.Result{}, nil
    76  	}
    77  
    78  	scheduler := chaos.GetScheduler()
    79  	duration, err := chaos.GetDuration()
    80  	if err != nil {
    81  		r.Log.Error(err, fmt.Sprintf("unable to get chaos[%s/%s]'s duration", chaos.GetChaos().Namespace, chaos.GetChaos().Name))
    82  		return ctrl.Result{}, err
    83  	}
    84  
    85  	var controller end.Endpoint
    86  	for _, end := range r.Endpoints {
    87  		if end.RouteFunc(chaos.(runtime.Object)) {
    88  			controller = end.NewEndpoint(ctx)
    89  		}
    90  	}
    91  	if controller == nil {
    92  		err := errors.Errorf("cannot route object to one of the endpoint")
    93  		r.Log.Error(err, "fail to route to endpoint", "object", chaos, "endpoints", r.Endpoints)
    94  		return ctrl.Result{}, err
    95  	}
    96  
    97  	var reconciler reconcile.Reconciler
    98  	if scheduler == nil && duration == nil {
    99  		reconciler = common.NewReconciler(req, controller, ctx)
   100  	} else if scheduler != nil {
   101  		// scheduler != nil && duration != nil
   102  		// but PodKill is an expection
   103  		reconciler = twophase.NewReconciler(req, controller, ctx)
   104  	} else {
   105  		err := errors.Errorf("both scheduler and duration should be nil or not nil")
   106  		r.Log.Error(err, "fail to construct reconciler", "scheduler", scheduler, "duration", duration)
   107  		return ctrl.Result{}, err
   108  	}
   109  
   110  	result, err = reconciler.Reconcile(req)
   111  	if err != nil {
   112  		if chaos.IsDeleted() || chaos.IsPaused() {
   113  			r.Event(chaos, v1.EventTypeWarning, events.ChaosRecoverFailed, err.Error())
   114  		} else {
   115  			r.Event(chaos, v1.EventTypeWarning, events.ChaosInjectFailed, err.Error())
   116  		}
   117  	}
   118  	return result, nil
   119  }

   121  // NewReconciler creates a new reconciler
   122  func NewReconciler(name string, object runtime.Object, c client.Client, mgr ctrl.Manager, endpoints []routeEndpoint, clusterScoped bool, targetNamespace string) *Reconciler {
   123  	return &Reconciler{
   124  		Name:            name,
   125  		Object:          object,
   126  		Endpoints:       endpoints,
   127  		ClusterScoped:   clusterScoped,
   128  		TargetNamespace: targetNamespace,
   129  
   130  		Context: ctx.Context{
   131  			Client:        c,
   132  			Reader:        mgr.GetAPIReader(),
   133  			EventRecorder: mgr.GetEventRecorderFor(name + "-controller"),
   134  			Log:           ctrl.Log.WithName("controllers").WithName(name),
   135  		},
   136  	}
   137  }

   139  // SetupWithManager registers controller to manager
   140  func (r *Reconciler) SetupWithManager(mgr ctrl.Manager) error {
   141  	err := ctrl.NewControllerManagedBy(mgr).
   142  		For(r.Object.DeepCopyObject()).
   143  		WithEventFilter(predicate.Funcs{
   144  			UpdateFunc: func(e event.UpdateEvent) bool {
   145  				old, _ := e.ObjectOld.(v1alpha1.InnerObject).GetSpecAndMetaString()
   146  				new, _ := e.ObjectNew.(v1alpha1.InnerObject).GetSpecAndMetaString()
   147  
   148  				return old != new
   149  			},
   150  		}).
   151  		Complete(r)
   152  
   153  	if err != nil {
   154  		return err
   155  	}
   156  
   157  	kind, err := apiutil.GVKForObject(r.Object.DeepCopyObject(), mgr.GetScheme())
   158  	if err != nil {
   159  		return err
   160  	}
   161  
   162  	return ctrl.NewControllerManagedBy(mgr).
   163  		For(r.Object.DeepCopyObject()).
   164  		Named(strings.ToLower(kind.Kind) + "-scheduler-updater").
   165  		WithEventFilter(predicate.Funcs{
   166  			CreateFunc: func(_ event.CreateEvent) bool {
   167  				return false
   168  			},
   169  			DeleteFunc: func(_ event.DeleteEvent) bool {
   170  				return false
   171  			},
   172  			GenericFunc: func(_ event.GenericEvent) bool {
   173  				return false
   174  			},
   175  			UpdateFunc: func(e event.UpdateEvent) bool {
   176  				old := e.ObjectOld.(v1alpha1.InnerSchedulerObject).GetScheduler()
   177  				new := e.ObjectNew.(v1alpha1.InnerSchedulerObject).GetScheduler()
   178  
   179  				if (old == nil) || (new == nil) {
   180  					return false
   181  				}
   182  
   183  				return old.Cron != new.Cron
   184  			},
   185  		}).
   186  		Complete(&twophase.SchedulerUpdater{
   187  			Context: r.Context,
   188  			Object:  r.Object,
   189  		})
   190  }
