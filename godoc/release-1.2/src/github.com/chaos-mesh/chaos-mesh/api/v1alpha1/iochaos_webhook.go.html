    72  // Validate validates chaos object
    73  func (in *IoChaos) Validate() error {
    74  	specField := field.NewPath("spec")
    75  	allErrs := in.ValidateScheduler(specField)
    76  	allErrs = append(allErrs, in.ValidatePodMode(specField)...)
    77  	allErrs = append(allErrs, in.Spec.validateDelay(specField.Child("delay"))...)
    78  	allErrs = append(allErrs, in.Spec.validateErrno(specField.Child("errno"))...)
    79  	allErrs = append(allErrs, in.Spec.validatePercent(specField.Child("percent"))...)
    80  	allErrs = append(allErrs, in.Spec.validateMistake(specField.Child("mistake"))...)
    81  
    82  	if len(allErrs) > 0 {
    83  		return fmt.Errorf(allErrs.ToAggregate().Error())
    84  	}
    85  	return nil
    86  }

   103  func (in *IoChaosSpec) validateDelay(delay *field.Path) field.ErrorList {
   104  	allErrs := field.ErrorList{}
   105  	if in.Action == IoLatency {
   106  		_, err := time.ParseDuration(in.Delay)
   107  		if err != nil {
   108  			allErrs = append(allErrs, field.Invalid(delay, in.Delay,
   109  				fmt.Sprintf("parse delay field error:%s for action:%s", err, in.Action)))
   110  		}
   111  	}
   112  	return allErrs
   113  }
