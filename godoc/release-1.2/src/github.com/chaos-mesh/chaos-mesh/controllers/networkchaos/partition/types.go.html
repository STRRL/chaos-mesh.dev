    65  // Apply applies the chaos operation
    66  func (e *endpoint) Apply(ctx context.Context, req ctrl.Request, chaos v1alpha1.InnerObject) error {
    67  	e.Log.Info("Applying network partition")
    68  
    69  	networkchaos, ok := chaos.(*v1alpha1.NetworkChaos)
    70  	if !ok {
    71  		err := errors.New("chaos is not NetworkChaos")
    72  		e.Log.Error(err, "chaos is not NetworkChaos", "chaos", chaos)
    73  
    74  		return err
    75  	}
    76  
    77  	source := networkchaos.Namespace + "/" + networkchaos.Name
    78  	m := podnetworkchaosmanager.New(source, e.Log, e.Client)
    79  
    80  	sources, err := selector.SelectAndFilterPods(ctx, e.Client, e.Reader, &networkchaos.Spec, config.ControllerCfg.ClusterScoped, config.ControllerCfg.TargetNamespace, config.ControllerCfg.EnableFilterNamespace)
    81  
    82  	if err != nil {
    83  		e.Log.Error(err, "failed to select and filter source pods")
    84  		return err
    85  	}
    86  
    87  	var targets []v1.Pod
    88  
    89  	if networkchaos.Spec.Target != nil {
    90  		targets, err = selector.SelectAndFilterPods(ctx, e.Client, e.Reader, networkchaos.Spec.Target, config.ControllerCfg.ClusterScoped, config.ControllerCfg.TargetNamespace, config.ControllerCfg.EnableFilterNamespace)
    91  		if err != nil {
    92  			e.Log.Error(err, "failed to select and filter target pods")
    93  			return err
    94  		}
    95  	}
    96  
    97  	sourceSet := ipset.BuildIPSet(sources, []string{}, networkchaos, sourceIPSetPostFix, source)
    98  	externalCidrs, err := netutils.ResolveCidrs(networkchaos.Spec.ExternalTargets)
    99  	if err != nil {
   100  		e.Log.Error(err, "failed to resolve external targets")
   101  		return err
   102  	}
   103  	targetSet := ipset.BuildIPSet(targets, externalCidrs, networkchaos, targetIPSetPostFix, source)
   104  
   105  	allPods := append(sources, targets...)
   106  
   107  	keyPodMap := make(map[types.NamespacedName]podPositionTuple)
   108  	for index, pod := range allPods {
   109  		position := ""
   110  		if index < len(sources) {
   111  			position = "source"
   112  		} else {
   113  			position = "target"
   114  		}
   115  		keyPodMap[types.NamespacedName{
   116  			Name:      pod.Name,
   117  			Namespace: pod.Namespace,
   118  		}] = podPositionTuple{
   119  			Pod:      pod,
   120  			Position: position,
   121  		}
   122  	}
   123  
   124  	// Set up ipset in every related pods
   125  	for index := range allPods {
   126  		pod := allPods[index]
   127  		e.Log.Info("PODS", "name", pod.Name, "namespace", pod.Namespace)
   128  
   129  		t := m.WithInit(types.NamespacedName{
   130  			Name:      pod.Name,
   131  			Namespace: pod.Namespace,
   132  		})
   133  
   134  		t.Append(sourceSet)
   135  		t.Append(targetSet)
   136  	}
   137  
   138  	sourcesChains := []v1alpha1.RawIptables{}
   139  	targetsChains := []v1alpha1.RawIptables{}
   140  	if networkchaos.Spec.Direction == v1alpha1.To || networkchaos.Spec.Direction == v1alpha1.Both {
   141  		sourcesChains = append(sourcesChains, v1alpha1.RawIptables{
   142  			Name:      iptable.GenerateName(pb.Chain_OUTPUT, networkchaos),
   143  			Direction: v1alpha1.Output,
   144  			IPSets:    []string{targetSet.Name},
   145  			RawRuleSource: v1alpha1.RawRuleSource{
   146  				Source: source,
   147  			},
   148  		})
   149  
   150  		targetsChains = append(targetsChains, v1alpha1.RawIptables{
   151  			Name:      iptable.GenerateName(pb.Chain_INPUT, networkchaos),
   152  			Direction: v1alpha1.Input,
   153  			IPSets:    []string{sourceSet.Name},
   154  			RawRuleSource: v1alpha1.RawRuleSource{
   155  				Source: source,
   156  			},
   157  		})
   158  	}
   159  
   160  	if networkchaos.Spec.Direction == v1alpha1.From || networkchaos.Spec.Direction == v1alpha1.Both {
   161  		sourcesChains = append(sourcesChains, v1alpha1.RawIptables{
   162  			Name:      iptable.GenerateName(pb.Chain_INPUT, networkchaos),
   163  			Direction: v1alpha1.Input,
   164  			IPSets:    []string{targetSet.Name},
   165  			RawRuleSource: v1alpha1.RawRuleSource{
   166  				Source: source,
   167  			},
   168  		})
   169  
   170  		targetsChains = append(targetsChains, v1alpha1.RawIptables{
   171  			Name:      iptable.GenerateName(pb.Chain_OUTPUT, networkchaos),
   172  			Direction: v1alpha1.Output,
   173  			IPSets:    []string{sourceSet.Name},
   174  			RawRuleSource: v1alpha1.RawRuleSource{
   175  				Source: source,
   176  			},
   177  		})
   178  	}
   179  	e.Log.Info("chains prepared", "sourcesChains", sourcesChains, "targetsChains", targetsChains)
   180  
   181  	err = e.SetChains(ctx, sources, sourcesChains, m, networkchaos)
   182  	if err != nil {
   183  		return err
   184  	}
   185  
   186  	err = e.SetChains(ctx, targets, targetsChains, m, networkchaos)
   187  	if err != nil {
   188  		return err
   189  	}
   190  
   191  	responses := m.Commit(ctx)
   192  
   193  	var errors error
   194  	networkchaos.Status.Experiment.PodRecords = make([]v1alpha1.PodStatus, 0, len(allPods))
   195  	for _, keyErrorTuple := range responses {
   196  		key := keyErrorTuple.Key
   197  		err := keyErrorTuple.Err
   198  		if err != nil {
   199  			if err != podnetworkchaosmanager.ErrPodNotFound && err != podnetworkchaosmanager.ErrPodNotRunning {
   200  				e.Log.Error(err, "fail to commit")
   201  				errors = multierror.Append(errors, err)
   202  			} else {
   203  				e.Log.Info("pod is not found or not running", "key", key)
   204  			}
   205  			continue
   206  		}
   207  
   208  		pod := keyPodMap[keyErrorTuple.Key]
   209  		ps := v1alpha1.PodStatus{
   210  			Namespace: pod.Pod.Namespace,
   211  			Name:      pod.Pod.Name,
   212  			HostIP:    pod.Pod.Status.HostIP,
   213  			PodIP:     pod.Pod.Status.PodIP,
   214  			Action:    string(networkchaos.Spec.Action),
   215  		}
   216  		if pod.Position == "source" {
   217  			ps.Message = networkChaosSourceMsg
   218  		} else {
   219  			ps.Message = networkChaosTargetMsg
   220  		}
   221  
   222  		// TODO: add source, target and tc action message
   223  		if networkchaos.Spec.Duration != nil {
   224  			ps.Message += fmt.Sprintf(networkPartitionActionMsg, *networkchaos.Spec.Duration)
   225  		}
   226  		networkchaos.Status.Experiment.PodRecords = append(networkchaos.Status.Experiment.PodRecords, ps)
   227  	}
   228  
   229  	if errors != nil {
   230  		return errors
   231  	}
   232  
   233  	e.Event(networkchaos, v1.EventTypeNormal, events.ChaosInjected, "")
   234  	return nil
   235  }

   237  // SetChains sets iptables chains for pods
   238  func (e *endpoint) SetChains(ctx context.Context, pods []v1.Pod, chains []v1alpha1.RawIptables, m *podnetworkchaosmanager.PodNetworkManager, networkchaos *v1alpha1.NetworkChaos) error {
   239  	for index := range pods {
   240  		pod := &pods[index]
   241  
   242  		t := m.WithInit(types.NamespacedName{
   243  			Name:      pod.Name,
   244  			Namespace: pod.Namespace,
   245  		})
   246  		for _, chain := range chains {
   247  			t.Append(chain)
   248  		}
   249  	}
   250  	return nil
   251  }

   253  // Recover means the reconciler recovers the chaos action
   254  func (e *endpoint) Recover(ctx context.Context, req ctrl.Request, chaos v1alpha1.InnerObject) error {
   255  	networkchaos, ok := chaos.(*v1alpha1.NetworkChaos)
   256  	if !ok {
   257  		err := errors.New("chaos is not NetworkChaos")
   258  		e.Log.Error(err, "chaos is not NetworkChaos", "chaos", chaos)
   259  		return err
   260  	}
   261  
   262  	if err := e.cleanFinalizersAndRecover(ctx, networkchaos); err != nil {
   263  		return err
   264  	}
   265  	e.Event(networkchaos, v1.EventTypeNormal, events.ChaosRecovered, "")
   266  
   267  	return nil
   268  }

   270  func (e *endpoint) cleanFinalizersAndRecover(ctx context.Context, chaos *v1alpha1.NetworkChaos) error {
   271  	var result error
   272  
   273  	source := chaos.Namespace + "/" + chaos.Name
   274  	m := podnetworkchaosmanager.New(source, e.Log, e.Client)
   275  
   276  	keyRecordMap := make(map[types.NamespacedName]v1alpha1.PodStatus)
   277  	for _, podRecord := range chaos.Status.Experiment.PodRecords {
   278  		key := types.NamespacedName{
   279  			Namespace: podRecord.Namespace,
   280  			Name:      podRecord.Name,
   281  		}
   282  		_ = m.WithInit(key)
   283  		keyRecordMap[key] = podRecord
   284  	}
   285  	responses := m.Commit(ctx)
   286  
   287  	restRecord := []v1alpha1.PodStatus{}
   288  	for _, response := range responses {
   289  		key := response.Key
   290  		err := response.Err
   291  		// if pod not found or not running, directly return and giveup recover.
   292  		if err != nil {
   293  			if err != podnetworkchaosmanager.ErrPodNotFound && err != podnetworkchaosmanager.ErrPodNotRunning {
   294  				e.Log.Error(err, "fail to commit", "key", key)
   295  
   296  				result = multierror.Append(result, err)
   297  				restRecord = append(restRecord, keyRecordMap[key])
   298  				continue
   299  			}
   300  
   301  			e.Log.Info("pod is not found or not running", "key", key)
   302  		}
   303  	}
   304  
   305  	if len(restRecord) == 0 {
   306  		chaos.Finalizers = []string{}
   307  	}
   308  	e.Log.Info("After recovering", "finalizers", chaos.Finalizers)
   309  
   310  	if chaos.Annotations[common.AnnotationCleanFinalizer] == common.AnnotationCleanFinalizerForced {
   311  		e.Log.Info("Force cleanup all finalizers", "chaos", chaos)
   312  		chaos.Finalizers = make([]string, 0)
   313  		return nil
   314  	}
   315  
   316  	return result
   317  }
