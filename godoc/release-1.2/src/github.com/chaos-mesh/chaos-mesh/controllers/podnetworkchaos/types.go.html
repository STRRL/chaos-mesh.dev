    48  // Apply flushes network configuration on pod
    49  func (h *Handler) Apply(ctx context.Context, chaos *v1alpha1.PodNetworkChaos) error {
    50  	h.Log.Info("updating network chaos", "pod", chaos.Namespace+"/"+chaos.Name, "spec", chaos.Spec)
    51  
    52  	pod := &corev1.Pod{}
    53  
    54  	err := h.Client.Get(ctx, types.NamespacedName{
    55  		Name:      chaos.Name,
    56  		Namespace: chaos.Namespace,
    57  	}, pod)
    58  	if err != nil {
    59  		h.Log.Error(err, "fail to find pod")
    60  		return err
    61  	}
    62  
    63  	if !h.AllowHostNetworkTesting {
    64  		if pod.Spec.HostNetwork {
    65  			err := errors.Errorf("it's dangerous to inject network chaos on a pod(%s/%s) with `hostNetwork`", pod.Namespace, pod.Name)
    66  			return err
    67  		}
    68  	}
    69  
    70  	err = h.SetIPSets(ctx, pod, chaos)
    71  	if err != nil {
    72  		return err
    73  	}
    74  
    75  	err = h.SetIptables(ctx, pod, chaos)
    76  	if err != nil {
    77  		return err
    78  	}
    79  
    80  	err = h.SetTcs(ctx, pod, chaos)
    81  	if err != nil {
    82  		return err
    83  	}
    84  
    85  	return nil
    86  }

   100  // SetIptables sets iptables on pod
   101  func (h *Handler) SetIptables(ctx context.Context, pod *corev1.Pod, chaos *v1alpha1.PodNetworkChaos) error {
   102  	chains := []*pb.Chain{}
   103  	for _, chain := range chaos.Spec.Iptables {
   104  		var direction pb.Chain_Direction
   105  		if chain.Direction == v1alpha1.Input {
   106  			direction = pb.Chain_INPUT
   107  		} else if chain.Direction == v1alpha1.Output {
   108  			direction = pb.Chain_OUTPUT
   109  		} else {
   110  			err := fmt.Errorf("unknown direction %s", string(chain.Direction))
   111  			h.Log.Error(err, "unknown direction")
   112  			return err
   113  		}
   114  		chains = append(chains, &pb.Chain{
   115  			Name:      chain.Name,
   116  			Ipsets:    chain.IPSets,
   117  			Direction: direction,
   118  			Target:    "DROP",
   119  		})
   120  	}
   121  	return iptable.SetIptablesChains(ctx, h.Client, pod, chains)
   122  }

   124  // SetTcs sets traffic control related chaos on pod
   125  func (h *Handler) SetTcs(ctx context.Context, pod *corev1.Pod, chaos *v1alpha1.PodNetworkChaos) error {
   126  	tcs := []*pb.Tc{}
   127  	for _, tc := range chaos.Spec.TrafficControls {
   128  		if tc.Type == v1alpha1.Bandwidth {
   129  			tbf, err := netem.FromBandwidth(tc.Bandwidth)
   130  			if err != nil {
   131  				return err
   132  			}
   133  			tcs = append(tcs, &pb.Tc{
   134  				Type:  pb.Tc_BANDWIDTH,
   135  				Tbf:   tbf,
   136  				Ipset: tc.IPSet,
   137  			})
   138  		} else if tc.Type == v1alpha1.Netem {
   139  			netem, err := mergeNetem(tc.TcParameter)
   140  			if err != nil {
   141  				return err
   142  			}
   143  			tcs = append(tcs, &pb.Tc{
   144  				Type:  pb.Tc_NETEM,
   145  				Netem: netem,
   146  				Ipset: tc.IPSet,
   147  			})
   148  		} else {
   149  			return fmt.Errorf("unknown tc type")
   150  		}
   151  	}
   152  
   153  	h.Log.Info("setting tcs", "tcs", tcs)
   154  	return tcpkg.SetTcs(ctx, h.Client, pod, tcs)
   155  }

   162  // mergeNetem calls ToNetem on all non nil network emulation specs and merges them into one request.
   163  func mergeNetem(spec v1alpha1.TcParameter) (*pb.Netem, error) {
   164  	// NOTE: a cleaner way like
   165  	// emSpecs = []NetemSpec{spec.Delay, spec.Loss} won't work.
   166  	// Because in the for _, spec := range emSpecs loop,
   167  	// spec != nil would always be true.
   168  	// See https://stackoverflow.com/questions/13476349/check-for-nil-and-nil-interface-in-go
   169  	// And https://groups.google.com/forum/#!topic/golang-nuts/wnH302gBa4I/discussion
   170  	// > In short: If you never store (*T)(nil) in an interface, then you can reliably use comparison against nil
   171  	var emSpecs []*pb.Netem
   172  	if spec.Delay != nil {
   173  		em, err := netem.FromDelay(spec.Delay)
   174  		if err != nil {
   175  			return nil, err
   176  		}
   177  		emSpecs = append(emSpecs, em)
   178  	}
   179  	if spec.Loss != nil {
   180  		em, err := netem.FromLoss(spec.Loss)
   181  		if err != nil {
   182  			return nil, err
   183  		}
   184  		emSpecs = append(emSpecs, em)
   185  	}
   186  	if spec.Duplicate != nil {
   187  		em, err := netem.FromDuplicate(spec.Duplicate)
   188  		if err != nil {
   189  			return nil, err
   190  		}
   191  		emSpecs = append(emSpecs, em)
   192  	}
   193  	if spec.Corrupt != nil {
   194  		em, err := netem.FromCorrupt(spec.Corrupt)
   195  		if err != nil {
   196  			return nil, err
   197  		}
   198  		emSpecs = append(emSpecs, em)
   199  	}
   200  	if len(emSpecs) == 0 {
   201  		return nil, errors.New(invalidNetemSpecMsg)
   202  	}
   203  
   204  	merged := &pb.Netem{}
   205  	for _, em := range emSpecs {
   206  		merged = pbutils.MergeNetem(merged, em)
   207  	}
   208  	return merged, nil
   209  }
