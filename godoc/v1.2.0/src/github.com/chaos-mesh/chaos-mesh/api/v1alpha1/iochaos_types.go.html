    35  // IoChaosSpec defines the desired state of IoChaos
    36  type IoChaosSpec struct {
    37  	// Selector is used to select pods that are used to inject chaos action.
    38  	Selector SelectorSpec `json:"selector"`
    39  
    40  	// Mode defines the mode to run chaos action.
    41  	// Supported mode: one / all / fixed / fixed-percent / random-max-percent
    42  	// +kubebuilder:validation:Enum=one;all;fixed;fixed-percent;random-max-percent
    43  	Mode PodMode `json:"mode"`
    44  
    45  	// Value is required when the mode is set to `FixedPodMode` / `FixedPercentPodMod` / `RandomMaxPercentPodMod`.
    46  	// If `FixedPodMode`, provide an integer of pods to do chaos action.
    47  	// If `FixedPercentPodMod`, provide a number from 0-100 to specify the percent of pods the server can do chaos action.
    48  	// IF `RandomMaxPercentPodMod`,  provide a number from 0-100 to specify the max percent of pods to do chaos action
    49  	// +optional
    50  	Value string `json:"value"`
    51  
    52  	// Action defines the specific pod chaos action.
    53  	// Supported action: latency / fault / attrOverride / mistake
    54  	// +kubebuilder:validation:Enum=latency;fault;attrOverride;mistake
    55  	Action IoChaosType `json:"action"`
    56  
    57  	// Delay defines the value of I/O chaos action delay.
    58  	// A delay string is a possibly signed sequence of
    59  	// decimal numbers, each with optional fraction and a unit suffix,
    60  	// such as "300ms".
    61  	// Valid time units are "ns", "us" (or "µs"), "ms", "s", "m", "h".
    62  	// +optional
    63  	Delay string `json:"delay,omitempty"`
    64  
    65  	// Errno defines the error code that returned by I/O action.
    66  	// refer to: https://www-numi.fnal.gov/offline_software/srt_public_context/WebDocs/Errors/unix_system_errors.html
    67  	// +optional
    68  	Errno uint32 `json:"errno,omitempty"`
    69  
    70  	// Attr defines the overrided attribution
    71  	// +optional
    72  	Attr *AttrOverrideSpec `json:"attr,omitempty"`
    73  
    74  	// Mistake defines what types of incorrectness are injected to IO operations
    75  	// +optional
    76  	Mistake *MistakeSpec `json:"mistake,omitempty"`
    77  
    78  	// Path defines the path of files for injecting I/O chaos action.
    79  	// +optional
    80  	Path string `json:"path,omitempty"`
    81  
    82  	// Methods defines the I/O methods for injecting I/O chaos action.
    83  	// default: all I/O methods.
    84  	// +optional
    85  	Methods []IoMethod `json:"methods,omitempty" faker:"ioMethods"`
    86  
    87  	// Percent defines the percentage of injection errors and provides a number from 0-100.
    88  	// default: 100.
    89  	// +optional
    90  	Percent int `json:"percent,omitempty"`
    91  
    92  	// VolumePath represents the mount path of injected volume
    93  	VolumePath string `json:"volumePath"`
    94  
    95  	// ContainerName indicates the target container to inject iochaos in
    96  	// +optional
    97  	ContainerName *string `json:"containerName,omitempty"`
    98  
    99  	// Scheduler defines some schedule rules to
   100  	// control the running time of the chaos experiment about pods.
   101  	Scheduler *SchedulerSpec `json:"scheduler,omitempty"`
   102  
   103  	// Duration represents the duration of the chaos action.
   104  	// It is required when the action is `PodFailureAction`.
   105  	// A duration string is a possibly signed sequence of
   106  	// decimal numbers, each with optional fraction and a unit suffix,
   107  	// such as "300ms", "-1.5h" or "2h45m".
   108  	// Valid time units are "ns", "us" (or "µs"), "ms", "s", "m", "h".
   109  	// +optional
   110  	Duration *string `json:"duration,omitempty"`
   111  }
