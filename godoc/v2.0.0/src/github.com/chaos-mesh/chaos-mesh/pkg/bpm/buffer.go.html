    71  func (br *blockingBuffer) Write(b []byte) (ln int, err error) {
    72  	if br.closed {
    73  		return 0, nil
    74  	}
    75  	ln, err = br.buf.Write(b)
    76  	br.cond.Broadcast()
    77  	return
    78  }

    80  func (br *blockingBuffer) Read(b []byte) (ln int, err error) {
    81  	if br.closed {
    82  		return 0, io.EOF
    83  	}
    84  	ln, err = br.buf.Read(b)
    85  	for err == io.EOF {
    86  		br.cond.L.Lock()
    87  		if br.closed {
    88  			return 0, io.EOF
    89  		}
    90  		br.cond.Wait()
    91  		br.cond.L.Unlock()
    92  		ln, err = br.buf.Read(b)
    93  	}
    94  	return
    95  }

   103  func (cb *concurrentBuffer) start() {
   104  	for {
   105  		if cb.buf.Len() == 0 {
   106  			select {
   107  			case data := <-cb.writeChan:
   108  				cb.buf.Write(data)
   109  				continue
   110  			case <-cb.ctx.Done():
   111  				return
   112  			}
   113  		}
   114  
   115  		select {
   116  		case data := <-cb.writeChan:
   117  			cb.buf.Write(data)
   118  		case <-cb.ctx.Done():
   119  			return
   120  		case ctx := <-cb.readChan:
   121  			ln, err := cb.buf.Read(ctx.data)
   122  			ctx.ret <- readRet{
   123  				ln:  ln,
   124  				err: err,
   125  			}
   126  		}
   127  	}
   128  }

   131  	select {
   132  	case <-cb.ctx.Done():

   139  func (cb *concurrentBuffer) Read(b []byte) (ln int, err error) {
   140  	ret := make(chan readRet)
   141  
   142  	select {
   143  	case <-cb.ctx.Done():
   144  		return 0, io.EOF

   151  			return 0, io.EOF
   152  		case result := <-ret:
   153  			return result.ln, result.err
   154  		}
   155  	}
   156  }
