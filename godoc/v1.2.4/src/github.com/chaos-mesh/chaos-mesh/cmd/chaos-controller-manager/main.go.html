    93  func main() {
    94  	parseFlags()
    95  	version.PrintVersionInfo("Controller manager")
    96  	if printVersion {
    97  		os.Exit(0)
    98  	}
    99  
   100  	// set RPCTimeout config
   101  	grpcUtils.RPCTimeout = ccfg.ControllerCfg.RPCTimeout
   102  
   103  	ctrl.SetLogger(zap.New(zap.UseDevMode(true)))
   104  
   105  	options := ctrl.Options{
   106  		Scheme:             scheme,
   107  		MetricsBindAddress: ccfg.ControllerCfg.MetricsAddr,
   108  		LeaderElection:     ccfg.ControllerCfg.EnableLeaderElection,
   109  		Port:               9443,
   110  	}
   111  
   112  	if ccfg.ControllerCfg.ClusterScoped {
   113  		setupLog.Info("Chaos controller manager is running in cluster scoped mode.")
   114  		// will not specific a certain namespace
   115  	} else {
   116  		setupLog.Info("Chaos controller manager is running in namespace scoped mode.", "targetNamespace", ccfg.ControllerCfg.TargetNamespace)
   117  		options.Namespace = ccfg.ControllerCfg.TargetNamespace
   118  	}
   119  
   120  	cfg := ctrl.GetConfigOrDie()
   121  	setRestConfig(cfg)
   122  	mgr, err := ctrl.NewManager(cfg, options)
   123  	if err != nil {
   124  		setupLog.Error(err, "unable to start manager")
   125  		os.Exit(1)
   126  	}
   127  
   128  	updatedClient, err := updatedclient.New(mgr, scheme)
   129  	if err != nil {
   130  		setupLog.Error(err, "fail to setup updated client")
   131  		os.Exit(1)
   132  	}
   133  
   134  	authCli, err := authorizationv1.NewForConfig(cfg)
   135  	if err != nil {
   136  		setupLog.Error(err, "unable to get authorization client")
   137  		os.Exit(1)
   138  	}
   139  
   140  	err = router.SetupWithManagerAndConfigs(updatedClient, mgr, ccfg.ControllerCfg)
   141  	if err != nil {
   142  		setupLog.Error(err, "fail to setup with manager")
   143  		os.Exit(1)
   144  	}
   145  
   146  	// We only setup webhook for podiochaos, and the logic of applying chaos are in the mutation
   147  	// webhook, because we need to get the running result synchronously in io chaos reconciler
   148  	v1alpha1.RegisterPodIoHandler(&podiochaos.Handler{
   149  		Client: updatedClient,
   150  		Log:    ctrl.Log.WithName("handler").WithName("PodIOChaos"),
   151  	})
   152  	if err = (&v1alpha1.PodIoChaos{}).SetupWebhookWithManager(mgr); err != nil {
   153  		setupLog.Error(err, "unable to create webhook", "webhook", "PodIOChaos")
   154  		os.Exit(1)
   155  	}
   156  
   157  	// We only setup webhook for podnetworkchaos, and the logic of applying chaos are in the validation
   158  	// webhook, because we need to get the running result synchronously in network chaos reconciler
   159  	v1alpha1.RegisterRawPodNetworkHandler(&podnetworkchaos.Handler{
   160  		Client:                  updatedClient,
   161  		Reader:                  mgr.GetAPIReader(),
   162  		Log:                     ctrl.Log.WithName("handler").WithName("PodNetworkChaos"),
   163  		AllowHostNetworkTesting: ccfg.ControllerCfg.AllowHostNetworkTesting,
   164  	})
   165  	if err = (&v1alpha1.PodNetworkChaos{}).SetupWebhookWithManager(mgr); err != nil {
   166  		setupLog.Error(err, "unable to create webhook", "webhook", "PodNetworkChaos")
   167  		os.Exit(1)
   168  	}
   169  
   170  	// Init metrics collector
   171  	metricsCollector := metrics.NewChaosCollector(mgr.GetCache(), controllermetrics.Registry)
   172  
   173  	setupLog.Info("Setting up webhook server")
   174  	hookServer := mgr.GetWebhookServer()
   175  	hookServer.CertDir = ccfg.ControllerCfg.CertsDir
   176  	conf := config.NewConfigWatcherConf()
   177  
   178  	stopCh := ctrl.SetupSignalHandler()
   179  
   180  	if ccfg.ControllerCfg.PprofAddr != "0" {
   181  		go func() {
   182  			if err := http.ListenAndServe(ccfg.ControllerCfg.PprofAddr, nil); err != nil {
   183  				setupLog.Error(err, "unable to start pprof server")
   184  				os.Exit(1)
   185  			}
   186  		}()
   187  	}
   188  
   189  	if err = ccfg.ControllerCfg.WatcherConfig.Verify(); err != nil {
   190  		setupLog.Error(err, "invalid environment configuration")
   191  		os.Exit(1)
   192  	}
   193  	configWatcher, err := watcher.New(*ccfg.ControllerCfg.WatcherConfig, metricsCollector)
   194  	if err != nil {
   195  		setupLog.Error(err, "unable to create config watcher")
   196  		os.Exit(1)
   197  	}
   198  
   199  	go watchConfig(configWatcher, conf, stopCh)
   200  	hookServer.Register("/inject-v1-pod", &webhook.Admission{
   201  		Handler: &apiWebhook.PodInjector{
   202  			Config:        conf,
   203  			ControllerCfg: ccfg.ControllerCfg,
   204  			Metrics:       metricsCollector,
   205  		}},
   206  	)
   207  
   208  	hookServer.Register("/validate-auth", &webhook.Admission{
   209  		Handler: apiWebhook.NewAuthValidator(ccfg.ControllerCfg.SecurityMode, updatedClient, mgr.GetAPIReader(), authCli,
   210  			ccfg.ControllerCfg.ClusterScoped, ccfg.ControllerCfg.TargetNamespace, ccfg.ControllerCfg.EnableFilterNamespace),
   211  	},
   212  	)
   213  
   214  	// +kubebuilder:scaffold:builder
   215  
   216  	setupLog.Info("Starting manager")
   217  	if err := mgr.Start(stopCh); err != nil {
   218  		setupLog.Error(err, "unable to start manager")
   219  		os.Exit(1)
   220  	}
   221  }

   232  func setupWatchQueue(stopCh <-chan struct{}, configWatcher *watcher.K8sConfigMapWatcher) workqueue.Interface {
   233  	// watch for reconciliation signals, and grab configmaps, then update the running configuration
   234  	// for the server
   235  	sigChan := make(chan interface{}, 10)
   236  
   237  	queue := workqueue.NewRateLimitingQueue(&workqueue.BucketRateLimiter{Limiter: rate.NewLimiter(rate.Limit(0.5), 1)})
   238  
   239  	go func() {
   240  		for {
   241  			select {
   242  			case <-stopCh:
   243  				queue.ShutDown()
   244  				return
   245  			case <-sigChan:
   246  				queue.AddRateLimited(struct{}{})
   247  			}
   248  		}
   249  	}()
   250  
   251  	go func() {
   252  		for {
   253  			setupLog.Info("Launching watcher for ConfigMaps")
   254  			if err := configWatcher.Watch(sigChan, stopCh); err != nil {
   255  				switch err {
   256  				case watcher.ErrWatchChannelClosed:
   257  					// known issue: https://github.com/kubernetes/client-go/issues/334
   258  					setupLog.Info("watcher channel has closed, restart watcher")
   259  				default:
   260  					setupLog.Error(err, "unable to watch new ConfigMaps")
   261  					os.Exit(1)
   262  				}
   263  			}
   264  
   265  			select {
   266  			case <-stopCh:
   267  				close(sigChan)
   268  				return
   269  			default:
   270  				// sleep 2 seconds to prevent excessive log due to infinite restart
   271  				time.Sleep(2 * time.Second)
   272  			}
   273  		}
   274  	}()
   275  
   276  	return queue
   277  }

   279  func watchConfig(configWatcher *watcher.K8sConfigMapWatcher, cfg *config.Config, stopCh <-chan struct{}) {
   280  	queue := setupWatchQueue(stopCh, configWatcher)
   281  
   282  	for {
   283  		item, shutdown := queue.Get()
   284  		if shutdown {
   285  			break
   286  		}
   287  		func() {
   288  			defer queue.Done(item)
   289  
   290  			setupLog.Info("Triggering ConfigMap reconciliation")
   291  			updatedInjectionConfigs, err := configWatcher.GetInjectionConfigs()
   292  			if err != nil {
   293  				setupLog.Error(err, "unable to get ConfigMaps")
   294  				return
   295  			}
   296  
   297  			setupLog.Info("Updating server with newly loaded configurations",
   298  				"original configs count", len(cfg.Injections), "updated configs count", len(updatedInjectionConfigs))
   299  			cfg.ReplaceInjectionConfigs(updatedInjectionConfigs)
   300  			setupLog.Info("Configuration replaced")
   301  		}()
   302  	}
