    44  func (e *endpoint) Apply(ctx context.Context, req ctrl.Request, chaos v1alpha1.InnerObject) error {
    45  	awschaos, ok := chaos.(*v1alpha1.AwsChaos)
    46  	if !ok {
    47  		err := errors.New("chaos is not awschaos")
    48  		e.Log.Error(err, "chaos is not AwsChaos", "chaos", chaos)
    49  		return err
    50  	}
    51  
    52  	opts := []func(*awscfg.LoadOptions) error{
    53  		awscfg.WithRegion(awschaos.Spec.AwsRegion),
    54  	}
    55  
    56  	if awschaos.Spec.Endpoint != nil {
    57  		opts = append(opts, awscfg.WithEndpointResolver(aws.EndpointResolverFunc(func(service, region string) (aws.Endpoint, error) {
    58  			return aws.Endpoint{URL: *awschaos.Spec.Endpoint, SigningRegion: region}, nil
    59  		})))
    60  	}
    61  
    62  	if awschaos.Spec.SecretName != nil {
    63  		secret := &v1.Secret{}
    64  		err := e.Client.Get(ctx, types.NamespacedName{
    65  			Name:      *awschaos.Spec.SecretName,
    66  			Namespace: awschaos.Namespace,
    67  		}, secret)
    68  		if err != nil {
    69  			e.Log.Error(err, "fail to get cloud secret")
    70  			return err
    71  		}
    72  		opts = append(opts, awscfg.WithCredentialsProvider(credentials.NewStaticCredentialsProvider(
    73  			string(secret.Data["aws_access_key_id"]),
    74  			string(secret.Data["aws_secret_access_key"]),
    75  			"",
    76  		)))
    77  	}
    78  	cfg, err := awscfg.LoadDefaultConfig(ctx, opts...)
    79  	if err != nil {
    80  		e.Log.Error(err, "unable to load aws SDK config")
    81  		return err
    82  	}
    83  	ec2client := ec2.NewFromConfig(cfg)
    84  
    85  	awschaos.Finalizers = []string{AwsFinalizer}
    86  	_, err = ec2client.StopInstances(context.TODO(), &ec2.StopInstancesInput{
    87  		InstanceIds: []string{awschaos.Spec.Ec2Instance},
    88  	})
    89  
    90  	if err != nil {
    91  		awschaos.Finalizers = make([]string, 0)
    92  		e.Log.Error(err, "fail to stop the instance")
    93  		return err
    94  	}
    95  
    96  	return nil
    97  }
