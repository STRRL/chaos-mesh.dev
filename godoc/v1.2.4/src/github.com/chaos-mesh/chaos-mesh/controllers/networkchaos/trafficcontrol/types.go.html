    58  // Apply implements the reconciler.InnerReconciler.Apply
    59  func (r *endpoint) Apply(ctx context.Context, req ctrl.Request, chaos v1alpha1.InnerObject) error {
    60  	r.Log.Info("traffic control Apply", "req", req, "chaos", chaos)
    61  
    62  	networkchaos, ok := chaos.(*v1alpha1.NetworkChaos)
    63  	if !ok {
    64  		err := errors.New("chaos is not NetworkChaos")
    65  		r.Log.Error(err, "chaos is not NetworkChaos", "chaos", chaos)
    66  		return err
    67  	}
    68  
    69  	source := networkchaos.Namespace + "/" + networkchaos.Name
    70  	m := podnetworkchaosmanager.New(source, r.Log, r.Client)
    71  
    72  	sources, err := selector.SelectAndFilterPods(ctx, r.Client, r.Reader, &networkchaos.Spec, config.ControllerCfg.ClusterScoped, config.ControllerCfg.TargetNamespace, config.ControllerCfg.EnableFilterNamespace)
    73  	if err != nil {
    74  		r.Log.Error(err, "failed to select and filter source pods")
    75  		return err
    76  	}
    77  
    78  	var targets []v1.Pod
    79  
    80  	// We should only apply filter when we specify targets
    81  	if networkchaos.Spec.Target != nil {
    82  		targets, err = selector.SelectAndFilterPods(ctx, r.Client, r.Reader, networkchaos.Spec.Target, config.ControllerCfg.ClusterScoped, config.ControllerCfg.TargetNamespace, config.ControllerCfg.EnableFilterNamespace)
    83  		if err != nil {
    84  			r.Log.Error(err, "failed to select and filter target pods")
    85  			return err
    86  		}
    87  	}
    88  
    89  	pods := append(sources, targets...)
    90  	type podPositionTuple struct {
    91  		Pod      v1.Pod
    92  		Position string
    93  	}
    94  	keyPodMap := make(map[types.NamespacedName]podPositionTuple)
    95  	for index, pod := range pods {
    96  		position := ""
    97  		if index < len(sources) {
    98  			position = "source"
    99  		} else {
   100  			position = "target"
   101  		}
   102  		keyPodMap[types.NamespacedName{
   103  			Name:      pod.Name,
   104  			Namespace: pod.Namespace,
   105  		}] = podPositionTuple{
   106  			Pod:      pod,
   107  			Position: position,
   108  		}
   109  	}
   110  
   111  	externalCidrs, err := netutils.ResolveCidrs(networkchaos.Spec.ExternalTargets)
   112  	if err != nil {
   113  		r.Log.Error(err, "failed to resolve external targets")
   114  		return err
   115  	}
   116  
   117  	switch networkchaos.Spec.Direction {
   118  	case v1alpha1.To:
   119  		err = r.applyTc(ctx, sources, targets, externalCidrs, targetIPSetPostFix, m, networkchaos)
   120  		if err != nil {
   121  			r.Log.Error(err, "failed to apply traffic control", "sources", sources, "targets", targets)
   122  			return err
   123  		}
   124  	case v1alpha1.From:
   125  		err = r.applyTc(ctx, targets, sources, []string{}, sourceIPSetPostFix, m, networkchaos)
   126  		if err != nil {
   127  			r.Log.Error(err, "failed to apply traffic control", "sources", targets, "targets", sources)
   128  			return err
   129  		}
   130  	case v1alpha1.Both:
   131  		err = r.applyTc(ctx, sources, targets, externalCidrs, targetIPSetPostFix, m, networkchaos)
   132  		if err != nil {
   133  			r.Log.Error(err, "failed to apply traffic control", "sources", sources, "targets", targets)
   134  			return err
   135  		}
   136  
   137  		err = r.applyTc(ctx, targets, sources, []string{}, sourceIPSetPostFix, m, networkchaos)
   138  		if err != nil {
   139  			r.Log.Error(err, "failed to apply traffic control", "sources", targets, "targets", sources)
   140  			return err
   141  		}
   142  	default:
   143  		err = fmt.Errorf("unknown direction %s", networkchaos.Spec.Direction)
   144  		r.Log.Error(err, "unknown direction", "direction", networkchaos.Spec.Direction)
   145  		return err
   146  	}
   147  
   148  	responses := m.Commit(ctx)
   149  
   150  	var errors error
   151  	networkchaos.Status.Experiment.PodRecords = make([]v1alpha1.PodStatus, 0, len(pods))
   152  	for _, keyErrorTuple := range responses {
   153  		key := keyErrorTuple.Key
   154  		err := keyErrorTuple.Err
   155  		if err != nil {
   156  			if err != podnetworkchaosmanager.ErrPodNotFound && err != podnetworkchaosmanager.ErrPodNotRunning {
   157  				r.Log.Error(err, "fail to commit")
   158  				errors = multierror.Append(errors, err)
   159  			} else {
   160  				r.Log.Info("pod is not found or not running", "key", key)
   161  			}
   162  
   163  			continue
   164  		}
   165  
   166  		pod := keyPodMap[keyErrorTuple.Key]
   167  		ps := v1alpha1.PodStatus{
   168  			Namespace: pod.Pod.Namespace,
   169  			Name:      pod.Pod.Name,
   170  			HostIP:    pod.Pod.Status.HostIP,
   171  			PodIP:     pod.Pod.Status.PodIP,
   172  			Action:    string(networkchaos.Spec.Action),
   173  		}
   174  		if pod.Position == "source" {
   175  			ps.Message = networkChaosSourceMsg
   176  		} else {
   177  			ps.Message = networkChaosTargetMsg
   178  		}
   179  
   180  		// TODO: add source, target and tc action message
   181  		if networkchaos.Spec.Duration != nil {
   182  			ps.Message += fmt.Sprintf(networkTcActionMsg, *networkchaos.Spec.Duration)
   183  		}
   184  		networkchaos.Status.Experiment.PodRecords = append(networkchaos.Status.Experiment.PodRecords, ps)
   185  	}
   186  	if errors != nil {
   187  		return errors
   188  	}
   189  
   190  	r.Event(networkchaos, v1.EventTypeNormal, events.ChaosInjected, "")
   191  	return nil
   192  }

   194  // Recover means the reconciler recovers the chaos action
   195  func (r *endpoint) Recover(ctx context.Context, req ctrl.Request, chaos v1alpha1.InnerObject) error {
   196  	networkchaos, ok := chaos.(*v1alpha1.NetworkChaos)
   197  	if !ok {
   198  		err := errors.New("chaos is not NetworkChaos")
   199  		r.Log.Error(err, "chaos is not NetworkChaos", "chaos", chaos)
   200  		return err
   201  	}
   202  
   203  	if err := r.cleanFinalizersAndRecover(ctx, networkchaos); err != nil {
   204  		return err
   205  	}
   206  	r.Event(networkchaos, v1.EventTypeNormal, events.ChaosRecovered, "")
   207  	return nil
   208  }

   210  func (r *endpoint) cleanFinalizersAndRecover(ctx context.Context, chaos *v1alpha1.NetworkChaos) error {
   211  	var result error
   212  
   213  	source := chaos.Namespace + "/" + chaos.Name
   214  	m := podnetworkchaosmanager.New(source, r.Log, r.Client)
   215  
   216  	keyRecordMap := make(map[types.NamespacedName]v1alpha1.PodStatus)
   217  	for _, podRecord := range chaos.Status.Experiment.PodRecords {
   218  		key := types.NamespacedName{
   219  			Namespace: podRecord.Namespace,
   220  			Name:      podRecord.Name,
   221  		}
   222  		_ = m.WithInit(key)
   223  		keyRecordMap[key] = podRecord
   224  	}
   225  	responses := m.Commit(ctx)
   226  
   227  	restRecord := []v1alpha1.PodStatus{}
   228  	for _, response := range responses {
   229  		key := response.Key
   230  		err := response.Err
   231  		// if pod not found or not running, directly return and giveup recover.
   232  		if err != nil {
   233  			if err != podnetworkchaosmanager.ErrPodNotFound && err != podnetworkchaosmanager.ErrPodNotRunning {
   234  				r.Log.Error(err, "fail to commit", "key", key)
   235  
   236  				result = multierror.Append(result, err)
   237  				restRecord = append(restRecord, keyRecordMap[key])
   238  				continue
   239  			}
   240  
   241  			r.Log.Info("pod is not found or not running", "key", key)
   242  		}
   243  	}
   244  
   245  	if len(restRecord) == 0 {
   246  		chaos.Finalizers = []string{}
   247  	}
   248  	r.Log.Info("After recovering", "finalizers", chaos.Finalizers)
   249  
   250  	if chaos.Annotations[common.AnnotationCleanFinalizer] == common.AnnotationCleanFinalizerForced {
   251  		r.Log.Info("Force cleanup all finalizers", "chaos", chaos)
   252  		chaos.Finalizers = make([]string, 0)
   253  		return nil
   254  	}
   255  
   256  	return result
   257  }

   259  func (r *endpoint) applyTc(
   260  	ctx context.Context,
   261  	sources, targets []v1.Pod,
   262  	externalTargets []string,
   263  	ipSetPostFix string,
   264  	m *podnetworkchaosmanager.PodNetworkManager,
   265  	networkchaos *v1alpha1.NetworkChaos,
   266  ) error {
   267  	tcType := v1alpha1.Bandwidth
   268  	switch networkchaos.Spec.Action {
   269  	case v1alpha1.NetemAction, v1alpha1.DelayAction, v1alpha1.DuplicateAction, v1alpha1.CorruptAction, v1alpha1.LossAction:
   270  		tcType = v1alpha1.Netem
   271  	case v1alpha1.BandwidthAction:
   272  		tcType = v1alpha1.Bandwidth
   273  	default:
   274  		return fmt.Errorf("unknown action %s", networkchaos.Spec.Action)
   275  	}
   276  
   277  	// if we don't specify targets, then sources pods apply traffic control on all egress traffic
   278  	if len(targets)+len(externalTargets) == 0 {
   279  		r.Log.Info("apply traffic control", "sources", sources)
   280  		for index := range sources {
   281  			pod := &sources[index]
   282  
   283  			t := m.WithInit(types.NamespacedName{
   284  				Name:      pod.Name,
   285  				Namespace: pod.Namespace,
   286  			})
   287  			t.Append(v1alpha1.RawTrafficControl{
   288  				Type:        tcType,
   289  				TcParameter: networkchaos.Spec.TcParameter,
   290  				Source:      m.Source,
   291  			})
   292  		}
   293  		return nil
   294  	}
   295  
   296  	// create ipset contains all target ips
   297  	dstIpset := ipset.BuildIPSet(targets, externalTargets, networkchaos, string(tcType[0:2])+ipSetPostFix, m.Source)
   298  	r.Log.Info("apply traffic control with filter", "sources", sources, "ipset", dstIpset)
   299  
   300  	for index := range sources {
   301  		pod := &sources[index]
   302  
   303  		t := m.WithInit(types.NamespacedName{
   304  			Name:      pod.Name,
   305  			Namespace: pod.Namespace,
   306  		})
   307  		t.Append(dstIpset)
   308  		t.Append(v1alpha1.RawTrafficControl{
   309  			Type:        tcType,
   310  			TcParameter: networkchaos.Spec.TcParameter,
   311  			Source:      m.Source,
   312  			IPSet:       dstIpset.Name,
   313  		})
   314  	}
   315  
   316  	return nil
   317  }
