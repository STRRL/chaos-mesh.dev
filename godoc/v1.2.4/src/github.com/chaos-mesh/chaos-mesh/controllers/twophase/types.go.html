    54  // Reconcile is twophase reconcile implement
    55  func (r *Reconciler) Reconcile(req ctrl.Request) (ctrl.Result, error) {
    56  	var err error
    57  	now := time.Now()
    58  
    59  	r.Log.Info("Reconciling a two phase chaos", "name", req.Name, "namespace", req.Namespace, "time", time.Now())
    60  	ctx, cancel := context.WithCancel(context.Background())
    61  	defer cancel()
    62  
    63  	_chaos := r.Object()
    64  	if err = r.Client.Get(ctx, req.NamespacedName, _chaos); err != nil {
    65  		r.Log.Error(err, "unable to get chaos")
    66  		return ctrl.Result{}, err
    67  	}
    68  	chaos := _chaos.(v1alpha1.InnerSchedulerObject)
    69  
    70  	if chaos.GetStatus().Experiment.Phase == v1alpha1.ExperimentPhaseFinished {
    71  		// This chaos was finished, we should remove the finalizer
    72  		r.Log.Info("Removing pre-finalizer")
    73  
    74  		chaos.GetMeta().SetFinalizers(finalizer.RemoveFromFinalizer(chaos.GetMeta().GetFinalizers(), common.PreFinalizer))
    75  
    76  		updateError := r.Update(ctx, chaos)
    77  
    78  		if updateError != nil {
    79  			r.Log.Error(err, "unable to update chaos")
    80  			return ctrl.Result{}, err
    81  		}
    82  	} else if !finalizer.ContainsFinalizer(chaos.GetMeta().GetFinalizers(), common.PreFinalizer) {
    83  		chaos.GetMeta().SetFinalizers(finalizer.InsertFinalizer(chaos.GetMeta().GetFinalizers(), common.PreFinalizer))
    84  
    85  		updateError := r.Update(ctx, chaos)
    86  
    87  		if updateError != nil {
    88  			r.Log.Error(err, "unable to update chaos")
    89  			return ctrl.Result{}, err
    90  		}
    91  	}
    92  
    93  	status := chaos.GetStatus()
    94  
    95  	targetPhase := status.Experiment.Phase
    96  
    97  	if !chaos.GetNextRecover().IsZero() && chaos.GetNextRecover().Before(now) {
    98  		targetPhase = v1alpha1.ExperimentPhaseWaiting
    99  	}
   100  
   101  	if chaos.GetNextStart().Before(now) {
   102  		targetPhase = v1alpha1.ExperimentPhaseRunning
   103  	}
   104  
   105  	// For standard cron(* * * * *), judge the phase when start
   106  	if chaos.GetNextStart().IsZero() && chaos.GetNextRecover().IsZero() {
   107  		phase, nextStart, err := ifStartAtWaitingPhase(chaos, now, r.Log)
   108  		if err != nil {
   109  			return ctrl.Result{}, err
   110  		}
   111  		if phase == v1alpha1.ExperimentPhaseWaiting {
   112  			targetPhase = phase
   113  			chaos.SetNextStart(*nextStart)
   114  			duration, err := chaos.GetDuration()
   115  			if err != nil {
   116  				return ctrl.Result{}, err
   117  			}
   118  			if duration == nil {
   119  				zero := 0 * time.Second
   120  				duration = &zero
   121  			}
   122  			chaos.SetNextRecover(nextStart.Add(*duration))
   123  		}
   124  	}
   125  
   126  	if chaos.IsPaused() {
   127  		targetPhase = v1alpha1.ExperimentPhasePaused
   128  	}
   129  
   130  	// TODO: find a better way to solve the pause and resume problem.
   131  	// Or pause is a bad design for the scheduler :(
   132  	if !chaos.IsPaused() && status.Experiment.Phase == v1alpha1.ExperimentPhasePaused {
   133  		// Running and Waiting has the same logic for resuming
   134  		targetPhase = v1alpha1.ExperimentPhaseRunning
   135  	}
   136  
   137  	if chaos.IsDeleted() {
   138  		targetPhase = v1alpha1.ExperimentPhaseFinished
   139  	}
   140  
   141  	r.Log.Info("decide target phase", "target phase", targetPhase)
   142  
   143  	machine := chaosStateMachine{
   144  		Chaos:      chaos,
   145  		Req:        req,
   146  		Reconciler: r,
   147  	}
   148  	err = machine.Into(ctx, targetPhase, now)
   149  	if err != nil {
   150  		r.Log.Error(err, "fail to step into the phase", "target phase", targetPhase)
   151  		return ctrl.Result{}, err
   152  	}
   153  
   154  	// the reconciliation of Finished and Paused resource shouldn't be triggered by time
   155  	if chaos.GetStatus().Experiment.Phase == v1alpha1.ExperimentPhaseFinished ||
   156  		chaos.GetStatus().Experiment.Phase == v1alpha1.ExperimentPhasePaused {
   157  		return ctrl.Result{}, nil
   158  	}
   159  
   160  	requeueAfter, err := calcRequeueAfterTime(chaos, now)
   161  	if err != nil {
   162  		r.Log.Error(err, "unexpected time", "now", now, "nextStart", chaos.GetNextStart(), "nextRecover", chaos.GetNextRecover())
   163  
   164  		// will not return error and retry
   165  		// because nothing will be better with retrying
   166  		return ctrl.Result{}, nil
   167  	}
   168  
   169  	r.Log.Info("requeue", "requeue after", requeueAfter)
   170  	if requeueAfter == time.Duration(0) {
   171  		return ctrl.Result{Requeue: true}, nil
   172  	}
   173  
   174  	return ctrl.Result{
   175  		RequeueAfter: requeueAfter,
   176  	}, nil
   177  }

   179  func calcRequeueAfterTime(chaos v1alpha1.InnerSchedulerObject, now time.Time) (time.Duration, error) {
   180  	requeueAfter := time.Duration(math.MaxInt64)
   181  	// requeueAfter = min(filter([nextRecoverAfter, nextStartAfter], >0))
   182  	nextRecoverAfter := chaos.GetNextRecover().Sub(now)
   183  	nextStartAfter := chaos.GetNextStart().Sub(now)
   184  
   185  	// If the `nextRecoverAfter` is less zero, we reset `nextRecoverAfter` to zero,
   186  	// which represents that the chaos action should be recovered immediately.
   187  	// `chaos.GetNextRecover()` ignores millisecond field, which will cause the `nextRecoverAfter` is less zero
   188  	// when the duration is zero or less 1 second.
   189  	if nextRecoverAfter < 0 {
   190  		nextRecoverAfter = 0 * time.Second
   191  	}
   192  
   193  	if requeueAfter > nextRecoverAfter {
   194  		requeueAfter = nextRecoverAfter
   195  	}
   196  
   197  	if nextStartAfter > 0 && requeueAfter > nextStartAfter {
   198  		requeueAfter = nextStartAfter
   199  	}
   200  
   201  	var err error
   202  	if requeueAfter == math.MaxInt64 {
   203  		err = errors.Errorf("unexpected behavior, now is greater than nextRecover and nextStart")
   204  	}
   205  
   206  	return requeueAfter, err
   207  }

   209  func ifStartAtWaitingPhase(chaos v1alpha1.InnerSchedulerObject, now time.Time, log logr.Logger) (v1alpha1.ExperimentPhase, *time.Time, error) {
   210  	duration, err := chaos.GetDuration()
   211  	if err != nil {
   212  		log.Error(err, "failed to get chaos duration")
   213  		return "", nil, err
   214  	}
   215  	if duration == nil {
   216  		zero := 0 * time.Second
   217  		duration = &zero
   218  	}
   219  
   220  	scheduler := chaos.GetScheduler()
   221  	if scheduler == nil {
   222  		log.Info("Scheduler should be defined currently")
   223  		return "", nil, fmt.Errorf("misdefined scheduler")
   224  	}
   225  
   226  	lastStart, err := sch.LastTime(*scheduler, now)
   227  	if err != nil {
   228  		log.Error(err, "failed to get the last start time")
   229  		return "", nil, err
   230  	}
   231  	if lastStart.Add(*duration).Before(now) {
   232  		nextStart, err := nextTime(*scheduler, now)
   233  		if err != nil {
   234  			log.Error(err, "failed to get the next start time")
   235  			return "", nil, err
   236  		}
   237  		return v1alpha1.ExperimentPhaseWaiting, nextStart, nil
   238  	}
   239  	return "", nil, nil
   240  }

   242  func nextTime(spec v1alpha1.SchedulerSpec, now time.Time) (*time.Time, error) {
   243  	scheduler, err := cron.ParseStandard(spec.Cron)
   244  	if err != nil {
   245  		return nil, fmt.Errorf("fail to parse runner rule %s, %v", spec.Cron, err)
   246  	}
   247  
   248  	next := scheduler.Next(now)
   249  	return &next, nil
   250  }
