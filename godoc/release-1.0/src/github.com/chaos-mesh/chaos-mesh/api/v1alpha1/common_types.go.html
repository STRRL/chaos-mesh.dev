    29  // SelectorSpec defines the some selectors to select objects.
    30  // If the all selectors are empty, all objects will be used in chaos experiment.
    31  type SelectorSpec struct {
    32  	// Namespaces is a set of namespace to which objects belong.
    33  	// +optional
    34  	Namespaces []string `json:"namespaces,omitempty"`
    35  
    36  	// Nodes is a set of node name and objects must belong to these nodes.
    37  	// +optional
    38  	Nodes []string `json:"nodes,omitempty"`
    39  
    40  	// Pods is a map of string keys and a set values that used to select pods.
    41  	// The key defines the namespace which pods belong,
    42  	// and the each values is a set of pod names.
    43  	// +optional
    44  	Pods map[string][]string `json:"pods,omitempty"`
    45  
    46  	// Map of string keys and values that can be used to select nodes.
    47  	// Selector which must match a node's labels,
    48  	// and objects must belong to these selected nodes.
    49  	// +optional
    50  	NodeSelectors map[string]string `json:"nodeSelectors,omitempty"`
    51  
    52  	// Map of string keys and values that can be used to select objects.
    53  	// A selector based on fields.
    54  	// +optional
    55  	FieldSelectors map[string]string `json:"fieldSelectors,omitempty"`
    56  
    57  	// Map of string keys and values that can be used to select objects.
    58  	// A selector based on labels.
    59  	// +optional
    60  	LabelSelectors map[string]string `json:"labelSelectors,omitempty"`
    61  
    62  	// Map of string keys and values that can be used to select objects.
    63  	// A selector based on annotations.
    64  	// +optional
    65  	AnnotationSelectors map[string]string `json:"annotationSelectors,omitempty"`
    66  
    67  	// PodPhaseSelectors is a set of condition of a pod at the current time.
    68  	// supported value: Pending / Running / Succeeded / Failed / Unknown
    69  	// +optional
    70  	PodPhaseSelectors []string `json:"podPhaseSelectors,omitempty"`
    71  }

    73  // SchedulerSpec defines information about schedule of the chaos experiment.
    74  type SchedulerSpec struct {
    75  	// Cron defines a cron job rule.
    76  	//
    77  	// Some rule examples:
    78  	// "0 30 * * * *" means to "Every hour on the half hour"
    79  	// "@hourly"      means to "Every hour"
    80  	// "@every 1h30m" means to "Every hour thirty"
    81  	//
    82  	// More rule info: https://godoc.org/github.com/robfig/cron
    83  	Cron string `json:"cron"`
    84  }

    86  // PodMode represents the mode to run pod chaos action.
    87  type PodMode string
    88  
    89  const (
    90  	// OnePodMode represents that the system will do the chaos action on one pod selected randomly.
    91  	OnePodMode PodMode = "one"
    92  	// AllPodMode represents that the system will do the chaos action on all pods
    93  	// regardless of status (not ready or not running pods includes).
    94  	// Use this label carefully.
    95  	AllPodMode PodMode = "all"
    96  	// FixedPodMode represents that the system will do the chaos action on a specific number of running pods.
    97  	FixedPodMode PodMode = "fixed"
    98  	// FixedPercentPodMode to specify a fixed % that can be inject chaos action.
    99  	FixedPercentPodMode PodMode = "fixed-percent"
   100  	// RandomMaxPercentPodMode to specify a maximum % that can be inject chaos action.
   101  	RandomMaxPercentPodMode PodMode = "random-max-percent"
   102  )
