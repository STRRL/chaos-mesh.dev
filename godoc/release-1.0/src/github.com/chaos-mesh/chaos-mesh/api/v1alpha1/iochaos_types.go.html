    38  // IoChaosSpec defines the desired state of IoChaos
    39  type IoChaosSpec struct {
    40  	// Selector is used to select pods that are used to inject chaos action.
    41  	Selector SelectorSpec `json:"selector"`
    42  
    43  	// Mode defines the mode to run chaos action.
    44  	// Supported mode: one / all / fixed / fixed-percent / random-max-percent
    45  	// +kubebuilder:validation:Enum=one;all;fixed;fixed-percent;random-max-percent
    46  	Mode PodMode `json:"mode"`
    47  
    48  	// Value is required when the mode is set to `FixedPodMode` / `FixedPercentPodMod` / `RandomMaxPercentPodMod`.
    49  	// If `FixedPodMode`, provide an integer of pods to do chaos action.
    50  	// If `FixedPercentPodMod`, provide a number from 0-100 to specify the percent of pods the server can do chaos action.
    51  	// IF `RandomMaxPercentPodMod`,  provide a number from 0-100 to specify the max percent of pods to do chaos action
    52  	// +optional
    53  	Value string `json:"value"`
    54  
    55  	// Action defines the specific pod chaos action.
    56  	// Supported action: latency / fault / attrOverride
    57  	// +kubebuilder:validation:Enum=latency;fault;attrOverride
    58  	Action IoChaosType `json:"action"`
    59  
    60  	// Delay defines the value of I/O chaos action delay.
    61  	// A delay string is a possibly signed sequence of
    62  	// decimal numbers, each with optional fraction and a unit suffix,
    63  	// such as "300ms".
    64  	// Valid time units are "ns", "us" (or "µs"), "ms", "s", "m", "h".
    65  	// +optional
    66  	Delay string `json:"delay,omitempty"`
    67  
    68  	// Errno defines the error code that returned by I/O action.
    69  	// refer to: https://www-numi.fnal.gov/offline_software/srt_public_context/WebDocs/Errors/unix_system_errors.html
    70  	// +optional

    74  	// +optional

    78  	// +optional
    79  	Path string `json:"path,omitempty"`
    80

    83  	// +optional
    84  	Methods []IoMethod `json:"methods,omitempty"`
    85  
    86  	// Percent defines the percentage of injection errors and provides a number from 0-100.

    88  	// +optional
    89  	Percent int `json:"percent,omitempty"`
    90  
    91  	// VolumePath represents the mount path of injected volume
