    71  // Validate validates chaos object
    72  func (in *PodChaos) Validate() error {
    73  	specField := field.NewPath("spec")
    74  	allErrs := in.ValidateScheduler(specField)
    75  	allErrs = append(allErrs, in.ValidatePodMode(specField)...)
    76  	allErrs = append(allErrs, in.Spec.validateContainerName(specField.Child("containerName"))...)
    77  
    78  	if len(allErrs) > 0 {
    79  		return fmt.Errorf(allErrs.ToAggregate().Error())
    80  	}
    81  	return nil
    82  }

    84  // ValidateScheduler validates the scheduler and duration
    85  func (in *PodChaos) ValidateScheduler(spec *field.Path) field.ErrorList {
    86  	allErrs := field.ErrorList{}
    87  	schedulerField := spec.Child("scheduler")
    88  
    89  	switch in.Spec.Action {
    90  	case PodFailureAction:
    91  		allErrs = append(allErrs, ValidateScheduler(in, spec)...)
    92  		break
    93  	case PodKillAction:
    94  		// We choose to ignore the Duration property even user define it
    95  		if in.Spec.Scheduler == nil {
    96  			allErrs = append(allErrs, field.Invalid(schedulerField, in.Spec.Scheduler, ValidatePodchaosSchedulerError))
    97  		} else {
    98  			_, err := ParseCron(in.Spec.Scheduler.Cron, schedulerField.Child("cron"))
    99  			allErrs = append(allErrs, err...)
   100  		}
   101  		break
   102  	case ContainerKillAction:
   103  		// We choose to ignore the Duration property even user define it
   104  		if in.Spec.Scheduler == nil {
   105  			allErrs = append(allErrs, field.Invalid(schedulerField, in.Spec.Scheduler, ValidatePodchaosSchedulerError))
   106  		} else {
   107  			_, err := ParseCron(in.Spec.Scheduler.Cron, schedulerField.Child("cron"))
   108  			allErrs = append(allErrs, err...)
   109  		}
   110  		break
   111  	default:
   112  		err := fmt.Errorf("podchaos[%s/%s] have unknown action type", in.Namespace, in.Name)
   113  		log.Error(err, "Wrong PodChaos Action type")
   114  
   115  		actionField := spec.Child("action")
   116  		allErrs = append(allErrs, field.Invalid(actionField, in.Spec.Action, err.Error()))
   117  		break
   118  	}
   119  	return allErrs
   120  }

   127  // validateContainerName validates the ContainerName
   128  func (in *PodChaosSpec) validateContainerName(containerField *field.Path) field.ErrorList {
   129  	allErrs := field.ErrorList{}
   130  	if in.Action == ContainerKillAction {
   131  		if in.ContainerName == "" {
   132  			err := fmt.Errorf("the name of container should not be empty on %s action", in.Action)
   133  			allErrs = append(allErrs, field.Invalid(containerField, in.ContainerName, err.Error()))
   134  		}
   135  	}
   136  	return allErrs
