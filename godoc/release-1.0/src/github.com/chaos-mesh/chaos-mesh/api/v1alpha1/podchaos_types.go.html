   152  // PodChaosSpec defines the attributes that a user creates on a chaos experiment about pods.
   153  type PodChaosSpec struct {
   154  	// Selector is used to select pods that are used to inject chaos action.
   155  	Selector SelectorSpec `json:"selector"`
   156  
   157  	// Scheduler defines some schedule rules to
   158  	// control the running time of the chaos experiment about pods.
   159  	Scheduler *SchedulerSpec `json:"scheduler,omitempty"`
   160  
   161  	// Action defines the specific pod chaos action.
   162  	// Supported action: pod-kill / pod-failure / container-kill
   163  	// Default action: pod-kill
   164  	// +kubebuilder:validation:Enum=pod-kill;pod-failure;container-kill
   165  	Action PodChaosAction `json:"action"`
   166  
   167  	// Mode defines the mode to run chaos action.
   168  	// Supported mode: one / all / fixed / fixed-percent / random-max-percent
   169  	// +kubebuilder:validation:Enum=one;all;fixed;fixed-percent;random-max-percent
   170  	Mode PodMode `json:"mode"`
   171  
   172  	// Value is required when the mode is set to `FixedPodMode` / `FixedPercentPodMod` / `RandomMaxPercentPodMod`.
   173  	// If `FixedPodMode`, provide an integer of pods to do chaos action.
   174  	// If `FixedPercentPodMod`, provide a number from 0-100 to specify the percent of pods the server can do chaos action.
   175  	// IF `RandomMaxPercentPodMod`,  provide a number from 0-100 to specify the max percent of pods to do chaos action
   176  	// +optional
   177  	Value string `json:"value"`
   178  
   179  	// Duration represents the duration of the chaos action.
   180  	// It is required when the action is `PodFailureAction`.
   181  	// A duration string is a possibly signed sequence of
   182  	// decimal numbers, each with optional fraction and a unit suffix,
   183  	// such as "300ms", "-1.5h" or "2h45m".
   184  	// Valid time units are "ns", "us" (or "µs"), "ms", "s", "m", "h".
   185  	// +optional
   186  	Duration *string `json:"duration,omitempty"`
   187  
   188  	// ContainerName indicates the name of the container.
   189  	// Needed in container-kill.
   190  	// +optional
   191  	ContainerName string `json:"containerName"`
   192  
   193  	// GracePeriod is used in pod-kill action. It represents the duration in seconds before the pod should be deleted.
   194  	// Value must be non-negative integer. The default value is zero that indicates delete immediately.
   195  	// +optional
   196  	// +kubebuilder:validation:Minimum=0
   197  	GracePeriod int64 `json:"gracePeriod"`
   198  }
