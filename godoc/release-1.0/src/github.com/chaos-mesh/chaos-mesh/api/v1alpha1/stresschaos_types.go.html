   224  // Stressors defines plenty of stressors supported to stress system components out.
   225  // You can use one or more of them to make up various kinds of stresses
   226  type Stressors struct {
   227  	// MemoryStressor stresses virtual memory out
   228  	// +optional
   229  	MemoryStressor *MemoryStressor `json:"memory,omitempty"`
   230  	// CPUStressor stresses CPU out
   231  	// +optional
   232  	CPUStressor *CPUStressor `json:"cpu,omitempty"`
   233  }
   234  
   235  // Normalize the stressors to comply with stress-ng
   236  func (in *Stressors) Normalize() (string, error) {
   237  	stressors := ""
   238  	if in.MemoryStressor != nil {
   239  		stressors += fmt.Sprintf(" --bigheap %d", in.MemoryStressor.Workers)
   240  
   241  		if in.MemoryStressor.Options != nil {
   242  			for _, v := range in.MemoryStressor.Options {
   243  				stressors += fmt.Sprintf(" %v ", v)
   244  			}
   245  		}
   246  	}
   247  	if in.CPUStressor != nil {
   248  		stressors += fmt.Sprintf(" --cpu %d", in.CPUStressor.Workers)
   249  		if in.CPUStressor.Load != nil {
   250  			stressors += fmt.Sprintf(" --cpu-load %d",
   251  				*in.CPUStressor.Load)
   252  		}
   253  
   254  		if in.CPUStressor.Options != nil {
   255  			for _, v := range in.CPUStressor.Options {
   256  				stressors += fmt.Sprintf(" %v ", v)
   257  			}
   258  		}
   259  	}
   260  	return stressors, nil
   261  }
