    75  func newGRPCServer(containerRuntime string, reg prometheus.Registerer) (*grpc.Server, error) {
    76  	ds, err := newDaemonServer(containerRuntime)
    77  	if err != nil {
    78  		return nil, err
    79  	}
    80  
    81  	grpcMetrics := grpc_prometheus.NewServerMetrics()
    82  	grpcMetrics.EnableHandlingTimeHistogram(
    83  		grpc_prometheus.WithHistogramBuckets([]float64{0.001, 0.01, 0.1, 0.3, 0.6, 1, 3, 6, 10}),
    84  	)
    85  	reg.MustRegister(grpcMetrics)
    86  
    87  	grpcOpts := []grpc.ServerOption{
    88  		grpc_middleware.WithUnaryServerChain(
    89  			utils.TimeoutServerInterceptor,
    90  			grpcMetrics.UnaryServerInterceptor(),
    91  		),
    92  	}
    93  
    94  	s := grpc.NewServer(grpcOpts...)
    95  	grpcMetrics.InitializeMetrics(s)
    96  
    97  	pb.RegisterChaosDaemonServer(s, ds)
    98  	reflection.Register(s)
    99  
   100  	return s, nil
   101  }

   109  // StartServer starts chaos-daemon.
   110  func StartServer(conf *Config, reg RegisterGatherer) error {
   111  	g := &errgroup.Group{}
   112  
   113  	httpBindAddr := conf.HttpAddr()
   114  	httpServer := newHTTPServerBuilder().Addr(httpBindAddr).Metrics(reg).Profiling(conf.Profiling).Build()
   115  
   116  	grpcBindAddr := conf.GrpcAddr()
   117  	grpcListener, err := net.Listen("tcp", grpcBindAddr)
   118  	if err != nil {
   119  		log.Error(err, "failed to listen grpc address", "grpcBindAddr", grpcBindAddr)
   120  		return err
   121  	}
   122  
   123  	grpcServer, err := newGRPCServer(conf.Runtime, reg)
   124  	if err != nil {
   125  		log.Error(err, "failed to create grpc server")
   126  		return err
   127  	}
   128  
   129  	g.Go(func() error {
   130  		log.Info("Starting http endpoint", "address", httpBindAddr)
   131  		if err := httpServer.ListenAndServe(); err != nil {
   132  			log.Error(err, "failed to start http endpoint")
   133  			httpServer.Shutdown(context.Background())
   134  			return err
   135  		}
   136  		return nil
   137  	})
   138  
   139  	g.Go(func() error {
   140  		log.Info("Starting grpc endpoint", "address", grpcBindAddr, "runtime", conf.Runtime)
   141  		if err := grpcServer.Serve(grpcListener); err != nil {
   142  			log.Error(err, "failed to start grpc endpoint")
   143  			grpcServer.Stop()
   144  			return err
   145  		}
   146  		return nil
   147  	})
   148  
   149  	return g.Wait()
   150  }
