    68  // FormatContainerID strips protocol prefix from the container ID
    69  func (c DockerClient) FormatContainerID(ctx context.Context, containerID string) (string, error) {
    70  	if len(containerID) < len(dockerProtocolPrefix) {
    71  		return "", fmt.Errorf("container id %s is not a docker container id", containerID)
    72  	}
    73  	if containerID[0:len(dockerProtocolPrefix)] != dockerProtocolPrefix {
    74  		return "", fmt.Errorf("expected %s but got %s", dockerProtocolPrefix, containerID[0:len(dockerProtocolPrefix)])
    75  	}
    76  	return containerID[len(dockerProtocolPrefix):], nil
    77  }

   159  // CreateContainerRuntimeInfoClient creates a container runtime information client.
   160  func CreateContainerRuntimeInfoClient(containerRuntime string) (ContainerRuntimeInfoClient, error) {
   161  	// TODO: support more container runtime
   162  
   163  	var cli ContainerRuntimeInfoClient
   164  	switch containerRuntime {
   165  	case containerRuntimeDocker:
   166  		client, err := newDockerClient(defaultDockerSocket, "", nil, nil)
   167  		if err != nil {
   168  			return nil, err
   169  		}
   170  		cli = DockerClient{client}
   171  
   172  	case containerRuntimeContainerd:
   173  		// TODO(yeya24): add more options?
   174  		client, err := newContainerdClient(defaultContainerdSocket, containerd.WithDefaultNamespace(containerdDefaultNS))
   175  		if err != nil {
   176  			return nil, err
   177  		}
   178  		cli = ContainerdClient{client}
   179  
   180  	default:
   181  		return nil, fmt.Errorf("only docker and containerd is supported, but got %s", containerRuntime)
   182  	}
   183  
   184  	return cli, nil
   185  }

   212  	}

   214  	container, err := c.client.LoadContainer(ctx, containerID)
   215  	if err != nil {
   216  		return err
   217  	}
   218  	task, err := container.Task(ctx, nil)
   219  	if err != nil {
   220  		return err
   221  	}
   222  
   223  	err = task.Kill(ctx, syscall.SIGKILL)

   243  // GetChildProcesses will return all child processes's pid. Include all generations.
   244  // only return error when /proc/pid/tasks cannot be read
   245  func GetChildProcesses(ppid uint32) ([]uint32, error) {
   246  	procs, err := ioutil.ReadDir(defaultProcPrefix)
   247  	if err != nil {
   248  		return nil, err
   249  	}
   250  
   251  	type processPair struct {
   252  		Pid  uint32
   253  		Ppid uint32
   254  	}
   255  
   256  	pairs := make(chan processPair)
   257  	done := make(chan bool)
   258  
   259  	go func() {
   260  		var wg sync.WaitGroup
   261  
   262  		for _, proc := range procs {
   263  			_, err := strconv.ParseUint(proc.Name(), 10, 32)
   264  			if err != nil {
   265  				continue
   266  			}
   267  
   268  			statusPath := defaultProcPrefix + "/" + proc.Name() + "/stat"
   269  
   270  			wg.Add(1)
   271  			go func() {
   272  				defer wg.Done()
   273  
   274  				reader, err := os.Open(statusPath)
   275  				if err != nil {
   276  					log.Error(err, "read status file error", "path", statusPath)
   277  					return
   278  				}
   279  
   280  				var (
   281  					pid    uint32
   282  					comm   string
   283  					state  string
   284  					parent uint32
   285  				)
   286  				// according to procfs's man page
   287  				fmt.Fscanf(reader, "%d %s %s %d", &pid, &comm, &state, &parent)
   288  
   289  				pairs <- processPair{
   290  					Pid:  pid,
   291  					Ppid: parent,
   292  				}
   293  			}()
   294  		}
   295  
   296  		wg.Wait()
   297  		done <- true
   298  	}()
   299  
   300  	processGraph := utils.NewGraph()
   301  	for {
   302  		select {
   303  		case pair := <-pairs:
   304  			processGraph.Insert(pair.Ppid, pair.Pid)
   305  		case <-done:
   306  			return processGraph.Flatten(ppid), nil
   307  		}
   308  	}
   309  }
