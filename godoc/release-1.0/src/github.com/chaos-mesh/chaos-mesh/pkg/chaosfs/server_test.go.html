    31  	Context("randomErrno", func() {

    33  			errSets := make(map[string]bool)
    34  			for i := 0; i < 5; i++ {
    35  				e := randomErrno()
    36  				errSets[e.Error()] = true
    37  			}
    38  			Expect(len(errSets) > 1).Should(Equal(true))
    39  		})
    40  	})
    41  
    42  	Context("probab", func() {
    43  		It("should be true", func() {
    44  			Expect(probab(100)).To(Equal(true))
    45  			Expect(probab(101)).To(Equal(true))
    46  			Expect(probab(100000000)).To(Equal(true))
    47  		})
    48  
    49  		It("should be false", func() {
    50  			Expect(probab(0)).To(Equal(false))
    51  		})
    52  	})

    54  	Context("faultInject", func() {
    55  		It("should work", func() {
    56  			faultMap.Store(faultInjectMethod, &faultContext{
    57  				pct:    100,
    58  				random: true,
    59  			})
    60  			err := faultInject(faultInjectPath, faultInjectMethod)
    61  			Expect(err).ToNot(BeNil())
    62  		})
    63  
    64  		It("should skip unknow method", func() {
    65  			err := faultInject(faultInjectPath, "unknow method")
    66  			Expect(err).To(BeNil())
    67  		})
    68  
    69  		It("should skip on wrong regex", func() {
    70  			faultMap.Store(faultInjectMethod, &faultContext{
    71  				pct:    100,

    73  				path:   `^\/(?!\/)(.*?)`,
    74  			})
    75  			err := faultInject(faultInjectPath, faultInjectMethod)
    76  			Expect(err).To(BeNil())
    77  		})
    78  
    79  		It("should skip on mismatch path", func() {
    80  			faultMap.Store(faultInjectMethod, &faultContext{
    81  				pct:    100,

    83  				path:   `mismatch-path`,
    84  			})
    85  			err := faultInject(faultInjectPath, faultInjectMethod)

    88  
    89  		It("should return specified errno", func() {

    98  	})

   120  				return true

   124  	})
