   121  // @Summary Create a new chaos experiment.
   122  // @Description Create a new chaos experiment.
   123  // @Tags experiments
   124  // @Produce json
   125  // @Param request body core.ExperimentInfo true "Request body"
   126  // @Success 200 {object} core.ExperimentInfo
   127  // @Failure 400 {object} utils.APIError
   128  // @Failure 500 {object} utils.APIError
   129  // @Router /experiments/new [post]
   130  func (s *Service) createExperiment(c *gin.Context) {
   131  	exp := &core.ExperimentInfo{}
   132  	if err := c.ShouldBindJSON(exp); err != nil {
   133  		c.Status(http.StatusBadRequest)
   134  		_ = c.Error(utils.ErrInvalidRequest.WrapWithNoMessage(err))
   135  		return
   136  	}
   137  
   138  	createFuncs := map[string]actionFunc{
   139  		v1alpha1.KindPodChaos:     s.createPodChaos,
   140  		v1alpha1.KindNetworkChaos: s.createNetworkChaos,
   141  		v1alpha1.KindIOChaos:      s.createIOChaos,
   142  		v1alpha1.KindStressChaos:  s.createStressChaos,
   143  		v1alpha1.KindTimeChaos:    s.createTimeChaos,
   144  		v1alpha1.KindKernelChaos:  s.createKernelChaos,
   145  	}
   146  
   147  	f, ok := createFuncs[exp.Target.Kind]
   148  	if !ok {
   149  		c.Status(http.StatusBadRequest)
   150  		_ = c.Error(utils.ErrInvalidRequest.New(exp.Target.Kind + " is not supported"))
   151  		return
   152  	}
   153  
   154  	if err := f(exp); err != nil {
   155  		c.Status(http.StatusInternalServerError)
   156  		_ = c.Error(utils.ErrInternalServer.WrapWithNoMessage(err))
   157  		return
   158  	}
   159  
   160  	c.JSON(http.StatusOK, exp)
   161  }

   227  	}
   228

   787  				continue
