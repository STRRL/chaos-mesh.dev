    76  // @Description Get pods from Kubernetes cluster.
    77  // @Tags common
    78  // @Produce json
    79  // @Param request body core.SelectorInfo true "Request body"

    83  func (s *Service) listPods(c *gin.Context) {
    84  	exp := &core.SelectorInfo{}
    85  	if err := c.ShouldBindJSON(exp); err != nil {
    86  		c.Status(http.StatusBadRequest)
    87  		_ = c.Error(utils.ErrInvalidRequest.WrapWithNoMessage(err))
    88  		return
    89  	}
    90  	ctx := context.TODO()
    91  	filteredPods, err := pkgutils.SelectPods(ctx, s.kubeCli, s.reader, exp.ParseSelector())
    92  	if err != nil {
    93  		c.Status(http.StatusInternalServerError)
    94  		_ = c.Error(utils.ErrInternalServer.WrapWithNoMessage(err))
    95  		return
    96  	}
    97  
    98  	pods := make([]Pod, 0, len(filteredPods))
    99  	for _, pod := range filteredPods {
   100  		pods = append(pods, Pod{
   101  			Name:      pod.Name,
   102  			IP:        pod.Status.PodIP,
   103  			Namespace: pod.Namespace,
   104  			State:     string(pod.Status.Phase),
   105  		})
   106  	}
   107  
   108  	c.JSON(http.StatusOK, pods)
   109  }

   112  // @Description Get all from Kubernetes cluster.
   113  // @Deprecated This API only works within cluster scoped mode. Please use /common/chaos-available-namespaces instead.
   114  // @Tags common

   119  func (s *Service) listNamespaces(c *gin.Context) {
   120  
   121  	var namespaces sort.StringSlice
   122  
   123  	var nsList v1.NamespaceList
   124  	if err := s.kubeCli.List(context.Background(), &nsList); err != nil {
   125  		c.Status(http.StatusInternalServerError)
   126  		_ = c.Error(utils.ErrInternalServer.WrapWithNoMessage(err))
   127  		return
   128  	}
   129  	namespaces = make(sort.StringSlice, 0, len(nsList.Items))
   130  	for _, ns := range nsList.Items {
   131  		namespaces = append(namespaces, ns.Name)
   132  	}
   133  
   134  	sort.Sort(namespaces)
   135  	c.JSON(http.StatusOK, namespaces)
   136  }

   190  // @Summary Get the labels of the pods in the specified namespace from Kubernetes cluster.
   191  // @Description Get the labels of the pods in the specified namespace from Kubernetes cluster.
   192  // @Tags common
   193  // @Produce json
   194  // @Param podNamespaceList query string true "The pod's namespace list, split by ,"
   195  // @Success 200 {object} MapSlice
   196  // @Router /common/labels [get]
   197  // @Failure 500 {object} utils.APIError
   198  func (s *Service) getLabels(c *gin.Context) {
   199  	podNamespaceList := c.Query("podNamespaceList")
   200  
   201  	if podNamespaceList == "" {
   202  		c.Status(http.StatusInternalServerError)
   203  		_ = c.Error(utils.ErrInternalServer.WrapWithNoMessage(fmt.Errorf("podNamespaceList cannot be empty")))
   204  		return
   205  	}
   206  
   207  	exp := &core.SelectorInfo{}
   208  	nsList := strings.Split(podNamespaceList, ",")
   209  	exp.NamespaceSelectors = nsList
   210  
   211  	ctx := context.TODO()
   212  	filteredPods, err := pkgutils.SelectPods(ctx, s.kubeCli, s.reader, exp.ParseSelector())
   213  	if err != nil {
   214  		c.Status(http.StatusInternalServerError)
   215  		_ = c.Error(utils.ErrInternalServer.WrapWithNoMessage(err))
   216  		return
   217  	}
   218  
   219  	labels := make(map[string][]string)
   220  
   221  	for _, pod := range filteredPods {
   222  		for k, v := range pod.Labels {
   223  			if _, ok := labels[k]; ok {
   224  				if !inSlice(v, labels[k]) {
   225  					labels[k] = append(labels[k], v)
   226  				}
   227  			} else {
   228  				labels[k] = []string{v}
   229  			}
   230  		}
   231  	}
   232  	c.JSON(http.StatusOK, labels)
   233  }

   280  // inSlice checks given string in string slice or not.
   281  func inSlice(v string, sl []string) bool {
   282  	for _, vv := range sl {
   283  		if vv == v {
   284  			return true
   285  		}
   286  	}
   287  	return false
