    44  // Reconcile reconciles a chaos collector.
    45  func (r *ChaosCollector) Reconcile(req ctrl.Request) (ctrl.Result, error) {
    46  	if r.apiType == nil {
    47  		r.Log.Error(nil, "apiType has not been initialized")
    48  		return ctrl.Result{}, nil
    49  	}
    50  	ctx := context.Background()
    51  
    52  	obj, ok := r.apiType.DeepCopyObject().(v1alpha1.InnerObject)
    53  	if !ok {
    54  		r.Log.Error(nil, "it's not a stateful object")
    55  		return ctrl.Result{}, nil
    56  	}
    57  
    58  	err := r.Get(ctx, req.NamespacedName, obj)
    59  	if apierrors.IsNotFound(err) {
    60  		if err = r.archiveExperiment(req.Namespace, req.Name); err != nil {
    61  			r.Log.Error(err, "failed to archive experiment")
    62  		}
    63  		return ctrl.Result{}, nil
    64  	}
    65  
    66  	if err != nil {
    67  		r.Log.Error(err, "failed to get chaos object", "request", req.NamespacedName)
    68  		return ctrl.Result{}, nil
    69  	}
    70  
    71  	if obj.IsDeleted() {
    72  		if err = r.archiveExperiment(req.Namespace, req.Name); err != nil {
    73  			r.Log.Error(err, "failed to archive experiment")
    74  		}
    75  		return ctrl.Result{}, nil
    76  	}
    77  
    78  	if err := r.setUnarchivedExperiment(req, obj); err != nil {
    79  		r.Log.Error(err, "failed to archive experiment")
    80  		// ignore error here
    81  	}
    82  
    83  	if err := r.recordEvent(req, obj); err != nil {
    84  		r.Log.Error(err, "failed to record event")
    85  	}
    86  
    87  	return ctrl.Result{}, nil
    88  }

    99  func (r *ChaosCollector) recordEvent(req ctrl.Request, obj v1alpha1.InnerObject) error {
   100  	var (
   101  		chaosMeta metav1.Object
   102  		ok        bool
   103  	)
   104  
   105  	if chaosMeta, ok = obj.(metav1.Object); !ok {
   106  		return errors.New("failed to get chaos meta information")
   107  	}
   108  
   109  	UID := chaosMeta.GetUID()
   110  	status := obj.GetStatus()
   111  	kind := obj.GetObjectKind().GroupVersionKind().Kind
   112  
   113  	switch status.Experiment.Phase {
   114  	case v1alpha1.ExperimentPhaseRunning:
   115  		return r.createEvent(req, kind, status, string(UID))
   116  	case v1alpha1.ExperimentPhaseFinished, v1alpha1.ExperimentPhasePaused, v1alpha1.ExperimentPhaseWaiting:
   117  		return r.updateOrCreateEvent(req, kind, status, string(UID))
   118  	}
   119  
   120  	return nil
   121  }

   123  func (r *ChaosCollector) createEvent(req ctrl.Request, kind string, status *v1alpha1.ChaosStatus, UID string) error {
   124  	event := &core.Event{
   125  		Experiment:   req.Name,
   126  		Namespace:    req.Namespace,
   127  		Kind:         kind,
   128  		StartTime:    &status.Experiment.StartTime.Time,
   129  		ExperimentID: UID,
   130  	}
   131  
   132  	if _, err := r.event.FindByExperimentAndStartTime(
   133  		context.Background(), event.Experiment, event.Namespace, event.StartTime); err == nil {
   134  		r.Log.Info("event has been created")
   135  		return nil
   136  	}
   137  
   138  	for _, pod := range status.Experiment.PodRecords {
   139  		podRecord := &core.PodRecord{
   140  			EventID:   event.ID,
   141  			PodIP:     pod.PodIP,
   142  			PodName:   pod.Name,
   143  			Namespace: pod.Namespace,
   144  			Message:   pod.Message,
   145  			Action:    pod.Action,
   146  		}
   147  		event.Pods = append(event.Pods, podRecord)
   148  	}
   149  	if err := r.event.Create(context.Background(), event); err != nil {
   150  		r.Log.Error(err, "failed to store event", "event", event)
   151  		return err
   152  	}
   153  
   154  	return nil
   155  }

   157  func (r *ChaosCollector) updateOrCreateEvent(req ctrl.Request, kind string, status *v1alpha1.ChaosStatus, UID string) error {
   158  	if status.Experiment.StartTime == nil || status.Experiment.EndTime == nil {
   159  		return fmt.Errorf("failed to get experiment time, startTime or endTime is empty")
   160  	}
   161  
   162  	event := &core.Event{
   163  		Experiment:   req.Name,
   164  		Namespace:    req.Namespace,
   165  		Kind:         kind,
   166  		StartTime:    &status.Experiment.StartTime.Time,
   167  		FinishTime:   &status.Experiment.EndTime.Time,
   168  		Duration:     status.Experiment.Duration,
   169  		ExperimentID: UID,
   170  	}
   171  
   172  	if _, err := r.event.FindByExperimentAndStartTime(
   173  		context.Background(), event.Experiment, event.Namespace, event.StartTime); err != nil && gorm.IsRecordNotFoundError(err) {
   174  		if err := r.createEvent(req, kind, status, UID); err != nil {
   175  			return err
   176  		}
   177  	}
   178  
   179  	if err := r.event.Update(context.Background(), event); err != nil {
   180  		r.Log.Error(err, "failed to update event", "event", event)
   181  		return err
   182  	}
   183  
   184  	return nil
   185  }

   187  func (r *ChaosCollector) setUnarchivedExperiment(req ctrl.Request, obj v1alpha1.InnerObject) error {
   188  	var (
   189  		chaosMeta metav1.Object
   190  		ok        bool
   191  	)
   192  
   193  	if chaosMeta, ok = obj.(metav1.Object); !ok {
   194  		r.Log.Error(nil, "failed to get chaos meta information")
   195  	}
   196  	UID := string(chaosMeta.GetUID())
   197  
   198  	archive := &core.ArchiveExperiment{
   199  		ArchiveExperimentMeta: core.ArchiveExperimentMeta{
   200  			Namespace: req.Namespace,
   201  			Name:      req.Name,
   202  			Kind:      obj.GetObjectKind().GroupVersionKind().Kind,
   203  			UID:       UID,
   204  			Archived:  false,
   205  		},
   206  	}
   207  
   208  	switch chaos := obj.(type) {
   209  	case *v1alpha1.PodChaos:
   210  		archive.Action = string(chaos.Spec.Action)
   211  	case *v1alpha1.NetworkChaos:
   212  		archive.Action = string(chaos.Spec.Action)
   213  	case *v1alpha1.IoChaos:
   214  		archive.Action = string(chaos.Spec.Action)
   215  	case *v1alpha1.TimeChaos, *v1alpha1.KernelChaos, *v1alpha1.StressChaos:
   216  		archive.Action = ""
   217  	default:
   218  		return errors.New("unsupported chaos type " + archive.Kind)
   219  	}
   220  
   221  	archive.StartTime = chaosMeta.GetCreationTimestamp().Time
   222  	if chaosMeta.GetDeletionTimestamp() != nil {
   223  		archive.FinishTime = chaosMeta.GetDeletionTimestamp().Time
   224  	}
   225  
   226  	data, err := json.Marshal(chaosMeta)
   227  	if err != nil {
   228  		r.Log.Error(err, "failed to marshal chaos", "kind", archive.Kind,
   229  			"namespace", archive.Namespace, "name", archive.Name)
   230  		return err
   231  	}
   232  
   233  	archive.Experiment = string(data)
   234  
   235  	find, err := r.archive.FindByUID(context.Background(), UID)
   236  	if err != nil && !gorm.IsRecordNotFoundError(err) {
   237  		r.Log.Error(err, "failed to find experiment", "UID", UID)
   238  		return err
   239  	}
   240  
   241  	if find != nil {
   242  		archive.ID = find.ID
   243  		archive.CreatedAt = find.CreatedAt
   244  		archive.UpdatedAt = find.UpdatedAt
   245  	}
   246  
   247  	if err := r.archive.Set(context.Background(), archive); err != nil {
   248  		r.Log.Error(err, "failed to update experiment", "archive", archive)
   249  		return err
   250  	}
   251  
   252  	return nil
   253  }

   255  func (r *ChaosCollector) archiveExperiment(ns, name string) error {
   256  	if err := r.event.UpdateIncompleteEvents(context.Background(), ns, name); err != nil {
   257  		r.Log.Error(err, "failed to update incomplete events", "namespace", ns, "name", name)
   258  		return err
   259  	}
   260  
   261  	if err := r.archive.Archive(context.Background(), ns, name); err != nil {
   262  		r.Log.Error(err, "failed to archive experiment", "namespace", ns, "name", name)
   263  		return err
   264  	}
   265  
   266  	return nil
   267  }
