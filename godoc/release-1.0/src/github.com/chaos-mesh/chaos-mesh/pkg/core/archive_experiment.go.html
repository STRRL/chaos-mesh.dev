   110  	PhaseSelector       []string          `json:"phase_selectors" binding:"PhaseSelectorsValid"`
   111  
   112  	// Pods is a map of string keys and a set values that used to select pods.
   113  	// The key defines the namespace which pods belong,

   116  }

   118  // ParseSelector parses SelectorInfo to v1alpha1.SelectorSpec
   119  func (s *SelectorInfo) ParseSelector() v1alpha1.SelectorSpec {
   120  	selector := v1alpha1.SelectorSpec{}
   121  
   122  	for _, ns := range s.NamespaceSelectors {
   123  		selector.Namespaces = append(selector.Namespaces, ns)
   124  	}
   125  
   126  	selector.LabelSelectors = make(map[string]string)
   127  	for key, val := range s.LabelSelectors {
   128  		selector.LabelSelectors[key] = val
   129  	}
   130  
   131  	selector.AnnotationSelectors = make(map[string]string)
   132  	for key, val := range s.AnnotationSelectors {
   133  		selector.AnnotationSelectors[key] = val
   134  	}
   135  
   136  	selector.FieldSelectors = make(map[string]string)
   137  	for key, val := range s.FieldSelectors {
   138  		selector.FieldSelectors[key] = val
   139  	}
   140  
   141  	for _, ph := range s.PhaseSelector {
   142  		selector.PodPhaseSelectors = append(selector.PodPhaseSelectors, ph)
   143  	}
   144  
   145  	if s.Pods != nil {
   146  		selector.Pods = s.Pods
   147  	}
   148  
   149  	return selector
   150  }

   218  func (e *ArchiveExperiment) ParsePodChaos() (ExperimentInfo, error) {
   219  	chaos := &v1alpha1.PodChaos{}
   220  	if err := json.Unmarshal([]byte(e.Experiment), &chaos); err != nil {
   221  		return ExperimentInfo{}, err
   222  	}
   223  
   224  	info := ExperimentInfo{
   225  		Name:        chaos.Name,
   226  		Namespace:   chaos.Namespace,
   227  		Labels:      chaos.Labels,
   228  		Annotations: chaos.Annotations,
   229  		Scope: ScopeInfo{
   230  			SelectorInfo: SelectorInfo{
   231  				NamespaceSelectors:  chaos.Spec.Selector.Namespaces,
   232  				LabelSelectors:      chaos.Spec.Selector.LabelSelectors,
   233  				AnnotationSelectors: chaos.Spec.Selector.AnnotationSelectors,
   234  				FieldSelectors:      chaos.Spec.Selector.FieldSelectors,
   235  				PhaseSelector:       chaos.Spec.Selector.PodPhaseSelectors,
   236  				Pods:                chaos.Spec.Selector.Pods,
   237  			},
   238  			Mode:  string(chaos.Spec.Mode),
   239  			Value: chaos.Spec.Value,
   240  		},
   241  		Target: TargetInfo{
   242  			Kind: v1alpha1.KindPodChaos,
   243  			PodChaos: &PodChaosInfo{
   244  				Action:        string(chaos.Spec.Action),
   245  				ContainerName: chaos.Spec.ContainerName,
   246  			},
   247  		},
   248  	}
   249  
   250  	if chaos.Spec.Scheduler != nil {
   251  		info.Scheduler.Cron = chaos.Spec.Scheduler.Cron
   252  	}
   253  
   254  	if chaos.Spec.Duration != nil {
   255  		info.Scheduler.Duration = *chaos.Spec.Duration
   256  	}

   258  	return info, nil
