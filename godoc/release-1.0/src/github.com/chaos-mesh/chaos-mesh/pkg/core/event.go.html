    21  // EventStore defines operations for working with event.
    22  type EventStore interface {
    23  	// List returns an event list from the datastore.
    24  	List(context.Context) ([]*Event, error)
    25  
    26  	// ListByFilter returns an event list by podName, podNamespace, experimentName, experimentNamespace, uid, kind, startTime and finishTime.
    27  	ListByFilter(context.Context, Filter) ([]*Event, error)
    28  
    29  	// ListByExperiment returns an event list by the name and namespace of the experiment.
    30  	ListByExperiment(context.Context, string, string) ([]*Event, error)
    31  
    32  	// ListByNamespace returns an event list by the namespace of the pod.
    33  	ListByNamespace(context.Context, string) ([]*Event, error)
    34  
    35  	// ListByPod returns an event list by the name and namespace of the pod.
    36  	ListByPod(context.Context, string, string) ([]*Event, error)
    37  
    38  	// ListByUID returns an event list by the UID.
    39  	ListByUID(context.Context, string) ([]*Event, error)
    40  
    41  	// DryListByFilter returns an event list by experimentName, experimentNamespace, uid, kind, startTime and finishTime.
    42  	DryListByFilter(context.Context, Filter) ([]*Event, error)
    43  
    44  	// Find returns an event from the datastore by ID.
    45  	Find(context.Context, uint) (*Event, error)
    46  
    47  	// FindByExperimentAndStartTime returns an event by the experiment and start time.
    48  	FindByExperimentAndStartTime(context.Context, string, string, *time.Time) (*Event, error)
    49  
    50  	// Create persists a new event to the datastore.
    51  	Create(context.Context, *Event) error
    52  
    53  	// Update persists an updated event to the datastore.
    54  	Update(context.Context, *Event) error
    55  
    56  	// DeleteIncompleteEvent deletes all incomplete events.
    57  	// If the chaos-dashboard was restarted, some incomplete events would be stored in datastore,
    58  	// which means the event would never save the finish_time.
    59  	// DeleteIncompleteEvent can be used to delete all incomplete events to avoid this case.
    60  	DeleteIncompleteEvents(context.Context) error
    61  
    62  	// DeleteByFinishTime deletes events and podrecords whose time difference is greater than the given time from FinishTime.
    63  	DeleteByFinishTime(context.Context, time.Duration) error
    64  
    65  	// UpdateIncompleteEvents updates the incomplete event by the namespace and name
    66  	// If chaos is deleted before an event is over, then the incomplete event would be stored in datastore,
    67  	// which means the event would never save the finish_time.
    68  	// UpdateIncompleteEvents can update the finish_time when the chaos is deleted.
    69  	UpdateIncompleteEvents(context.Context, string, string) error
    70  }

    72  // Event represents an event instance.
    73  type Event struct {
    74  	ID           uint         `gorm:"primary_key" json:"id"`
    75  	CreatedAt    time.Time    `json:"created_at"`
    76  	UpdatedAt    time.Time    `json:"updated_at"`
    77  	DeletedAt    *time.Time   `sql:"index" json:"deleted_at"`
    78  	Experiment   string       `gorm:"index:experiment" json:"experiment"`
    79  	Namespace    string       `json:"namespace"`
    80  	Kind         string       `json:"kind"`
    81  	Message      string       `json:"message"`
    82  	StartTime    *time.Time   `gorm:"index:start_time" json:"start_time"`
    83  	FinishTime   *time.Time   `json:"finish_time"`
    84  	Duration     string       `json:"duration"`
    85  	Pods         []*PodRecord `gorm:"-" json:"pods"`
    86  	ExperimentID string       `gorm:"index:experiment_id" json:"experiment_id"`
    87  }
    88

    92  	CreatedAt time.Time  `json:"created_at"`
    93  	UpdatedAt time.Time  `json:"updated_at"`

    96  	PodIP     string     `gorm:"index:pod_id" json:"pod_ip"`
    97  	PodName   string     `json:"pod_name"`
    98  	Namespace string     `json:"namespace"`
    99  	Message   string     `json:"message"`
   100  	Action    string     `json:"action"`
   101  }
