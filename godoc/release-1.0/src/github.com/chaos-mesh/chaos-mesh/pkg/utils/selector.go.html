    78  // SelectPods returns the list of pods that are available for pod chaos action.
    79  // It returns all pods that match the configured label, annotation and namespace selectors.
    80  // If pods are specifically specified by `selector.Pods`, it just returns the selector.Pods.
    81  func SelectPods(ctx context.Context, c client.Client, r client.Reader, selector v1alpha1.SelectorSpec) ([]v1.Pod, error) {
    82  	// TODO: refactor: make different selectors to replace if-else logics
    83  	var pods []v1.Pod
    84  
    85  	// pods are specifically specified
    86  	if len(selector.Pods) > 0 {
    87  		for ns, names := range selector.Pods {
    88  			if !common.ControllerCfg.ClusterScoped {
    89  				if common.ControllerCfg.TargetNamespace != ns {
    90  					log.Info("skip namespace because ns is out of scope within namespace scoped mode", "namespace", ns)
    91  					continue
    92  				}
    93  			}
    94  			if !IsAllowedNamespaces(ns) {
    95  				log.Info("filter pod by namespaces", "namespace", ns)
    96  				continue
    97  			}
    98  			for _, name := range names {
    99  				var pod v1.Pod
   100  				err := c.Get(ctx, types.NamespacedName{
   101  					Namespace: ns,
   102  					Name:      name,
   103  				}, &pod)
   104  				if err == nil {
   105  					pods = append(pods, pod)
   106  					continue
   107  				}
   108  
   109  				if apierrors.IsNotFound(err) {
   110  					log.Error(err, "Pod is not found", "namespace", ns, "pod name", name)
   111  					continue
   112  				}
   113  
   114  				return nil, err
   115  			}
   116  		}
   117  
   118  		return pods, nil
   119  	}
   120  
   121  	if !common.ControllerCfg.ClusterScoped {
   122  		if len(selector.Namespaces) > 1 {
   123  			return nil, fmt.Errorf("could NOT use more than 1 namespace selector within namespace scoped mode")
   124  		} else if len(selector.Namespaces) == 1 {
   125  			if selector.Namespaces[0] != common.ControllerCfg.TargetNamespace {
   126  				return nil, fmt.Errorf("could NOT list pods from out of scoped namespace: %s", selector.Namespaces[0])
   127  			}
   128  		}
   129  	}
   130  
   131  	var podList v1.PodList
   132  
   133  	var listOptions = client.ListOptions{}
   134  	if !common.ControllerCfg.ClusterScoped {
   135  		listOptions.Namespace = common.ControllerCfg.TargetNamespace
   136  	}
   137  	if len(selector.LabelSelectors) > 0 {
   138  		listOptions.LabelSelector = labels.SelectorFromSet(selector.LabelSelectors)
   139  	}
   140  	if len(selector.FieldSelectors) > 0 {
   141  		listOptions.FieldSelector = fields.SelectorFromSet(selector.FieldSelectors)
   142  		if err := r.List(ctx, &podList, &listOptions); err != nil {
   143  			return nil, err
   144  		}
   145  	} else {
   146  		if err := c.List(ctx, &podList, &listOptions); err != nil {
   147  			return nil, err
   148  		}
   149  	}
   150  	pods = append(pods, podList.Items...)
   151  	var (
   152  		nodes           []v1.Node
   153  		nodeList        v1.NodeList
   154  		nodeListOptions = client.ListOptions{}
   155  	)
   156  	// if both setting Nodes and NodeSelectors, the node list will be combined.
   157  	if len(selector.Nodes) > 0 || len(selector.NodeSelectors) > 0 {
   158  		if len(selector.Nodes) > 0 {
   159  			for _, nodename := range selector.Nodes {
   160  				var node v1.Node
   161  				if err := c.Get(ctx, types.NamespacedName{Name: nodename}, &node); err != nil {
   162  					return nil, err
   163  				}
   164  				nodes = append(nodes, node)
   165  			}
   166  		}
   167  		if len(selector.NodeSelectors) > 0 {
   168  			nodeListOptions.LabelSelector = labels.SelectorFromSet(selector.NodeSelectors)
   169  			if err := c.List(ctx, &nodeList, &nodeListOptions); err != nil {
   170  				return nil, err
   171  			}
   172  			nodes = append(nodes, nodeList.Items...)
   173  		}
   174  		pods = filterPodByNode(pods, nodes)
   175  	}
   176  	pods = filterByNamespaces(pods)
   177  
   178  	namespaceSelector, err := parseSelector(strings.Join(selector.Namespaces, ","))
   179  	if err != nil {
   180  		return nil, err
   181  	}
   182  	pods, err = filterByNamespaceSelector(pods, namespaceSelector)
   183  	if err != nil {
   184  		return nil, err
   185  	}
   186  
   187  	annotationsSelector, err := parseSelector(label.Label(selector.AnnotationSelectors).String())
   188  	if err != nil {
   189  		return nil, err
   190  	}
   191  	pods = filterByAnnotations(pods, annotationsSelector)
   192  
   193  	phaseSelector, err := parseSelector(strings.Join(selector.PodPhaseSelectors, ","))
   194  	if err != nil {
   195  		return nil, err
   196  	}
   197  	pods, err = filterByPhaseSelector(pods, phaseSelector)
   198  	if err != nil {
   199  		return nil, err
   200  	}
   201  
   202  	return pods, nil
   203  }

   207  func CheckPodMeetSelector(pod v1.Pod, selector v1alpha1.SelectorSpec) (bool, error) {
   208  	if len(selector.Pods) > 0 {
   209  		meet := false
   210  		for ns, names := range selector.Pods {
   211  			if pod.Namespace != ns {
   212  				continue
   213  			}
   214  
   215  			for _, name := range names {
   216  				if pod.Name == name {
   217  					meet = true
   218  				}
   219  			}
   220  
   221  			if !meet {
   222  				return false, nil
   223  			}
   224  		}
   225  	}
   226  
   227  	// check pod labels.
   228  	if pod.Labels == nil {
   229  		pod.Labels = make(map[string]string)
   230  	}
   231  
   232  	if selector.LabelSelectors == nil {
   233  		selector.LabelSelectors = make(map[string]string)
   234  	}
   235  
   236  	if len(selector.LabelSelectors) > 0 {
   237  		ls := labels.SelectorFromSet(selector.LabelSelectors)
   238  		podLabels := labels.Set(pod.Labels)
   239  		if len(pod.Labels) == 0 || !ls.Matches(podLabels) {
   240  			return false, nil
   241  		}
   242  	}
   243  
   244  	pods := []v1.Pod{pod}
   245  
   246  	namespaceSelector, err := parseSelector(strings.Join(selector.Namespaces, ","))
   247  	if err != nil {
   248  		return false, err
   249  	}
   250  
   251  	pods, err = filterByNamespaceSelector(pods, namespaceSelector)
   252  	if err != nil {
   253  		return false, err
   254  	}
   255  
   256  	annotationsSelector, err := parseSelector(label.Label(selector.AnnotationSelectors).String())
   257  	if err != nil {
   258  		return false, err
   259  	}
   260  
   261  	pods = filterByAnnotations(pods, annotationsSelector)
   262  
   263  	phaseSelector, err := parseSelector(strings.Join(selector.PodPhaseSelectors, ","))
   264  	if err != nil {
   265  		return false, err
   266  	}
   267  	pods, err = filterByPhaseSelector(pods, phaseSelector)
   268  	if err != nil {
   269  		return false, err
   270  	}
   271  
   272  	if len(pods) > 0 {
   273  		return true, nil
   274  	}
   275  
   276  	return false, nil
   277  }

   294  // filterPodsByMode filters pods by mode from pod list
   295  func filterPodsByMode(pods []v1.Pod, mode v1alpha1.PodMode, value string) ([]v1.Pod, error) {
   296  	if len(pods) == 0 {
   297  		return nil, errors.New("cannot generate pods from empty list")
   298  	}
   299  
   300  	switch mode {
   301  	case v1alpha1.OnePodMode:
   302  		index := rand.Intn(len(pods))
   303  		pod := pods[index]
   304  
   305  		return []v1.Pod{pod}, nil
   306  	case v1alpha1.AllPodMode:
   307  		return pods, nil
   308  	case v1alpha1.FixedPodMode:
   309  		num, err := strconv.Atoi(value)
   310  		if err != nil {
   311  			return nil, err
   312  		}
   313  
   314  		if len(pods) < num {
   315  			num = len(pods)
   316  		}
   317  
   318  		if num <= 0 {
   319  			return nil, errors.New("cannot select any pod as value below or equal 0")
   320  		}
   321  
   322  		return getFixedSubListFromPodList(pods, num), nil
   323  	case v1alpha1.FixedPercentPodMode:
   324  		percentage, err := strconv.Atoi(value)
   325  		if err != nil {
   326  			return nil, err
   327  		}
   328  
   329  		if percentage == 0 {
   330  			return nil, errors.New("cannot select any pod as value below or equal 0")
   331  		}
   332  
   333  		if percentage < 0 || percentage > 100 {
   334  			return nil, fmt.Errorf("fixed percentage value of %d is invalid, Must be (0,100]", percentage)
   335  		}
   336  
   337  		num := int(math.Floor(float64(len(pods)) * float64(percentage) / 100))
   338  
   339  		return getFixedSubListFromPodList(pods, num), nil
   340  	case v1alpha1.RandomMaxPercentPodMode:
   341  		maxPercentage, err := strconv.Atoi(value)
   342  		if err != nil {
   343  			return nil, err
   344  		}
   345  
   346  		if maxPercentage == 0 {
   347  			return nil, errors.New("cannot select any pod as value below or equal 0")
   348  		}
   349  
   350  		if maxPercentage < 0 || maxPercentage > 100 {
   351  			return nil, fmt.Errorf("fixed percentage value of %d is invalid, Must be [0-100]", maxPercentage)
   352  		}
   353  
   354  		percentage := rand.Intn(maxPercentage + 1) // + 1 because Intn works with half open interval [0,n) and we want [0,n]
   355  		num := int(math.Floor(float64(len(pods)) * float64(percentage) / 100))
   356  
   357  		return getFixedSubListFromPodList(pods, num), nil
   358  	default:
   359  		return nil, fmt.Errorf("mode %s not supported", mode)
   360  	}
   361  }

   382  	return filteredList

   385  // filterByPhaseSet filters a list of pods by a given PodPhase selector.
   386  func filterByPhaseSelector(pods []v1.Pod, phases labels.Selector) ([]v1.Pod, error) {
   387  	if phases.Empty() {
   388  		return pods, nil
   389  	}
   390  
   391  	reqs, _ := phases.Requirements()
   392  	var (
   393  		reqIncl []labels.Requirement
   394  		reqExcl []labels.Requirement
   395  
   396  		filteredList []v1.Pod
   397  	)
   398  
   399  	for _, req := range reqs {
   400  		switch req.Operator() {
   401  		case selection.Exists:
   402  			reqIncl = append(reqIncl, req)
   403  		case selection.DoesNotExist:
   404  			reqExcl = append(reqExcl, req)
   405  		default:
   406  			return nil, fmt.Errorf("unsupported operator: %s", req.Operator())
   407  		}
   408  	}
   409  
   410  	for _, pod := range pods {
   411  		included := len(reqIncl) == 0
   412  		selector := labels.Set{string(pod.Status.Phase): ""}
   413  
   414  		// include pod if one including requirement matches
   415  		for _, req := range reqIncl {
   416  			if req.Matches(selector) {
   417  				included = true
   418  				break
   419  			}
   420  		}
   421  
   422  		// exclude pod if it is filtered out by at least one excluding requirement
   423  		for _, req := range reqExcl {
   424  			if !req.Matches(selector) {
   425  				included = false
   426  				break
   427  			}
   428  		}
   429  
   430  		if included {
   431  			filteredList = append(filteredList, pod)
   432  		}
   433  	}
   434  
   435  	return filteredList, nil
   436  }

   553  // RandomFixedIndexes returns the `count` random indexes between `start` and `end`.
   554  // [start, end)
   555  func RandomFixedIndexes(start, end, count uint) []uint {
   556  	var indexes []uint
   557  	m := make(map[uint]uint, count)
   558  
   559  	if end < start {
   560  		return indexes
   561  	}
   562  
   563  	if count > end-start {
   564  		for i := start; i < end; i++ {
   565  			indexes = append(indexes, i)
   566  		}
   567  
   568  		return indexes
   569  	}
   570  
   571  	for i := 0; i < int(count); {
   572  		index := uint(rand.Intn(int(end-start))) + start
   573  
   574  		_, exist := m[index]
   575  		if exist {
   576  			continue
   577  		}
   578  
   579  		m[index] = index
   580  		indexes = append(indexes, index)
   581  		i++
   582  	}
   583  
   584  	return indexes
   585  }
