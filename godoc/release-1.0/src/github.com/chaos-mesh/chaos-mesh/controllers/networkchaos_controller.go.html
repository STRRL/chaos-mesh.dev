    44  // Reconcile reconciles a NetworkChaos resource
    45  func (r *NetworkChaosReconciler) Reconcile(req ctrl.Request) (result ctrl.Result, err error) {
    46  	logger := r.Log.WithValues("reconciler", "networkchaos")
    47  
    48  	if !common.ControllerCfg.ClusterScoped && req.Namespace != common.ControllerCfg.TargetNamespace {
    49  		// NOOP
    50  		logger.Info("ignore chaos which belongs to an unexpected namespace within namespace scoped mode",
    51  			"chaosName", req.Name, "expectedNamespace", common.ControllerCfg.TargetNamespace, "actualNamespace", req.Namespace)
    52  		return ctrl.Result{}, nil
    53  	}
    54  
    55  	reconciler := networkchaos.Reconciler{
    56  		Client:        r.Client,
    57  		Reader:        r.Reader,
    58  		EventRecorder: r.EventRecorder,
    59  		Log:           logger,
    60  	}
    61  
    62  	chaos := &v1alpha1.NetworkChaos{}
    63  	if err := r.Client.Get(context.Background(), req.NamespacedName, chaos); err != nil {
    64  		if apierrors.IsNotFound(err) {
    65  			r.Log.Info("network chaos not found")
    66  		} else {
    67  			r.Log.Error(err, "unable to get network chaos")
    68  		}
    69  		return ctrl.Result{}, nil
    70  	}
    71  
    72  	result, err = reconciler.Reconcile(req, chaos)
    73  	if err != nil {
    74  		if chaos.IsDeleted() || chaos.IsPaused() {
    75  			r.Event(chaos, v1.EventTypeWarning, utils.EventChaosRecoverFailed, err.Error())
    76  		} else {
    77  			r.Event(chaos, v1.EventTypeWarning, utils.EventChaosInjectFailed, err.Error())
    78  		}
    79  	}
    80  
    81  	return result, nil
    82  
    83  }
