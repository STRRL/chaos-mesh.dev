    45  // Reconcile reconciles an IOChaos resource
    46  func (r *IoChaosReconciler) Reconcile(req ctrl.Request) (result ctrl.Result, err error) {
    47  	logger := r.Log.WithValues("reconciler", "iochaos")
    48  
    49  	if !common.ControllerCfg.ClusterScoped && req.Namespace != common.ControllerCfg.TargetNamespace {
    50  		// NOOP
    51  		logger.Info("ignore chaos which belongs to an unexpected namespace within namespace scoped mode",
    52  			"chaosName", req.Name, "expectedNamespace", common.ControllerCfg.TargetNamespace, "actualNamespace", req.Namespace)
    53  		return ctrl.Result{}, nil
    54  	}
    55  
    56  	chaos := &v1alpha1.IoChaos{}
    57  	if err := r.Client.Get(context.Background(), req.NamespacedName, chaos); err != nil {
    58  		if apierrors.IsNotFound(err) {
    59  			r.Log.Info("io chaos not found")
    60  		} else {
    61  			r.Log.Error(err, "unable to get io chaos")
    62  		}
    63  		return ctrl.Result{}, nil
    64  	}
    65  
    66  	scheduler := chaos.GetScheduler()
    67  	duration, err := chaos.GetDuration()
    68  	if err != nil {
    69  		r.Log.Error(err, fmt.Sprintf("unable to get iochaos[%s/%s]'s duration", chaos.Namespace, chaos.Name))
    70  		return ctrl.Result{}, err
    71  	}
    72  	if scheduler == nil && duration == nil {
    73  		return r.commonIoChaos(chaos, req)
    74  	} else if scheduler != nil && duration != nil {
    75  		return r.scheduleIoChaos(chaos, req)
    76  	}
    77  
    78  	if err != nil {
    79  		if chaos.IsDeleted() || chaos.IsPaused() {
    80  			r.Event(chaos, v1.EventTypeWarning, utils.EventChaosRecoverFailed, err.Error())
    81  		} else {
    82  			r.Event(chaos, v1.EventTypeWarning, utils.EventChaosInjectFailed, err.Error())
    83  		}
    84  	}
    85  	return result, nil
    86  }

    88  func (r *IoChaosReconciler) commonIoChaos(chaos *v1alpha1.IoChaos, req ctrl.Request) (ctrl.Result, error) {
    89  	cr := iochaos.NewCommonReconciler(r.Client, r.Reader, r.Log.WithValues("iochaos", req.NamespacedName), req, r.EventRecorder)
    90  	return cr.Reconcile(req)
    91  }

    93  func (r *IoChaosReconciler) scheduleIoChaos(chaos *v1alpha1.IoChaos, req ctrl.Request) (ctrl.Result, error) {
    94  	sr := iochaos.NewTwoPhaseReconciler(r.Client, r.Reader, r.Log.WithValues("iochaos", req.NamespacedName), req, r.EventRecorder)
    95  	return sr.Reconcile(req)
    96  }
