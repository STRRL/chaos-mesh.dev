    91  // Apply implements the reconciler.InnerReconciler.Apply
    92  func (r *Reconciler) Apply(ctx context.Context, req ctrl.Request, chaos v1alpha1.InnerObject) error {
    93  	r.Log.Info("Applying network partition")
    94  
    95  	networkchaos, ok := chaos.(*v1alpha1.NetworkChaos)
    96  	if !ok {
    97  		err := errors.New("chaos is not NetworkChaos")
    98  		r.Log.Error(err, "chaos is not NetworkChaos", "chaos", chaos)
    99  
   100  		return err
   101  	}
   102  
   103  	source := networkchaos.Namespace + "/" + networkchaos.Name
   104  	m := podnetworkmanager.New(source, r.Log, r.Client, r.Reader)
   105  
   106  	sources, err := utils.SelectAndFilterPods(ctx, r.Client, r.Reader, &networkchaos.Spec)
   107  
   108  	if err != nil {
   109  		r.Log.Error(err, "failed to select and filter pods")
   110  		return err
   111  	}
   112  
   113  	var targets []v1.Pod
   114  
   115  	if networkchaos.Spec.Target != nil {
   116  		targets, err = utils.SelectAndFilterPods(ctx, r.Client, r.Reader, networkchaos.Spec.Target)
   117  		if err != nil {
   118  			r.Log.Error(err, "failed to select and filter pods")
   119  			return err
   120  		}
   121  	}
   122  
   123  	sourceSet := ipset.BuildIPSet(sources, []string{}, networkchaos, sourceIPSetPostFix, source)
   124  	externalCidrs, err := netutils.ResolveCidrs(networkchaos.Spec.ExternalTargets)
   125  	if err != nil {
   126  		r.Log.Error(err, "failed to resolve external targets")
   127  		return err
   128  	}
   129  	targetSet := ipset.BuildIPSet(targets, externalCidrs, networkchaos, targetIPSetPostFix, source)
   130  
   131  	allPods := append(sources, targets...)
   132  
   133  	type podPositionTuple struct {
   134  		Pod      v1.Pod
   135  		Position string
   136  	}
   137  	keyPodMap := make(map[types.NamespacedName]podPositionTuple)
   138  	for index, pod := range allPods {
   139  		position := ""
   140  		if index < len(sources) {
   141  			position = "source"
   142  		} else {
   143  			position = "target"
   144  		}
   145  		keyPodMap[types.NamespacedName{
   146  			Name:      pod.Name,
   147  			Namespace: pod.Namespace,
   148  		}] = podPositionTuple{
   149  			Pod:      pod,
   150  			Position: position,
   151  		}
   152  	}
   153  
   154  	// Set up ipset in every related pods
   155  	for index := range allPods {
   156  		pod := allPods[index]
   157  		r.Log.Info("PODS", "name", pod.Name, "namespace", pod.Namespace)
   158  
   159  		t := m.WithInit(types.NamespacedName{
   160  			Name:      pod.Name,
   161  			Namespace: pod.Namespace,
   162  		})
   163  
   164  		t.Append(sourceSet)
   165  		t.Append(targetSet)
   166  	}
   167  
   168  	sourcesChains := []v1alpha1.RawIptables{}
   169  	targetsChains := []v1alpha1.RawIptables{}
   170  	if networkchaos.Spec.Direction == v1alpha1.To || networkchaos.Spec.Direction == v1alpha1.Both {
   171  		sourcesChains = append(sourcesChains, v1alpha1.RawIptables{
   172  			Name:      iptable.GenerateName(pb.Chain_OUTPUT, networkchaos),
   173  			Direction: v1alpha1.Output,
   174  			IPSets:    []string{targetSet.Name},
   175  			RawRuleSource: v1alpha1.RawRuleSource{
   176  				Source: source,
   177  			},
   178  		})
   179  
   180  		targetsChains = append(targetsChains, v1alpha1.RawIptables{
   181  			Name:      iptable.GenerateName(pb.Chain_INPUT, networkchaos),
   182  			Direction: v1alpha1.Input,
   183  			IPSets:    []string{sourceSet.Name},
   184  			RawRuleSource: v1alpha1.RawRuleSource{
   185  				Source: source,
   186  			},
   187  		})
   188  	}
   189  
   190  	if networkchaos.Spec.Direction == v1alpha1.From || networkchaos.Spec.Direction == v1alpha1.Both {
   191  		sourcesChains = append(sourcesChains, v1alpha1.RawIptables{
   192  			Name:      iptable.GenerateName(pb.Chain_INPUT, networkchaos),
   193  			Direction: v1alpha1.Input,
   194  			IPSets:    []string{targetSet.Name},
   195  			RawRuleSource: v1alpha1.RawRuleSource{
   196  				Source: source,
   197  			},
   198  		})
   199  
   200  		targetsChains = append(targetsChains, v1alpha1.RawIptables{
   201  			Name:      iptable.GenerateName(pb.Chain_OUTPUT, networkchaos),
   202  			Direction: v1alpha1.Output,
   203  			IPSets:    []string{sourceSet.Name},
   204  			RawRuleSource: v1alpha1.RawRuleSource{
   205  				Source: source,
   206  			},
   207  		})
   208  	}
   209  	r.Log.Info("chains prepared", "sourcesChains", sourcesChains, "targetsChains", targetsChains)
   210  
   211  	err = r.SetChains(ctx, sources, sourcesChains, m, networkchaos)
   212  	if err != nil {
   213  		return err
   214  	}
   215  
   216  	err = r.SetChains(ctx, targets, targetsChains, m, networkchaos)
   217  	if err != nil {
   218  		return err
   219  	}
   220  
   221  	responses := m.Commit(ctx)
   222  
   223  	networkchaos.Status.Experiment.PodRecords = make([]v1alpha1.PodStatus, 0, len(allPods))
   224  	for _, keyErrorTuple := range responses {
   225  		key := keyErrorTuple.Key
   226  		err := keyErrorTuple.Err
   227  		if err != nil {
   228  			if err != podnetworkmanager.ErrPodNotFound && err != podnetworkmanager.ErrPodNotRunning {
   229  				r.Log.Error(err, "fail to commit")
   230  			} else {
   231  				r.Log.Info("pod is not found or not running", "key", key)
   232  			}
   233  			return err
   234  		}
   235  
   236  		pod := keyPodMap[keyErrorTuple.Key]
   237  		ps := v1alpha1.PodStatus{
   238  			Namespace: pod.Pod.Namespace,
   239  			Name:      pod.Pod.Name,
   240  			HostIP:    pod.Pod.Status.HostIP,
   241  			PodIP:     pod.Pod.Status.PodIP,
   242  			Action:    string(networkchaos.Spec.Action),
   243  		}
   244  		if pod.Position == "source" {
   245  			ps.Message = networkChaosSourceMsg
   246  		} else {
   247  			ps.Message = networkChaosTargetMsg
   248  		}
   249  
   250  		// TODO: add source, target and tc action message
   251  		if networkchaos.Spec.Duration != nil {
   252  			ps.Message += fmt.Sprintf(networkPartitionActionMsg, *networkchaos.Spec.Duration)
   253  		}
   254  		networkchaos.Status.Experiment.PodRecords = append(networkchaos.Status.Experiment.PodRecords, ps)
   255  	}
   256  
   257  	r.Event(networkchaos, v1.EventTypeNormal, utils.EventChaosInjected, "")
   258  	return nil
   259  }

   261  // SetChains sets iptables chains for pods
   262  func (r *Reconciler) SetChains(ctx context.Context, pods []v1.Pod, chains []v1alpha1.RawIptables, m *podnetworkmanager.PodNetworkManager, networkchaos *v1alpha1.NetworkChaos) error {
   263  	for index := range pods {
   264  		pod := &pods[index]
   265  
   266  		key, err := cache.MetaNamespaceKeyFunc(pod)
   267  		if err != nil {
   268  			return err
   269  		}
   270  
   271  		t := m.WithInit(types.NamespacedName{
   272  			Name:      pod.Name,
   273  			Namespace: pod.Namespace,
   274  		})
   275  		for _, chain := range chains {
   276  			t.Append(chain)
   277  		}
   278  
   279  		networkchaos.Finalizers = utils.InsertFinalizer(networkchaos.Finalizers, key)
   280  
   281  	}
   282  	return nil
   283  }

   285  // Recover implements the reconciler.InnerReconciler.Recover
   286  func (r *Reconciler) Recover(ctx context.Context, req ctrl.Request, chaos v1alpha1.InnerObject) error {
   287  	networkchaos, ok := chaos.(*v1alpha1.NetworkChaos)
   288  	if !ok {
   289  		err := errors.New("chaos is not NetworkChaos")
   290  		r.Log.Error(err, "chaos is not NetworkChaos", "chaos", chaos)
   291  
   292  		return err
   293  	}
   294  
   295  	if err := r.cleanFinalizersAndRecover(ctx, networkchaos); err != nil {
   296  		r.Log.Error(err, "cleanFinalizersAndRecover failed")
   297  		return err
   298  	}
   299  	r.Event(networkchaos, v1.EventTypeNormal, utils.EventChaosRecovered, "")
   300  
   301  	return nil
   302  }

   304  func (r *Reconciler) cleanFinalizersAndRecover(ctx context.Context, networkchaos *v1alpha1.NetworkChaos) error {
   305  	var result error
   306  
   307  	source := networkchaos.Namespace + "/" + networkchaos.Name
   308  	m := podnetworkmanager.New(source, r.Log, r.Client, r.Reader)
   309  
   310  	for _, key := range networkchaos.Finalizers {
   311  		ns, name, err := cache.SplitMetaNamespaceKey(key)
   312  		if err != nil {
   313  			result = multierror.Append(result, err)
   314  			continue
   315  		}
   316  
   317  		_ = m.WithInit(types.NamespacedName{
   318  			Namespace: ns,
   319  			Name:      name,
   320  		})
   321  	}
   322  	responses := m.Commit(ctx)
   323  	for _, response := range responses {
   324  		key := response.Key
   325  		err := response.Err
   326  		// if pod not found or not running, directly return and giveup recover.
   327  		if err != nil {
   328  			if err != podnetworkmanager.ErrPodNotFound && err != podnetworkmanager.ErrPodNotRunning {
   329  				r.Log.Error(err, "fail to commit", "key", key)
   330  				continue
   331  			}
   332  
   333  			r.Log.Info("pod is not found or not running", "key", key)
   334  		}
   335  
   336  		networkchaos.Finalizers = utils.RemoveFromFinalizer(networkchaos.Finalizers, response.Key.String())
   337  	}
   338  	r.Log.Info("After recovering", "finalizers", networkchaos.Finalizers)
   339  
   340  	if networkchaos.Annotations[common.AnnotationCleanFinalizer] == common.AnnotationCleanFinalizerForced {
   341  		r.Log.Info("Force cleanup all finalizers", "chaos", networkchaos)
   342  		networkchaos.Finalizers = networkchaos.Finalizers[:0]
   343  		return nil
   344  	}
   345  
   346  	return result
   347  }
