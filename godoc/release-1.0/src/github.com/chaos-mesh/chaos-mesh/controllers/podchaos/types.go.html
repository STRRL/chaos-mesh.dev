    40  // Reconcile reconciles a PodChaos resource
    41  func (r *Reconciler) Reconcile(req ctrl.Request, chaos *v1alpha1.PodChaos) (ctrl.Result, error) {
    42  	r.Log.Info("Reconciling podchaos")
    43  	scheduler := chaos.GetScheduler()
    44  	duration, err := chaos.GetDuration()
    45  	if err != nil {
    46  		r.Log.Error(err, fmt.Sprintf("unable to get podchaos[%s/%s]'s duration", chaos.Namespace, chaos.Name))
    47  		return ctrl.Result{}, err
    48  	}
    49  	if scheduler == nil && duration == nil {
    50  		return r.commonPodChaos(chaos, req)
    51  	} else if scheduler != nil {
    52  		return r.schedulePodChaos(chaos, req)
    53  	}
    54  
    55  	// This should be ensured by admission webhook in the future
    56  	r.Log.Error(fmt.Errorf("podchaos[%s/%s] spec invalid", chaos.Namespace, chaos.Name), "scheduler and duration should be omitted or defined at the same time")
    57  	return ctrl.Result{}, fmt.Errorf("invalid scheduler and duration")
    58  }

    60  func (r *Reconciler) commonPodChaos(podchaos *v1alpha1.PodChaos, req ctrl.Request) (ctrl.Result, error) {
    61  	var pr *common.Reconciler
    62  	switch podchaos.Spec.Action {
    63  	case v1alpha1.PodKillAction:
    64  		return r.notSupportedResponse(podchaos)
    65  	case v1alpha1.ContainerKillAction:
    66  		return r.notSupportedResponse(podchaos)
    67  	case v1alpha1.PodFailureAction:
    68  		pr = podfailure.NewCommonReconciler(r.Client, r.Reader, r.Log.WithValues("action",
    69  			"pod-failure"), r.EventRecorder)
    70  	default:
    71  		return r.invalidActionResponse(podchaos)
    72  	}
    73  	return pr.Reconcile(req)
    74  }
    75  
    76  func (r *Reconciler) schedulePodChaos(podchaos *v1alpha1.PodChaos, req ctrl.Request) (ctrl.Result, error) {
    77  	var tr *twophase.Reconciler
    78  	switch podchaos.Spec.Action {
    79  	case v1alpha1.PodKillAction:
    80  		tr = podkill.NewTwoPhaseReconciler(r.Client, r.Reader, r.Log.WithValues("action",
    81  			"pod-kill"), r.EventRecorder)
    82  	case v1alpha1.PodFailureAction:
    83  		tr = podfailure.NewTwoPhaseReconciler(r.Client, r.Reader, r.Log.WithValues("action",
    84  			"pod-failure"), r.EventRecorder)
    85  	case v1alpha1.ContainerKillAction:
    86  		tr = containerkill.NewTwoPhaseReconciler(r.Client, r.Reader, r.Log.WithValues("action",
    87  			"container-kill"), r.EventRecorder)
    88  	default:
    89  		return r.invalidActionResponse(podchaos)
    90  	}
    91  	return tr.Reconcile(req)
    92  }

    94  func (r *Reconciler) invalidActionResponse(podchaos *v1alpha1.PodChaos) (ctrl.Result, error) {
    95  	r.Log.Error(nil, "podchaos action is invalid", "action", podchaos.Spec.Action)

    99  func (r *Reconciler) notSupportedResponse(podchaos *v1alpha1.PodChaos) (ctrl.Result, error) {
   100  	r.Log.Error(nil, "podchaos action hasn't support duration chaos yet", "action", podchaos.Spec.Action)
   101  	return ctrl.Result{}, fmt.Errorf("unsupported chaos action")
