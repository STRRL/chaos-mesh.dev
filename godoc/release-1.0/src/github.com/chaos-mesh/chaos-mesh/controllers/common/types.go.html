    70  func validate(config *config.ChaosControllerConfig) error {
    71  
    72  	if config.WatcherConfig == nil {
    73  		return fmt.Errorf("required WatcherConfig is missing")
    74  	}
    75  
    76  	if config.ClusterScoped != config.WatcherConfig.ClusterScoped {
    77  		return fmt.Errorf("K8sConfigMapWatcher config ClusterScoped is not same with controller-manager ClusterScoped. k8s configmap watcher: %t, controller manager: %t", config.WatcherConfig.ClusterScoped, config.ClusterScoped)
    78  	}
    79  
    80  	if !config.ClusterScoped {
    81  		if strings.TrimSpace(config.TargetNamespace) == "" {
    82  			return fmt.Errorf("no target namespace specified with namespace scoped mode")
    83  		}
    84  		if !isAllowedNamespaces(config.TargetNamespace, config.AllowedNamespaces, config.IgnoredNamespaces) {
    85  			return fmt.Errorf("target namespace %s is not allowed with filter, please check config AllowedNamespaces and IgnoredNamespaces", config.TargetNamespace)
    86  		}
    87  
    88  		if config.TargetNamespace != config.WatcherConfig.TargetNamespace {
    89  			return fmt.Errorf("K8sConfigMapWatcher config TargertNamespace is not same with controller-manager TargetNamespace. k8s configmap watcher: %s, controller manager: %s", config.WatcherConfig.TargetNamespace, config.TargetNamespace)
    90  		}
    91  	}
    92  
    93  	return nil
    94  }

    97  func isAllowedNamespaces(namespace, allowedNamespace, ignoredNamespace string) bool {
    98  	if allowedNamespace != "" {
    99  		matched, err := regexp.MatchString(allowedNamespace, namespace)
   100  		if err != nil {
   101  			return false
   102  		}
   103  		return matched
   104  	}
   105  
   106  	if ignoredNamespace != "" {
   107  		matched, err := regexp.MatchString(ignoredNamespace, namespace)
   108  		if err != nil {
   109  			return false
   110  		}
   111  		return !matched
   112  	}
   113  
   114  	return true
   115  }

   135  // Reconcile the common chaos
   136  func (r *Reconciler) Reconcile(req ctrl.Request) (ctrl.Result, error) {
   137  	var err error
   138  
   139  	r.Log.Info("Reconciling a common chaos", "name", req.Name, "namespace", req.Namespace)
   140  	ctx := context.Background()
   141  
   142  	chaos := r.Object()
   143  	if err = r.Client.Get(ctx, req.NamespacedName, chaos); err != nil {
   144  		r.Log.Error(err, "unable to get chaos")
   145  		return ctrl.Result{}, err
   146  	}
   147  
   148  	status := chaos.GetStatus()
   149  
   150  	if chaos.IsDeleted() {
   151  		// This chaos was deleted
   152  		r.Log.Info("Removing self")
   153  		if err = r.Recover(ctx, req, chaos); err != nil {
   154  			r.Log.Error(err, "failed to recover chaos")
   155  			updateFailedMessage(ctx, r, chaos, err.Error())
   156  			return ctrl.Result{Requeue: true}, err
   157  		}
   158  		status.Experiment.Phase = v1alpha1.ExperimentPhaseFinished
   159  		status.FailedMessage = emptyString
   160  	} else if chaos.IsPaused() {
   161  		if status.Experiment.Phase == v1alpha1.ExperimentPhaseRunning {
   162  			r.Log.Info("Pausing")
   163  
   164  			if err = r.Recover(ctx, req, chaos); err != nil {
   165  				r.Log.Error(err, "failed to pause chaos")
   166  				updateFailedMessage(ctx, r, chaos, err.Error())
   167  				return ctrl.Result{Requeue: true}, err
   168  			}
   169  			now := time.Now()
   170  			status.Experiment.EndTime = &metav1.Time{
   171  				Time: now,
   172  			}
   173  			if status.Experiment.StartTime != nil {
   174  				status.Experiment.Duration = now.Sub(status.Experiment.StartTime.Time).String()
   175  			}
   176  		}
   177  		status.Experiment.Phase = v1alpha1.ExperimentPhasePaused
   178  		status.FailedMessage = emptyString
   179  	} else if status.Experiment.Phase == v1alpha1.ExperimentPhaseRunning {
   180  		r.Log.Info("The common chaos is already running", "name", req.Name, "namespace", req.Namespace)
   181  		return ctrl.Result{}, nil
   182  	} else {
   183  		// Start chaos action
   184  		r.Log.Info("Performing Action")
   185  
   186  		if err = r.Apply(ctx, req, chaos); err != nil {
   187  			r.Log.Error(err, "failed to apply chaos action")
   188  			updateFailedMessage(ctx, r, chaos, err.Error())
   189  
   190  			status.Experiment.Phase = v1alpha1.ExperimentPhaseFailed
   191  
   192  			updateError := retry.RetryOnConflict(retry.DefaultRetry, func() error {
   193  				return r.Update(ctx, chaos)
   194  			})
   195  			if updateError != nil {
   196  				r.Log.Error(updateError, "unable to update chaos finalizers")
   197  				updateFailedMessage(ctx, r, chaos, updateError.Error())
   198  			}
   199  
   200  			return ctrl.Result{Requeue: true}, err
   201  		}
   202  		status.Experiment.StartTime = &metav1.Time{
   203  			Time: time.Now(),
   204  		}
   205  		status.Experiment.Phase = v1alpha1.ExperimentPhaseRunning
   206  		status.FailedMessage = emptyString
   207  	}
   208  
   209  	if err := r.Update(ctx, chaos); err != nil {
   210  		r.Log.Error(err, "unable to update chaos status")
   211  		return ctrl.Result{}, err
   212  	}
   213  
   214  	return ctrl.Result{}, nil
   215  }

   217  func updateFailedMessage(
